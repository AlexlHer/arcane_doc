95void CartesianMeshTestUtils::
96_computeCenters()
97{
98 IMesh* mesh = m_mesh;
99
100 // Calcule le centre des mailles
101 {
102 VariableNodeReal3& nodes_coord = mesh->nodesCoordinates();
103 ENUMERATE_CELL(icell,m_mesh->allCells()){
104 Cell cell = *icell;
105 Real3 center;
106 for( NodeEnumerator inode(cell.nodes()); inode.hasNext(); ++inode )
107 center += nodes_coord[inode];
108 center /= cell.nbNode();
109 m_cell_center[icell] = center;
110 }
111 }
112
113 // Calcule le centre des faces
114 {
115 VariableNodeReal3& nodes_coord = mesh->nodesCoordinates();
116 ENUMERATE_FACE(iface,m_mesh->allFaces()){
117 Face face = *iface;
118 Real3 center;
119 for( NodeEnumerator inode(face.nodes()); inode.hasNext(); ++inode )
120 center += nodes_coord[inode];
121 center /= face.nbNode();
122 m_face_center[iface] = center;
123 }
124 }
125}

164void CartesianMeshTestUtils::
165_testDirCell()
166{
167 info() << "TEST_DIR_CELL";
168
169 IMesh* mesh = m_mesh;
170 Integer nb_dir = mesh->dimension();
171 Integer nb_print = m_nb_print;
172 CellDirectionMng cdm2;
173 CellDirectionMng cdm3;
174 for( Integer idir=0; idir<nb_dir; ++idir){
175 CellDirectionMng cdm(m_cartesian_mesh->cellDirection(idir));
176 cdm2 = m_cartesian_mesh->cellDirection(idir);
177 cdm3 = cdm;
178 eMeshDirection md = cdm.direction();
179 Integer iprint = 0;
180 info() << "DIRECTION=" << idir << " Cells=" << cdm.allCells().name();
181 _checkItemGroupIsSorted(cdm.allCells());
182 ENUMERATE_CELL(icell,cdm.allCells()){
183 Cell cell = *icell;
184 DirCell dir_cell(cdm[icell]);
185 Cell prev_cell = dir_cell.previous();
186 Cell next_cell = dir_cell.next();
187 if (prev_cell.null() && next_cell.null())
188 ARCANE_FATAL("Null previous and next cell");
189 DirCell dir_cell2(cdm2[icell]);
190 Cell prev_cell2 = dir_cell2.previous();
191 Cell next_cell2 = dir_cell2.next();
192 DirCell dir_cell3(cdm3[icell]);
193 Cell prev_cell3 = dir_cell3.previous();
194 Cell next_cell3 = dir_cell3.next();
195 _checkSameId(prev_cell,prev_cell2);
196 _checkSameId(next_cell,next_cell2);
197 _checkSameId(prev_cell,prev_cell3);
198 _checkSameId(next_cell,next_cell3);
199 if (nb_print<0 || iprint<nb_print){
200 ++iprint;
201 if (!prev_cell.null() && !next_cell.null()){
202 info() << "Cell uid=" << ItemPrinter(cell) << " dir=" << md
203 << " prev=" << ItemPrinter(prev_cell) << " xyz=" << m_cell_center[prev_cell]
204 << " next=" << ItemPrinter(next_cell) << " xyz=" << m_cell_center[next_cell];
205 }
206 else{
207 info() << "Cell uid=" << ItemPrinter(cell) << " dir=" << md;
208 if (!prev_cell.null())
209 info() << " prev=" << ItemPrinter(prev_cell) << " xyz=" << m_cell_center[prev_cell];
210 if (!next_cell.null())
211 info() << " next=" << ItemPrinter(next_cell) << " xyz=" << m_cell_center[next_cell];
212 }
213 }
214 }
215 }

233void CartesianMeshTestUtils::
234_testDirFace(int idir)
235{
236 // Teste l'utilisation de DirFace et vérifie que
237 // les valeurs sont correctes: pour une face donnée,
238 // il faut que selon la direction choisie, la coordonnée du centre
239 // de la maille avant (previousCell()) soit inférieure à celle
240 // de la maille après (nextCell()) et aussi celle de la maille avant
241 // soit inférieure à ce
242 Integer nb_print = m_nb_print;
243 Integer nb_error = 0;
244 Integer iprint = 0;
245
246 FaceDirectionMng fdm2;
247 FaceDirectionMng fdm(m_cartesian_mesh->faceDirection(idir));
248 fdm2 = fdm;
249 eMeshDirection md = fdm.direction();
250 info() << "TEST_DIR_FACE for direction=" << idir << " -> " << (eMeshDirection)idir;
251 _checkItemGroupIsSorted(fdm.allFaces());
252 ENUMERATE_FACE(iface,fdm.allFaces()){
253 Face face = *iface;
254 DirFace dir_face(fdm[iface]);
255 DirFace dir_face2(fdm2[iface]);
256 //Integer nb_node = cell.nbNode();
257 Cell prev_cell = dir_face.previousCell();
258 Cell next_cell = dir_face.nextCell();
259 Cell prev_cell2 = dir_face2.previousCell();
260 Cell next_cell2 = dir_face2.nextCell();
261 _checkSameId(prev_cell,prev_cell2);
262 _checkSameId(next_cell,next_cell2);
263 _checkSameId(prev_cell,dir_face.previousCellId());
264 _checkSameId(next_cell,dir_face.nextCellId());
265 bool is_print = (nb_print<0 || iprint<nb_print);
266 ++iprint;
267 Real face_coord = m_face_center[iface][idir];
268 if (!prev_cell.null() && !next_cell.null()){
269 Real next_coord = m_cell_center[next_cell][idir];
270 Real prev_coord = m_cell_center[prev_cell][idir];
271 if (next_coord<prev_coord){
272 info() << "Bad ordering for face";
273 is_print = true;
274 ++nb_error;
275 }
276 if (is_print)
277 info() << "Face uid=" << ItemPrinter(face) << " dir=" << md << " xyz=" << m_face_center[iface]
278 << " prev=" << prev_cell.uniqueId() << " xyz=" << m_cell_center[prev_cell]
279 << " next=" << next_cell.uniqueId() << " xyz=" << m_cell_center[next_cell];
280 }
281 else{
282 if (!prev_cell.null()){
283 Real prev_coord = m_cell_center[prev_cell][idir];
284 if (face_coord<prev_coord){
285 info() << "Bad ordering for face";
286 is_print = true;
287 ++nb_error;
288 }
289 if (is_print)
290 info() << "Face uid=" << ItemPrinter(face) << " dir=" << md << " xyz=" << m_face_center[iface]
291 << " prev=" << prev_cell.uniqueId() << " xyz=" << m_cell_center[prev_cell];
292 }
293 if (!next_cell.null()){
294 Real next_coord = m_cell_center[next_cell][idir];
295 if (next_coord<face_coord){
296 info() << "Bad ordering for face";
297 is_print = true;
298 ++nb_error;
299 }
300 if (is_print)
301 info() << "Face uid=" << ItemPrinter(face) << " dir=" << md << " xyz=" << m_face_center[iface]
302 << " next=" << next_cell.uniqueId() << " xyz=" << m_cell_center[next_cell];
303 }
304 }
305 }
306 if (nb_error!=0)
307 ARCANE_FATAL("Bad connectivity for DirFace nb_error={0}",nb_error);
308}

313void CartesianMeshTestUtils::
314_testDirNode()
315{
316 info() << "TEST_DIR_NODE";
317 m_node_density.fill(0.0);
318
319 IMesh* mesh = m_mesh;
320 Integer nb_dir = mesh->dimension();
321 Integer nb_print = m_nb_print;
322 Integer nb_error = 0;
323 VariableNodeReal3& nodes_coord = mesh->nodesCoordinates();
324 // Pour tester l'opérateur 'operator='
325 NodeDirectionMng node_dm2;
326 NodeDirectionMng node_dm3;
327 for( Integer idir=0; idir<nb_dir; ++idir){
328 NodeDirectionMng node_dm(m_cartesian_mesh->nodeDirection(idir));
329 node_dm2 = m_cartesian_mesh->nodeDirection(idir);
330 node_dm3 = node_dm;
331 eMeshDirection md = node_dm.direction();
332 Integer iprint = 0;
333 info() << "DIRECTION=" << idir;
334 NodeGroup dm_all_nodes = node_dm.allNodes();
335 _checkItemGroupIsSorted(dm_all_nodes);
336 ENUMERATE_NODE(inode,dm_all_nodes){
337 Node node = *inode;
338 m_node_density[inode] += 1.0;
339 DirNode dir_node(node_dm[inode]);
340 DirNode dir_node2(node_dm2[inode]);
341 DirNode dir_node3(node_dm3[inode]);
342 Node prev_node = dir_node.previous();
343 Node next_node = dir_node.next();
344 Node prev_node2 = dir_node2.previous();
345 Node next_node2 = dir_node2.next();
346 Node prev_node3 = dir_node3.previous();
347 Node next_node3 = dir_node3.next();
348 _checkSameId(prev_node,prev_node2);
349 _checkSameId(next_node,next_node2);
350 _checkSameId(prev_node,prev_node3);
351 _checkSameId(next_node,next_node3);
352 _checkSameId(prev_node,dir_node.previousId());
353 _checkSameId(next_node,dir_node.nextId());
354 Real my_coord = nodes_coord[inode][idir];
355 bool is_print = (nb_print<0 || iprint<nb_print);
356 ++iprint;
357 if (is_print){
358 Int32 node_nb_cell = node.nbCell();
359 info() << "DirNode node= " << ItemPrinter(node) << " nb_cell=" << node_nb_cell << " pos=" << nodes_coord[node];
360 for( Integer k=0; k<node_nb_cell; ++k ){
361 Real3 cell_pos = m_cell_center[node.cell(k)];
362 info() << "Node k=" << k << " cell_pos=" << cell_pos << " cell=" << ItemPrinter(node.cell(k));
363 }
364 for( Integer k=0; k<8; ++k ){
365 Int32 cell_index = dir_node.cellIndex(k);
366 Real3 cell_pos;
367 if (cell_index!=(-1)){
368 if ((1+cell_index)>node_nb_cell)
369 ARCANE_FATAL("Bad value for cell_index '{0}' node_nb_cell={1}",cell_index,node_nb_cell);
370 cell_pos = m_cell_center[node.cell(cell_index)];
371 }
372 info() << "DirNode cellIndex k=" << k << " index=" << cell_index << " pos=" << cell_pos
373 << " cell_lid=" << dir_node.cellId(k)
374 << " cell=" << dir_node.cell(k);
375 }
376 info() << "DirNode direct "
377 << " " << dir_node.nextLeftCellId()
378 << " " << dir_node.nextRightCellId()
379 << " " << dir_node.previousRightCellId()
380 << " " << dir_node.previousLeftCellId()
381 << " " << dir_node.topNextLeftCellId()
382 << " " << dir_node.topNextRightCellId()
383 << " " << dir_node.topPreviousRightCellId()
384 << " " << dir_node.topPreviousLeftCellId();
385 info() << "DirNode direct "
386 << " " << ItemPrinter(dir_node.nextLeftCell())
387 << " " << ItemPrinter(dir_node.nextRightCell())
388 << " " << ItemPrinter(dir_node.previousRightCell())
389 << " " << ItemPrinter(dir_node.previousLeftCell())
390 << " " << ItemPrinter(dir_node.topNextLeftCell())
391 << " " << ItemPrinter(dir_node.topNextRightCell())
392 << " " << ItemPrinter(dir_node.topPreviousRightCell())
393 << " " << ItemPrinter(dir_node.topPreviousLeftCell());
394 }
395 if (prev_node.null() && next_node.null())
396 ARCANE_FATAL("Null previous and next node for node {0}",ItemPrinter(node));
397 //TODO: Vérifier que les coordonnées autres que celle de idir sont bien les mêmes pour next,prev et my_coord.
398 if (!prev_node.null() && !next_node.null()){
399 Real next_coord = nodes_coord[next_node][idir];
400 Real prev_coord = nodes_coord[prev_node][idir];
401 if (next_coord<prev_coord){
402 info() << "Bad ordering for node";
403 is_print = true;
404 ++nb_error;
405 }
406 if (is_print)
407 info() << "Node uid=" << ItemPrinter(node) << " dir=" << md
408 << " prev=" << ItemPrinter(prev_node) << " xyz=" << prev_coord
409 << " next=" << ItemPrinter(next_node) << " xyz=" << next_coord;
410 }
411 else{
412 if (!prev_node.null()){
413 Real prev_coord = nodes_coord[prev_node][idir];
414 if (my_coord<prev_coord){
415 info() << "Bad ordering for node";
416 is_print = true;
417 ++nb_error;
418 }
419 if (is_print)
420 info() << "Node uid=" << ItemPrinter(node) << " dir=" << md
421 << " prev=" << ItemPrinter(prev_node) << " xyz=" << nodes_coord[prev_node];
422 }
423 if (!next_node.null()){
424 Real next_coord = nodes_coord[next_node][idir];
425 if (next_coord<my_coord){
426 info() << "Bad ordering for node";
427 is_print = true;
428 ++nb_error;
429 }
430 if (is_print)
431 info() << "Node uid=" << ItemPrinter(node) << " dir=" << md
432 << " next=" << ItemPrinter(next_node) << " xyz=" << nodes_coord[next_node];
433 }
434 }
435 }
436 }
437
438 if (nb_error!=0)
439 ARCANE_FATAL("Bad connectivity for DirNode");
440
441 // Vérifie qu'on a parcouru tous les noeuds
442 ENUMERATE_NODE(inode,m_mesh->allNodes()){
443 Node node = *inode;
444 if (node.cell(0).level()!=0)
445 continue;
446 if (m_node_density[inode]==0.0){
447 ++nb_error;
448 info() << "Bad value for node " << ItemPrinter(node);
449 }
450 }
451 if (nb_error!=0)
452 ARCANE_FATAL("Bad values for DirNode");
453}

533void CartesianMeshTestUtils::
534_testNodeToCellConnectivity3D()
535{
536 info() << "Test NodeToCell Connectivity3D";
537 IMesh* mesh = m_mesh;
538 VariableNodeReal3& nodes_coord = mesh->nodesCoordinates();
539 CartesianConnectivity cc = m_cartesian_mesh->connectivity();
540 ENUMERATE_NODE(inode,m_mesh->allNodes()){
541 Node node = *inode;
542 Real3 node_coord = nodes_coord[inode];
543 info() << "node_uid=" << node.uniqueId()
544 << " UL=" << cc.upperLeft(node).localId()
545 << " UR=" << cc.upperRight(node).localId()
546 << " LR=" << cc.lowerRight(node).localId()
547 << " LL=" << cc.lowerLeft(node).localId()
548 << " TUL=" << cc.topZUpperLeft(node).localId()
549 << " TUR=" << cc.topZUpperRight(node).localId()
550 << " TLR=" << cc.topZLowerRight(node).localId()
551 << " TLL=" << cc.topZLowerLeft(node).localId();
552 {
553 Cell upper_left = cc.upperLeft(node);
554 if (!upper_left.null()){
555 Real3 c = m_cell_center[upper_left];
556 if (c.y<=node_coord.y || c.x>=node_coord.x || c.z>=node_coord.z)
557 ARCANE_FATAL("Bad upperLeft cell for node={0} c={1} n={2}",ItemPrinter(node),c,node_coord);
558 }
559 }
560 {
561 Cell upper_right = cc.upperRight(node);
562 if (!upper_right.null()){
563 Real3 c = m_cell_center[upper_right];
564 if (c.y<=node_coord.y || c.x<=node_coord.x || c.z>=node_coord.z)
565 ARCANE_FATAL("Bad upperRight cell for node={0} c={1} n={2}",ItemPrinter(node),c,node_coord);
566 }
567 }
568 {
569 Cell lower_right = cc.lowerRight(node);
570 if (!lower_right.null()){
571 Real3 c = m_cell_center[lower_right];
572 if (c.y>=node_coord.y || c.x<=node_coord.x || c.z>=node_coord.z)
573 ARCANE_FATAL("Bad lowerRight cell for node={0} c={1} n={2}",ItemPrinter(node),c,node_coord);
574 }
575 }
576 {
577 Cell lower_left = cc.lowerLeft(node);
578 if (!lower_left.null()){
579 Real3 c = m_cell_center[lower_left];
580 if (c.y>=node_coord.y || c.x>=node_coord.x || c.z>=node_coord.z)
581 ARCANE_FATAL("Bad lowerLeft cell for node={0} c={1} n={2}",ItemPrinter(node),c,node_coord);
582 }
583 }
584 {
585 Cell top_upper_left = cc.topZUpperLeft(node);
586 if (!top_upper_left.null()){
587 Real3 c = m_cell_center[top_upper_left];
588 if (c.y<=node_coord.y || c.x>=node_coord.x || c.z<=node_coord.z)
589 ARCANE_FATAL("Bad topZUpperLeft cell for node={0} c={1} n={2}",ItemPrinter(node),c,node_coord);
590 }
591 }
592 {
593 Cell top_upper_right = cc.topZUpperRight(node);
594 if (!top_upper_right.null()){
595 Real3 c = m_cell_center[top_upper_right];
596 if (c.y<=node_coord.y || c.x<=node_coord.x || c.z<=node_coord.z)
597 ARCANE_FATAL("Bad topZUpperRight cell for node={0} c={1} n={2}",ItemPrinter(node),c,node_coord);
