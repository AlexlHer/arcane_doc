80void DynamicMeshChecker::
81checkValidMesh()
82{
83 info(4) << "Check mesh coherence.";
84 checkValidConnectivity();
85
86 for( IItemFamilyCollection::Enumerator i(m_mesh->itemFamilies()); ++i; ){
87 IItemFamily* family = *i;
88 //GG: regarder pourquoi il y a ce test.
89 if (family->itemKind()>=NB_ITEM_KIND)
90 continue;
91 family->checkValid();
92 }
93 mesh_utils::checkMeshProperties(m_mesh,false,false,true);
94
95 // Sauve dans une variable aux mailles la liste des faces et
96 // noeuds qui la composent.
97 // Ceci n'est utile que pour des vérifications
98 // NOTE: pour l'instant mis a false car fait planter si utilise
99 // partitionnement initial dans Arcane
100 bool global_check = false;
101 if (m_check_level>=1 && global_check){
102 if (!m_var_cells_faces)
103 m_var_cells_faces = new VariableCellArrayInt64(VariableBuildInfo(m_mesh->subDomain(),
104 "MeshTestCellFaces"));
105
106 if (!m_var_cells_nodes)
107 m_var_cells_nodes = new VariableCellArrayInt64(VariableBuildInfo(m_mesh->subDomain(),
108 "MeshTestCellNodes"));
109
110 CellGroup all_cells(m_mesh->allCells());
111 Integer max_nb_face = 0;
112 Integer max_nb_node = 0;
113 ENUMERATE_CELL(i,all_cells){
114 Cell cell = *i;
115 Integer nb_face = cell.nbFace();
116 Integer nb_node = cell.nbNode();
117 if (nb_face>max_nb_face)
118 max_nb_face = nb_face;
119 if (nb_node>max_nb_node)
120 max_nb_node = nb_node;
121 }
122 m_var_cells_faces->resize(max_nb_face);
123 m_var_cells_nodes->resize(max_nb_node);
124 ENUMERATE_CELL(i,all_cells){
125 Cell cell = *i;
126 Integer nb_face = cell.nbFace();
127 for( Integer z=0; z<nb_face; ++z )
128 (*m_var_cells_faces)[cell][z] = cell.face(z).uniqueId().asInt64();
129 Integer nb_node = cell.nbNode();
130 for( Integer z=0; z<nb_node; ++z )
131 (*m_var_cells_nodes)[cell][z] = cell.node(z).uniqueId().asInt64();
132 }
133 }
134 checkValidMeshFull();
135}

140void DynamicMeshChecker::
141checkValidMeshFull()
142{
143 String func_name("DynamicMesh::checkValidMeshFull");
144 info() << func_name << " on " << m_mesh->name();
145 Integer nb_error = 0;
146 VariableFaceInt64 var_faces(VariableBuildInfo(m_mesh,"MeshCheckFaces"));
147 debug() << " VAR_FACE GROUP=" << var_faces.variable()->itemFamily()
148 << " NAME=" << var_faces.itemGroup().name()
149 << " FAMILY_NAME=" << var_faces.variable()->itemFamilyName()
150 << " GROUP_NAME=" << var_faces.variable()->itemGroupName();
151 ENUMERATE_FACE(iface,m_mesh->allFaces()){
152 Face face = *iface;
153 Cell c = face.backCell();
154 Int64 uid = c.null() ? NULL_ITEM_UNIQUE_ID : c.uniqueId();
155 var_faces[face] = uid;
156 }
157 var_faces.synchronize();
158 ENUMERATE_FACE(iface,m_mesh->allFaces()){
159 Face face = *iface;
160 Cell c = face.backCell();
161 Int64 uid = c.null() ? NULL_ITEM_UNIQUE_ID : c.uniqueId();
162 if (uid!=var_faces[face] && uid!=NULL_ITEM_ID && var_faces[face]!=NULL_ITEM_ID){
163 error() << func_name << " bad back cell in face uid=" << face.uniqueId();
164 ++nb_error;
165 }
166 }
167
168 ENUMERATE_FACE(iface,m_mesh->allFaces()){
169 Face face = *iface;
170 Cell c = face.frontCell();
171 Int64 uid = c.null() ? NULL_ITEM_UNIQUE_ID : c.uniqueId();
172 var_faces[face] = uid;
173 }
174 var_faces.synchronize();
175 ENUMERATE_FACE(iface,m_mesh->allFaces()){
176 Face face = *iface;
177 Cell c = face.frontCell();
178 Int64 uid = c.null() ? NULL_ITEM_UNIQUE_ID : c.uniqueId();
179 if (uid!=var_faces[face] && uid!=NULL_ITEM_ID && var_faces[face]!=NULL_ITEM_ID){
180 error() << func_name << " bad front cell in face uid=" << face.uniqueId();
181 ++nb_error;
182 }
183 }
184
185 ENUMERATE_CELL(icell,m_mesh->allCells()){
186 Cell cell = *icell;
187 Integer cell_type = cell.type();
188 if (cell_type == IT_Line2) {
189 error() << func_name << " bad cell type in face uid=" << cell.uniqueId();
190 ++nb_error;
191 }
192 }
193
194 ENUMERATE_FACE(iface,m_mesh->allFaces()){
195 Face face = *iface;
196 Integer face_type = face.type();
197 if (face_type == IT_Vertex) {
198 error() << func_name << " bad face type in face uid=" << face.uniqueId();
199 ++nb_error;
200 }
201 }
202
203 if (nb_error!=0)
204 ARCANE_FATAL("Invalid mesh");
205}

207/*---------------------------------------------------------------------------*/
208/*---------------------------------------------------------------------------*/
209/*
210 * \brief Vérifie que la connectivité est valide.
211 *
212 * Les vérifications portent sur les points suivants:
213 * - pas d'entités du maillage ayant un indice nul.
214 * - pour les faces, vérifie qu'il existe au moins une frontCell ou une
215 * backCell. De plus, si elle a deux mailles, la backCell doit
216 * toujours être la première.
217 * - les noeuds et les faces doivent avoir le même propriétaire
218 * qu'une des mailles attenante.
219 * - vérifie que les faces sont bien ordonnées et orientées
220 */
221void DynamicMeshChecker::
222checkValidConnectivity()
223{
224 String func_name = "DynamicMesh::checkValidConnectivity";
225 debug() << func_name << " check";
226
227 // Appelle la méthode de vérification de chaque famille.
228 IItemFamilyCollection item_families = m_mesh->itemFamilies();
229 for( auto x = item_families.enumerator(); ++x; ){
230 IItemFamily* f = *x;
231 f->checkValidConnectivity();
232 }
233
234 if (!m_mesh->parentMesh()){
235 Integer index = 0;
236 ENUMERATE_FACE(i,m_mesh->allFaces()){
237 Face elem = *i;
238 Cell front_cell = elem.frontCell();
239 Cell back_cell = elem.backCell();
240 if (front_cell.null() && back_cell.null())
241 ARCANE_FATAL("Local face '{0}' has two null cell face=",index,ItemPrinter(elem));
242 index++;
243 }
244 }
245
246 // Vérifie que la connective maille<->noeud
247 // est réciproque
248 ENUMERATE_NODE(inode,m_mesh->allNodes()){
249 Node node = *inode;
250 for( CellEnumerator icell(node.cells()); icell(); ++icell ){
251 Cell cell = *icell;
252 bool has_found = false;
253 for( NodeEnumerator inode2(cell.nodes()); inode2(); ++inode2 ){
254 Node node2 = *inode2;
255 if (node2==node){
256 has_found = true;
257 break;
258 }
259 }
260 if (!has_found){
261 ARCANE_FATAL("Node uid={0} is connected to the cell uid={1} but the cell"
262 " is not connected to this node.",ItemPrinter(node),ItemPrinter(cell));
263 }
264 }
265 }
266
267 // ATT: les verifications suivantes ne sont pas compatible avec l'AMR
268 // TODO : etendre les verifs au cas AMR
270 if(!m_mesh->isAmrActivated()){
271 // Vérifie que la connective maille<->arêtes
272 // est réciproque
273 ENUMERATE_EDGE(iedge,m_mesh->allEdges()){
274 Edge edge = *iedge;
275 for( CellEnumerator icell(edge.cells()); icell(); ++icell ){
276 Cell cell = *icell;
277 bool has_found = false;
278 for( EdgeEnumerator iedge2(cell.edges()); iedge2(); ++iedge2 ){
279 Edge edge2 = *iedge2;
280 if (edge2==edge){
281 has_found = true;
282 break;
283 }
284 }
285 if (!has_found){
286 ARCANE_FATAL("Edge uid={0} is connected to the cell uid={1} but the cell"
287 " is not connected to this edge.",ItemPrinter(edge),ItemPrinter(cell));
288 }
289 }
290 }
291 }
292 // Vérifie que la connective maille<->face
293 // est réciproque
294 ENUMERATE_FACE(iface,m_mesh->allFaces()){
295 Face face = *iface;
296 if(!m_mesh->isAmrActivated()){
297 for( CellEnumerator icell(face.cells()); icell(); ++icell ){
298 Cell cell = *icell;
299 bool has_found = false;
300 for( FaceEnumerator iface2(cell.faces()); iface2(); ++iface2 ){
301 Face face2 = *iface2;
302 if (face2==face){
303 has_found = true;
304 break;
305 }
306 }
307 if (!has_found){
308 ARCANE_FATAL("Face uid={0} is connected to the cell uid={1} but the cell"
309 " is not connected to this face.",ItemPrinter(face),ItemPrinter(cell));
310 }
311 }
312 }
313 for( CellEnumerator icell(face.cells()); icell(); ++icell ){
314 Cell cell = *icell;
315 bool has_found = false;
316 for( FaceEnumerator iface2(cell.faces()); iface2(); ++iface2 ){
317 Face face2 = *iface2;
318 if (face2==face){
319 has_found = true;
320 break;
321 }
322 }
323 if (!has_found && (face.backCell().level() == face.frontCell().level())){
324 warning() << func_name << ". The face " << FullItemPrinter(face)
325 << " is connected to the cell " << FullItemPrinter(cell)
326 << " but the cell (level "<< cell.level()<< ")"
327 << " is not connected to the face.";
328 }
329 }
330
331 Integer nb_cell = face.nbCell();
332 Cell back_cell = face.backCell();
333 Cell front_cell = face.frontCell();
334 if ((back_cell.null() || front_cell.null()) && nb_cell==2)
335 ARCANE_FATAL("Face uid='{0}' bad number of cells face",ItemPrinter(face));
336 // Si on a deux mailles connectées, alors la back cell doit être la première
337 if (nb_cell==2 && back_cell!=face.cell(0))
338 ARCANE_FATAL("Bad face face.backCell()!=face.cell(0) face={0} back_cell={1} from_cell={2} cell0={3}",
339 ItemPrinter(face),ItemPrinter(back_cell),ItemPrinter(front_cell),ItemPrinter(face.cell(0)));
340 }
341
342 _checkFacesOrientation();
343
344 if (m_mesh->parentMesh()){
345 Integer nerror = 0;
346 eItemKind kinds[] = { IK_Node, IK_Edge, IK_Face, IK_Cell };
347 Integer nb_kind = sizeof(kinds)/sizeof(eItemKind);
348
349 for(Integer i_kind=0;i_kind<nb_kind;++i_kind){
350 const eItemKind kind = kinds[i_kind];
351 IItemFamily * family = m_mesh->itemFamily(kind);
352 if (!family->parentFamily()){
353 error() << "Mesh " << m_mesh->name() << " : Family " << kind << " does not exist in mesh";
354 ++nerror;
355 }
356 else{
357 ENUMERATE_ITEM(iitem,family->allItems()){
358 const Item & item = *iitem;
359 const Item & parent_item = item.parent();
360 if (parent_item.internal()->isSuppressed()){
361 error() << "Mesh " << m_mesh->name() << " : Inconsistent suppresssed parent item uid : "
362 << ItemPrinter(item) << " / " << ItemPrinter(parent_item);
363 ++nerror;
364 }
365 if (parent_item.uniqueId() != item.uniqueId()){
366 error() << "Mesh " << m_mesh->name() << " : Inconsistent item/parent item uid : " << ItemPrinter(item);
367 ++nerror;
368 }
369 }
370 }
371 }
372 if (nerror > 0)
373 ARCANE_FATAL("Mesh name={0} has {1} (see above)",m_mesh->name(),String::plural(nerror, "error"));
374
375 // Vérifie la consistence parallèle du groupe parent
376 nerror = 0;
377 {
378 ItemGroup parent_group = m_mesh->parentGroup();
379 String var_name = String::format("SubMesh_{0}_GroupConsistencyChecker",parent_group.name());
380 TemporaryVariableBuildInfo tvbi(m_mesh->parentMesh(),var_name,parent_group.itemFamily()->name());
381 ItemVariableScalarRefT<Integer> var(tvbi,m_mesh->parentGroup().itemKind());
382 var.fill(-1);
383 ENUMERATE_ITEM(iitem, m_mesh->parentGroup()){
384 var[iitem] = iitem->owner();
385 }
386 nerror += var.checkIfSync(10);
387 }
388 if (nerror > 0)
389 ARCANE_FATAL("Mesh name={0} has parent group consistency {1}\n"
390 "This usually means that parent group was not symmetrically built",
391 m_mesh->name(),String::plural(nerror, "error", false));
392 }
393
394 _checkValidItemOwner(m_mesh->nodeFamily());
395 _checkValidItemOwner(m_mesh->edgeFamily());
396 _checkValidItemOwner(m_mesh->faceFamily());
397 _checkValidItemOwner(m_mesh->cellFamily());
398}

444 fr.checkAndChangeOrientationAMR(face);
445 }
446 }
447 }

456void DynamicMeshChecker::
457_checkFacesOrientation()
458{
459 bool is_1d = (m_mesh->dimension()==1);
460 //Temporaire: pour l'instant, ne teste pas l'orientation en 1D.
461 if (is_1d)
462 return;
463
464 String func_name = "DynamicMesh::_checkFacesOrientation";
465
466 Int64UniqueArray m_work_face_sorted_nodes;
467 IntegerUniqueArray m_work_face_nodes_index;
468 Int64UniqueArray m_work_face_orig_nodes_uid;
469
470
471 ENUMERATE_CELL(icell,m_mesh->allCells()){
472 Cell cell = *icell;
473 const ItemTypeInfo* type = cell.typeInfo();
474 Integer cell_nb_face = type->nbLocalFace();
475 for( Integer i_face=0; i_face<cell_nb_face; ++i_face ){
476 const ItemTypeInfo::LocalFace& lf = type->localFace(i_face);
477 Integer face_nb_node = lf.nbNode();
478
479 m_work_face_sorted_nodes.resize(face_nb_node);
480 m_work_face_nodes_index.resize(face_nb_node);
481 m_work_face_orig_nodes_uid.resize(face_nb_node);
482 for( Integer z=0; z<face_nb_node; ++z )
483 m_work_face_orig_nodes_uid[z] = cell.node(lf.node(z)).uniqueId();
484
485 bool is_reorder = false;
486 if (is_1d){
487 is_reorder = (i_face==1);
488 m_work_face_nodes_index[0] = 0;
489 }
490 else
491 is_reorder = mesh_utils::reorderNodesOfFace2(m_work_face_orig_nodes_uid,m_work_face_nodes_index);
492 for( Integer z=0; z<face_nb_node; ++z )
493 m_work_face_sorted_nodes[z] = m_work_face_orig_nodes_uid[m_work_face_nodes_index[z]];
494
495 Face cell_face = cell.face(i_face);
496 if (cell_face.nbNode()!=face_nb_node)
497 ARCANE_FATAL("Incoherent number of node for 'face' and 'localFace'"
498 " cell={0} face={1} nb_local_node={2} nb_face_node={3}",
499 ItemPrinter(cell),ItemPrinter(cell_face),face_nb_node,cell_face.nbNode());
500
501 for( Integer z=0; z<face_nb_node; ++z ){
502 if (cell_face.node(z).uniqueId()!=m_work_face_sorted_nodes[z])
503 ARCANE_FATAL("Bad unique id for face: cell={0} face={1} cell_node_uid={2} face_node_uid={3}",
504 ItemPrinter(cell),ItemPrinter(cell_face),m_work_face_sorted_nodes[z],
505 cell_face.node(z).uniqueId());
506 }
507 if (is_reorder){
508 Cell front_cell = cell_face.frontCell();
509 if (front_cell!=cell){
510 if (!front_cell.null())
511 ARCANE_FATAL("Bad orientation for face. Should be front cell: cell={0} face={1} front_cell={2}",
512 ItemPrinter(cell),ItemPrinter(cell_face),ItemPrinter(front_cell));
513 else
514 ARCANE_FATAL("Bad orientation for face. Should be front cell (no front cell) cell={0} face={1}",
515 ItemPrinter(cell),ItemPrinter(cell_face));
516 }
517 }
518 else{
519 Cell back_cell = cell_face.backCell();
520 if (back_cell!=cell){
521 if (!back_cell.null())
522 ARCANE_FATAL("Bad orientation for face. Should be back cell: cell={0} face={1} front_cell={2}",
523 ItemPrinter(cell),ItemPrinter(cell_face),ItemPrinter(back_cell));
524 else
525 ARCANE_FATAL("Bad orientation for face. Should be back cell (no back cell) cell={0} face={1}",
526 ItemPrinter(cell),ItemPrinter(cell_face));
527 }
528 }
529 }
530 }

536void DynamicMeshChecker::
537_checkValidItemOwner(IItemFamily* family)
538{
539 String func_name = "DynamicMesh::_checkValidItemOwner";
540
541 // Pour les maillages non sous-maillages, il faut que tout sub-item est une cellule voisine de même propriétaire
542 // Pour les sous-maillages, il faut, en plus, que tout item soit de même propriétaire que son parent
543
544 Integer nerror = 0;
545
546 if (!m_mesh->parentMesh()){
547
548 if (family->itemKind() == IK_Cell)
549 return; // implicitement valide pour les cellules
550
551 ItemGroup own_items = family->allItems().own();
552 ENUMERATE_ITEM(iitem,own_items){
553 ItemInternal* item = (*iitem).internal();
554 Int32 owner = item->owner();
555 bool is_ok = false;
556 ItemInternalVectorView cells = item->internalCells();
557 for( ItemInternal* cell : cells ){
558 if (cell->owner()==owner){
559 is_ok = true;
560 break;
561 }
562 }
563 if (!is_ok) {
564 OStringStream ostr;
565 Integer index = 0;
566 for( ItemInternal* cell : cells ){
567 ostr() << " SubCell i=" << index << " cell=" << ItemPrinter(cell);
568 ++index;
569 }
570 error() << "Mesh " << m_mesh->name() << " Item" << ItemPrinter(item) << " has no cell with same owner:"
571 << ostr.str();
572 ++nerror;
573 }
574 }
575 }
576 else {
577 ENUMERATE_ITEM(iitem,family->allItems()){
578 Item item = *iitem;
579 Item parent_item = item.parent();
580 if (parent_item.owner() != item.owner()) {
581 error() << "Mesh " << m_mesh->name() << " : Inconsistent item/parent item owner : "
582 << ItemPrinter(item) << " / " << ItemPrinter(parent_item);
583 ++nerror;
584 }
585 }
586 }
587
588 if (nerror>0)
589 ARCANE_FATAL("mesh {0} family={1} has {2}",m_mesh->name(),family->name(),
590 String::plural(nerror,"owner error"));
591}

596void DynamicMeshChecker::
597checkVariablesSynchronization()
598{
599 Int64 nb_diff = 0;
600 VariableCollection used_vars(m_mesh->subDomain()->variableMng()->usedVariables());
601 for( VariableCollection::Enumerator i_var(used_vars); ++i_var; ){
602 IVariable* var = *i_var;
603 switch (var->itemKind()){
604 case IK_Node:
605 case IK_Edge:
606 case IK_Face:
607 case IK_Cell:
608 case IK_DoF:
609 nb_diff += var->checkIfSync(10);
610 break;
611 case IK_Particle:
612 case IK_Unknown:
613 break;
614 }
615 }
616 if (nb_diff!=0)
617 ARCANE_FATAL("Error in checkVariablesSynchronization() nb_diff=",nb_diff);
618}

651void DynamicMeshChecker::
652checkGhostCells()
653{
654 pwarning() << "CHECK GHOST CELLS";
655 Integer sid = m_mesh->meshRank();
656 ENUMERATE_CELL(icell,m_mesh->cellFamily()->allItems()){
657 Cell cell = *icell;
658 if (cell.isOwn())
659 continue;
660 bool is_ok = false;
661 for( NodeEnumerator inode(cell.nodes()); inode.hasNext(); ++inode ){
662 Node node = *inode;
663 for( CellEnumerator icell2(node.cells()); icell2.hasNext(); ++icell2 ){
664 Cell cell2 = *icell2;
665 if (cell2.owner()==sid){
666 is_ok = true;
667 }
668 }
669 }
670 if (!is_ok)
671 info() << "WARNING: Cell " << ItemPrinter(cell) << " should not be a ghost cell";
672 }

708
709void DynamicMeshChecker::
710checkValidReplication()
711{
712 info() << "Checking valid replication";
713 IParallelMng* mesh_pm = m_mesh->parallelMng();
714 IParallelReplication* pr = mesh_pm->replication();
715 if (!pr->hasReplication())
716 return;
717
718 IParallelMng* pm = pr->replicaParallelMng();
719
720 // Vérifie que toutes les familles (sauf les particules) sont les mêmes
721 UniqueArray<IItemFamily*> wanted_same_family;
722
723 for( IItemFamilyCollection::Enumerator i(m_mesh->itemFamilies()); ++i; ){
724 IItemFamily* family = *i;
725 if (family->itemKind()!=IK_Particle)
726 wanted_same_family.add(family);
727 }
728 ValueChecker vc(A_FUNCINFO);
729
730 // Vérifie que tout le monde à le même nombre de famille.
731 Integer nb_family = wanted_same_family.size();
732 Integer max_nb_family = pm->reduce(Parallel::ReduceMax,nb_family);
733 vc.areEqual(nb_family,max_nb_family,"Bad number of family");
734
735 // Vérifie que toutes les familles ont le même nombre d'éléments.
736 //TODO: il faudrait vérifier aussi que les noms des familles correspondent.
737 {
738 UniqueArray<Int32> families_size(nb_family);
739 for( Integer i=0; i<nb_family; ++i )
740 families_size[i] = wanted_same_family[i]->nbItem();
741 UniqueArray<Int32> global_families_size(families_size);
742 pm->reduce(Parallel::ReduceMax,global_families_size.view());
743 vc.areEqualArray(global_families_size,families_size,"Bad family");
744 }
745
746 // Vérifie que toutes les familles ont les mêmes entités (même uniqueId())
747 {
748 UniqueArray<Int64> unique_ids;
749 UniqueArray<Int64> global_unique_ids;
750 for( Integer i=0; i<nb_family; ++i ){
751 ItemGroup group = wanted_same_family[i]->allItems();
752 unique_ids.resize(group.size());
753 Integer index = 0;
754 ENUMERATE_ITEM(iitem,group){
755 unique_ids[index] = iitem->uniqueId();
756 ++index;
757 }
758 global_unique_ids.resize(group.size());
759 global_unique_ids.copy(unique_ids);
760 pm->reduce(Parallel::ReduceMax,global_unique_ids.view());
761 String message = String::format("Bad unique ids for family '{0}'",
762 wanted_same_family[i]->name());
763 vc.areEqualArray(global_unique_ids,unique_ids,message);
764 }
765 }
766}
