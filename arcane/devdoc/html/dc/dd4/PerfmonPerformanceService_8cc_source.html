405void PerfmonPerformanceService::
406startProfiling()
407{
408 info() << " START PROFILING\n";
409
410 if (!global_infos)
411 global_infos = new PerfmonInfos(traceMng());
412
413 int retval;
414 caddr_t start,end;
415 unsigned long length;
416 unsigned int *profbuf;
417
418 char **p;
419 pfarg_context_t ctx[1];
420 pfmlib_input_param_t inp;
421 pfmlib_output_param_t outp;
422 pfarg_reg_t pc[NUM_PMCS];
423 pfarg_load_t load_args;
424 pfmlib_options_t pfmlib_options;
425 struct sigaction act;
426 unsigned int i;
427 int ret;
428
429 int sampling_period = 1000000;
430 String s = platform::getEnvironmentVariable("ARCANE_PROFILING_SAMPLING_PERIOD");
431 if (!s.null()){
432 info() << " SAMPLING PERIOD FROM ENVIRONMENT '" << s << "'\n";
433 Integer sp = 0;
434 bool is_bad = builtInGetValue(sp,s);
435 if (!is_bad && sp>100000){
436 sampling_period = sp;
437 }
438 }
439 info() << " SAMPLING_PERIOD '" << sampling_period << "'\n";
440
441 /*
442 * Initialize pfm library (required before we can use it)
443 */
444 if (pfm_initialize() != PFMLIB_SUCCESS) {
445 fatal() << "pfm_initialize() failed. Can't initialize library\n";
446 }
447
448 /*
449 * Install the signal handler (SIGIO)
450 */
451 memset(&act, 0, sizeof(act));
452 act.sa_handler = (sig_t)arcane_sigio_handler;
453 sigaction (SIGPROF, &act, 0);
454
455 /*
456 * pass options to library (optional)
457 */
458 memset(&pfmlib_options, 0, sizeof(pfmlib_options));
459 pfmlib_options.pfm_debug = 0; /* set to 1 for debug */
460 pfm_set_options(&pfmlib_options);
461
462 memset(pc, 0, sizeof(pc));
463 memset(ctx, 0, sizeof(ctx));
464 memset(&load_args, 0, sizeof(load_args));
465 memset(&inp,0, sizeof(inp));
466 memset(&outp,0, sizeof(outp));
467
468 p = event_list;
469 for (i=0; i < N_EVENTS ; i++, p++) {
470 if (pfm_find_event(*p, &inp.pfp_events[i].event) != PFMLIB_SUCCESS) {
471 fatal() << "Cannot find " << *p << "event\n";
472 }
473 }
474
475 /*
476 * set the default privilege mode for all counters:
477 * PFM_PLM3 : user level only
478 */
479 inp.pfp_dfl_plm = PFM_PLM3;
480
481 /*
482 * how many counters we use
483 */
484 inp.pfp_event_count = i;
485
486 /*
487 * let the library figure out the values for the PMCS
488 */
489 if ((ret=pfm_dispatch_events(&inp, NULL, &outp, NULL)) != PFMLIB_SUCCESS) {
490 fatal() << "Cannot configure events: " << pfm_strerror(ret);
491 }
492
493 /*
494 * when we know we are self-monitoring and we have only one context, then
495 * when we get an overflow we know where it is coming from. Therefore we can
496 * save the call to the kernel to extract the notification message. By default,
497 * a message is generated. The queue of messages has a limited size, therefore
498 * it is important to clear the queue by reading the message on overflow. Failure
499 * to do so may result in a queue full and you will lose notification messages.
500 *
501 * With the PFM_FL_OVFL_NO_MSG, no message will be queue, but you will still get
502 * the signal. Similarly, the PFM_MSG_END will be generated.
503 */
504 ctx->ctx_flags = PFM_FL_OVFL_NO_MSG;
505 /*
506 * now create the context for self monitoring/per-task
507 */
508 if (perfmonctl(0, PFM_CREATE_CONTEXT, ctx, 1) == -1 ) {
509 if (errno == ENOSYS) {
510 fatal() << "Your kernel does not have performance monitoring support!";
511 }
512 fatal() << "Can't create PFM context " << strerror(errno);
513 }
514 ctx_fd = ctx->ctx_fd;
515
516 /*
517 * Now prepare the argument to initialize the PMDs and PMCS.
518 * We use pfp_pmc_count to determine the number of registers to
519 * setup. Note that this field can be >= pfp_event_count.
520 */
521
522 for (i=0; i < outp.pfp_pmc_count; i++) {
523 pc[i].reg_num = outp.pfp_pmcs[i].reg_num;
524 pc[i].reg_value = outp.pfp_pmcs[i].reg_value;
525 }
526
527 for (i=0; i < inp.pfp_event_count; i++) {
528 pd[i].reg_num = pc[i].reg_num;
529 }
530 /*
531 * We want to get notified when the counter used for our first
532 * event overflows
533 */
534 pc[0].reg_flags |= PFM_REGFL_OVFL_NOTIFY;
535 pc[0].reg_reset_pmds[0] |= 1UL << outp.pfp_pmcs[1].reg_num;
536
537 /*
538 * we arm the first counter, such that it will overflow
539 * after SMPL_PERIOD events have been observed
540 */
541 pd[0].reg_value = (~0UL) - SMPL_PERIOD + 1;
542 pd[0].reg_long_reset = (~0UL) - SMPL_PERIOD + 1;
543 pd[0].reg_short_reset = (~0UL) - SMPL_PERIOD + 1;
544
545 /*
546 * Now program the registers
547 *
548 * We don't use the save variable to indicate the number of elements passed to
549 * the kernel because, as we said earlier, pc may contain more elements than
550 * the number of events we specified, i.e., contains more than counting monitors.
551 */
552 if (perfmonctl(ctx_fd, PFM_WRITE_PMCS, pc, outp.pfp_pmc_count) == -1) {
553 fatal() << "perfmonctl error PFM_WRITE_PMCS errno " << errno;
554 }
555
556 if (perfmonctl(ctx_fd, PFM_WRITE_PMDS, pd, inp.pfp_event_count) == -1) {
557 fatal() << "perfmonctl error PFM_WRITE_PMDS errno " << errno;
558 }
559
560 /*
561 * we want to monitor ourself
562 */
563 load_args.load_pid = getpid();
564
565 if (perfmonctl(ctx_fd, PFM_LOAD_CONTEXT, &load_args, 1) == -1) {
566 fatal() << "perfmonctl error PFM_WRITE_PMDS errno " << errno;
567 }
568
569 /*
570 * setup asynchronous notification on the file descriptor
571 */
572 ret = fcntl(ctx_fd, F_SETFL, fcntl(ctx_fd, F_GETFL, 0) | O_ASYNC);
573 if (ret == -1) {
574 fatal() << "cannot set ASYNC: " << strerror(errno);
575 }
576
577 /*
578 * get ownership of the descriptor
579 */
580 ret = fcntl(ctx_fd, F_SETOWN, getpid());
581 if (ret == -1) {
582 fatal() << "cannot setown: " << strerror(errno);
583 }
584
585 /*
586 * when you explicitely declare that you want a particular signal,
587 * even with you use the default signal, the kernel will send more
588 * information concerning the event to the signal handler.
589 *
590 * In particular, it will send the file descriptor from which the
591 * event is originating which can be quite useful when monitoring
592 * multiple tasks from a single thread.
593 */
594 ret = fcntl(ctx_fd, F_SETSIG, SIGPROF);
595 if (ret == -1)
596 fatal() << "cannot setsig: " << strerror(errno);
597
598 global_infos->startProfiling();
599}

622void PerfmonPerformanceService::
623stopProfiling()
624{
625 if (!global_infos)
626 return;
627 info() << " STOP PROFILING\n";
628 global_infos->stopProfiling();
629
630 Int64 total_event = 1;
631 {
632 info() << " NB ADDRESS MAP = " << global_infos->m_addr_map.size();
633 info() << " NB FUNC MAP = " << global_infos->m_func_map.size();
634 info() << " NB STACK MAP = " << global_infos->m_stack_map.size();
635 info() << " TOTAL STACK = " << global_infos->m_total_stack;
636 {
637 PerfmonInfos::AddrMap::const_iterator begin = global_infos->m_addr_map.begin();
638 for( ; begin!=global_infos->m_addr_map.end(); ++begin ){
639 PerfmonFuncInfo* pf = begin->second.m_func_info;
640 Int64 nb_event = begin->second.m_counters[0];
641 total_event += nb_event;
642 }
643 }
644
645 {
646 String file_name = "profiling.addr";
647 file_name += platform::getProcessId();
648 file_name += ".xml";
649 ofstream ofile(file_name.localstr());
650 ofile << "<?xml version='1.0'?>\n";
651 ofile << "<addresses>\n";
652 PerfmonInfos::AddrMap::const_iterator begin = global_infos->m_addr_map.begin();
653 for( ; begin!=global_infos->m_addr_map.end(); ++begin ){
654 void* addr = begin->first;
655 const PerfmonAddrInfo& pa = begin->second;
656 ofile << "<addr addr='" << addr << "'"
657 << " count='" << pa.m_counters[0] << "'"
658 << " func='" << pa.m_func_info->m_func_name << "'"
659 << ">\n";
660 }
661 ofile << "/<addresses>\n";
662 }
663 }
664 {
665 std::set<PerfmonFuncInfo*,PerfmonFuncComparer> sorted_func;
666 {
667 PerfmonInfos::FuncMap::const_iterator begin = global_infos->m_func_map.begin();
668 for( ; begin!=global_infos->m_func_map.end(); ++begin ){
669 if (begin->second)
670 sorted_func.insert(begin->second);
671 }
672 }
673
674 std::set<PerfmonFuncInfo*,PerfmonFuncComparer>::const_iterator begin = sorted_func.begin();
675 char demangled_func_name[512];
676 info() << " TOTAL EVENT = " << total_event;
677 info() << " event % function";
678 Integer index = 0;
679 for( ; begin!=sorted_func.end(); ++begin ){
680 PerfmonFuncInfo* pfi = *begin;
681 const char* func_name = pfi->m_func_name;
682 size_t len = 512;
683 int dstatus = 0;
684 const char* buf = abi::__cxa_demangle(func_name,demangled_func_name,&len,&dstatus);
685 if (!buf)
686 buf = func_name;
687 Int64 nb_event = pfi->m_counters[0];
688 Int64 total_percent = (nb_event * 1000) / total_event;
689 Int64 percent = (total_percent/10);
690 info() << " " << Trace::Width(10) << nb_event
691 << " " << Trace::Width(3) << percent << "." << (total_percent % 10)
692 << " " << Trace::Width(12) << pfi->m_counters[0]
693 << " " << Trace::Width(12) << pfi->m_counters[1]
694 << " " << Trace::Width(12) << pfi->m_counters[2]
695 << " " << pfi->m_do_stack
696 << " " << buf;
697 if (total_percent<5 && index>20)
698 break;
699 ++index;
700 }
