104class VariableNameInfo
105{
106 public:
107 String m_local_name;
108 String m_family_name;
109 String m_mesh_name;
110 bool operator==(const VariableNameInfo& vni) const
111 {
112 return m_local_name==vni.m_local_name
113 && m_family_name==vni.m_family_name && m_mesh_name==vni.m_mesh_name;
114 }
115 public:
116 Int32 hash() const
117 {
118 Int32 h = 0;
119 if (!m_mesh_name.null()){
120 h = hash(h,m_mesh_name.localstr());
121 h = hash(h,"_");
122 }
123 if (!m_family_name.null()){
124 h = hash(h,m_family_name.localstr());
125 h = hash(h,"_");
126 }
127 h = hash(h,m_local_name.localstr());
128 return h;
129 }
130 public:
131 static Int32 hash(Int32 h,const char* p)
132 {
133 if (!p)
134 return h;
135 for(; *p != '\0'; ++p )
136 h = (h << 5) - h + *p;
137 return h;
138 }
139};

453initialize()
454{
455 VariableFactoryRegisterer* vff = VariableFactoryRegisterer::firstVariableFactory();
456 while (vff){
457 IVariableFactory* vf = vff->createFactory();
458 String full_name = vf->fullTypeName();
459 // Vérifie qu'aucune fabrique avec le même nom n'existe.
460 if (m_variable_factory_map.find(full_name)!=m_variable_factory_map.end()){

462 full_name);
463 }
464 m_variable_factory_map.insert(VariableFactoryPair(full_name,vf));
465 m_variable_factories.add(vf);

467 << " data_type=" << vff->dataType()
468 << " dim=" << vff->dimension()
469 << " multi_tag=" << vff->multiTag()
470 << " full_name=" << full_name
471 << " addr=" << vf;
472 vff = vff->nextVariableFactory();
473 }

479void VariableMng::
480removeAllVariables()
481{
482 //ATTENTION: ceci entraine des appels à removeVariable()
483 m_auto_create_variables.each(Deleter());
484
485 OStringStream var_str;
486 UniqueArray<IVariable*> remaining_vars;
487 for( auto i : m_full_name_variable_map ){
488 IVariable* v = i.second;
489 if (v->nbReference()==0)
490 delete v;
491 else{
492 remaining_vars.add(v);
493 var_str() << " " << v->fullName() << " (" << v->nbReference() << ")";
494 }
495 }
496 bool is_check = arcaneIsCheck();
497 const bool has_remaining_vars = !remaining_vars.empty();
498 if (has_remaining_vars && is_check)
499 pwarning() << "The following variables are still referenced: "
500 << var_str.str()
501 << " (set the environment variable ARCANE_TRACE_VARIABLE_CREATION"
502 << " to get the stack trace)";
503 bool has_trace = VariableRef::hasTraceCreation();
504 if (has_trace){
505 for( auto i : remaining_vars ){
506 for( VarRefEnumerator ivar(i); ivar.hasNext(); ++ivar ){
507 VariableRef* var = *ivar;
508 info() << " variable name=" << var->name()
509 << " stack=" << var->assignmentStackTrace();
510 }
511 }
512 }
513
514 // Appelle explicitement 'unregisterVariable()' sur les variables restantes.
515 // Sans cela, si ensuite l'instance 'this' est détruite avant que les variables
516 // restantent ne le soit cela va provoquer un plantage (Read after free). Cela
517 // n'arrive normalement pas pour le C++ mais peut arriver pour le wrapping.
518 if (has_remaining_vars){
519 // Recopie les références dans un tableau temporaire
520 // car les appels à unregisterVariable() modifient l'itérateur ivar
521 // et aussi m_full_name_variable_map.
522 UniqueArray<VariableRef*> remaining_refs;
523 for( auto i : remaining_vars )
524 for( VarRefEnumerator ivar(i); ivar.hasNext(); ++ivar )
525 remaining_refs.add(*ivar);
526 for( VariableRef* r : remaining_refs )
527 r->unregisterVariable();
528 if (is_check)
529 info() << "Remaining variables after cleanup n=" << m_full_name_variable_map.size();
530 }
531
532 m_full_name_variable_map.clear();
533}

574void VariableMng::
575addVariable(IVariable* var)
576{
577 const String& full_name = var->fullName();
578 subDomain()->checkId("VariableMng::checkVariable()",full_name);
579 if (var->itemKind()!=IK_Unknown && var->itemFamilyName().null())
580 ARCANE_FATAL("Bad Variable full-name={0} name={1}",var->fullName(),var->name());
581
582 info(5) << "Add variable"
583 << " name=" << var->name()
584 << " full_name=" << full_name
585 << " datatype=" << var->dataType()
586 << " kind=" << var->itemKind();
587
588 VariableNameInfo vni;
589 vni.m_local_name = var->name();
590 vni.m_family_name = var->itemFamilyName();
591 vni.m_mesh_name = var->meshName();
592 m_vni_map.add(vni,var);
593
594 m_full_name_variable_map.insert(FullNameVariablePair(full_name,var));
595 m_variables_changed = true;
596 m_used_variables_changed = true;
597 ++m_nb_created_variable;
598 IEntryPoint* ep = subDomain()->timeLoopMng()->currentEntryPoint();
599 IModule* module = nullptr;
600 if (ep)
601 module = ep->module();
602 m_variable_creation_modules.insert(std::make_pair(var,module));
603 VariableStatusChangedEventArgs eargs(var,VariableStatusChangedEventArgs::Status::Added);
604 m_on_variable_added.notify(eargs);
605}

610void VariableMng::
611removeVariable(IVariable* var)
612{
613 int p = var->property();
614 if (p & IVariable::PTemporary){
615 debug() << "** ** REMOVE " << var->name() << " " << var->nbReference()
616 << " property=" << p
617 << " nodump=" << (p & IVariable::PNoDump)
618 << " tmp=" << (p & IVariable::PTemporary)
619 << " norestore=" << (p & IVariable::PNoRestore);
620 }
621 VariableStatusChangedEventArgs eargs(var,VariableStatusChangedEventArgs::Status::Removed);
622 m_on_variable_removed.notify(eargs);
623 {
624 ItemGroup var_group = var->itemGroup();
625 // Retire cette variable de tous les groupes [bien défini mais sur ItemGroupImplNull]
626 if (!var_group.null())
627 var_group.internal()->detachObserver(var);
628 // Retire cette variable des variables existantes, puis la détruit
629 m_full_name_variable_map.erase(var->fullName());
630 {
631 VariableNameInfo vni;
632 vni.m_local_name = var->name();
633 vni.m_family_name = var->itemFamilyName();
634 vni.m_mesh_name = var->meshName();
635 m_vni_map.remove(vni);
636 }
637 m_variables_changed = true;
638 m_used_variables_changed = true;
639 m_variable_creation_modules.erase(var);
640 delete var;
641 }

647IVariable* VariableMng::
648checkVariable(const VariableInfo& infos)
649{
650 VariableNameInfo vni;
651 vni.m_local_name = infos.localName();
652 vni.m_family_name = infos.itemFamilyName();
653 vni.m_mesh_name = infos.meshName();
654
655 if (infos.itemKind()!=IK_Unknown && infos.meshName().null())
656 ARCANE_FATAL("Mesh variable without a mesh full-name={0} name={1}",infos.fullName(),infos.localName());
657
658 // Si variable du maillage, vérifie qu'aucune variable globale non lié à un maillage
659 // ne porte le même nom.
660 if (arcaneIsCheck()){
661 if (!infos.meshName().null()){
662 String check_name = infos.localName();
663 if (findVariableFullyQualified(check_name)){
664 ARCANE_FATAL("Mesh variable has the same name that a global variable (name={0})",check_name);
665 }
666 }
667 else{
668 // Si variable globale, vérifie qu'aucune variable du maillage ne porte le même nom.
669 String check_name = String("Mesh0_")+infos.localName();
670 if (findVariableFullyQualified(check_name)){
671 ARCANE_FATAL("Global variable has the same name that a mesh variable (name={0})",check_name);
672 }
673 }
674 }
675
676 IVariable* var = nullptr;
677 //VNIMap::Data* var_data = m_vni_map.lookup(vni);
678
679 //cerr << "** CHECK " << name << ' ' << infos.dataType() << ' ' << infos.kind() << '\n';
680 VNIMap::Data* var_data = m_vni_map.lookup(vni);
681 if (var_data){
682 // Une variable de même nom que \a var existe déjà.
683 // Il faut dans ce cas vérifier que son genre et son type sont les même.
684 // elle a le même genre et le même type.
685 var = var_data->value();
686 //cerr << "** FIND " << prv->name() << ' ' << prv->dataType() << ' ' << prv->kind() << '\n';
687 if (infos.dataType()!=var->dataType() ||
688 infos.itemKind()!=var->itemKind() ||
689 infos.dimension()!=var->dimension()){
690 throw BadVariableKindTypeException(A_FUNCINFO,var,infos.itemKind(),
691 infos.dataType(),infos.dimension());
692 }
693 }
694 return var;
695}

1570 // toujours le cas, sauf si la protection vient d'une ancienne

1584 if (hash_str!=reference_hash){
1585 ++nb_error;
1586 error() << "Hash values are different. Corrumpted values."
1587 << " name=" << var->fullName()
1588 << " ref=" << reference_hash
1589 << " current=" << hash_str;

1592 String fname = listing_dir.file(String::format("dump-{0}-sid_{1}",var->fullName(),sid));
1593 std::ofstream ofile(fname.localstr());
1594 ofile.write(reinterpret_cast<const char*>(buf.data()),buf.size());
1595 }
1596 }
