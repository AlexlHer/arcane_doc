95void FaceDirectionMng::
96_internalComputeInfos(const CellDirectionMng& cell_dm,const VariableCellReal3& cells_center,
97 const VariableFaceReal3& faces_center)
98{
99 IItemFamily* face_family = m_p->m_cartesian_mesh->mesh()->faceFamily();
100 int dir = (int)m_direction;
101 String base_group_name = String("Direction")+dir;
102 if (m_p->m_patch_index>=0)
103 base_group_name = base_group_name + String("AMRPatch")+m_p->m_patch_index;
104
105 // Calcule la liste des faces dans une direction donnée.
106 // Il faut pour chaque maille ajouter dans la liste des faces
107 // les deux faces de la direction souhaitées en prenant bien soin
108 // de ne pas ajouter deux fois la même face.
109 UniqueArray<Int32> faces_lid;
110 {
111 CellGroup all_cells = cell_dm.allCells();
112 faces_lid.reserve(all_cells.size());
113 // Ensemble des faces déjà ajoutées
114 std::set<Int32> done_faces;
115 ENUMERATE_CELL(icell,all_cells){
116 DirCellFace dcf(cell_dm.cellFace(*icell));
117 Face next_face = dcf.next();
118 Face prev_face = dcf.previous();
119
121 Int32 prev_lid = prev_face.localId();
122 if (done_faces.find(prev_lid)==done_faces.end()){
123 faces_lid.add(prev_lid);
124 done_faces.insert(prev_lid);
125 }
126 Int32 next_lid = next_face.localId();
127 if (done_faces.find(next_lid)==done_faces.end()){
128 faces_lid.add(next_lid);
129 done_faces.insert(next_lid);
130 }
131 }
132 }
133
134 FaceGroup all_faces = face_family->createGroup(String("AllFaces")+base_group_name,Int32ConstArrayView(),true);
135 all_faces.setItems(faces_lid,true);
136
137 UniqueArray<Int32> inner_lids;
138 UniqueArray<Int32> outer_lids;
139 ENUMERATE_FACE(iitem,all_faces){
140 Int32 lid = iitem.itemLocalId();
141 Face face = *iitem;
142 // TODO: ne pas utiser nbCell() mais faire cela via le std::set utilisé précédemment
143 if (face.nbCell()==1)
144 outer_lids.add(lid);
145 else
146 inner_lids.add(lid);
147 }
148 m_p->m_inner_all_items = face_family->createGroup(String("AllInner")+base_group_name,inner_lids,true);
149 m_p->m_outer_all_items = face_family->createGroup(String("AllOuter")+base_group_name,outer_lids,true);
150 m_p->m_all_items = all_faces;
151
152 _computeCellInfos(cell_dm,cells_center,faces_center);
153}

158bool FaceDirectionMng::
159_hasFace(Cell cell,Int32 face_local_id) const
160{
161 for( FaceEnumerator iface(cell.faces()); iface.hasNext(); ++iface ){
162 if (iface->localId()==face_local_id)
163 return true;
164 }
165 return false;
166}

176void FaceDirectionMng::
177_computeCellInfos(const CellDirectionMng& cell_dm,const VariableCellReal3& cells_center,
178 const VariableFaceReal3& faces_center)
179{
180 eMeshDirection dir = m_direction;
181
182 // Créé l'ensemble des mailles du patch et s'en sert
183 // pour être sur que chaque maille devant/derrière est dans
184 // cet ensemble
185 std::set<Int32> patch_cells_set;
186 ENUMERATE_CELL(icell,cell_dm.allCells()){
187 patch_cells_set.insert(icell.itemLocalId());
188 }
189
190 ENUMERATE_FACE(iface,m_p->m_all_items){
191 Face face = *iface;
192 Int32 face_lid = iface.itemLocalId();
193 Real3 face_coord = faces_center[iface];
194 Cell front_cell = face.frontCell();
195 Cell back_cell = face.backCell();
196
197 // Vérifie que les mailles sont dans notre patch.
198 if (!front_cell.null())
199 if (patch_cells_set.find(front_cell.localId())==patch_cells_set.end())
200 front_cell = Cell();
201 if (!back_cell.null())
202 if (patch_cells_set.find(back_cell.localId())==patch_cells_set.end())
203 back_cell = Cell();
204
205 bool is_inverse = false;
206 if (!front_cell.null()){
207 Real3 front_coord = cells_center[front_cell];
208 if (dir==MD_DirX){
209 if (front_coord.x<face_coord.x)
210 is_inverse = true;
211 }
212 else if (dir==MD_DirY){
213 if (front_coord.y<face_coord.y)
214 is_inverse = true;
215 }
216 else if (dir==MD_DirZ){
217 if (front_coord.z<face_coord.z)
218 is_inverse = true;
219 }
220 }
221 else{
222 Real3 back_coord = cells_center[back_cell];
223 if (dir==MD_DirX){
224 if (back_coord.x>face_coord.x)
225 is_inverse = true;
226 }
227 else if (dir==MD_DirY){
228 if (back_coord.y>face_coord.y)
229 is_inverse = true;
230 }
231 else if (dir==MD_DirZ){
232 if (back_coord.z>face_coord.z)
233 is_inverse = true;
234 }
235 }
236 // Si la face a deux mailles connectées, regarde le niveau AMR de ces
237 // deux mailles et s'il est différent, ne conserve que la maille
238 // dont le niveau AMR est celui de la face.
239 if (!back_cell.null() && !front_cell.null()){
240 Int32 back_level = back_cell.level();
241 Int32 front_level = front_cell.level();
242 if (back_level!=front_level){
243 // La face n'a pas l'information de son niveau mais si les deux
244 // mailles ne sont pas de même niveau la face n'appartient qu'à une
245 // seule des deux mailles. On ne garde donc que cette dernière.
246 if (!_hasFace(back_cell,face_lid))
247 back_cell = Cell();
248 if (!_hasFace(front_cell,face_lid))
249 front_cell = Cell();
250 }
251 }
252 ItemInternal* front_cell_i = front_cell.internal();
253 ItemInternal* back_cell_i = back_cell.internal();
254 ARCANE_CHECK_POINTER(back_cell_i);
255 ARCANE_CHECK_POINTER(front_cell_i);
256 if (is_inverse)
257 m_infos[face_lid] = ItemDirectionInfo(back_cell_i,front_cell_i);
258 else
259 m_infos[face_lid] = ItemDirectionInfo(front_cell_i,back_cell_i);
260 }
261}
