325ARCCORE_HOST_DEVICE inline Real
326doubleContraction(Real3x3 u,Real3x3 v)
327{
328 Real x1 = u.x.x * v.x.x;
329 Real x2 = u.x.y * v.x.y;
330 Real x3 = u.x.z * v.x.z;
331
332 Real y1 = u.y.x * v.y.x;
333 Real y2 = u.y.y * v.y.y;
334 Real y3 = u.y.z * v.y.z;
335
336 Real z1 = u.z.x * v.z.x;
337 Real z2 = u.z.y * v.z.y;
338 Real z3 = u.z.z * v.z.z;
339
340 return x1+x2+x3+y1+y2+y3+z1+z2+z3;
341}

590{
591 Real err = 0;
592 err = math::max(err,math::abs(relativeError(T1.x.x,T2.x.x)));
593 err = math::max(err,math::abs(relativeError(T1.x.y,T2.x.y)));
594 err = math::max(err,math::abs(relativeError(T1.x.z,T2.x.z)));
595 err = math::max(err,math::abs(relativeError(T1.y.x,T2.y.x)));
596 err = math::max(err,math::abs(relativeError(T1.y.y,T2.y.y)));
597 err = math::max(err,math::abs(relativeError(T1.y.z,T2.y.z)));
598 err = math::max(err,math::abs(relativeError(T1.z.x,T2.z.x)));
599 err = math::max(err,math::abs(relativeError(T1.z.y,T2.z.y)));
600 err = math::max(err,math::abs(relativeError(T1.z.z,T2.z.z)));
601
602 return (err);
603}

642inline bool
643searchExtrema(ConstArrayView<Real2> array,Real& xmin,Real& xmax,
644 Real& ymin,Real& ymax,bool need_init)
645{
646 if (need_init){
647 xmin = ymin = 0.;
648 xmax = ymax = 1.;
649 }
650
651 Integer size = array.size();
652 if (size==0)
653 return false;
654
655 if (need_init){
656 xmin = xmax = array[0].x;
657 ymin = ymax = array[0].y;
658 }
659
660 for( Integer i=1; i<size; ++i ){
661 if (array[i].x < xmin)
662 xmin = array[i].x;
663 if (array[i].x > xmax)
664 xmax = array[i].x;
665
666 if (array[i].y < ymin)
667 ymin = array[i].y;
668 if (array[i].y > ymax)
669 ymax = array[i].y;
670 }
671 return true;
672}
