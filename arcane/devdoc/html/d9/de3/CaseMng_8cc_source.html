287void CaseMng::
288readOptions(bool is_phase1)
289{
290 Trace::Setter mci(traceMng(),msgClassName());
291
292 if (is_phase1)
293 info() << "Reading the input data (phase1): language '"
294 << caseDocument()->language() << "', "
295 << m_case_options_list.count() << " input data.";
296 else
297 info() << "Reading the input data (phase2)";
298
299 if (is_phase1){
300 ICaseDocument* doc = _noNullCaseDocument();
301 doc->clearErrorsAndWarnings();
302 }
303
304 if (is_phase1)
305 readFunctions();
306
307 // Notifie du début de lecture des options.
308 if (is_phase1)
309 m_observables[eCaseMngEventType::BeginReadOptionsPhase1]->notifyAllObservers();
310 else
311 m_observables[eCaseMngEventType::BeginReadOptionsPhase2]->notifyAllObservers();
312
313 _readOptions(is_phase1);
314 _printErrors(is_phase1);
315
316 _checkTranslateDocument();
317}

341
342void CaseMng::
343_printErrors(bool is_phase1)
344{
345 ICaseDocument* doc = _noNullCaseDocument();
346
347 // Affiche les avertissements mais uniquement lors de la phase2 pour les avoir
348 // tous en une fois (certains avertissements ne sont générés que lors de la phase2)
349 if (!m_treat_warning_as_error){
350 if (!is_phase1){
351 if (doc->hasWarnings()){
352 OStringStream ostr;
353 doc->printWarnings(ostr());
354 pwarning() << "The input data are not coherent:\n\n" << ostr.str();
355 }
356 }
357 }
358 bool has_error = doc->hasError();
359 if (doc->hasWarnings() && m_treat_warning_as_error && !is_phase1)
360 has_error = true;
361
362 if (has_error){
363 OStringStream ostr;
364 if (m_treat_warning_as_error && doc->hasWarnings())
365 doc->printWarnings(ostr());
366 doc->printErrors(ostr());
367 ARCANE_FATAL("Input data are invalid:\n\n{0}",ostr.str());
368 }
369}

374void CaseMng::
375readFunctions()
376{
377 if (m_is_function_read)
378 return;
379
380 Trace::Setter mci(traceMng(),msgClassName());
381
382 // Enregistre les services fournissant les fonctions
383 {
384 typedef ServiceFinder2T<ICaseFunctionProvider,ISubDomain> FinderType;
385 ISubDomain* sd = subDomain();
386 FinderType finder(sd->application(),sd);
387 const Array<FinderType::FactoryType*>& factories = finder.factories();
388 info() << "NB_CASE_FUNCTION_PROVIDER_FACTORY = " << factories.size();
389 for( auto factory : factories ){
390 auto cfp = factory->createServiceReference(ServiceBuildInfoBase(sd));
391 if (cfp.get()){
392 info() << "FOUND CASE FUNCTION PROVIDER (V2)!";
393 cfp->registerCaseFunctions(this);
394 }
395 }
396 }
397
398 _readFunctions();
399
400 m_is_function_read = true;
401}

426
427void CaseMng::
428_readFunctions()
429{
430 bool has_error = false;
431
432 // Lecture des fonctions
433
434 XmlNode case_root_elem = caseDocument()->rootElement();
435
436 CaseNodeNames* cnn = caseDocument()->caseNodeNames();
437
438 // Récupère la liste des tables de marche.
439 XmlNode funcs_elem = case_root_elem.child(cnn->functions);
440 XmlNodeList functions_elem = funcs_elem.children();
441
442 String ustr_table(cnn->function_table);
443 String ustr_script(cnn->function_script);
444
445 if (functions_elem.empty())
446 return;
447
448 for( XmlNode node : functions_elem ){
449 bool is_bad = false;
450 if (node.type()!=XmlNode::ELEMENT)
451 continue;
452 if (node.isNamed(ustr_table)){
453 ErrorInfo err_info = _readOneTable(node);
454 if (err_info.hasError()){
455 is_bad = true;
456 String table_name = node.attrValue(cnn->name_attribute);
457 if (table_name.null())
458 error() << " Error in element '" << node.xpathFullName() << "' : "
459 << err_info.errorMessage();
460 else
461 error() << " Error in table named '" << table_name << "' : "
462 << err_info.errorMessage();
463 }
464 }
465 else
466 warning() << "Unknown node in functions: " << node.xpathFullName();
467 has_error |= is_bad;
468 }
469
470 if (has_error)
471 ARCANE_FATAL("Error while reading the functions");
472
473 // Affiche des informations sur le nombre de tables et leur nombre
474 // d'éléments
475 log() << "Number of functions: " << m_functions.count();
476 for( auto& icf_ref : m_functions ){
477 auto table = dynamic_cast<CaseTable*>(icf_ref.get());
478 if (table)
479 log() << "Table <" << table->name() << "> own "
480 << table->nbElement() << " element(s)";
481 }
482}

487CaseMng::ErrorInfo CaseMng::
488_checkValidFunction(const XmlNode& func_elem,CaseFunctionBuildInfo& cfbi)
489{
490 CaseNodeNames* cnn = caseDocument()->caseNodeNames();
491
492 String func_name = func_elem.attrValue(cnn->name_attribute);
493 if (func_name.null())
494 return ErrorInfo(String::format("missing attribute '{0}'",cnn->name_attribute));
495
496 cfbi.m_name = func_name;
497 if (findFunction(func_name))
498 return ErrorInfo(String::format("function '{0}' is defined several times",func_name));
499
500 String param_name = func_elem.attrValue(cnn->function_parameter);
501 if (param_name.null())
502 return ErrorInfo(String::format("missing attribute '{0}'",cnn->function_parameter));
503
504 String value_name = func_elem.attrValue(cnn->function_value);
505 if (value_name.null())
506 return ErrorInfo(String::format("missing attribute '{0}'",cnn->function_value));
507
508 String comul_name = func_elem.attrValue("comul");
509 String transform_x_name = func_elem.attrValue("comul-x");
510
511 cfbi.m_transform_param_func = transform_x_name;
512 cfbi.m_transform_value_func = comul_name;
513
514 String ustr_time(cnn->time_type);
515 String ustr_iteration(cnn->iteration_type);
516
517 String ustr_real(cnn->real_type);
518 String ustr_real3(cnn->real3_type);
519 String ustr_bool(cnn->bool_type);
520 String ustr_integer(cnn->integer_type);
521 String ustr_string(cnn->string_type);
522
523 UniqueArray<String> valid_param_strs = { ustr_time, ustr_real, ustr_iteration, ustr_integer };
524
525 // Vérifie que le type de paramètre spécifié est correct (réel ou entier)
526 ICaseFunction::eParamType param_type = ICaseFunction::ParamUnknown;
527 if (param_name==ustr_time || param_name==ustr_real)
528 param_type = ICaseFunction::ParamReal;
529 else if (param_name==ustr_iteration || param_name==ustr_integer)
530 param_type = ICaseFunction::ParamInteger;
531 if (param_type==ICaseFunction::ParamUnknown){
532 return ErrorInfo(String::format("invalid value '{0}' for attribute '{1}'. Valid values are '{2}'.",
533 param_name,cnn->function_parameter,String::join(", ",valid_param_strs)));
534 }
535 cfbi.m_param_type = param_type;
536
537 UniqueArray<String> valid_value_strs = { ustr_real, ustr_real3, ustr_integer, ustr_bool, ustr_string };
538
539 // Vérifie que le type de valeur spécifié est correct (réel ou entier)
540 ICaseFunction::eValueType value_type = ICaseFunction::ValueUnknown;
541 if (value_name==ustr_real)
542 value_type = ICaseFunction::ValueReal;
543 if (value_name==ustr_integer)
544 value_type = ICaseFunction::ValueInteger;
545 if (value_name==ustr_bool)
546 value_type = ICaseFunction::ValueBool;
547 if (value_name==ustr_string)
548 value_type = ICaseFunction::ValueString;
549 if (value_name==ustr_real3)
550 value_type = ICaseFunction::ValueReal3;
551 if (value_type==ICaseFunction::ValueUnknown)
552 return ErrorInfo(String::format("invalid value '{0}' for attribute '{1}'. Valid values are '{2}'.",
553 value_name,cnn->function_value,String::join(", ",valid_value_strs)));
554 cfbi.m_value_type = value_type;
555
556 // Regarde s'il y a un élément 'deltat-coef' et si oui récupère sa valeur.
557 String deltat_coef_str = func_elem.attrValue(cnn->function_deltat_coef);
558 Real deltat_coef = 0.0;
559 if (!deltat_coef_str.null()){
560 if (builtInGetValue(deltat_coef,deltat_coef_str))
561 return ErrorInfo(String::format("Invalid value '{0}' for attribute '{1}. Can not convert to 'Real' type",
562 deltat_coef_str,cnn->function_deltat_coef));
563 cfbi.m_deltat_coef = deltat_coef;
564 info() << "Coefficient deltat for the function '" << func_name << "' = " << cfbi.m_deltat_coef;
565 }
566
567 return ErrorInfo();
568}

573CaseMng::ErrorInfo CaseMng::
574_readOneTable(const XmlNode& func_elem)
575{
576 CaseFunctionBuildInfo cfbi(traceMng());
577
578 ErrorInfo err_info = _checkValidFunction(func_elem,cfbi);
579 if (err_info.hasError())
580 return err_info;
581
582 CaseNodeNames* cnn = caseDocument()->caseNodeNames();
583
584 String ustr_constant(cnn->function_constant);
585 String ustr_linear(cnn->function_linear);
586 String ustr_value(cnn->function_value);
587
588 String interpolation_name = func_elem.attrValue(cnn->function_interpolation);
589 if (interpolation_name.null())
590 return ErrorInfo(String::format("missing attribute '{0}'",cnn->function_interpolation));
591
592 // Vérifie que le type de la courbe spécifié est correct
593 CaseTable::eCurveType interpolation_type = CaseTable::CurveUnknown;
594 if (interpolation_name==ustr_constant)
595 interpolation_type = CaseTable::CurveConstant;
596 else if (interpolation_name==ustr_linear)
597 interpolation_type = CaseTable::CurveLinear;
598 if (interpolation_type==CaseTable::CurveUnknown){
599 return ErrorInfo(String::format("Invalid value for attribute '{0}'. Valid values are '{1}' or '{2}'",
600 cnn->function_interpolation,ustr_constant,ustr_linear));
601 }
602
603 // Récupère la liste des valeurs de la fonction
604 XmlNodeList func_value_list = func_elem.children(ustr_value);
605 Integer nb_func_value = func_value_list.size();
606 if (nb_func_value==0)
607 return ErrorInfo("The table has no values");
608
609 auto func = new CaseTable(cfbi,interpolation_type);
610 m_functions.add(Ref<ICaseFunction>::create(func));
611
612 Integer value_index = 0;
613
614 String ustr_x("x");
615 String ustr_y("y");
616 for( auto i : func_value_list ) {
617 String param_str = i.child(ustr_x).value();
618 String value_str = i.child(ustr_y).value();
619 param_str = String::collapseWhiteSpace(param_str);
620 value_str = String::collapseWhiteSpace(value_str);
621 if (param_str.null())
622 return ErrorInfo(String::format("index={0} element <x> is missing",value_index));
623 if (value_str.null())
624 return ErrorInfo(String::format("index={0} element <y> is missing",value_index));
625
626 CaseTable::eError error_number = func->appendElement(param_str,value_str);
627 // TODO: AJOUTER DOC DANS LE JDD sur ces informations.
628 if (error_number!=CaseTable::ErrNo){
629 String message = "No info";
630 switch(error_number){
631 case CaseTable::ErrCanNotConvertParamToRightType:
632 message = String::format("parameter '{0}' can not be converted to the 'param' type of the table",param_str);
633 break;
634 case CaseTable::ErrCanNotConvertValueToRightType:
635 message = String::format("value '{0}' can not be converted to the 'value' type of the table",value_str);
636 break;
637 case CaseTable::ErrNotGreaterThanPrevious:
638 message = "<x> lesser than previous <x>";
639 break;
640 case CaseTable::ErrNotLesserThanNext:
641 message = "<x> greater than next <x>";
642 break;
643 case CaseTable::ErrBadRange:
644 message = "bad interval";
645 break;
646 case CaseTable::ErrNo:
647 // Ne devrait jamais arriver
648 ARCANE_FATAL("Internal Error");
649 }
650 return ErrorInfo(String::format("index={0} : {1}",value_index,message));
651 }
652 ++value_index;
653 }
654 return ErrorInfo();
655}

662void CaseMng::
663_searchInvalidOptions()
664{
665 ICaseDocument* doc = _noNullCaseDocument();
666 // Cherche les éléments du jeu de données qui ne correspondent pas
667 // à une option connue.
668 XmlNodeList invalid_elems;
669 for( ICaseOptions* co : CaseOptionsFilterUsed(m_case_options_list))
670 co->addInvalidChildren(invalid_elems);
671
672 // Cherche les éléments racines qui ne correspondent pas à une option
673
674 // Temporairement pour des raisons de compatibilité, autorise les
675 // éléments à la racine qui ne sont pas utilisés.
676 if (!platform::getEnvironmentVariable("ARCANE_ALLOW_UNKNOWN_ROOT_ELEMENT").null())
677 m_allow_unknown_root_element = true;
678
679 String arcane_element_name = doc->arcaneElement().name();
680 String function_element_name = doc->functionsElement().name();
681 XmlNodeList mesh_elements = doc->meshElements();
682 String mesh_element_name;
683 if (mesh_elements.size()>0)
684 mesh_element_name = mesh_elements[0].name();
685 for( XmlNode node : doc->rootElement() ){
686 if (node.type()!=XmlNode::ELEMENT)
687 continue;
688 String name = node.name();
689 if (name=="comment")
690 continue;
691 if (name==arcane_element_name || name==function_element_name)
692 continue;
693 if (!mesh_element_name.null() && name==mesh_element_name)
694 continue;
695 bool is_found = false;
696 for( ICaseOptions* co : CaseOptionsFilterUsed(m_case_options_list)){
697 if (co->rootTagName()==name){
698 is_found = true;
699 break;
700 }
701 }
702 if (!is_found){
703 if (m_allow_unknown_root_element)
704 pwarning() << "-- Unknown root option '" << node.xpathFullName() << "'";
705 else
706 invalid_elems.add(node);
707 }
708 }
709
710 if (!doc->hasError()){
711 if (!invalid_elems.empty()){
712 for( XmlNode xnode : invalid_elems ){

726{

747 }
748 }

885/*---------------------------------------------------------------------------*/
886/*---------------------------------------------------------------------------*/
887
888void CaseMng::
889_readCaseDocument(const String& filename,ByteConstArrayView case_bytes)
890{
891 if (m_case_document.get())
892 // Déjà lu...
893 return;
894
895 IApplication* app = m_sub_domain->application();
896 IIOMng* io_mng = m_sub_domain->parallelMng()->ioMng();
897 String printed_filename = filename;
898
899 IXmlDocumentHolder* case_doc = io_mng->parseXmlBuffer(case_bytes,filename);
900 if (!case_doc){
901 info() << "XML Memory File:";
902 info() << filename;
903 pfatal() << "Failed to analyze the input data " << printed_filename;
904 }
905
906 m_case_document = app->mainFactory()->createCaseDocument(app,case_doc);
907
908 CaseNodeNames* cnn = caseDocument()->caseNodeNames();
909
910 String code_name = app->applicationInfo().codeName();
911 XmlNode root_elem = m_case_document->rootElement();
912 if (root_elem.name()!=cnn->root){
913 pfatal() << "The root element <" << root_elem.name() << "> has to be <" << cnn->root << ">";
914 }
915
916 if (root_elem.attrValue(cnn->code_name)!=code_name){
917 pfatal() << "The file is not a case of code '" << code_name << "'";
918 }
919
920 String case_codeversion = root_elem.attrValue(cnn->code_version);
921 String code_version(app->majorAndMinorVersionStr());
922 String code_version2(app->mainVersionStr());
923 if (case_codeversion!=code_version && case_codeversion!=code_version2){
924 if (!m_sub_domain->session()->checkIsValidCaseVersion(case_codeversion))
925 pfatal() << "The version number of the file (" << case_codeversion
926 << ") doesn't match the version of the code '"
927 << code_name << "' (" << code_version << ").";
928 }
929}
