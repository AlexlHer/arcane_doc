90void RayMeshIntersectionUnitTest::
91executeTest()
92{
93 IParallelMng* pm = mesh()->parallelMng();
94 Int32 my_rank = pm->commRank();
95 Int32 nb_rank = pm->commSize();
96 info() << "EXEC TEST rank=" << my_rank << " nb_rank=" << nb_rank;
97 // Creation de n rayons
98 Integer n0 = 15; // 40 pour un plus gros test.
99 Integer nb_segment = (n0*n0) / nb_rank;
100 Real3UniqueArray segments_position(nb_segment);
101 Real3UniqueArray segments_direction(nb_segment);
102 Real3UniqueArray segments_intersection(nb_segment);
103 RealUniqueArray segments_distance(nb_segment);
104 Int32UniqueArray segments_user_value(nb_segment);
105 Int32UniqueArray segments_orig_face(nb_segment,NULL_ITEM_LOCAL_ID);
106 for( Integer i=0; i<nb_segment; ++i ){
107 Integer index = i + nb_segment*my_rank;
108 Real px = (Real)(index / n0);
109 Real py = (Real)(index % n0);
110 segments_position[i] = Real3(7.0,py/10.0-py/5.0,px/10.0-px/5.0);
111 segments_direction[i] = Real3(-1.0-(px/5.0),(py/20.0)-(py/10.0),0.5-(px/10.0));
112 info() << " SEGMENT: pos=" << segments_position[i] << " dir=" << segments_direction[i];
113
114 }
115 Int32UniqueArray faces_lid(nb_segment);
116 //IRayMeshIntersection* mi = createBasicRayMeshIntersection(mesh());
117
118 //IServiceMng* sm = subDomain()->serviceMng();
119 //FactoryT<IRayMeshIntersection> factory(sm);
120 ServiceBuilder<IRayMeshIntersection> sb(subDomain());
121 auto mi(sb.createReference("BasicRayMeshIntersection"));
122
123 // Calcul avec les segments
124 mi->compute(segments_position,segments_direction,segments_orig_face,segments_user_value,
125 segments_intersection,segments_distance,faces_lid);
126
127 // Calcul avec les rayons sous forme de particule
128 IItemFamily* ray_family = mesh()->findItemFamily(IK_Particle,"Rays",true);
129 Int64UniqueArray uids(nb_segment);
130 Int32UniqueArray local_ids(nb_segment);
131 for( Integer i=0; i<nb_segment; ++i )
132 uids[i] = (i+(nb_segment*my_rank)+1);
133 IParticleFamily* pf = ray_family->toParticleFamily();
134 pf->addParticles(uids,local_ids);
135 ray_family->endUpdate();
136 VariableParticleReal3 rays_position(VariableBuildInfo(ray_family,"Position"));
137 VariableParticleReal3 rays_direction(VariableBuildInfo(ray_family,"Direction"));
138 VariableParticleReal3 rays_intersection(VariableBuildInfo(ray_family,"Intersection"));
139 VariableParticleReal rays_distance(VariableBuildInfo(ray_family,"Distance"));
140 VariableParticleInt32 user_values(VariableBuildInfo(ray_family,"UserValues"));
141 VariableParticleInt32 rays_orig_face(VariableBuildInfo(ray_family,"OriginalFace"));
142 VariableParticleInt32 rays_face(VariableBuildInfo(ray_family,"Face"));
143 // Pour test, supprime quelques rayons
144 {
145 Int32UniqueArray to_remove_rays;
146 for( Integer i=0; i<(nb_segment/10); ++i ){
147 to_remove_rays.add(i*5);
148 }
149 pf->removeParticles(to_remove_rays);
150 pf->endUpdate();
151 }
152
153 ENUMERATE_PARTICLE(iitem,ray_family->allItems()){
154 Particle p = *iitem;
155 pf->setParticleCell(p,Cell());
156 rays_position[p] = segments_position[p.localId()];
157 rays_direction[p] = segments_direction[p.localId()];
158 rays_orig_face[p] = NULL_ITEM_LOCAL_ID;
159 }
160 mi->compute(ray_family,rays_position,rays_direction,rays_orig_face,user_values,
161 rays_intersection,rays_distance,rays_face);
162 info() << "Print rays infos";
163 ItemInternalList faces_internal(mesh()->faceFamily()->itemsInternal());
164 ENUMERATE_PARTICLE(ipart,ray_family->allItems()){
165 Particle ray = *ipart;
166 Int32 face_lid = rays_face[ipart];
167 if (face_lid!=NULL_ITEM_ID)
168 info() << "Ray uid=" << ray.uniqueId()
169 << " pos=" << rays_position[ipart]
170 << " dir=" << rays_direction[ipart]
171 << " intersect_face_lid=" << face_lid
172 << " intersect_face_uid=" << faces_internal[face_lid]->uniqueId()
173 << " d=" << rays_distance[ipart]
174 << " p=" << rays_intersection[ipart];
175 }
176}
