87Real3 ItemRefinement::
88faceCenter(ItemInternal* iface,SharedVariableNodeReal3& nodes_coords) const
89{
90 Face face(iface);
91 Real3 pfc = Real3::null();
92 for( Node node : face.nodes() ){
93 pfc += nodes_coords[node];
94 }
95 pfc /= static_cast<Real> (face->nbNode());
96 return pfc ;
97}

102void ItemRefinement::
103initHMin()
104{
105 ENUMERATE_CELL(item,m_mesh->allCells()){
106 Real h_min=1.e30;
107 for (Integer i=0; i<item->nbNode(); i++)
108 for (Integer j=i+1; j<item->nbNode(); j++){
109 Real3 diff = (m_orig_nodes_coords[item->node(i)] - m_orig_nodes_coords[item->node(j)]) ;
110 h_min = std::min(h_min,diff.normL2());
111 }
112 m_cell_hmin[item] = h_min;
113 }
114}

156template <int typeID> void ItemRefinement::
157refineOneCell(Cell item, MeshRefinement& mesh_refinement)
158{
159 ItemTypeMng* itm = m_mesh->itemTypeMng();
160 const Int32 nb_hChildren = itm->nbHChildrenByItemType(item.type());
161 const Int32 nb_nodes = item.nbNode();
162 ARCANE_ASSERT((item.internal()->flags() & ItemInternal::II_Refine), ("Item is not flagged for refinement"));
163 ARCANE_ASSERT((item.isActive()), ("Refine non-active item is forbidden!"));
164
165 debug(Trace::High) << "[refineOneCell] nb_hChildren=" << nb_hChildren;
166 debug(Trace::High) << "[refineOneCell] nb_nodes=" << nb_nodes;
168
169 bool has_hChildren = item.hasHChildren();
170 debug(Trace::Highest) << "[refineOneCell] has_hChildren=" << has_hChildren;
171
172 m_nb_cell_to_add = 0;
173 m_nb_face_to_add = 0;
174 m_nb_node_to_add = 0;
175
176 if (!has_hChildren){
177 // Creation des enfants
178 computeHChildren<typeID>(item, mesh_refinement);
179 debug()<<"[ItemRefinement::refineOneCell] "<<m_nb_cell_to_add<<" new cells, "
180 <<m_nb_node_to_add<<" new nodes & "<<m_nb_face_to_add<<" faces";
181
182 // Créé les noeuds et positionne leur coordonnées
183 {
184 m_nodes_lid.resize(m_nb_node_to_add);
185 m_mesh->modifier()->addNodes(m_nodes_unique_id, m_nodes_lid);
186 m_mesh->nodeFamily()->endUpdate();
187 ItemInternalList nodes(m_mesh->nodeFamily()->itemsInternal());
188 for (Integer i = 0; i < m_nb_node_to_add; ++i) {
189 m_orig_nodes_coords[nodes[m_nodes_lid[i]]] = m_nodes_to_create_coords[i];
190 }
191 }
192
193 // Créé les faces
194 {
195 m_faces_lid.resize(m_nb_face_to_add);
196 m_mesh->modifier()->addFaces(m_nb_face_to_add, m_faces_infos, m_faces_lid);
197 }
198
199 // Créé les mailles
200 {
201 m_cells_lid.resize(m_nb_cell_to_add);
202 m_mesh->modifier()->addHChildrenCells(item.internal(), m_nb_cell_to_add, m_cells_infos, m_cells_lid);
204 ItemInternalList cells(m_mesh->cellFamily()->itemsInternal());
205 for (Integer i = 0; i < m_nb_cell_to_add; ++i){
206 ItemInternal* child = cells[m_cells_lid[i]];
207 Integer cf = child->flags();
208 cf |= ItemInternal::II_JustAdded;
209 child->setFlags(cf);
210 }
211 }
212 }
213 else{
214 for (Integer c = 0; c < nb_hChildren; c++){
215 Cell child = item.hChild(c);
216 //debug() << "[refineOneCell] child #"<<child->localId();
217 ARCANE_ASSERT((child.isSubactive()), ("child must be a sub active item!"));
218 Integer f = child.internal()->flags();
219 f |= ItemInternal::II_JustAdded;
220 f &= ~ItemInternal::II_Inactive;
221 child.internal()->setFlags(f);
222 }
223 }
224
225 // Maintenant, Unset le flag de raffinement de l'item
226 //debug(Trace::High) << "[refineOneCell] et on flush le flag";
227 Integer f = item.internal()->flags();
228 f &= ~ItemInternal::II_Refine;
229 f |= ItemInternal::II_Inactive;
230 f |= ItemInternal::II_JustRefined;
231 item.internal()->setFlags(f);
232#if defined(ARCANE_DEBUG_ASSERT)
233 for (Integer c = 0; c < nb_hChildren; c++){
234 Cell hParent = item.hChild(c).hParent();
235 //debug() << "[refineOneCell] child #"<<c;
236 ARCANE_ASSERT((hParent == item), ("parent-child relationship is not consistent"));
237 ARCANE_ASSERT((item.hChild(c).isActive()), ("children must be active"));
238 }
239 ARCANE_ASSERT((item.isAncestor()), ("current item must be an ancestor!"));
240#endif
241}

246template <int typeID> void ItemRefinement::
247coarsenOneCell(Cell item, const ItemRefinementPatternT<typeID>& rp)
248{
251
252 ARCANE_ASSERT ( (item->internal()->flags() & ItemInternal::II_CoarsenInactive), ("Item is not for coarsening!"));
253 ARCANE_ASSERT ( (!item->isActive()), ("Item is active!"));
254 //debug(Trace::High) << "[coarsenOneCell] "<<item_internal->uniqueId();
255 // ATT: Nous ne supprimons pas les enfants jusqu'à contraction via MeshRefinement::contract()
256
257 // re-calcul des noeuds hanging
258 IParallelMng* pm = m_mesh->parallelMng();
259 const Integer sid = pm->commRank();
260 computeOrigNodesCoords<typeID>(item,rp,sid);
261
262 for (Integer c=0; c<item.nbHChildren(); c++){
263 Cell mychild = item.hChild(c);
264 //debug(Trace::High) << "\t[coarsenOneCell] child #"<<c << ' ' << mychild->uniqueId() << " " << mychild->owner();
265
266 if (mychild->owner() != sid)
267 continue;
268 Integer f = mychild.internal()->flags();
269 ARCANE_ASSERT ((f & ItemInternal::II_Coarsen),("Item is not flagged for coarsening"));
270 f &= ~ItemInternal::II_Coarsen;
271 f |= ItemInternal::II_Inactive;
272 // f |= ItemInternal::II_NeedRemove; // TODO activer le flag de suppression
273 mychild.internal()->setFlags(f);
274 }
275 Integer f = item.internal()->flags();
276 f &= ~ItemInternal::II_Inactive; // TODO verify if this condition is needed
277 f &= ~ItemInternal::II_CoarsenInactive;
278 f |= ItemInternal::II_JustCoarsened;
279 //debug(Trace::High) << "[coarsenOneCell] item_internal->flags()="<<f;
280 item.internal()->setFlags(f);
281
282 ARCANE_ASSERT ( (item.isActive()), ("item must be active!"));
283 //debug() << "[coarsenOneCell] done";
284}

289template <int typeID> void ItemRefinement::
290computeHChildren(Cell item, MeshRefinement& mesh_refinement)
291{
292 debug(Trace::High) << "[refineOneCell] Computing new coordinates";
293
294 const ItemRefinementPatternT<typeID>& rp = mesh_refinement.getRefinementPattern<typeID>();
295 const Int32 nb_hChildren = rp.getNbHChildren();
296
297 m_p.resize(nb_hChildren);
298 m_nodes_uid.resize(nb_hChildren);
299
300 const Int32 nb_nodes = item.nbNode();
301 m_coord.resize(nb_nodes);
302 for (Integer i = 0; i < nb_nodes; ++i)
303 m_coord[i] = m_orig_nodes_coords[item.node(i)];
304
305 const Integer head_size = 2;
306 m_cells_infos.clear();
307 m_cells_infos.reserve(nb_hChildren * (head_size + nb_nodes));
308 debug(Trace::High) << "[refineOneCell] cells_infos reserved size is " << (nb_hChildren * (head_size + nb_nodes));
309
310 Int64 first_cell_uid = mesh_refinement.getFirstChildNewUid();
311 debug(Trace::Highest) << "[refineOneCell] first_cell_uid=" << first_cell_uid;
312
313 m_faces_infos.clear();
314 m_faces_infos.reserve(m_nb_face_to_add * ((m_mesh->dimension() == 3) ? 6 : (m_mesh->dimension() == 2) ? 4 : -1));
315
316 const Real tol = m_cell_hmin[item] * TOLERENCE;
317
318 typedef std::set<Int64> NodesSet;
319 NodesSet nodes_set;
320 Integer nb_cnodes_max_total = 0;
321
322 ItemTypeMng* itm = m_mesh->itemTypeMng();
323 for (Integer c = 0; c < nb_hChildren; c++){
324 const Integer c_type_id = rp.hChildrenTypeId(c);
325 ItemTypeInfo* c_type = itm->typeFromId(c_type_id);
326
327 const Integer nb_cnodes = c_type->nbLocalNode();
328 nb_cnodes_max_total += nb_cnodes;
329 m_p[c].resize(nb_cnodes);
330 m_nodes_uid[c].resize(nb_cnodes);
331
332 for (Integer nc = 0; nc < nb_cnodes; nc++){
333 // initialisation
334 m_p[c][nc] = Real3::null();
335 m_nodes_uid[c][nc] = NULL_ITEM_ID;
336
337 for (Integer n = 0; n < nb_nodes; n++){
338 // La valeur à partir de la matrice de raffinement
339 const Real em_val = rp.refine_matrix(c, nc, n);
340
341 if (em_val != 0.){
342 m_p[c][nc] += m_coord[n] * em_val;
343
344 //nous avons pu trouver le noeud, dans ce cas nous
345 //n'aurons pas besoin de le chercher plus tard ni de le creer.
346 if (em_val == 1.){
347 m_nodes_uid[c][nc] = item.node(n)->uniqueId().asInt64();
348 nodes_set.insert(m_nodes_uid[c][nc]);
349 }
350 }
351 }
352
353 // assignation des noeuds aux enfants
354 if (m_nodes_uid[c][nc] == NULL_ITEM_ID){
355 m_nodes_uid[c][nc] = mesh_refinement.findOrAddNodeUid(m_p[c][nc], tol);
356 debug(Trace::Highest) << "\t[refineOneCell] assigning node " << nc << " to l'uid:" << m_nodes_uid[c][nc];
357 }
358 m_nb_node_to_add = m_nb_node_to_add + 1;
359 }
360
361 // Création des mailles
362
363 // Infos pour la création des mailles
364 // par maille: 1 pour son unique id,
365 // 1 pour son type,
366 // nb_nodes pour les uid de ses noeuds
367
368 Int64 cell_unique_id = first_cell_uid + c;
369 debug(Trace::Highest) << "[refineOneCell] CELL TYPE:" << c_type_id << ", uid=" << cell_unique_id;
370
371 m_cells_infos.add(c_type_id);
372 m_cells_infos.add(cell_unique_id);
373 for (Integer nc = 0; nc < nb_cnodes; nc++)
374 m_cells_infos.add(m_nodes_uid[c][nc]);
375 m_nb_cell_to_add = m_nb_cell_to_add + 1;
376
377 // Création des faces
378
379 // Infos pour la création des faces
380 // par face: 1 pour son unique id,
381 // 1 pour son type,
382 // nb_nodes pour ses noeuds
383
384 const Integer nb_cface = c_type->nbLocalFace();
385 debug(Trace::High) << "[refineOneCell] nb faces à créer:" << nb_cface;
386
387 for (Integer f = 0; f < nb_cface; f++){
388 const Integer nb_node_face = c_type->localFace(f).nbNode();
389 m_face.resize(nb_node_face);
390 Real3 pfc = Real3::null();
391 for (Integer nc = 0; nc < nb_node_face; nc++){
392 const Integer node_face_rank = c_type->localFace(f).node(nc);
393 m_face[nc] = m_nodes_uid[c][node_face_rank];
394 pfc += m_p[c][node_face_rank];
395 }
396 pfc /= static_cast<Real>(nb_node_face);
397
398 bool is_added = false;
399 Int64 new_face_uid = mesh_refinement.findOrAddFaceUid(pfc, tol, is_added);
400 debug(Trace::Highest) << "\t[refineOneCell] FACE TYPE:" << c_type->localFace(f).typeId();
401 if (is_added) {
402 m_faces_infos.add(c_type->localFace(f).typeId());
403 m_faces_infos.add(new_face_uid);
404 for (Integer nc = 0; nc < nb_node_face; nc++)
405 m_faces_infos.add(m_face[nc]);
406 m_nb_face_to_add = m_nb_face_to_add + 1;
407 }
408 }
409 }
410
411 m_nodes_to_create_coords.clear();
412 m_nodes_to_create_coords.reserve(nb_cnodes_max_total);
413 m_nodes_unique_id.clear();
414 m_nodes_unique_id.reserve(nb_cnodes_max_total);
415
416 Integer node_local_id = 0;
417 debug(Trace::High) << "[refineOneCell] Create nodes and set their coordinates";
418 for (Integer c = 0; c < nb_hChildren; c++){
419 const Integer c_type_id = rp.hChildrenTypeId(c);
420 ItemTypeInfo* c_type = itm->typeFromId(c_type_id);
421
422 const Int32 nb_cnodes = c_type->nbLocalNode();
423 for (Integer nc = 0; nc < nb_cnodes; nc++){
424 const Int64 uid = m_nodes_uid[c][nc];
425 if (nodes_set.find(uid) != nodes_set.end())
426 continue;
427 else{
428 nodes_set.insert(uid);
429 m_nodes_to_create_coords.add(m_p[c][nc]);
430 m_nodes_unique_id.add(uid);
431 ++node_local_id;
432 }
433 }
434 }
435 m_nb_node_to_add = node_local_id;
436}

441template <int typeID> void ItemRefinement::
442computeOrigNodesCoords(Cell item, const ItemRefinementPatternT<typeID>& rp, const Integer sid)
443{
444 const Integer nb_nodes = item.nbNode();
445 m_coord.resize(nb_nodes);
446 for (Integer i = 0; i < nb_nodes; ++i)
447 m_coord[i] = m_orig_nodes_coords[item.node(i)];
448
449 for (Integer c = 0; c < item.nbHChildren(); c++){
450 //debug() << "[coarsenOneCell] child #"<<c;
451 Cell mychild = item.hChild(c);
452 if (mychild.owner() != sid)
453 continue;
454 for (Integer nc = 0; nc < mychild.nbNode(); nc++){
455 //debug() << "\t[coarsenOneCell] node #"<<nc;
456 Real3 new_pos = Real3::null();
457 bool calculated_new_pos = false;
458
459 for (Integer n = 0; n < nb_nodes; n++){
460 //debug() << "\t\t[coarsenOneCell] pos #"<<n;
461 // La valeur à partir de la matrice de raffinement
462 const Real em_val = rp.refine_matrix(c, nc, n);
463
464 // La position du noeud est quelque part entre les sommets existants
465 if ((em_val != 0.) && (em_val != 1.)){
466 new_pos += em_val * m_coord[n];
467 calculated_new_pos = true;
468 }
469 }
470
471 if (calculated_new_pos)
472 //Déplacement du noeud existant de nouveau dans sa position d'origine
473 m_orig_nodes_coords[mychild.node(nc)] = new_pos;
474 }
475 }
476}
