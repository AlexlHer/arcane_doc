48class FamilyInternal
49{
50private:
51 typedef std::set<IFamilyObserver*> FamilyObservers;
52
53public:
54
55 FamilyInternal() : m_max_local_id(0) {}
56 ~FamilyInternal()
57 {
58 clear();
59 }
60
61public:
62
64 FamilyInternal& operator<<(GroupBuilder builder)
65 {
66 ItemGroup group = builder.group();
67 const Integer size = m_groups.size();
68
69 if (m_groups.findGroupInfo(group.internal()) != NULL)
70 throw FatalErrorException(String::format("Group '{0}' already registered",group.name()));
71
72 m_groups.resize(size+1);
73 Private::GroupIndexInfo & info = m_groups[size];
74 info.group = group.internal();
75 info.group_index = size;
76 info.local_id_offset = m_max_local_id;
77 m_max_local_id += group.size();
78 info.is_partial = builder.isPartial();
79 info.group->attachObserver(this, newItemGroupObserverT(this, &FamilyInternal::_notifyGroupHasChanged));
80 // std::cout << "Attach " << this << " observer on group " << group.name() << "\n";
81
82 // On previent les observeurs
83 _notifyFamilyIsIncreased();
84 return *this;
85 }
86
88 inline bool contains(const ItemGroup& group) const
89 {
90 return (m_groups.findGroupInfo(group.internal()) != NULL);
91 }
92
94 inline bool isPartial(const ItemGroup& group) const
95 {
96 const Private::GroupIndexInfo * info = m_groups.findGroupInfo(group.internal());
97 if (info == NULL)
98 throw Arcane::FatalErrorException(Arcane::String::format("Group '{0}' not registered",group.name()));
99 return info->is_partial;
100 }
101
103 // Ce groupe n'a pas besoin d'observer la famille car il partage les données
104 inline Group allItems() const {
105 return m_groups;
106 }
107
109 inline Integer groupIndex(const ItemGroup& group) const {
110 const Private::GroupIndexInfo * info = m_groups.findGroupInfo(group.internal());
111 if (info == NULL)
112 throw FatalErrorException(String::format("Group '{0}' not registered",group.name()));
113 return info->group_index;
114 }
115
117 inline Integer firstLocalId(const ItemGroup& group) const {
118 const Private::GroupIndexInfo * info = m_groups.findGroupInfo(group.internal());
119 if (info == NULL)
120 throw FatalErrorException(String::format("Group '{0}' not registered",group.name()));
121 return info->local_id_offset;
122 }
123
125 template<typename AnyItemT>
126 Item item(const AnyItemT & any_item) const {
127 const Integer group_index = any_item.groupIndex();
128 const Private::GroupIndexInfo & info = m_groups[group_index];
129 const ItemGroupImpl & group = *(info.group);
130 Integer index_in_group = any_item.localId() - info.local_id_offset;
131 ItemInternal * item = group.itemsInternal()[group.itemsLocalId()[index_in_group]];
132 // ARCANE_ASSERT((!info.is_partial || (item->localId() == any_item.varIndex())),("Inconsistent concrete item"));
133 // ARCANE_ASSERT((item->isOwn() == any_item.m_is_own),("Inconsistent concrete item isOwn"));
134 return item;
135 }
136
138 inline Integer groupSize() const {
139 return m_groups.size();
140 }
141
143
144 inline Integer maxLocalId() const {
145 return m_max_local_id;
146 }
147
149 ItemGroup group(Integer i) const {
150 return ItemGroup(m_groups[i].group);
151 }
152
154 void clear() {
155 for(Integer igrp=0;igrp<m_groups.size();++igrp) {
156 m_groups[igrp].group->detachObserver(this);
157 // std::cout << "Detach " << this << " observer on group " << m_groups[igrp].group->name() << "\n";
158 }
159 // On efface
160 m_groups.clear();
161 m_max_local_id = 0;
162 // On previent les observeurs
163 _notifyFamilyIsInvalidate();
164 }
165
167 void registerObserver(IFamilyObserver& observer) const
168 {
169 FamilyObservers::const_iterator it = m_observers.find(&observer);
170 if(it != m_observers.end())
171 throw FatalErrorException("FamilyObserver already registered");
172 m_observers.insert(&observer);
173 }
174
176 void removeObserver(IFamilyObserver& observer) const
177 {
178 FamilyObservers::const_iterator it = m_observers.find(&observer);
179 if(it == m_observers.end())
180 throw FatalErrorException("FamilyObserver not registered");
181 m_observers.erase(it);
182 }
183
184public:
185 const Private::GroupIndexInfo * findGroupInfo(ItemGroup agroup) {
186 return m_groups.findGroupInfo(agroup.internal());
187 }
188
189private:
190
191 void _notifyFamilyIsInvalidate() {
192 for(FamilyObservers::iterator it = m_observers.begin(); it != m_observers.end(); ++it)
193 (*it)->notifyFamilyIsInvalidate();
194 }
195
196 void _notifyFamilyIsIncreased() {
197 for(FamilyObservers::iterator it = m_observers.begin(); it != m_observers.end(); ++it)
198 (*it)->notifyFamilyIsIncreased();
199 }
200
201 void _notifyGroupHasChanged() {
202 throw FatalErrorException(A_FUNCINFO, "Group changes while registered in AnyItem::Family");
203 }
204
205private:
206
208 Private::GroupIndexMapping m_groups;
209
211 Integer m_max_local_id;
212
214 mutable FamilyObservers m_observers;
215};
