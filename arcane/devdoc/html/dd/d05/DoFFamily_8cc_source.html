82void
83DoFFamily::
84addItems(Int64ConstArrayView unique_ids, Int32ArrayView items)
85{
86 Integer nb_item = unique_ids.size();
87 if (nb_item==0)
88 return;
89 preAllocate(nb_item);
90 for( Integer i=0; i<nb_item; ++i ){
91 Int64 uid = unique_ids[i];
92 ItemInternal* ii = _allocDoF(uid);
93 items[i] = ii->localId();
94 }
95
96 m_need_prepare_dump = true;
97}

178DoFFamily::

180{
181 // Copy paste de particle, pas utilise pour l'instant

183 Integer wanted_size = 2*(nb_item+infos().nbItem());
184 if (nb_hash<wanted_size)

195ItemInternal* DoFFamily::
196_allocDoF(const Int64 uid)
197{
198 bool need_alloc; // given by alloc
199 ItemInternal* item_internal = ItemFamily::_allocOne(uid,need_alloc);
200 if (!need_alloc)
201 item_internal->setUniqueId(uid);
202 else{
203 _allocateInfos(item_internal,uid,m_shared_info);
204 }
205 // Un dof appartient de base au sous-domaine qui l'a créé (sauf ghost)
206 item_internal->setOwner(m_sub_domain_id,m_sub_domain_id);
207 return item_internal;
208}
