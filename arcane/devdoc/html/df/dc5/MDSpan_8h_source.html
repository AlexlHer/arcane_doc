55template<typename DataType,int RankValue,typename LayoutType>
56class MDSpanBase
57{
58 using UnqualifiedValueType = std::remove_cv_t<DataType>;
59 friend class NumArrayBase<UnqualifiedValueType,RankValue,LayoutType>;
60 // Pour que MDSpan<const T> ait accès à MDSpan<T>
61 friend class MDSpanBase<const UnqualifiedValueType,RankValue,LayoutType>;
62 public:
63 MDSpanBase() = default;
64 ARCCORE_HOST_DEVICE MDSpanBase(DataType* ptr,ArrayExtentsWithOffset<RankValue,LayoutType> extents)
65 : m_ptr(ptr), m_extents(extents)
66 {
67 }
68 // Constructeur MDSpan<const T> à partir d'un MDSpan<T>
69 template<typename X,typename = std::enable_if_t<std::is_same_v<X,UnqualifiedValueType>>>
70 ARCCORE_HOST_DEVICE MDSpanBase(const MDSpanBase<X,RankValue>& rhs)
71 : m_ptr(rhs.m_ptr), m_extents(rhs.m_extents){}
72 public:
73 ARCCORE_HOST_DEVICE DataType* _internalData() { return m_ptr; }
74 ARCCORE_HOST_DEVICE const DataType* _internalData() const { return m_ptr; }
75 public:
76 ArrayExtents<RankValue> extents() const
77 {
78 return m_extents.extents();
79 }
80 ArrayExtentsWithOffset<RankValue,LayoutType> extentsWithOffset() const
81 {
82 return m_extents;
83 }
84 Int32 extent(int i) const { return m_extents(i); }
85 public:
86 ARCCORE_HOST_DEVICE Int64 offset(ArrayBoundsIndex<RankValue> idx) const
87 {
88 return m_extents.offset(idx);
89 }
91 ARCCORE_HOST_DEVICE DataType& operator()(ArrayBoundsIndex<RankValue> idx) const
92 {
93 return m_ptr[offset(idx)];
94 }
96 ARCCORE_HOST_DEVICE DataType* ptrAt(ArrayBoundsIndex<RankValue> idx) const
97 {
98 return m_ptr+offset(idx);
99 }
100 public:
101 ARCCORE_HOST_DEVICE MDSpanBase<const DataType,RankValue> constSpan() const
102 { return MDSpanBase<const DataType,RankValue>(m_ptr,m_extents); }
103 ARCCORE_HOST_DEVICE Span<DataType> to1DSpan() const { return { m_ptr, m_extents.totalNbElement() }; }
104 private:
105 // Utilisé uniquement par NumArrayBase pour la copie
106 Span2<const DataType> _internalTo2DSpan() const
107 {
108 Int64 dim1_size = m_extents(0);
109 Int64 dim2_size = 1;
110 for (int i=1; i<RankValue; ++i )
111 dim2_size *= m_extents(i);
112 return { m_ptr, dim1_size, dim2_size };
113 }
114 protected:
115 DataType* m_ptr = nullptr;
116 ArrayExtentsWithOffset<RankValue,LayoutType> m_extents;
117};

145 {

147 m_ptr = ptr;
148 }

151
152 public:

155 public:

160 ARCCORE_HOST_DEVICE DataType* ptrAt(Int32 i) const { return m_ptr+offset(i); }
162 ARCCORE_HOST_DEVICE DataType operator[](Int32 i) const { return m_ptr[offset(i)]; }
164 template<typename X = DataType,typename SubType = typename NumericTraitsT<X>::SubscriptType >

168 ARCCORE_HOST_DEVICE Sub2Type operator()(Int32 i,Int32 a,Int32 b) const { return m_ptr[offset(i)][a][b]; }
169 public:
170 ARCCORE_HOST_DEVICE MDSpan<const DataType,1,LayoutType> constSpan() const
171 { return MDSpan<const DataType,1,LayoutType>(m_ptr,m_extents); }
172};
