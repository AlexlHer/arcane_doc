31void ItemFamilyNetwork::
32addDependency(IItemFamily* master_family, IItemFamily* slave_family, IIncrementalItemConnectivity* master_to_slave_connectivity, bool is_deep_connectivity)
33{
34 m_dependency_graph.addEdge(master_family,slave_family,master_to_slave_connectivity);
35 if(master_to_slave_connectivity)
36 {
37 m_connectivity_list.add(master_to_slave_connectivity);
38 m_connectivity_status[master_to_slave_connectivity] = std::make_pair(false,is_deep_connectivity); // connectivity not stored by default
39 }
40 m_families.insert(master_family);
41 m_families.insert(slave_family);
42
43 m_is_activated = true ;
44}

49void ItemFamilyNetwork::
50addRelation(IItemFamily* source_family, IItemFamily* target_family, IIncrementalItemConnectivity* source_to_target_connectivity)
51{
52 m_relation_graph.addEdge(source_family,target_family,source_to_target_connectivity);
53 m_connectivity_list.add(source_to_target_connectivity);
54 if(m_connectivity_status.find(source_to_target_connectivity)==m_connectivity_status.end())
55 m_connectivity_status[source_to_target_connectivity] = std::make_pair(false,true); // connectivity not stored by default
56 m_families.insert(source_family);
57 m_families.insert(target_family);
58 m_is_activated = true ;
59}

74IIncrementalItemConnectivity* ItemFamilyNetwork::
75getConnectivity(IItemFamily* source_family, IItemFamily* target_family, const String& name, bool& is_dependency)
76{
77 // Up to now DirectedGraph and DirectedAcyclicGraph are not hypergraph, we cannot have multiple edges between two same nodes.
78 // => we just check the name is OK
79 auto* connectivity = m_dependency_graph.getEdge(source_family,target_family);
80 is_dependency = false;
81 if (connectivity)
82 {
83 _checkConnectivityName(*connectivity,name);
84 is_dependency = true;
85 return *connectivity;
86 }
87 connectivity = m_relation_graph.getEdge(source_family,target_family);
88 if (connectivity)
89 {
90 _checkConnectivityName(*connectivity,name);
91 return *connectivity;
92 }
93 else return nullptr;
94}

184SharedArray<IIncrementalItemConnectivity*> ItemFamilyNetwork::
185_getConnectivitiesFromGraph(const ConnectivityGraph::ConnectedEdgeSet& connectivity_edges)
186{
187 SharedArray<IIncrementalItemConnectivity*> connectivities(connectivity_edges.size());

190 connectivities[index++] = connectivity_edge;
191 }

247SharedArray<IItemFamily*> ItemFamilyNetwork::
248getFamilies(eSchedulingOrder order) const
249{
250 SharedArray<IItemFamily*> families;
251
252 switch (order)
253 {
254 case TopologicalOrder:
255 for (auto family : m_dependency_graph.topologicalSort())
256 {
257 families.add(family) ;
258 }
259 break;
260 case InverseTopologicalOrder:
261 for (auto family : m_dependency_graph.topologicalSort().reverseOrder())
262 {
263 families.add(family) ;
264 }
265 break;
266 case Unknown:
267 throw m_trace_mng->fatal() << "Cannot schedule task, scheduling order is unkwnown. Set Scheduling order";
268 break;
269 }
270 return families ;
271}

273void ItemFamilyNetwork::
274schedule(IItemFamilyNetworkTask task, eSchedulingOrder scheduling_order)
275{
276 switch (scheduling_order) {
277 case TopologicalOrder:
278 for (auto family : m_dependency_graph.topologicalSort())
279 {
280 task(family);
281 }
282 break;
283 case InverseTopologicalOrder:
284 for (auto family : m_dependency_graph.topologicalSort().reverseOrder())
285 {
286 task(family);
287 }
288 break;
289 case Unknown:
290 throw m_trace_mng->fatal() << "Cannot schedule task, scheduling order is unkwnown. Set Scheduling order";
291 break;
292 }
293}

298std::pair<IIncrementalItemConnectivity* const,std::pair<bool,bool>>& ItemFamilyNetwork::
299_getConnectivityStatus(IIncrementalItemConnectivity* connectivity)
300{
301 auto connectivity_iterator = m_connectivity_status.find(connectivity);
302 if (connectivity_iterator == m_connectivity_status.end()) throw FatalErrorException(String::format("Cannot find connectivity {0} between families {1} and {2}",
303 connectivity->name(),connectivity->sourceFamily(),connectivity->targetFamily()));
304 return *(connectivity_iterator);
305}
