48class MeshEnvironmentObserver
49: public TraceAccessor
50, public IItemGroupObserver
51{
52 public:
53 MeshEnvironmentObserver(MeshEnvironment* env,ITraceMng* tm)
54 : TraceAccessor(tm), m_environment(env){}
55 public:
56 void executeExtend(const Int32ConstArrayView* info1) override
57 {
58 if (info1){
59 info(4) << "EXTEND_ENV " << m_environment->name() << " ids=" << (*info1);
60 if (m_environment->materialMng()->isInMeshMaterialExchange())
61 info() << "EXTEND_ENV_IN_LOADBALANCE " << m_environment->name()
62 << " ids=" << (*info1);
63 }
64 }
65 void executeReduce(const Int32ConstArrayView* info1) override
66 {
67 if (info1){
68 info(4) << "REDUCE_ENV " << m_environment->name() << " ids=" << (*info1);
69 if (m_environment->materialMng()->isInMeshMaterialExchange())
70 info() << "REDUCE_ENV_IN_LOADBALANCE " << m_environment->name()
71 << " ids=" << (*info1);
72 }
73 }
74 void executeCompact(const Int32ConstArrayView* info1) override
75 {
76 info(4) << "COMPACT_ENV " << m_environment->name();
77 if (!info1)
78 ARCANE_FATAL("No info available");
79 Int32ConstArrayView old_to_new_ids(*info1);
80 m_environment->notifyLocalIdsChanged(old_to_new_ids);
81 }
82 void executeInvalidate() override
83 {
84 info() << "WARNING: invalidate() is invalid on an environment group! partial values may be corrupted"
85 << " env=" << m_environment->name();
86 }
87 bool needInfo() const override { return true; }
88 private:
89 MeshEnvironment* m_environment;
90};

129void MeshEnvironment::
130build()
131{
132 IMesh* mesh = m_material_mng->mesh();
133 IItemFamily* cell_family = mesh->cellFamily();
134 String group_name = m_material_mng->name() + "_" + name();
135 CellGroup cells = cell_family->findGroup(group_name,true);
136
137 if (m_material_mng->isMeshModificationNotified()){
138 m_group_observer = new MeshEnvironmentObserver(this,traceMng());
139 cells.internal()->attachObserver(this,m_group_observer);
140 }
141
142 m_data.setItems(cells);
143}

158void MeshEnvironment::
159setVariableIndexer(MeshMaterialVariableIndexer* idx)
160{
161 m_data.setVariableIndexer(idx);
162 idx->setCells(m_data.items());
163 idx->setIsEnvironment(true);
164
165 // S'il n'y qu'un matériau, le variable indexer de ce matériau est
166 // aussi 'idx' mais avec un autre groupe associé. Pour que tout soit
167 // cohérent, il faut être sur que ce matériau a aussi le même groupe.
168 // TODO: pour garantir la cohérence, il faudrait supprimer
169 // dans m_data le groupe d'entité.
170 if (m_true_materials.size()==1)
171 m_true_materials[0]->componentData()->setItems(m_data.items());
172 m_data.buildPartData();
173 for( MeshMaterial* mat : m_true_materials.range() )
174 mat->componentData()->buildPartData();
175}

180void MeshEnvironment::
181computeNbMatPerCell()
182{
183 info(4) << "ComputeNbMatPerCell env=" << name();
184 Integer nb_mat = m_materials.size();
185 m_nb_mat_per_cell.fill(0);
186 Integer total = 0;
187 for( Integer i=0; i<nb_mat; ++i ){
188 IMeshMaterial* mat = m_materials[i];
189 CellGroup mat_cells = mat->cells();
190 total += mat_cells.size();
191 ENUMERATE_CELL(icell,mat_cells){
192 ++m_nb_mat_per_cell[icell];
193 }
194 }
195 m_total_nb_cell_mat = total;
196 ENUMERATE_CELL(icell,this->cells()){
197 Int32 n = m_nb_mat_per_cell[icell];
198 if (n==0)
199 ARCANE_FATAL("No material in cell '{0}' for environment '{1}'",
200 icell->uniqueId(),name());
201 }
202}

213void MeshEnvironment::
214computeMaterialIndexes()
215{
216 info(4) << "Compute (V2) indexes for environment name=" << name();
217
218 IItemFamily* cell_family = cells().itemFamily();
219 Integer max_local_id = cell_family->maxLocalId();
220 Integer total_nb_cell_mat = m_total_nb_cell_mat;
221 m_mat_items_internal.resize(total_nb_cell_mat);
222 for( Integer i=0; i<total_nb_cell_mat; ++i ){
223 ComponentItemInternal& ref_ii = m_mat_items_internal[i];
224 ref_ii.reset();
225 }
226
227 Int32UniqueArray cells_index(max_local_id);
228 Int32UniqueArray cells_pos(max_local_id);
229 //TODO: regarder comment supprimer ce tableau cells_env qui n'est normalement pas utile
230 // car on doit pouvoir directement utiliser les m_items_internal
231 UniqueArray<ComponentItemInternal*> cells_env(max_local_id);
232 Int32ArrayView nb_mat_per_cell = m_nb_mat_per_cell.asArray();
233 ConstArrayView<ComponentItemInternal*> items_internal = itemsInternalView();
234
235 {
236 Integer cell_index = 0;
237 Int32 env_id = this->id();
238 Int32ConstArrayView local_ids = variableIndexer()->localIds();
239
240 for( Integer z=0, nb=local_ids.size(); z<nb; ++z ){
241 Int32 lid = local_ids[z];
242 Int32 nb_mat = nb_mat_per_cell[lid];
243 ComponentItemInternal* env_item = items_internal[z];
244 cells_index[lid] = cell_index;
245 cells_pos[lid] = cell_index;
246 //info(4) << "XZ=" << z << " LID=" << lid << " POS=" << cell_index;
247 env_item->setNbSubItem(nb_mat);
248 env_item->setComponent(this,env_id);
249 if (nb_mat!=0){
250 env_item->setFirstSubItem(&m_mat_items_internal[cell_index]);
251 cells_env[lid] = env_item;
252 }
253 cell_index += nb_mat;
254 }
255 }
256 {
257 Integer nb_mat = m_true_materials.size();
258 ItemInternalList items_internal(cell_family->itemsInternal());
259 for( Integer i=0; i<nb_mat; ++i ){
260 MeshMaterial* mat = m_true_materials[i];
261 Int32 mat_id = mat->id();
262 const MeshMaterialVariableIndexer* var_indexer = mat->variableIndexer();
263 CellGroup mat_cells = mat->cells();
264 info(4) << "COMPUTE (V2) mat_cells mat=" << mat->name() << " nb_cell=" << mat_cells.size()
265 << " mat_id=" << mat_id << " index=" << var_indexer->index();
266
267 mat->resizeItemsInternal(var_indexer->nbItem());
268
269 ConstArrayView<MatVarIndex> matvar_indexes = var_indexer->matvarIndexes();
270
271 ArrayView<ComponentItemInternal*> mat_items_internal = mat->itemsInternalView();
272 Int32ConstArrayView local_ids = var_indexer->localIds();
273
274 for( Integer z=0, nb_id = matvar_indexes.size(); z<nb_id; ++z){
275 MatVarIndex mvi = matvar_indexes[z];
276 Int32 lid = local_ids[z];
277 Int32 pos = cells_pos[lid];
278 //info(4) << "Z=" << z << " LID=" << lid << " POS=" << pos;
279 ++cells_pos[lid];
280 ComponentItemInternal& ref_ii = m_mat_items_internal[pos];
281 mat_items_internal[z] = &m_mat_items_internal[pos];
282 ref_ii.setSuperAndGlobalItem(cells_env[lid],items_internal[lid]);
283 ref_ii.setComponent(mat,mat_id);
284 ref_ii.setVariableIndex(mvi);
285 ref_ii.setLevel(LEVEL_MATERIAL);
286 }
287 }
288 }
289}

297void MeshEnvironment::
298computeItemListForMaterials(const VariableCellInt32& nb_env_per_cell)
299{
300 info(4) << "ComputeItemListForMaterials (V2)";
301
302 // Calcul pour chaque matériau le nombre de mailles mixtes
303 // TODO: a faire dans MeshMaterialVariableIndexer
304 for( Integer i=0, n=m_true_materials.size(); i<n; ++i ){
305 MeshMaterialVariableIndexer* var_indexer = m_true_materials[i]->variableIndexer();
306 CellGroup cells = var_indexer->cells();
307 Integer var_nb_cell = cells.size();
308
309 ComponentItemListBuilder list_builder(var_indexer,0);
310
311 info(4) << "MAT_INDEXER i=" << i << " NB_CELL=" << var_nb_cell << " name=" << cells.name();
312 ENUMERATE_CELL(icell,var_indexer->cells()){
313 Int32 lid = icell.itemLocalId();
314 // On ne prend l'indice global que si on est le seul matériau et le seul
315 // milieu de la maille. Sinon, on prend un indice multiple
316 if (nb_env_per_cell[icell]>1 || m_nb_mat_per_cell[icell]>1)
317 list_builder.addPartialItem(lid);
318 else
319 list_builder.addPureItem(lid);
320 }
321
322 if (traceMng()->verbosityLevel()>=5)
323 info() << "MAT_NB_MULTIPLE_CELL (V2) mat=" << var_indexer->name()
324 << " nb_in_global=" << list_builder.pureMatVarIndexes().size()
325 << " (ids=" << list_builder.pureMatVarIndexes() << ")"
326 << " nb_in_multiple=" << list_builder.partialMatVarIndexes().size()
327 << " (ids=" << list_builder.partialLocalIds() << ")";
328 var_indexer->endUpdate(list_builder);
329 }
330}

335void MeshEnvironment::
336_addItemsToIndexer(const VariableCellInt32& nb_env_per_cell,
337 MeshMaterialVariableIndexer* var_indexer,
338 Int32ConstArrayView local_ids)
339{
340 IItemFamily* cell_family = cells().itemFamily();
341 ItemInternalList items_internal(cell_family->itemsInternal());
342
343 ComponentItemListBuilder list_builder(var_indexer,var_indexer->maxIndexInMultipleArray());
344
345 Integer nb_to_add = local_ids.size();
346 for( Integer i=0; i<nb_to_add; ++i ){
347 Int32 lid = local_ids[i];
348 Cell cell = items_internal[lid];
349 // On ne prend l'indice global que si on est le seul matériau et le seul
350 // milieu de la maille. Sinon, on prend un indice multiple
351 if (nb_env_per_cell[cell]>1 || m_nb_mat_per_cell[cell]>1)
352 list_builder.addPartialItem(lid);
353 else
354 list_builder.addPureItem(lid);
355 }
356
357 if (traceMng()->verbosityLevel()>=5)
358 info() << "MAT_NB_MULTIPLE_CELL (V2) mat=" << var_indexer->name()
359 << " nb_in_global=" << list_builder.pureMatVarIndexes().size()
360 << " (ids=" << list_builder.pureMatVarIndexes() << ")"
361 << " nb_in_multiple=" << list_builder.partialMatVarIndexes().size()
362 << " (ids=" << list_builder.partialLocalIds() << ")";
363 var_indexer->endUpdateAdd(list_builder);
364
365 // Maintenant que les nouveaux MatVar sont créés, il faut les
366 // initialiser avec les bonnes valeurs.
367 functor::apply(m_material_mng,&IMeshMaterialMng::visitVariables,
368 [=](IMeshMaterialVariable* mv) { mv->_initializeNewItems(list_builder); }
369 );
370}

382void MeshEnvironment::
383_addItemsDirect(const VariableCellInt32& nb_env_per_cell,MeshMaterial* mat,
384 Int32ConstArrayView local_ids,bool update_env_indexer)
385{
386 info(4) << "MeshEnvironment::addItemsDirect"
387 << " mat=" << mat->name();
388
389 IItemFamily* cell_family = cells().itemFamily();
390 ItemInternalList items_internal(cell_family->itemsInternal());
391
392 MeshMaterialVariableIndexer* var_indexer = mat->variableIndexer();
393 Integer nb_to_add = local_ids.size();
394
395 // Met à jour le nombre de matériaux par maille et le nombre total de mailles matériaux.
396 for( Integer i=0; i<nb_to_add; ++i ){
397 Int32 lid = local_ids[i];
398 Cell cell = items_internal[lid];
399 ++m_nb_mat_per_cell[cell];
400 }
401 m_total_nb_cell_mat += nb_to_add;
402
403 _addItemsToIndexer(nb_env_per_cell,var_indexer,local_ids);
404
405 if (update_env_indexer){
406 // Met aussi à jour les entités \a local_ids à l'indexeur du milieu.
407 // Cela n'est possible que si le nombre de matériaux du milieu
408 // est supérieur ou égal à 2 (car sinon c'est le matériau et le milieu
409 // ont le même indexeur)
410 _addItemsToIndexer(nb_env_per_cell,this->variableIndexer(),local_ids);
411 }
412}

415/*---------------------------------------------------------------------------*/
429void MeshEnvironment::
430_removeItemsDirect(MeshMaterial* mat,Int32ConstArrayView local_ids,
431 bool update_env_indexer)
432{
433 info(4) << "MeshEnvironment::removeItemsDirect mat=" << mat->name();
434
435 IItemFamily* cell_family = cells().itemFamily();
436 ItemInternalList items_internal(cell_family->itemsInternal());
437
438 Integer nb_to_remove = local_ids.size();
439
440 UniqueArray<bool> removed_local_ids_filter(cell_family->maxLocalId(),false);
441
442 // Met à jour le nombre de matériaux par maille et le nombre total de mailles matériaux.
443 for( Integer i=0; i<nb_to_remove; ++i ){
444 Int32 lid = local_ids[i];
445 Cell cell = items_internal[lid];
446 --m_nb_mat_per_cell[cell];
447 removed_local_ids_filter[lid] = true;
448 }
449 m_total_nb_cell_mat -= nb_to_remove;
450
451 mat->variableIndexer()->endUpdateRemove(removed_local_ids_filter,nb_to_remove);
452
453 if (update_env_indexer){
454 // Met aussi à jour les entités \a local_ids à l'indexeur du milieu.
455 // Cela n'est possible que si le nombre de matériaux du milieu
456 // est supérieur ou égal à 2 (car sinon le matériau et le milieu
457 // ont le même indexeur)
458 this->variableIndexer()->endUpdateRemove(removed_local_ids_filter,nb_to_remove);
459 }
460}

466updateItemsDirect(const VariableCellInt32& nb_env_per_cell,MeshMaterial* mat,
467 Int32ConstArrayView local_ids,eOperation operation,
468 bool remove_to_env_indexer)
469{
470 if (operation==eOperation::Add)

474 else
475 ARCANE_THROW(ArgumentException,"bad value for 'operation'");
476}

481void MeshEnvironment::
482notifyLocalIdsChanged(Int32ConstArrayView old_to_new_ids)
483{
484 // NOTE:
485 // Cette méthode est appelée lorsqu'il y a un compactage du maillage
486 // et le groupe d'entité associé à ce milieu vient d'être compacté.
487 // Comme actuellement il n'y a pas d'observeurs pour l'ajout
488 // ou la suppression de mailles du groupe, il est possible
489 // lorsque cette méthode est appelée que les infos des milieux et
490 // matériaux ne soient pas à jour (par exemple, la liste des local_ids
491 // du m_variable_indexer n'a pas les mêmes valeurs que cells().
492 // Pour l'instant ce n'est pas très grave car tout est écrasé après
493 // chaque modif sur un matériau ou un milieu.
494 // A terme, il faudra prendre cela en compte lorsque l'ajout
495 // où la suppression de mailles matériaux sera optimisée.
496 info(4) << "Changing (V3) local ids references env=" << name();
497 info(4) << "CurrentCells name=" << cells().name()
498 << " n=" << cells().view().localIds().size();
499 info(4) << "MatVarIndex name=" << cells().name()
500 << " n=" << variableIndexer()->matvarIndexes().size();
501 Integer nb_mat = m_true_materials.size();
502 info(4) << "NotifyLocalIdsChanged env=" << name() << " nb_mat=" << nb_mat
503 << " old_to_new_ids.size=" << old_to_new_ids.size();
504
505 // Si le milieu n'a qu'un seul matériau, ils partagent le même variable_indexer
506 // donc il ne faut changer les ids qu'une seule fois. Par contre, le
507 // tableau m_items_internal n'est pas partagé entre le matériau
508 // et le milieu donc il faut recalculer les infos séparément.
509 // Il faut le faire pour le milieu avant de mettre à jour les infos du matériau car
510 // une fois ceci fait la valeur m_variable_indexer->m_local_ids_in_indexes_view
511 // aura changé et il ne sera plus possible de déterminer la correspondance
512 // entre les nouveaux et les anciens localId
513
514 if (nb_mat==1){
515 m_data.changeLocalIdsForInternalList(old_to_new_ids);
516 MeshMaterial* true_mat = m_true_materials[0];
517 _changeIds(true_mat->componentData(),old_to_new_ids);
518 }
519 else{
520 // Change les infos des matériaux
521 for( Integer i=0; i<nb_mat; ++i ){
522 MeshMaterial* true_mat = m_true_materials[i];
523 info(4) << "ChangeIds MAT i=" << i << " MAT=" << true_mat->name();
524 _changeIds(true_mat->componentData(),old_to_new_ids);
525 }
526 // Change les infos du milieu
527 _changeIds(componentData(),old_to_new_ids);
528 }
529
530 // Reconstruit les infos sur les mailles pures et mixtes.
531 // Il faut le faire une fois que tous les valeurs sont à jour.
532 {
533 for( Integer i=0; i<nb_mat; ++i ){
534 MeshMaterial* true_mat = m_true_materials[i];
535 true_mat->componentData()->rebuildPartData();
536 }
537 componentData()->rebuildPartData();
538 }
539
540 checkValid();
541}
