95void MeshMaterialSyncUnitTest::
96initializeTest()
97{
98 IMeshMaterialMng* mm = m_material_mng;
99 Integer nb_mat = options()->nbMaterial();
100 info() << "Number of wanted materials: " << nb_mat;
101
102 // Lit les infos des matériaux du JDD et les enregistre dans le gestionnaire
103 for( Integer i=0; i<nb_mat; ++i ){
104 String mat_name = String("MAT_") + String::fromNumber(i);
105 mm->registerMaterialInfo(mat_name);
106 }
107
108 // Créé des milieux en fonction du nombre de matériaux du jeu de données.
109 {
110 Integer env_index = 1;
111 Integer mat_index = 0;
112
113 while(mat_index<nb_mat){
114 String env_name = "ENV_" + String::fromNumber(env_index);
115 Materials::MeshEnvironmentBuildInfo env_build(env_name);
116 // Utilise un std::set pour être sur qu'on n'ajoute pas 2 fois le même matériau.
117 std::set<String> mats_in_env;
118 for( Integer z=0; z<=env_index; ++z ){
119 String mat1_name = "MAT_" + String::fromNumber(mat_index);
120 mats_in_env.insert(mat1_name);
121 // Ajoute aussi des matériaux qui sont dans les milieux précédents
122 // pour être sur d'avoir des matériaux qui appartiennent à plusieurs milieux.
123 String mat2_name = "MAT_" + String::fromNumber(mat_index/2);
124 mats_in_env.insert(mat2_name);
125
126 ++mat_index;
127 if (mat_index>=nb_mat)
128 break;
129 }
130 for( String mat_name : mats_in_env ){
131 info() << "Add material " << mat_name << " for environment " << env_name;
132 env_build.addMaterial(mat_name);
133 }
134 mm->createEnvironment(env_build);
135 ++env_index;
136 }
137
138 mm->endCreate(false);
139 }
140}

162void MeshMaterialSyncUnitTest::
163_doPhase1()
164{
165 // Cette phase se fait juste après l'init donc on a déjà créé les
166 // matériaux mais on n'a pas encore de mailles dedans.
167
168 // Ajoute des mailles dans les matériaux.
169 // Pour tester la synchronisation, on n'ajoute que des matériaux aux
170 // mailles propres. Ainsi, il faudra que quelqu'un nous envoie les
171 // informations pour les mailles fantômes.
172
173 IMeshMaterialMng* mm = m_material_mng;
174 Integer nb_mat = options()->nbMaterial();
175
176 {
177 CellGroup cells = ownCells();
178 MeshMaterialModifier mmodifier(m_material_mng);
179 Int32UniqueArray ids;
180 // TODO: calculer en fonction du max des uid.
181 for( Integer imat=0; imat<nb_mat; ++imat ){
182 ids.clear();
183 Int64 min_uid = imat*10;
184 Int64 max_uid = min_uid + 10 + imat*10;
185 ENUMERATE_CELL(icell,cells){
186 Cell cell = *icell;
187 Int64 uid = cell.uniqueId();
188 if (uid<max_uid && uid>min_uid)
189 ids.add(cell.localId());
190 }
191 info() << "Adding cells n=" << ids.size() << " to mat " << imat << " (min_uid="
192 << min_uid << " max_uid=" << max_uid << ")";
193 mmodifier.addCells(mm->materials()[imat],ids);
194 }
195 }
196
197 m_material_mng->synchronizeMaterialsInCells();
198 m_material_mng->checkMaterialsInCells();
199 _checkVariableSync();
200}

205void MeshMaterialSyncUnitTest::
206_doPhase2()
207{
208 info() << "Begin phase2";
209 IMeshMaterialMng* mm = m_material_mng;
210 Integer nb_mat = options()->nbMaterial();
211 // Cette phase doit se faire après la phase1 et supprime
212 // des mailles qui ont été ajoutées dans la phase 1.
213
214 // Ajoute et supprime des mailles dans les matériaux.
215 // Pour tester la synchronisation, on ne travaille que sur
216 // les mailles dont on est propriétaires. Ainsi, il faudra que
217 // quelqu'un nous envoie les informations pour les mailles fantômes.
218 {
219 CellGroup cells = ownCells();
220 MeshMaterialModifier mmodifier(m_material_mng);
221 Int32UniqueArray add_ids;
222 Int32UniqueArray remove_ids;
223 // TODO: calculer en fonction du max des uid.
224 for( Integer imat=0; imat<nb_mat; ++imat ){
225 remove_ids.clear();
226 add_ids.clear();
227 Int64 phase1_min_uid = imat*10;
228 Int64 phase1_max_uid = phase1_min_uid + 10 + imat*10;
229
230 Int64 min_uid = phase1_max_uid + 5;
231 Int64 max_uid = min_uid + 20 + imat*5;
232 ENUMERATE_CELL(icell,cells){
233 Cell cell = *icell;
234 Int64 uid = cell.uniqueId();
235 if (uid<phase1_max_uid && uid>phase1_min_uid)
236 remove_ids.add(cell.localId());
237 else if (uid<max_uid && uid>min_uid)
238 add_ids.add(cell.localId());
239 }
240 info() << "Adding cells n=" << add_ids.size() << " to mat " << imat << " (min_uid="
241 << min_uid << " max_uid=" << max_uid << ")";
242 info() << "Removing cells n=" << remove_ids.size() << " to mat " << imat << " (min_uid="
243 << phase1_min_uid << " max_uid=" << phase1_max_uid << ")";
244 IMeshMaterial* mat = mm->materials()[imat];
245 mmodifier.removeCells(mat,remove_ids);
246 mmodifier.addCells(mat,add_ids);
247 }
248 }
249
250 m_material_mng->synchronizeMaterialsInCells();
251 m_material_mng->checkMaterialsInCells();
252 _checkVariableSync();
253}

258void MeshMaterialSyncUnitTest::
259_checkVariableSync()
260{
261 // Vérifie que la synchronisation des variables marche bien
262 MaterialVariableCellInt32 mat_indexes(MaterialVariableBuildInfo(m_material_mng,"SyncMatIndexes"));
263
264 ENUMERATE_ALLENVCELL(iallenvcell,m_material_mng,ownCells()){
265 AllEnvCell all_env_cell = *iallenvcell;
266 ENUMERATE_CELL_ENVCELL(ienvcell,all_env_cell){
267 ENUMERATE_CELL_MATCELL(imatcell,(*ienvcell)){
268 MatCell mc = *imatcell;
269 mat_indexes[mc] = mc.materialId() + 1;
270 }
271 }
272 }
273
274 mat_indexes.synchronize();
275
276 Integer nb_error = 0;
277 ENUMERATE_ALLENVCELL(iallenvcell,m_material_mng,allCells()){
278 AllEnvCell all_env_cell = *iallenvcell;
279 ENUMERATE_CELL_ENVCELL(ienvcell,all_env_cell){
280 ENUMERATE_CELL_MATCELL(imatcell,(*ienvcell)){
281 MatCell mc = *imatcell;
282 if (mat_indexes[mc] != (mc.materialId() + 1)){
283 ++nb_error;
284 if (nb_error<10)
285 error() << "VariableSync error mat=" << mc.materialId()
286 << " mat_index=" << mat_indexes[mc]
287 << " cell=" << ItemPrinter(mc.globalCell());
288 }
289 }
290 }
291 }
292 if (nb_error!=0)
293 fatal() << "Bad variable synchronization";
294}
