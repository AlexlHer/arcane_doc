57// Les sauts négatifs sont supportés.
58RandomNumberGeneratorSeed PDESRandomNumberGeneratorService::
59generateRandomSeed(Integer leap)
60{
61 // Pas besoin de faire de saut si leap == 0.
62 if (leap != 0) {
63 _ran4(&m_seed, leap - 1);
64 }
65 Int64 spawned_seed = _hashState(m_seed);
66 _ran4(&m_seed, 0);
67 return RandomNumberGeneratorSeed(spawned_seed);
68}

115void PDESRandomNumberGeneratorService::
116_breakupUInt64(uint64_t uint64_in, uint32_t* front_bits, uint32_t* back_bits)
117{
118 *front_bits = static_cast<uint32_t>(uint64_in >> 32);
119 *back_bits = static_cast<uint32_t>(uint64_in & 0xffffffff);

129uint64_t PDESRandomNumberGeneratorService::
130_reconstructUInt64(uint32_t front_bits, uint32_t back_bits)
131{
132 uint64_t reconstructed, temp;
133 reconstructed = static_cast<uint64_t>(front_bits);
134 temp = static_cast<uint64_t>(back_bits);
135
136 // shift first bits 32 bits to left
137 reconstructed = reconstructed << 32;
138
139 // temp must be masked to kill leading 1's. Then 'or' with reconstructed
140 // to get the last bits in
141 reconstructed |= (temp & 0x00000000ffffffff);
142
143 return reconstructed;
144}

157void PDESRandomNumberGeneratorService::
158_psdes(uint32_t* lword, uint32_t* irword)
159{
160 const Integer NITER = 4;
161 const uint32_t c1[] = { 0xbaa96887L, 0x1e17d32cL, 0x03bcdc3cL, 0x0f33d1b2L };
162 const uint32_t c2[] = { 0x4b0f3b58L, 0xe874f0c3L, 0x6955c5a6L, 0x55a7ca46L };
163
164 for (Integer i = 0; i < NITER; i++) {
165 uint32_t iswap = (*irword);
166 uint32_t ia = iswap ^ c1[i];
167 uint32_t itmpl = ia & 0xffff;
168 uint32_t itmph = ia >> 16;
169 uint32_t ib = itmpl * itmpl + ~(itmph * itmph);
170
171 *irword = (*lword) ^ (((ia = (ib >> 16) | ((ib & 0xffff) << 16)) ^ c2[i]) + itmpl * itmph);
172 *lword = iswap;
173 }
174}

183uint64_t PDESRandomNumberGeneratorService::
184_hashState(uint64_t initial_number)
185{
186 uint32_t front_bits, back_bits;
187 _breakupUInt64(initial_number, &front_bits, &back_bits);
188
189 _psdes(&front_bits, &back_bits);
190
191 uint64_t fin = _reconstructUInt64(front_bits, back_bits);
192 return fin;
193}

209Real PDESRandomNumberGeneratorService::
210_ran4(Int64* seed, Integer leap)
211{
212 uint32_t front_bits, back_bits, irword, lword;
213 _breakupUInt64((uint64_t)(*seed), &front_bits, &back_bits);
214 front_bits += leap;
215
216 irword = front_bits;
217 lword = back_bits;
218 _psdes(&lword, &irword);
219
220 *seed = (Int64)_reconstructUInt64(++front_bits, back_bits);
221
222 volatile Real fin = 5.4210108624275222e-20 * (Real)_reconstructUInt64(lword, irword);
223 return fin;
224}
