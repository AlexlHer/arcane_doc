121ItemInternal* OneMeshItemAdder::
122addOneFace(Int64 a_face_uid, Int64ConstArrayView a_node_list, Integer a_type)
123{
124 m_work_face_sorted_nodes.resize(a_node_list.size());
125 m_work_face_orig_nodes_uid.resize(a_node_list.size());
126 for( Integer z=0; z<a_node_list.size(); ++z )
127 m_work_face_orig_nodes_uid[z] = a_node_list[z];
128 mesh_utils::reorderNodesOfFace(m_work_face_orig_nodes_uid, m_work_face_sorted_nodes);
129
130 ItemTypeMng* itm = m_mesh->itemTypeMng();
131 ItemInternal *face = m_face_family.allocOne(a_face_uid,itm->typeFromId(a_type));
132 face->setOwner(m_mesh_info.rank(), m_mesh_info.rank());
133
134 for(Integer i_node=0; i_node<a_node_list.size(); ++i_node ){
135 ItemInternal *current_node_internal = addOneNode(m_work_face_sorted_nodes[i_node], m_mesh_info.rank());
136 m_face_family.replaceNode(ItemLocalId(face),i_node,ItemLocalId(current_node_internal));
137 m_node_family.addFaceToNode(current_node_internal, face);
138 }
139 ++m_mesh_info.nbFace();
140 return face;
141}

146ItemInternal* OneMeshItemAdder::
147addOneFace(ItemTypeInfo* type, Int64 face_uid, Int32 sub_domain_id, Int64ConstArrayView nodes_uid)
148{
149 const Integer face_nb_node = nodes_uid.size();
150
151 m_work_face_sorted_nodes.resize(face_nb_node);
152 m_work_face_orig_nodes_uid.resize(face_nb_node);
153 for( Integer z=0; z<face_nb_node; ++z )
154 m_work_face_orig_nodes_uid[z] = nodes_uid[z];
155 mesh_utils::reorderNodesOfFace(m_work_face_orig_nodes_uid,m_work_face_sorted_nodes);
156
157 bool is_add_face = false;
158 ItemInternal* face = m_face_family.findOrAllocOne(face_uid,type,is_add_face);
159
160 // La face n'existe pas
161 if (is_add_face) {
162 ++m_mesh_info.nbFace();
163 face->setOwner(sub_domain_id,sub_domain_id);
164 for(Integer i_node=0; i_node<face_nb_node; ++i_node ){
165 ItemInternal* node_internal = addOneNode(m_work_face_sorted_nodes[i_node], m_mesh_info.rank());
166 m_face_family.replaceNode(ItemLocalId(face),i_node,ItemLocalId(node_internal));
167 m_node_family.addFaceToNode(node_internal, face);
168 }
169 }
170
171 return face;
172}

177ItemInternal* OneMeshItemAdder::
178addOneEdge(Int64 edge_uid, Integer sub_domain_id, Int64ConstArrayView nodes_uid)
179{
180 m_work_edge_sorted_nodes.resize(2);
181 m_work_edge_orig_nodes_uid.resize(2);
182
183 for( Integer z=0; z<2; ++z )
184 m_work_edge_orig_nodes_uid[z] = nodes_uid[z];
185 // reorderNodesOfFace se comporte ici correctement pour des arêtes == face en 2D
186 mesh_utils::reorderNodesOfFace(m_work_edge_orig_nodes_uid,m_work_edge_sorted_nodes);
187
188 bool is_add_edge = false;
189 ItemInternal* edge = m_edge_family.findOrAllocOne(edge_uid,is_add_edge);
190
191 // L'arête n'existe pas
192 if (is_add_edge) {
193 ++m_mesh_info.nbEdge();
194 edge->setOwner(sub_domain_id,sub_domain_id);
195 for(Integer i_node=0; i_node<2; ++i_node ){
196 ItemInternal *current_node_internal = addOneNode(m_work_edge_sorted_nodes[i_node], m_mesh_info.rank());
197 m_edge_family.replaceNode(ItemLocalId(edge),i_node, ItemLocalId(current_node_internal));
198 m_node_family.addEdgeToNode(current_node_internal, edge);
199 }
200 }
201 return edge;
202}

328ItemInternal* OneMeshItemAdder::
329addOneItem(IItemFamily* family,
330 IItemFamilyModifier* family_modifier,
331 ItemTypeInfo* type,
332 Int64 item_uid,
333 Integer item_owner,
334 Integer sub_domain_id,
335 Integer nb_connected_family,
336 Int64ConstArrayView connectivity_info)
337{
338 ARCANE_ASSERT(m_mesh->itemFamilyNetwork(),("ItemFamilyNetwork is required to call OneMeshItemAdder::addOneItem"));
339 bool is_alloc = true;
340 ItemInternal* item = family_modifier->findOrAllocOne(item_uid,type,m_mesh_info,is_alloc); // don't forget to add print in the class method
341 item->setOwner(item_owner,sub_domain_id);
342 // Add connectivities if needed
343 Integer info_index = 0;
344 for (Integer family_index = 0; family_index < nb_connected_family; ++family_index)
345 {
346 // get connected family
347 eItemKind family_kind = static_cast<eItemKind>(connectivity_info[info_index++]); // another way ?
348 Int32 nb_connected_item = CheckedConvert::toInt32(connectivity_info[info_index++]);
349 IItemFamily* connected_family = m_mesh->itemFamily(family_kind);
350 // get connectivities family -> connected_family and reverse
351 String connectivity_name = mesh::connectivityName(family,connected_family);
352 String reverse_connectivity_name = mesh::connectivityName(connected_family,family);
353 bool is_dependency = false;
354 IIncrementalItemConnectivity* family_to_connected_family = m_mesh->itemFamilyNetwork()->getStoredConnectivity(family,connected_family,connectivity_name,is_dependency);
355 IIncrementalItemConnectivity* connected_family_to_family = m_mesh->itemFamilyNetwork()->getStoredConnectivity(connected_family,family,reverse_connectivity_name);
356 // Clear connectivities for already allocated items (except dependencies since replace is used)
357 if (!is_alloc) {
358 if(!is_dependency) _clearConnectivity(ItemLocalId(item), family_to_connected_family);
359 if (connected_family_to_family) _clearReverseConnectivity(ItemLocalId(item),family_to_connected_family,connected_family_to_family);
360 }
361 // get connected item lids
362 Int32UniqueArray connected_item_lids(nb_connected_item);
363 connected_family->itemsUniqueIdToLocalId(connected_item_lids,connectivity_info.subView(info_index,nb_connected_item),true);
364 for (Integer connected_item_index = 0; connected_item_index < nb_connected_item; ++connected_item_index)
365 {
366 if (family_to_connected_family) {
367 // pre-alloc are done (=> use replace) when a dependency relation ("owning relation") while not when only a relation (use add)
368 if (! is_dependency) family_to_connected_family->addConnectedItem(ItemLocalId(item),ItemLocalId(connected_item_lids[connected_item_index]));
369 else family_to_connected_family->replaceConnectedItem(ItemLocalId(item),connected_item_index,ItemLocalId(connected_item_lids[connected_item_index])); // does not work with face to edges...
370 }
371 if (connected_family_to_family) connected_family_to_family->addConnectedItem(ItemLocalId(connected_item_lids[connected_item_index]),ItemLocalId(item));
372 }
373 info_index+= nb_connected_item;
374 }
375// debug(Trace::Highest) << "[addItems] ADD_ITEM " << ItemPrinter(item) << " in " << family->name();
376 debug(Trace::Highest) << "[addItems] ADD_ITEM " << ItemPrinter(item) << " in " << family->name();
377 return item;
378}

381
382ItemInternal* OneMeshItemAdder::
383addOneItem2(IItemFamily* family,
384 IItemFamilyModifier* family_modifier,
385 ItemTypeInfo* type,
386 Int64 item_uid,
387 Integer item_owner,
388 Integer sub_domain_id,
389 Integer nb_connected_family,
390 Int64ConstArrayView connectivity_info)
391{
392 ARCANE_ASSERT(m_mesh->itemFamilyNetwork(),("ItemFamilyNetwork is required to call OneMeshItemAdder::addOneItem"));
393 bool is_alloc = true;
394 ItemInternal* item = family_modifier->findOrAllocOne(item_uid,type,m_mesh_info,is_alloc); // don't forget to add print in the class method
395 item->setOwner(item_owner,sub_domain_id);
396 // Add connectivities if needed
397 Integer info_index = 0;
398 for (Integer family_index = 0; family_index < nb_connected_family; ++family_index) {
399 // Prepare connection
400 // get connected family
401 eItemKind family_kind = static_cast<eItemKind>(connectivity_info[info_index++]); // another way ?
402 Int32 nb_connected_item = CheckedConvert::toInt32(connectivity_info[info_index++]);
403 if (nb_connected_item == 0) continue;
404 IItemFamily* connected_family = nullptr ;
405 switch(family_kind)
406 {
407 case IK_Particle:
408 connected_family = m_mesh->findItemFamily(family_kind, ParticleFamily::defaultFamilyName(), false,false);
409 break ;
410 case IK_DoF:
411 if(family->name()==GraphDoFs::dualNodeFamilyName())
412 connected_family = m_mesh->findItemFamily(family_kind, GraphDoFs::linkFamilyName(), false,false);
413 else
414 connected_family = m_mesh->findItemFamily(family_kind, GraphDoFs::dualNodeFamilyName(), false,false);
415 break ;
416 default:
417 connected_family = m_mesh->itemFamily(family_kind);
418 break ;
419 }
420 // get connectivities family -> connected_family and reverse
421 String connectivity_name = mesh::connectivityName(family,connected_family);
422 bool is_dependency = false;
423 IIncrementalItemConnectivity* family_to_connected_family = m_mesh->itemFamilyNetwork()->getConnectivity(family,connected_family,connectivity_name,is_dependency);
424 if (!family_to_connected_family)
425 fatal() << "Cannot find connectivity " << connectivity_name;
426 bool is_deep_connectivity = m_mesh->itemFamilyNetwork()->isDeep(family_to_connected_family) ;
427 bool is_relation = !(is_dependency && is_deep_connectivity);
428 // Build connection
429 // get connected item lids
430 Int32UniqueArray connected_item_lids(nb_connected_item);
431 bool do_fatal = is_relation ? false : true; // for relations, connected items may not be present and will be skipped.
432 connected_family->itemsUniqueIdToLocalId(connected_item_lids,connectivity_info.subView(info_index,nb_connected_item),do_fatal);
433 // if connection is relation, connected item not necessarily present: remove absent (ie null) items
434 Integer nb_connected_item_found = nb_connected_item;
435 if (is_relation)
436 {
437 for (Integer index = 0; index < connected_item_lids.size(); )
438 {
439 if (connected_item_lids[index] == NULL_ITEM_LOCAL_ID) {
440 connected_item_lids.remove(index);
441 --nb_connected_item_found;
442 }
443 else ++index;
444 }
445 }
446 for (Integer connected_item_index = 0; connected_item_index < nb_connected_item_found; ++connected_item_index)
447 {
448 if (family_to_connected_family)
449 {
450 // Only strategy : check and add
451 auto connected_item_lid = ItemLocalId{connected_item_lids[connected_item_index]};
452 if (is_relation)
453 {
454 if (!family_to_connected_family->hasConnectedItem(ItemLocalId(item),connected_item_lid))
455 {
456 family_to_connected_family->addConnectedItem(ItemLocalId(item),connected_item_lid);
457 }
458 }
459 else
460 {
461 family_to_connected_family->replaceConnectedItem(ItemLocalId(item),connected_item_index,connected_item_lid);
462 }
463 }
464 }
465 info_index+= nb_connected_item;
466 }
467 if (is_alloc)
468 debug(Trace::Highest) << "[addItems] ADD_ITEM " << ItemPrinter(item) << " in " << family->name();
469 // debug(Trace::Highest) << "[addItems] DEPENDENCIES for " << family->name() << FullItemPrinter(item) ; // debug info
470 // _printRelations(item); // debug info
471 return item;
472}

553template<typename CellInfo>
554ItemInternal* OneMeshItemAdder::
555_addOneCell(const CellInfo& cell_info)
556{
557 bool is_check = arcaneIsCheck();
558
559 // Regarde si la maille existe déjà (auquel cas on ne fait rien)
560 ItemInternal* inew_cell = nullptr;
561 {
562 bool is_add; // ce flag est toujours correctement positionné via les findOrAllocOne
563 inew_cell = m_cell_family.findOrAllocOne(cell_info.uniqueId(),
564 cell_info.typeInfo(),
565 is_add);
566 if (!is_add){
567 if (is_check){
568 Cell cell2(inew_cell);
569 // Vérifie que les noeuds correspondent bien à ceux existants
570 for( Integer i=0, is=cell_info.nbNode(); i<is; ++i )
571 if (cell_info.nodeUniqueId(i)!=cell2.node(i).uniqueId())
572 ARCANE_FATAL("trying to add existing cell (uid={0}) with different nodes",
573 cell_info.uniqueId());
574 }
575 return inew_cell;
576 }
577 }
578
579 bool is_verbose = false;
580
581 Cell new_cell(inew_cell);
582
583 const Integer cell_nb_face = cell_info.nbFace();
584 const Integer cell_nb_edge = cell_info.nbEdge();
585
586 ++m_mesh_info.nbCell();
587
588 inew_cell->setOwner(cell_info.owner(), m_mesh_info.rank());
589 // Vérifie la cohérence entre le type local et la maille créée.
590 if (is_check){
591 if (cell_info.nbNode()!=inew_cell->nbNode())
592 ARCANE_FATAL("Incoherent number of nodes v={0} expected={1}",inew_cell->nbNode(),cell_info.nbNode());
593 if (cell_nb_face!=inew_cell->nbFace())
594 ARCANE_FATAL("Incoherent number of faces v={0} expected={1}",inew_cell->nbFace(),cell_nb_face);
595 }
596
598 DynamicMeshIncrementalBuilder::ItemInternalMap& nodes_map = m_node_family.itemsMap();
599
600 _addNodesToCell(inew_cell,cell_info);
601
602 ItemTypeInfo* cell_type_info = cell_info.typeInfo();
603 if (m_mesh_builder->hasEdge()) {
604 // Ajoute les nouvelles arêtes ci-nécessaire
605 for( Integer i_edge=0; i_edge<cell_nb_edge; ++i_edge ){
606 const ItemTypeInfo::LocalEdge& le = cell_type_info->localEdge(i_edge);
607
608 Int64 first_node = cell_info.nodeUniqueId( le.beginNode() );
609 Int64 second_node = cell_info.nodeUniqueId( le.endNode() );
610 if (first_node > second_node)
611 std::swap(first_node,second_node);
612
613 bool is_add = false;
614 ItemInternal* edge_internal = _findInternalEdge(i_edge, cell_info, first_node, second_node, is_add);
615 if (is_add){
616 if (is_verbose)
617 info() << "Create edge " << edge_internal->uniqueId() << ' ' << edge_internal->localId();
618
619 edge_internal->setOwner(cell_info.edgeOwner(i_edge),m_mesh_info.rank());
620 {
621 ItemInternal* current_node_internal = nodes_map.lookupValue(first_node);
622 m_edge_family.replaceNode(ItemLocalId(edge_internal),0,ItemLocalId(current_node_internal));
623 m_node_family.addEdgeToNode(current_node_internal,edge_internal);
624 }
625 {
626 ItemInternal* current_node_internal = nodes_map.lookupValue(second_node);
627 m_edge_family.replaceNode(ItemLocalId(edge_internal),1,ItemLocalId(current_node_internal));
628 m_node_family.addEdgeToNode(current_node_internal,edge_internal);
629 }
630 ++m_mesh_info.nbEdge();
631 }
632
633 m_cell_family.replaceEdge(ItemLocalId(new_cell),i_edge,ItemLocalId(edge_internal));
634 m_edge_family.addCellToEdge(edge_internal,inew_cell);
635 }
636 }
637
638 // Ajoute les nouvelles faces ci-nécessaire
639 for( Integer i_face=0; i_face<cell_nb_face; ++i_face ){
640 const ItemTypeInfo::LocalFace& lf = cell_type_info->localFace(i_face);
641 const Integer face_nb_node = lf.nbNode();
642 const bool is_reorder = _isReorder(i_face, lf, cell_info);
643 // en effet de bord, _isReorder construit aussi m_work_face_sorted_nodes
644
645 bool is_add = false;
646 ItemInternal* face_internal = _findInternalFace(i_face, cell_info, is_add);
647 Face face(face_internal);
648 if (is_add){
649 if (is_verbose){
650 info() << "Create face " << face_internal->uniqueId() << ' ' << face_internal->localId();
651 info() << "AddCell (uid=" << new_cell.uniqueId() << ": Create face (index=" << i_face
652 << ") uid=" << face_internal->uniqueId()
653 << " lid=" << face_internal->localId();
654 }
655 face_internal->setOwner(cell_info.faceOwner(i_face),m_mesh_info.rank());
656
657 for( Integer i_node=0; i_node<face_nb_node; ++i_node ){
658 ItemInternal* current_node_internal = nodes_map.lookupValue(m_work_face_sorted_nodes[i_node]);
659 m_face_family.replaceNode(ItemLocalId(face_internal),i_node,ItemLocalId(current_node_internal));
660 m_node_family.addFaceToNode(current_node_internal,face_internal);
661 }
662
663 if (m_mesh_builder->hasEdge()) {
664 Integer face_nb_edge = lf.nbEdge();
665 for( Integer i_edge=0; i_edge<face_nb_edge; ++i_edge ){
666 Edge current_edge = new_cell.edge( lf.edge(i_edge) );
667 m_face_family.addEdgeToFace(face_internal,current_edge.internal());
668 m_edge_family.addFaceToEdge(current_edge.internal(),face_internal);
669 }
670 }
671 ++m_mesh_info.nbFace();
672 }
673 m_cell_family.replaceFace(new_cell,i_face,ItemLocalId(face_internal));
674
676 if(m_mesh->isAmrActivated()){
677 if (is_reorder){
678 if(face_internal->nbCell() == 2)
679 m_face_family.replaceFrontCellToFace(face,new_cell);
680 else
681 m_face_family.addFrontCellToFace(face_internal,inew_cell);
682 } else{
683 if(face_internal->nbCell() == 2)
684 m_face_family.replaceBackCellToFace(face,new_cell);
685 else
686 m_face_family.addBackCellToFace(face_internal,inew_cell);
687 }
688 } else {
689 if (is_reorder){
690 m_face_family.addFrontCellToFace(face_internal,inew_cell);
691 } else{
692 m_face_family.addBackCellToFace(face_internal,inew_cell);
693 }
694 }
695 }
696
697 _AMR_Patch(inew_cell, cell_info);
698 return inew_cell;
699}

719ItemInternal* OneMeshItemAdder::
720addOneParentItem(const Item & item, const eItemKind submesh_kind, const bool fatal_on_existing_item)
721{
722 //bool is_check = arcaneIsCheck();
723 ItemTypeMng* itm = m_mesh->itemTypeMng();
724 eItemKind kind = item.kind();
725 ItemTypeInfo * type = itm->typeFromId(item.type());
726 if (item.type() == IT_Line2 && submesh_kind == IK_Cell)
727 type = itm->typeFromId(IT_CellLine2);
728 if (item.type() == IT_Vertex && submesh_kind == IK_Face)
729 type = itm->typeFromId(IT_FaceVertex);
730 ItemInternal * item_internal = item.internal();
731
732 if (MeshToMeshTransposer::kindTranspose(submesh_kind, m_mesh, m_mesh->parentMesh()) != kind)
733 ARCANE_FATAL("Incompatible kind/sub-kind");
734
735 // Regarde si la maille existe déjà
736 bool is_add; // ce flag est toujours correctement positionné via les findOrAllocOne
737 ItemInternal* new_item = nullptr;
738
739 switch (submesh_kind) {
740 case IK_Node:
741 new_item = m_node_family.findOrAllocOne(item.uniqueId(),is_add);
742 ++m_mesh_info.nbNode();
743 break;
744 case IK_Edge:
745 new_item = m_edge_family.findOrAllocOne(item.uniqueId(),is_add);
746 ++m_mesh_info.nbEdge();
747 break;
748 case IK_Face:
749 new_item = m_face_family.findOrAllocOne(item.uniqueId(),type,is_add);
750 ++m_mesh_info.nbFace();
751 break;
752 case IK_Cell:
753 if (kind == IK_Face && !Face(item_internal).isSubDomainBoundary())
754 ARCANE_FATAL("Bad boundary face");
755 new_item = m_cell_family.findOrAllocOne(item.uniqueId(),type,is_add);
756 ++m_mesh_info.nbCell();
757 break;
758 default:
759 throw NotSupportedException(A_FUNCINFO,String::format("Kind {0} not supported", submesh_kind));
760 }
761
762 if (!is_add){
763 if (fatal_on_existing_item)
764 ARCANE_FATAL("Cannot add already existing parent item in submesh");
765 else
766 return new_item;
767 }
768
769 new_item->setParent(0,item.localId());
770 new_item->setOwner(item.owner(),m_mesh_info.rank());
771
772 // Localise vis-à-vis de l'item à insérer ces sous-items
773 // Par défaut tout à 0, qui correspond aussi au cas submesh_kind==IK_Node
774 Integer item_nb_node = 0;
775 Integer item_nb_face = 0;
776 Integer item_nb_edge = 0;
777
778 switch (submesh_kind) {
779 case IK_Cell:
780 item_nb_face = type->nbLocalFace();
781 item_nb_edge = type->nbLocalEdge();
782 item_nb_node = type->nbLocalNode();
783 break;
784 case IK_Face:
785 item_nb_edge = type->nbLocalEdge();
786 item_nb_node = type->nbLocalNode();
787 break;
788 case IK_Edge:
789 item_nb_node = type->nbLocalNode();
790 default: // les autres sont déjà filtrés avant avec une exception
791 break;
792 }
793
794 // Traitement du cas de désactivation des arêtes
795 if (!m_mesh_builder->hasEdge())
796 item_nb_edge = 0;
797
798#if OLD
799 // Ne fonctionne plus si on désactive les anciennes connectivités
800 if (is_check){
801 for( Integer z=0; z<item_nb_face; ++z )
802 new_item->_setFace(z,NULL_ITEM_ID);
803 for( Integer z=0; z<item_nb_edge; ++z )
804 new_item->_setEdge(z,NULL_ITEM_ID);
805 for( Integer z=0; z<item_nb_node; ++z )
806 new_item->_setNode(z,NULL_ITEM_ID);
807 }
808#endif
809
811 DynamicMeshKindInfos::ItemInternalMap& nodes_map = m_mesh->nodesMap();
812 auto* parent_mesh = ARCANE_CHECK_POINTER(dynamic_cast<DynamicMesh*>(m_mesh->parentMesh()));
813 DynamicMeshKindInfos::ItemInternalMap& parent_nodes_map = parent_mesh->nodesMap();
814
815 // Traitement des nouveaux noeuds
816
817 // Les sommets sont utilisés dans l'ordre de l'item sauf les cellules de sous-maillages surfacique.
818 const bool direct_node_order =
819 !(submesh_kind == IK_Cell
820 && kind == IK_Face
821 && !Face(item_internal).isSubDomainBoundaryOutside());
822
823 Int64UniqueArray nodes_uid(item_nb_node,NULL_ITEM_UNIQUE_ID);
824 for( Integer i_node=0; i_node<item_nb_node; ++i_node ){
825 Item parent_item;
826 if (type->typeId() == IT_FaceVertex)
827 parent_item = item;
828 else{
829 Int32 idx = ((direct_node_order) ? i_node : (item_nb_node-1-i_node));
830 parent_item = item.toItemWithNodes().node(idx);
831 }
832 Int64 new_node_uid = nodes_uid[i_node] = parent_item->uniqueId();
833 ItemInternal* node_internal = m_node_family.findOrAllocOne(new_node_uid,is_add);
834 if (is_add){
835#ifdef ARCANE_DEBUG_DYNAMIC_MESH
836 info() << "Création node " << new_node_uid << ' '
837 << node_internal->uniqueId() << ' ' << node_internal->localId();
838#endif /* ARCANE_DEBUG_DYNAMIC_MESH */
839 node_internal->setParent(0,parent_item.localId());
840 node_internal->setOwner(parent_item.owner(),m_mesh_info.rank());
841 ++m_mesh_info.nbNode();
842 }
843
844 // Connection de l'item aux sommets
845 ItemLocalId node_lid(node_internal);
846 ItemLocalId new_item_lid(new_item);
847 switch (submesh_kind) {
848 case IK_Cell:
849 m_cell_family.replaceNode(new_item_lid,i_node, node_lid);
850 m_node_family.addCellToNode(node_internal, new_item);
851 break;
852 case IK_Face:
853 m_face_family.replaceNode(new_item_lid,i_node, node_lid);
854 m_node_family.addFaceToNode(node_internal, new_item);
855 break;
856 case IK_Edge:
857 m_edge_family.replaceNode(new_item_lid,i_node, node_lid);
858 m_node_family.addEdgeToNode(node_internal, new_item);
859 break;
860 default: // les autres sont déjà filtrés avant
861 break;
862 }
863 }
864
865 // Traitement des nouvelles arêtes (le filtrage has_edge est déjà pris en compte dans item_nb_edge)
866 for( Integer i_edge=0; i_edge<item_nb_edge; ++i_edge ) {
867 const ItemTypeInfo::LocalEdge& le = type->localEdge(i_edge);
868
869 Int64 first_node = nodes_uid[ le.beginNode() ];
870 Int64 second_node = nodes_uid[ le.endNode() ];
871 if (first_node > second_node)
872 std::swap(first_node,second_node);
873
874 Edge parent_item = item_internal->internalEdge(i_edge);
875 if (parent_item.null())
876 ARCANE_FATAL("Cannot find parent edge");
877
878 Int64 new_edge_uid = parent_item.uniqueId();
879
880 ItemInternal* edge_internal = m_edge_family.findOrAllocOne(new_edge_uid,is_add);
881 if (is_add) {
882#ifdef ARCANE_DEBUG_DYNAMIC_MESH
883 info() << "Création edge " << new_edge_uid << ' '
884 << edge_internal->uniqueId() << ' ' << edge_internal->localId();
885#endif /* ARCANE_DEBUG_DYNAMIC_MESH */
886 edge_internal->setParent(0,parent_item->localId());
887 edge_internal->setOwner(parent_item->owner(),m_mesh_info.rank());
888
889 {
890 ItemInternal* current_node_internal = nodes_map.lookupValue(first_node);
891 m_edge_family.replaceNode(ItemLocalId(edge_internal),0,ItemLocalId(current_node_internal));
892 m_node_family.addEdgeToNode(current_node_internal,edge_internal);
893 }
894 {
895 ItemInternal* current_node_internal = nodes_map.lookupValue(second_node);
896 m_edge_family.replaceNode(ItemLocalId(edge_internal),1,ItemLocalId(current_node_internal));
897 m_node_family.addEdgeToNode(current_node_internal,edge_internal);
898 }
899 ++m_mesh_info.nbEdge();
900 }
901
902
903 // Connection de l'item aux sommets
904 switch (submesh_kind) {
905 case IK_Cell: {
906 m_cell_family.replaceEdge(ItemLocalId(new_item),i_edge,ItemLocalId(edge_internal));
907 m_edge_family.addCellToEdge(edge_internal, new_item);
908 } break;
909 case IK_Face: {
910 m_face_family.replaceEdge(ItemLocalId(new_item),i_edge,ItemLocalId(edge_internal));
911 m_edge_family.addFaceToEdge(edge_internal, new_item);
912 } break;
913 default: // les autres sont déjà filtrés avant
914 break;
915 }
916 }
917
918 // Traitement des nouvelles faces
919 // item_nb_face matérialise déjà que ce contexte ne se produit qu'avec submesh_kind==IK_Cell
920 for( Integer i_face=0; i_face<item_nb_face; ++i_face ) {
921 const ItemTypeInfo::LocalFace& lf = type->localFace(i_face);
922 Integer face_nb_node = lf.nbNode();
923
924 m_work_face_sorted_nodes.resize(face_nb_node);
925 m_work_face_orig_nodes_uid.resize(face_nb_node);
926 for( Integer z=0; z<face_nb_node; ++z )
927 m_work_face_orig_nodes_uid[z] = nodes_uid[ lf.node(z) ];
928 bool is_reorder = false;
929 if (m_mesh->dimension() == 1) { // is 1d mesh
930 is_reorder = (i_face==1);
931 m_work_face_sorted_nodes[0] = m_work_face_orig_nodes_uid[0];
932 }
933 else
934 is_reorder = mesh_utils::reorderNodesOfFace(m_work_face_orig_nodes_uid,m_work_face_sorted_nodes);
935
936 // find parent item
937 ItemInternal * parent_item = NULL;
938 if (kind==IK_Cell) {
939 parent_item = Cell(item_internal).face(i_face).internal();
940 } else if (kind==IK_Face) {
941 if (m_mesh->dimension() == 1) { // is 1d mesh
942 parent_item = parent_nodes_map.lookupValue(m_work_face_sorted_nodes[0]);
943 } else {
944 // Algo sans CT_FaceToEdge
945 Int64 first_node = m_work_face_sorted_nodes[0];
946 Int64 second_node = m_work_face_sorted_nodes[1];
947 if (first_node > second_node)
948 std::swap(first_node,second_node);
949 ItemInternal* nbi = parent_nodes_map.lookupValue(first_node);
950 parent_item = ItemTools::findEdgeInNode(nbi,first_node,second_node);
951 }
952 }
953
954 if (parent_item == NULL) fatal() << "Cannot find parent face";
955 Int64 new_face_uid = parent_item->uniqueId();
956 ItemTypeInfo* face_type = itm->typeFromId(lf.typeId());
957
958 ItemInternal* face_internal = m_face_family.findOrAllocOne(new_face_uid,face_type,is_add);
959 if (is_add){
960#ifdef ARCANE_DEBUG_DYNAMIC_MESH
961 info() << "Création face " << new_face_uid << ' '
962 << face_internal->uniqueId() << ' ' << face_internal->localId();
963#endif /* ARCANE_DEBUG_DYNAMIC_MESH */
964 face_internal->setParent(0,parent_item->localId());
965 face_internal->setOwner(parent_item->owner(),m_mesh_info.rank());
966
967 for( Integer i_node=0; i_node<face_nb_node; ++i_node ){
968 ItemInternal* current_node_internal = nodes_map.lookupValue(m_work_face_sorted_nodes[i_node]);
969 m_face_family.replaceNode(ItemLocalId(face_internal),i_node,ItemLocalId(current_node_internal));
970 m_node_family.addFaceToNode(current_node_internal,face_internal);
971 }
972
973 if (m_mesh_builder->hasEdge()) {
974 Integer face_nb_edge = lf.nbEdge();
975 for( Integer i_edge=0; i_edge<face_nb_edge; ++i_edge ){
976 Int32 edge_idx = lf.edge(i_edge);
977 Edge current_edge = new_item->internalEdge(edge_idx);
978 m_face_family.addEdgeToFace(face_internal,current_edge.internal());
979 m_edge_family.addFaceToEdge(current_edge.internal(),face_internal);
980 }
981 }
982 ++m_mesh_info.nbFace();
983 }
984 m_cell_family.replaceFace(ItemLocalId(new_item),i_face,ItemLocalId(face_internal));
985 if (is_reorder){
986 m_face_family.addFrontCellToFace(face_internal,new_item);
987 } else{
988 m_face_family.addBackCellToFace(face_internal,new_item);
989 }
990 }
991
992 return new_item;
993}
