257template<typename DataType> Ref<ISerializedData> ArrayDataT<DataType>::
258createSerializedDataRef(bool use_basic_type) const
259{
260 typedef typename DataTypeTraitsT<DataType>::BasicType BasicType;
261
262 Integer nb_count = 1;
263 eDataType data_type = dataType();
264 Integer type_size = sizeof(DataType);
265
266 if (use_basic_type){
267 nb_count = DataTypeTraitsT<DataType>::nbBasicType();
268 data_type = DataTypeTraitsT<BasicType>::type();
269 type_size = sizeof(BasicType);
270 }
271
272 Int64 nb_element = m_value.largeSize();
273 Int64 nb_base_element = nb_element * nb_count;
274 Int64 full_size = nb_base_element * type_size;
275 Span<const Byte> base_values(reinterpret_cast<const Byte*>(m_value.data()),full_size);
276 UniqueArray<Int64> extents;
277 extents.add(nb_element);
278 auto sd = arcaneCreateSerializedDataRef(data_type,base_values.size(),1,nb_element,
279 nb_base_element,false,extents);
280 sd->setConstBytes(base_values);
281 return sd;
282}

287template<typename DataType> void ArrayDataT<DataType>::
288allocateBufferForSerializedData(ISerializedData* sdata)
289{
290 typedef typename DataTypeTraitsT<DataType>::BasicType BasicType;
291
292 eDataType data_type = sdata->baseDataType();
293 eDataType base_data_type = DataTypeTraitsT<BasicType>::type();
294
295 if (data_type!=dataType() && data_type==base_data_type)
296 ARCANE_THROW(ArgumentException,"Bad serialized type");
297
298 Int64 nb_element = sdata->nbElement();
299
300 //m_trace->info() << " ASSIGN DATA nb_element=" << nb_element
301 // << " this=" << this;
302 m_value.resize(nb_element);
303
304 Byte* byte_data = reinterpret_cast<Byte*>(m_value.data());
305 Span<Byte> buffer(byte_data,sdata->memorySize());
306 sdata->setWritableBytes(buffer);
307}

322template<typename DataType> void ArrayDataT<DataType>::
323serialize(ISerializer* sbuf,IDataOperation* operation)
324{
325 Integer nb_count = DataTypeTraitsT<DataType>::nbBasicType();
326 typedef typename DataTypeTraitsT<DataType>::BasicType BasicType;
327 eDataType data_type = DataTypeTraitsT<BasicType>::type();
328 bool is_debug = arcaneIsDebug();
329
330 switch(sbuf->mode()){
331 case ISerializer::ModeReserve:
332 {
333 Int64 nb_value = m_value.largeSize();
334 Int64 total_size = nb_value*nb_count;
335 m_trace->debug(Trace::High) << " ArrayDataT::serialize (full) reserve datatype="
336 << dataTypeName(data_type)
337 << " ids=" << nb_value << " totalsize=" << total_size;
338 sbuf->reserve(DT_Int64,2); // 1 pour magic number et 1 pour la taille
339 sbuf->reserveSpan(data_type,total_size);
340 }
341 break;
342 case ISerializer::ModePut:
343 {
344 Int64 nb_value = m_value.largeSize();
345 Int64 total_size = nb_value*nb_count;
346 m_trace->debug(Trace::High) << " ArrayDataT::serialize (full) put datatype="
347 << dataTypeName(data_type)
348 << " ids=" << nb_value << " totalsize=" << total_size;
349 if (is_debug)
350 for( Int64 i=0; i<nb_value; ++i )
351 m_trace->debug(Trace::Highest) << "Put i=" << i << " value =" << m_value[i];
352
353 Span<const BasicType> base_value(reinterpret_cast<BasicType*>(m_value.data()),total_size);
354 sbuf->putInt64(SERIALIZE_MAGIC_NUMBER);
355 sbuf->putInt64(nb_value);
356 sbuf->putSpan(base_value);
357 }
358 break;
359 case ISerializer::ModeGet:
360 {
361 Int64 saved_magic_number = sbuf->getInt64();
362 Int64 nb_value = sbuf->getInt64();
363
364 if (saved_magic_number!=SERIALIZE_MAGIC_NUMBER)
365 ARCANE_FATAL("Internal errror: bad magic number for serialisation expected={0} current={1}",
366 SERIALIZE_MAGIC_NUMBER,saved_magic_number);
367
368 Int64 total_size = nb_value * nb_count;
369
370 m_trace->debug(Trace::High) << " ArrayDataT::serialize (full) get mode=" << sbuf->readMode()
371 << " datatype=" << dataTypeName(data_type)
372 << " ids=" << nb_value << " totalsize=" << total_size;
373 switch(sbuf->readMode()){
374 case ISerializer::ReadReplace:
375 {
376 m_value.resize(nb_value); // must resize using resizeFromGroup ?
377
378 if (operation) {
379 UniqueArray<BasicType> base_value(total_size);
380 sbuf->getSpan(base_value);
381 Span<const DataType> data_value(reinterpret_cast<DataType*>(base_value.data()),nb_value);
382 operation->applySpan(m_value,data_value);
383 if (is_debug)
384 for( Int64 i=0; i<nb_value; ++i )
385 m_trace->debug(Trace::Highest) << "Get i=" << i << " value="
386 << data_value[i] << " transformed value=" << m_value[i];
387 }
388 else{
389 Span<BasicType> base_value(reinterpret_cast<BasicType*>(m_value.data()), total_size);
390 sbuf->getSpan(base_value);
391 if (is_debug)
392 for( Int64 i=0; i<nb_value; ++i )
393 m_trace->debug(Trace::Highest) << "Get i=" << i << " value=" << m_value[i];
394 }
395 }
396 break;
397 case ISerializer::ReadAdd:
398 {
399 Int64 current_size = m_value.largeSize();
400 m_value.resize(current_size+nb_value); // must resize using resizeFromGroup ?
401 Span<BasicType> base_value(reinterpret_cast<BasicType*>(m_value.data()+current_size),total_size);
402 if (operation)
403 ARCANE_THROW(NotImplementedException,"ArrayData::serialize : Cannot deserialize using operation in ReadAdd mode");
404 sbuf->getSpan(base_value);
405 if (is_debug)
406 for( Int64 i=0; i<nb_value; ++i )
407 m_trace->debug(Trace::Highest) << "Get i=" << i << " value=" << m_value[i];
408 }
409 }
410 break;
411 }
412 break;
413 }
414}

428template<typename DataType> void ArrayDataT<DataType>::
429_serialize(ISerializer* sbuf,Span<const Int32> ids,IDataOperation* operation)
430{
431 Integer nb_count = DataTypeTraitsT<DataType>::nbBasicType();
432 typedef typename DataTypeTraitsT<DataType>::BasicType BasicType;
433 eDataType data_type = DataTypeTraitsT<BasicType>::type();
434 bool is_debug = arcaneIsDebug();
435
436 Int64 nb_value = ids.size();
437 Int64 total_size = nb_value*nb_count;
438
439 switch(sbuf->mode()){
440 case ISerializer::ModeReserve:
441 {
442 m_trace->debug(Trace::High) << " ArrayDataT::serialize (partial) reserve datatype="
443 << dataTypeName(data_type)
444 << " ids=" << nb_value << " totalsize=" << total_size;
445 sbuf->reserve(DT_Int64,2);
446 sbuf->reserveSpan(data_type,total_size);
447 }
448 break;
449 case ISerializer::ModePut:
450 {
451 m_trace->debug(Trace::High) << " ArrayDataT::serialize (partial) put datatype="
452 << dataTypeName(data_type)
453 << " ids=" << nb_value << " totalsize=" << total_size;
454 if (is_debug){
455 // Vérifie les valeurs
456 for( Integer i=0, max_value=m_value.size(); i<nb_value; ++i )
457 if (ids[i]>max_value)
458 throw IndexOutOfRangeException(A_FUNCINFO,
459 String::format(" put,serialize : bad sizes i={0} ids[i]={1} nb_value={2} this={3}",
460 i,ids[i], max_value, this),
461 i,0,max_value);
462 }
463
464 UniqueArray<BasicType> base_value;
465 base_value.reserve(total_size);
466 for( Int64 i=0; i<nb_value; ++i ){
467#ifdef ARCANE_DEBUG
468 m_trace->debug(Trace::Highest) << "Put i=" << i << " index=" << ids[i] << " value=" << m_value[ids[i]];
469#endif /* ARCANE_DEBUG */
470 ConstArrayView<BasicType> current_value(nb_count, reinterpret_cast<BasicType*>(&m_value[ids[i]]));
471 base_value.addRange(current_value);
472 }
473 sbuf->putInt64(SERIALIZE_MAGIC_NUMBER);
474 sbuf->putInt64(nb_value);
475 sbuf->putSpan(base_value);
476 }
477 break;
478 case ISerializer::ModeGet:
479 {
480 m_trace->debug(Trace::High) << " ArrayDataT::serialize (partial) get mode=" << sbuf->readMode()
481 << " datatype=" << dataTypeName(data_type)
482 << " ids=" << nb_value << " totalsize=" << total_size;
483 if (is_debug){
484 // Vérifie les valeurs
485 for( Integer i=0, max_value=m_value.size(); i<nb_value; ++i )
486 if (ids[i]>max_value)
487 throw IndexOutOfRangeException(A_FUNCINFO,
488 String::format(" put,serialize : bad sizes i={0} ids[i]={1} nb_value={2} this={3}",
489 i,ids[i], max_value, this),
490 i,0,max_value);
491 }
492
493 switch(sbuf->readMode()){
494 case ISerializer::ReadReplace:
495 {
496 Int64 saved_magic_number = sbuf->getInt64();
497 Int64 saved_nb_value = sbuf->getInt64();
498
499 if (saved_magic_number!=SERIALIZE_MAGIC_NUMBER)
500 ARCANE_FATAL("Internal errror: bad magic number for serialisation expected={0} current={1}",
501 SERIALIZE_MAGIC_NUMBER,saved_magic_number);
502
503 if (saved_nb_value!=nb_value)
504 ARCANE_FATAL("Internal errror: bad size for serialisation expected={0} found={1}",
505 nb_value,saved_nb_value);
506
507 UniqueArray<BasicType> base_value(total_size);
508 sbuf->getSpan(base_value);
509
510 Span<DataType> data_value(reinterpret_cast<DataType*>(base_value.data()),nb_value);
511 UniqueArray<DataType> current_value;
512 Span<DataType> transformed_value;
513
514 if (operation && nb_value!=0) {
515 current_value.resize(ids.size());
516 Arccore::sampleSpan(m_value.constSpan(),ids,current_value.span());
517 transformed_value = current_value.view();
518 operation->applySpan(transformed_value,data_value);
519 }
520 else {
521 transformed_value = data_value;
522 }
523
524 if (is_debug){
525 if (operation)
526 for( Int64 i=0; i<nb_value; ++i )
527 m_trace->debug(Trace::Highest) << "Get i=" << i << " index=" << ids[i]
528 << " value=" << data_value[i] << " transformed value=" << transformed_value[i];
529 else
530 for( Int64 i=0; i<nb_value; ++i )
531 m_trace->debug(Trace::Highest) << "Get i=" << i << " index=" << ids[i]
532 << " value=" << data_value[i];
533 }
534
535 for( Int64 i=0; i<nb_value; ++i ) {
536 m_value[ids[i]] = transformed_value[i];
537 }
538 }
539 break;
540 case ISerializer::ReadAdd:
541 ARCANE_THROW(NotImplementedException,"ArrayData::serialize : Cannot deserialize with ReadAdd mode");
542 break;
543 }
544 }
545 break;
546 }
547}
