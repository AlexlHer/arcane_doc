70ItemSharedInfo::
71ItemSharedInfo(IItemFamily* family,ItemTypeInfo* item_type,MeshItemInternalList* items,
72 ItemInternalConnectivityList* connectivity,Int64ArrayView* unique_ids,
73 Int32ConstArrayView buffer)
74: m_items(items)
75, m_connectivity(connectivity)
76, m_item_family(family)
77, m_unique_ids(unique_ids)
78, m_item_type(item_type)
79, m_item_kind(family->itemKind())
80, m_type_id(item_type->typeId())
81{
82 // La taille du buffer dépend des versions de Arcane.
83 // Avant la 3.2 (Octobre 2021), la taille du buffer est 9 (non AMR) ou 13 (AMR)
84 // Entre la 3.2 et la 3.6 (Mai 2022), la taille vaut toujours 13
85 // A partir de la 3.6, la taille vaut 6.
86 //
87 // On ne cherche pas à faire de reprise avec des versions
88 // de Arcane antérieures à 3.2 donc on peut supposer que la taille
89 // du buffer vaut 13. Ces versions utilisent la nouvelle connectivité
90 // et donc le nombre des éléments est toujours 0 (ainsi que les *allocated)
91 // sauf pour m_nb_node.
92 //
93 // A partir de la 3.6, le nombre de noeuds n'est plus utilisé non
94 // plus et vaut toujours 0. On pourra donc pour les versions de fin
95 // 2022 supprimer ces champs de ItemSharedInfo.
96 Int32 buf_size = buffer.size();
97 if (buf_size>=9){
98 m_nb_node = item_type->nbLocalNode();
99 m_nb_edge = buffer[1];
100 m_nb_face = buffer[2];
101 m_nb_cell = buffer[3];
102 m_edge_allocated = buffer[4];
103 m_face_allocated = buffer[5];
104 m_cell_allocated = buffer[6];
105 m_index = buffer[7];
106 m_nb_reference = buffer[8];
108 if (buf_size>=13){
109 m_nb_hParent = buffer[9];
110 m_nb_hChildren = buffer[10];
111 m_hParent_allocated = buffer[11];
112 m_hChild_allocated = buffer[12];
113 }
114 }
115 else if (buf_size>=4){
116 m_index = buffer[2];
117 m_nb_reference = buffer[3];
118 }
119
120 _init(m_item_kind);
121}

216void ItemSharedInfo::
217_init(eItemKind ik)
218{
219 ARCANE_ASSERT(m_hParent_allocated==0,("m_hParent_allocated should be zero"));
220 ARCANE_ASSERT(m_hChild_allocated==0,("m_hChild_allocated should be zero"));
221 ARCANE_ASSERT(m_nb_edge==0,("m_edge should be zero"));
222 ARCANE_ASSERT(m_nb_face==0,("m_face should be zero"));
223 ARCANE_ASSERT(m_nb_cell==0,("m_cell should be zero"));
224 ARCANE_ASSERT(m_edge_allocated==0,("m_edge_allocated should be zero"));
225 ARCANE_ASSERT(m_face_allocated==0,("m_face_allocated should be zero"));
226 ARCANE_ASSERT(m_cell_allocated==0,("m_cell_allocated should be zero"));
227
228 bool is_amr_activated = false;
229
230 if (ik==IK_Node || ik==IK_Edge || ik==IK_Face || ik==IK_Cell){
231 IItemFamily* base_family = m_item_family;
232 m_nb_parent = 0;
233 if (base_family)
234 m_nb_parent = base_family->parentFamilyDepth();
235 ARCANE_ASSERT((m_nb_parent<=1),("More than one parent level: not implemented"));
236 is_amr_activated = m_items->mesh->isAmrActivated();
237 }
238
239 m_first_node = FIRST_NODE_INDEX;
240 m_first_edge = m_first_node + m_nb_node;
241 m_first_face = m_first_edge;
242 m_first_cell = m_first_edge;
243 m_first_parent = m_first_edge;
244
246 if (is_amr_activated){
247 m_first_hParent = m_first_edge + m_nb_parent;
248 m_first_hChild = m_first_edge + m_nb_parent;
249 }
250
251 m_needed_memory = m_first_edge + m_nb_parent;
252 m_minimum_needed_memory = COMMON_BASE_MEMORY + m_nb_node + m_nb_parent;
253}
