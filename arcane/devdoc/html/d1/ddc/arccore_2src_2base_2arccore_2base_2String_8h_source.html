423 template<typename StringContainer> void
424 split(StringContainer& str_array,char c) const
425 {
426 const String& str = *this;
427 //TODO: passer par String::bytes().
428 const char* str_str = str.localstr();
429 Int64 offset = 0;
430 Int64 len = str.length();
431 for( Int64 i=0; i<len; ++i ){
432 // GG: remet temporairement l'ancienne sémantique (équivalente à strtok())
433 // et supprime la modif IFPEN car cela cause trop d'incompatibilités avec
434 // le code existant. A noter que l'implémentation de l'ancienne sémantique
435 // a plusieurs bugs:
436 // 1. dans le cas où on répète 3 fois ou plus consécutivement le
437 // délimiteur. Par exemple 'X:::Y' retourne {'X',':','Y'} au lieu de
438 // {'X','Y'}
439 // 2. Si on commence par le délimiteur, ce dernier est retourné:
440 // Avec ':X:Y', on retourne {':X','Y'} au lieu de {'X','Y'}
441 //if (str_str[i]==c){
442 if (str_str[i]==c && i!=offset){
443 str_array.push_back(std::string_view(str_str+offset,i-offset));
444 offset = i+1;
445 }
446 }
447 if (len!=offset)
448 str_array.push_back(std::string_view(str_str+offset,len-offset));
449 }
