198void MeshPartitionConstraintMng::Helper::
199merge(Int64Array& linked_items,Int32Array& linked_owners)
200{
201 //TODO: ne pas faire un allGather mais plutot dispatcher la liste
202 // sur plusieurs processeurs, chaque processeur faisant
203 // le tri sur ses entités locales et ensuite faire
204 // plusieurs itérations pour fusionner.
205 IParallelMng* pm = m_mesh->parallelMng();
206 Int64UniqueArray global_linked_items;
207 pm->allGatherVariable(linked_items.view(),global_linked_items);
208 Int32UniqueArray global_linked_owners;
209 pm->allGatherVariable(linked_owners.view(),global_linked_owners);
210 Integer nb_global = global_linked_items.size() / 2;
211 info() << "NB_GLOBAL_LINKED_CELL=" << nb_global;
212 //ItemLinkedList linked_list;
213 m_true_first_uid.clear();
214 m_uids_owner.clear();
215 Integer nb_changed = 0;
216 Integer nb_iter = 0;
217 if (m_is_debug){
218 for( Integer i=0; i<nb_global; ++i ){
219 Int64 first_uid = global_linked_items[(i*2)];
220 Int64 second_uid = global_linked_items[(i*2)+1];
221 Int32 first_owner = global_linked_owners[i];
222 info() << "LinkedItem: first_uid=" << first_uid
223 << " second_uid=" << second_uid
224 << " first_owner=" << first_owner;
225 }
226 }
227 do {
228 nb_changed = 0;
229 ++nb_iter;
230 for( Integer i=0; i<nb_global; ++i ){
231 Int64 first_uid = global_linked_items[(i*2)];
232 Int64 second_uid = global_linked_items[(i*2)+1];
233 if (nb_iter==1){
234 Int32 first_owner = global_linked_owners[i];
235 m_uids_owner.add(first_uid,first_owner);
236 }
237 // IL FAUT TOUJOURS first_uid < second_uid
238 ItemFirstUidData* first_data = m_true_first_uid.lookup(first_uid);
239 ItemFirstUidData* second_data = m_true_first_uid.lookup(second_uid);
240 Int64 first_data_uid = first_uid;
241 Int64 second_data_uid = second_uid;
242 if (first_data)
243 first_data_uid = first_data->value();
244 if (second_data)
245 second_data_uid = second_data->value();
246 Int64 smallest_uid = math::min(first_data_uid,second_data_uid);
247 if (second_data){
248 if (second_data_uid>smallest_uid){
249 if (m_is_debug)
250 info() << "Changed second current=" << second_data->value() << " to " << smallest_uid
251 << " (first=" << first_uid << ",second=" << second_uid << ")";
252 second_data->setValue(smallest_uid);
253 ++nb_changed;
254 }
255 }
256 else{
257 if (m_is_debug)
258 info() << "Changed second add current=" << second_uid << " to " << smallest_uid
259 << " (first=" << first_uid << ",second=" << second_uid << ")";
260 // Attention, cela peut invalider first_data et second_data
261 m_true_first_uid.add(second_uid,smallest_uid);
262 ++nb_changed;
263 }
264
265 if (first_data){
266 if (first_data_uid>smallest_uid){
267 if (m_is_debug)
268 info() << "Changed first current=" << first_data->value() << " to " << smallest_uid
269 << " (first=" << first_uid << ",second=" << second_uid << ")";
270 first_data->setValue(smallest_uid);
271 ++nb_changed;
272 }
273 }
274 else{
275 if (m_is_debug)
276 info() << "Changed first add current=" << first_uid << " to " << smallest_uid
277 << " (first=" << first_uid << ",second=" << second_uid << ")";
278 // Attention, cela peut invalider first_data et second_data
279 m_true_first_uid.add(first_uid,smallest_uid);
280 ++nb_changed;
281 }
282 //linked_list.add(first_uid,second_uid);
283 }
284 debug(Trace::High) << "NB_CHANGED=" << nb_changed << " iter=" << nb_iter;
285 } while (nb_changed!=0 && nb_iter<100);
286 if (nb_iter>=100)
287 fatal() << "Too many iterations";
288 // Ajoute tous les uids qui ne sont pas encore dans la table de hashage.
289 // Leur propriétaire est celui directement spécifié dans linked_cells
290 for( Integer i=0; i<nb_global; ++i ){
291 Int64 first_uid = global_linked_items[(i*2)];
292 Int64 second_uid = global_linked_items[(i*2)+1];
293 ItemFirstUidData* first_data = m_true_first_uid.lookup(first_uid);
294 if (!first_data)
295 m_true_first_uid.add(first_uid,first_uid);
296 ItemFirstUidData* second_data = m_true_first_uid.lookup(second_uid);
297 if (!second_data)
298 m_true_first_uid.add(second_uid,first_uid);
299 }
300}

305void MeshPartitionConstraintMng::Helper::
306changeOwners()
307{
308 // Positionne les propriétaires pour chaque maille
309 // de notre sous-domaine.
310 Int64UniqueArray cells_uid;
311 Int32UniqueArray cells_owner;
312
313 for( auto i : m_true_first_uid.buckets().range() )
314 for( ItemFirstUidData* nbid = i; nbid; nbid = nbid->next() ){
315 cells_uid.add(nbid->key());
316 cells_owner.add(m_uids_owner[nbid->value()]);
317 }
318 Integer nb_cell = cells_uid.size();
319 Int32UniqueArray cells_local_id(nb_cell);
320 IItemFamily* cell_family = m_mesh->cellFamily();
321 cell_family->itemsUniqueIdToLocalId(cells_local_id,cells_uid,false);
322 VariableItemInt32& cells_new_owner = cell_family->itemsNewOwner();
323 ItemInternalList cells_internal = cell_family->itemsInternal();
324 for( Integer i=0; i<nb_cell; ++i ){
325 Int32 lid = cells_local_id[i];
326 if (lid!=NULL_ITEM_LOCAL_ID){
327 Cell cell = cells_internal[lid];
328 if (m_is_debug)
329 info() << "Change cell owner uid=" << cell.uniqueId() << " old=" << cell.owner()
330 << " new=" << cells_owner[i];
331 cells_new_owner[cell] = cells_owner[i];
332 }
333 }
334 // Pas besoin de synchroniser car le maillage s'en occupe, ni de mettre
335 // à jour les autres types d'éléments.
336 cells_new_owner.synchronize();
337}

342void MeshPartitionConstraintMng::Helper::
343computeList(MultiArray2<Int64> & tied_uids)
344{
345 IntegerUniqueArray nb_tieds;
346 nb_tieds.reserve(1000);
347 HashTableMapT<Int64,Integer> uids_tied_index(5000,true);
348
349 for( auto i : m_true_first_uid.buckets().range() )
350 for( ItemFirstUidData* nbid = i; nbid; nbid = nbid->next() ){
351 Int64 tied_uid = nbid->value();
352 HashTableMapT<Int64,Integer>::Data* d = uids_tied_index.lookup(tied_uid);
353 if (d)
354 ++nb_tieds[d->value()];
355 else{
356 Integer index = nb_tieds.size();
357 uids_tied_index.add(tied_uid,index);
358 nb_tieds.add(1);
359 }
360 }
361
362 info() << "NB_TIED=" << nb_tieds.size();
363 tied_uids.resize(nb_tieds);
364 nb_tieds.fill(0);
365
366 for( auto i : m_true_first_uid.buckets().range() )
367 for( ItemFirstUidData* nbid = i; nbid; nbid = nbid->next() ){
368 Int64 tied_uid = nbid->value();
369 Integer index = uids_tied_index[tied_uid];
370 Integer index2 = nb_tieds[index];
371 ++nb_tieds[index];
372 tied_uids.setAt(index,index2,nbid->key());
373 }
374#ifdef ARCANE_DEBUG
375 for( Integer i=0, n=tied_uids.dim1Size(); i<n; ++i ){
376 Int64ConstArrayView uids(tied_uids[i]);
377 debug(Trace::Highest) << " N=" << uids.size();
378 for( Integer j=0, js=uids.size(); j<js; ++j ){
379 debug(Trace::Highest) << " " << uids[j];
380 }
381 }
382#endif
383}
