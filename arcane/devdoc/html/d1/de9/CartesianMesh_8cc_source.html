213void CartesianMeshImpl::
214recreateFromDump()
215{
216 info() << "Creating 'CartesianMesh' infos from dump";
217
218 // Sauve le numéro de version pour être sur que c'est OK en reprise
219 Int32 v = m_properties->getInt32("Version");
220 if (v!=SERIALIZE_VERSION)
221 ARCANE_FATAL("Bad serializer version: trying to read from incompatible checkpoint v={0} expected={1}",
222 v,SERIALIZE_VERSION);
223
224 // Récupère les noms des groupes des patchs
225 UniqueArray<String> patch_group_names;
226 m_properties->get("PatchGroupNames",patch_group_names);
227 info(4) << "Found n=" << patch_group_names.size() << " patchs";
228 m_amr_patch_cell_groups.clear();
229 IItemFamily* cell_family = m_mesh->cellFamily();
230 for( const String& x : patch_group_names ){
231 CellGroup group = cell_family->findGroup(x);
232 if (group.null())
233 ARCANE_FATAL("Can not find cell group '{0}'",x);
234 m_amr_patch_cell_groups.add(group);
235 }
236
237 computeDirections();
238}

242
243void CartesianMeshImpl::
244computeDirections()
245{
246 m_amr_patches.clear();
247 m_amr_patches.add(m_all_items_direction_info);
248
249 m_is_amr = m_mesh->isAmrActivated();
250
251 VariableCellReal3 cells_center(VariableBuildInfo(m_mesh,"TemporaryCartesianMeshCellCenter"));
252 VariableFaceReal3 faces_center(VariableBuildInfo(m_mesh,"TemporaryCartesianMeshFaceCenter"));
253
254 // Calcule les coordonnées du centre des mailles.
255 VariableNodeReal3& nodes_coord = m_mesh->nodesCoordinates();
256 ENUMERATE_CELL(icell,m_mesh->allCells()){
257 Cell cell = *icell;
258 Real3 center;
259 for( NodeEnumerator inode(cell.nodes()); inode.hasNext(); ++inode )
260 center += nodes_coord[inode];
261 center /= cell.nbNode();
262 cells_center[icell] = center;
263 }
264 ENUMERATE_FACE(iface,m_mesh->allFaces()){
265 Face face = *iface;
266 Real3 center;
267 for( NodeEnumerator inode(face.nodes()); inode.hasNext(); ++inode )
268 center += nodes_coord[inode];
269 center /= face.nbNode();
270 faces_center[iface] = center;
271 }
272
273 IItemFamily* cell_family = m_mesh->cellFamily();
274 IItemFamily* node_family = m_mesh->nodeFamily();
275 Int32 next_face_x = -1;
276 Int32 next_face_y = -1;
277 Int32 next_face_z = -1;
278
279 CellVectorView cell_view = cell_family->allItems().view();
280 Cell cell0 = cell_view[0];
281 Integer nb_face = cell0.nbFace();
282 Integer nb_node = cell0.nbNode();
283 Real3 cell_center = cells_center[cell0];
284
285 info(4) << "Cartesian mesh compute directions is_amr=" << m_is_amr;
286
287 for( Integer i=0; i<nb_node; ++i ){
288 Node node = cell0.node(i);
289 info(4) << "Node I=" << i << " node=" << ItemPrinter(node) << " pos=" << nodes_coord[node];
290 }
291
292 // On suppose que toutes les mailles ont le même sens de numérotation dans le maillage.
293 // Par exemple, pour toutes les mailles, la face d'indice 0 est celle du haut, celle
294 // d'indice 1 celle de droite.
295 for( Integer i=0; i<nb_face; ++i ){
296 Face f = cell0.face(i);
297 if (f.isSubDomainBoundary())
298 continue;
299 Cell next_cell = (f.backCell()==cell0) ? f.frontCell() : f.backCell();
300 Real3 next_center = cells_center[next_cell];
301 info(4) << "NEXT_CELL=" << ItemPrinter(next_cell) << " center=" << next_center \
302 << " back=" << f.backCell().uniqueId()
303 << " front=" << f.frontCell().uniqueId();
304
305 Real diff_x = math::abs(next_center.x - cell_center.x);
306 Real diff_y = math::abs(next_center.y - cell_center.y);
307 Real diff_z = math::abs(next_center.z - cell_center.z);
308 info(4) << "NEXT_CELL=" << ItemPrinter(next_cell) << " diff=" << Real3(diff_x,diff_y,diff_z);
309 //TODO: Verifier qu'il s'agit bien de la maille apres et pas avant.
310 // (tenir compte du signe de diff)
311 if (diff_x>diff_y && diff_x>diff_z){
312 // INC X
313 next_face_x = i;
314 info() << "Advance in direction X -> " << next_face_x;
315 }
316 else if (diff_y>diff_x && diff_y>diff_z){
317 // INC Y
318 next_face_y = i;
319 info() << "Advance in direction Y -> " << next_face_y;
320 }
321 else if (diff_z>diff_x && diff_z>diff_y){
322 // INC Z
323 next_face_z = i;
324 info() << "Advance in direction Z -> " << next_face_z;
325 }
326 else
327 ARCANE_FATAL("Bad value for next cell");
328 }
329
330 bool is_3d = m_mesh->dimension()==3;
331 m_all_items_direction_info->_internalComputeNodeCellInformations(cell0,cells_center[cell0],nodes_coord);
332
333 info() << "Informations from IMesh properties:";
334
335 auto* cmgi = ICartesianMeshGenerationInfo::getReference(m_mesh,true);
336
337 info() << "GlobalNbCell = " << cmgi->globalNbCells();
338 info() << "OwnNbCell: " << cmgi->ownNbCells();
339 info() << "SubDomainOffset: " << cmgi->subDomainOffsets();
340 info() << "OwnCellOffset: " << cmgi->ownCellOffsets();
341
342 CellGroup all_cells = cell_family->allItems();
343 NodeGroup all_nodes = node_family->allItems();
344 if (m_is_amr){
345 auto x = _buildPatchGroups(mesh()->allLevelCells(0),0);
346 all_cells = std::get<0>(x);
347 all_nodes = std::get<1>(x);
348 }
349
350 if (next_face_x!=(-1)){
351 m_local_face_direction[MD_DirX] = next_face_x;
352 _computeMeshDirection(*m_all_items_direction_info.get(),MD_DirX,cells_center,faces_center,all_cells,all_nodes);
353 }
354 if (next_face_y!=(-1)){
355 m_local_face_direction[MD_DirY] = next_face_y;
356 _computeMeshDirection(*m_all_items_direction_info.get(),MD_DirY,cells_center,faces_center,all_cells,all_nodes);
357 }
358 if (next_face_z!=(-1)){
359 m_local_face_direction[MD_DirZ] = next_face_z;
360 _computeMeshDirection(*m_all_items_direction_info.get(),MD_DirZ,cells_center,faces_center,all_cells,all_nodes);
361 }
362
363 // Positionne les informations par direction
364 for( Integer idir=0, nb_dir=mesh()->dimension(); idir<nb_dir; ++idir ){
365 CellDirectionMng& cdm = m_all_items_direction_info->cellDirection(idir);
366 cdm.m_global_nb_cell = cmgi->globalNbCells()[idir];
367 cdm.m_own_nb_cell = cmgi->ownNbCells()[idir];
368 cdm.m_sub_domain_offset = cmgi->subDomainOffsets()[idir];
369 cdm.m_own_cell_offset = cmgi->ownCellOffsets()[idir];
370 }
371
372 info() << "Compute cartesian connectivity";
373
374 m_nodes_to_cell_storage.resize(mesh()->nodeFamily()->maxLocalId());
375 m_cells_to_node_storage.resize(mesh()->cellFamily()->maxLocalId());
376 m_connectivity.setStorage(m_nodes_to_cell_storage,m_cells_to_node_storage);
377 m_connectivity.computeInfos(mesh(),nodes_coord,cells_center);
378
379 // Ajoute informations de connectivités pour les patchs AMR
380 // TODO: supporter plusieurs appels à cette méthode
381 for( const CellGroup& cells : m_amr_patch_cell_groups ){
382 Integer patch_index = m_amr_patches.size();
383 info() << "AMR Patch name=" << cells.name() << " size=" << cells.size() << " index=" << patch_index;
384 auto* cdi = new CartesianMeshPatch(this,patch_index);
385 m_amr_patches.add(makeRef(cdi));
386 cdi->_internalComputeNodeCellInformations(cell0,cells_center[cell0],nodes_coord);
387 auto [ patch_cells, patch_nodes ] = _buildPatchGroups(cells,patch_index);
388 _computeMeshDirection(*cdi,MD_DirX,cells_center,faces_center,patch_cells,patch_nodes);
389 _computeMeshDirection(*cdi,MD_DirY,cells_center,faces_center,patch_cells,patch_nodes);
390 if (is_3d)
391 _computeMeshDirection(*cdi,MD_DirZ,cells_center,faces_center,patch_cells,patch_nodes);
392 }
393
394 if (arcaneIsCheck())
395 checkValid();
396
397 _saveInfosInProperties();
398}

403std::tuple<CellGroup,NodeGroup> CartesianMeshImpl::
404_buildPatchGroups(const CellGroup& cells,Integer patch_level)
405{
406 // On créé un groupe pour chaque patch en garantissant que l'ordre de parcours
407 // est celui des uniqueId() des entités
408 // TODO: à terme, il faudrait que l'ordre de parcours soit le même que
409 // celui du maillage cartésien. Pour cela, il faut soit que les uniqueId()
410 // des mailles/noeuds créés soient dans le même ordre que le maillage cartésien,
411 // soit que la fonction de tri soit spécifique à ce type de maillage.
412 NodeGroup nodes = cells.nodeGroup();
413 IItemFamily* cell_family = cells.itemFamily();
414 IItemFamily* node_family = nodes.itemFamily();
415
416 String cell_group_name = String("AMRPatchCells") + patch_level;
417 CellGroup patch_cells = cell_family->createGroup(cell_group_name,Int32ConstArrayView(),true);
418 // Met les mêmes mailles que \a cells mais force le tri
419 patch_cells.setItems(cells.view().localIds(),true);
420
421 String node_group_name = String("AMRPatchNodes") + patch_level;
422 NodeGroup patch_nodes = node_family->createGroup(node_group_name,Int32ConstArrayView(),true);
423 // Met les mêmes noeuds que \a nodes mais force le tri
424 patch_nodes.setItems(nodes.view().localIds(),true);
425 info(4) << "PATCH_CELLS name=" << patch_cells.name() << " size=" << patch_cells.size();
426 info(4) << "PATCH_NODES name=" << patch_nodes.name() << " size=" << patch_nodes.size();
427 return { patch_cells, patch_nodes };
428}

433void CartesianMeshImpl::
434_computeMeshDirection(CartesianMeshPatch& cdi,eMeshDirection dir,VariableCellReal3& cells_center,
435 VariableFaceReal3& faces_center,CellGroup all_cells,NodeGroup all_nodes)
436{
437 IItemFamily* cell_family = m_mesh->cellFamily();
438 IItemFamily* face_family = m_mesh->faceFamily();
439 IItemFamily* node_family = m_mesh->nodeFamily();
440
441 Int32 max_cell_id = cell_family->maxLocalId();
442 Int32 max_face_id = face_family->maxLocalId();
443 Int32 max_node_id = node_family->maxLocalId();
444
445 CellDirectionMng& cell_dm = cdi.cellDirection(dir);
446 cell_dm.m_infos.resize(max_cell_id);
447
448 FaceDirectionMng& face_dm = cdi.faceDirection(dir);
449 face_dm.m_infos.resize(max_face_id);
450
451 NodeDirectionMng& node_dm = cdi.nodeDirection(dir);
452 node_dm.m_infos.resize(max_node_id);
453
454 //TODO: attention à remettre à jour après changement de maillage.
455 info(4) << "COMPUTE DIRECTION dir=" << dir;
456
457 Int32 prev_local_face = -1;
458 Int32 next_local_face = m_local_face_direction[dir];
459 Integer mesh_dim = m_mesh->dimension();
460 // Calcul le numero local de face oppose à la face suivante.
461 if (mesh_dim==2)
462 prev_local_face = (next_local_face + 2) % 4;
463 else if (mesh_dim==3)
464 prev_local_face = (next_local_face + 3) % 6;
465
466 cell_dm._internalSetLocalFaceIndex(next_local_face,prev_local_face);
467
468 // Positionne pour chaque maille les faces avant et après dans la direction.
469 // On s'assure que ces entités sont dans le groupe des entités de la direction correspondante
470 std::set<Int32> cells_set;
471 ENUMERATE_CELL(icell,all_cells){
472 cells_set.insert(icell.itemLocalId());
473 }
474
475 // Calcule les mailles devant/derrière. En cas de patch AMR, il faut que ces deux mailles
476 // soient de même niveau
477 ENUMERATE_CELL(icell,all_cells){
478 Cell cell = *icell;
479 Int32 my_level = cell.level();
480 Face next_face = cell.face(next_local_face);
481 Cell next_cell = next_face.backCell()==cell ? next_face.frontCell() : next_face.backCell();
482 if (cells_set.find(next_cell.localId())==cells_set.end())
483 next_cell = Cell();
484 else if (next_cell.level()!=my_level)
485 next_cell = Cell();
486
487 Face prev_face = cell.face(prev_local_face);
488 Cell prev_cell = prev_face.backCell()==cell ? prev_face.frontCell() : prev_face.backCell();
489 if (cells_set.find(prev_cell.localId())==cells_set.end())
490 prev_cell = Cell();
491 else if (prev_cell.level()!=my_level)
492 prev_cell = Cell();
493 cell_dm.m_infos[icell.itemLocalId()] = CellDirectionMng::ItemDirectionInfo(next_cell.internal(),prev_cell.internal());
494 }
495 cell_dm._internalComputeInnerAndOuterItems(all_cells);
496 face_dm._internalComputeInfos(cell_dm,cells_center,faces_center);
497 node_dm._internalComputeInfos(cell_dm,all_nodes,cells_center);
498}

503void CartesianMeshImpl::
504refinePatch2D(Real2 position,Real2 length)
505{
506 VariableNodeReal3& nodes_coord = m_mesh->nodesCoordinates();
507 UniqueArray<Int32> cells_local_id;
508 // Parcours les mailles actives et ajoute dans la liste des mailles
509 // à raffiner celles qui sont contenues dans le boîte englobante
510 // spécifiée dans le jeu de données.
511 info() << "REFINEMENT ..." << position << " L=" << length;
512 Real2 min_pos = position;
513 Real2 max_pos = min_pos + length;
514 cells_local_id.clear();
515 ENUMERATE_CELL(icell,m_mesh->allActiveCells()){
516 Cell cell = *icell;
517 Real3 center;
518 for( NodeEnumerator inode(cell.nodes()); inode.hasNext(); ++inode )
519 center += nodes_coord[inode];
520 center /= cell.nbNode();
521 if (center.x>min_pos.x && center.x<max_pos.x && center.y>min_pos.y && center.y<max_pos.y)
522 cells_local_id.add(icell.itemLocalId());
523 }
524 _applyRefine(cells_local_id);
525 _saveInfosInProperties();
526}

531void CartesianMeshImpl::
532_applyRefine(ConstArrayView<Int32> cells_local_id)
533{
534 IItemFamily* cell_family = m_mesh->cellFamily();
535 Integer nb_cell = cells_local_id.size();
536 info(4) << "Local_NbCellToRefine = " << nb_cell;
537 Integer index = m_amr_patch_cell_groups.size();
538 String parent_group_name = String("CartesianMeshPatchParentCells")+index;
539 CellGroup parent_cells = cell_family->createGroup(parent_group_name,cells_local_id,true);
540
541 IParallelMng* pm = m_mesh->parallelMng();
542 Int64 total_nb_cell = pm->reduce(Parallel::ReduceSum,nb_cell);
543 info(4) << "Global_NbCellToRefine = " << total_nb_cell;
544 if (total_nb_cell==0)
545 return;
546 m_mesh->modifier()->flagCellToRefine(cells_local_id);
547 m_mesh->modifier()->adapt();
548 {
549 MeshStats ms(traceMng(),m_mesh,m_mesh->parallelMng());
550 ms.dumpStats();
551 }
552 {
553 // Créé le groupe contenant les mailles AMR
554 // Il s'agit des mailles filles de \a parent_cells
555 String children_group_name = String("CartesianMeshPatchCells")+index;
556 UniqueArray<Int32> children_local_id;
557 ENUMERATE_CELL(icell,parent_cells){
558 Cell c = *icell;
559 for(Integer k=0; k<c.nbHChildren(); ++k ){
560 Cell child = c.hChild(k);
561 children_local_id.add(child.localId());
562 }
563 }
564 CellGroup children_cells = cell_family->createGroup(children_group_name,children_local_id,true);
565 m_amr_patch_cell_groups.add(children_cells);
566 }
567}

586void CartesianMeshImpl::
587renumberItemsUniqueId(const CartesianMeshRenumberingInfo& v)
588{
589 auto* cmgi = ICartesianMeshGenerationInfo::getReference(m_mesh,true);
590
591 // Regarde d'abord si on renumérote les faces
592 Int32 face_method = v.renumberFaceMethod();
593 if (face_method!=0 && face_method!=1)
594 ARCANE_FATAL("Invalud value '{0}' for renumberFaceMethod(). Valid values are 0 or 1");
595 if (face_method==1)
596 ARCANE_THROW(NotImplementedException,"Method 1 for face renumbering");
597
598 // Regarde ensuite les patchs si demandé.
599 Int32 patch_method = v.renumberPatchMethod();
600 if (patch_method!=0 && patch_method!=1)
601 ARCANE_FATAL("Invalud value '{0}' for renumberPatchMethod(). Valid values are 0 or 1");
602 if (patch_method==1){
603 CartesianMeshUniqueIdRenumbering renumberer(this,cmgi);
604 renumberer.renumber();
605 }
606
607 // Termine par un tri éventuel.
608 if (v.isSortAfterRenumbering()){
609 info() << "Compacting and Sorting after renumbering";
610 m_mesh->nodeFamily()->compactItems(true);
611 m_mesh->faceFamily()->compactItems(true);
612 m_mesh->cellFamily()->compactItems(true);
613 computeDirections();
614 }
615}

623ICartesianMesh* ICartesianMesh::
624getReference(IMesh* mesh,bool create)
625{
626 //TODO: faire lock pour multi-thread
627 const char* name = "CartesianMesh";
628 IUserDataList* udlist = mesh->userDataList();
629
630 IUserData* ud = udlist->data(name,true);
631 if (!ud){
632 if (!create)
