566
567void CaseOptions::
568read(eCaseOptionReadPhase read_phase)
569{
570 ITraceMng* tm = traceMng();
571 bool is_phase1 = read_phase==eCaseOptionReadPhase::Phase1;
572 if (is_phase1 && m_p->m_is_phase1_read)
573 return;
574
575 if (is_phase1 && !m_p->m_is_translated_name_set)
576 _setTranslatedName();
577
578 m_p->m_config_list->readChildren(is_phase1);
579
580 if (is_phase1){
581 ICaseDocument* doc = caseDocument();
582 // Lit la fonction d'activation (si elle est présente)
583 XmlNode velem = m_p->m_config_list->rootElement();
584 CaseNodeNames* cnn = doc->caseNodeNames();
585 String func_activation_name = velem.attrValue(cnn->function_activation_ref);
586 if (!func_activation_name.null()){
587 ICaseFunction* func = caseMng()->findFunction(func_activation_name);
588 if (!func){
589 CaseOptionError::addError(doc,A_FUNCINFO,velem.xpathFullName(),
590 String::format("No function with the name '{0}' exists",
591 func_activation_name));
592 }
593 else if (func->paramType()!=ICaseFunction::ParamReal){
594 CaseOptionError::addError(doc,A_FUNCINFO,velem.xpathFullName(),
595 String::format("The function '{0}' requires a parameter of type 'time'",
596 func_activation_name));
597 }
598 else if (func->valueType()!=ICaseFunction::ValueBool){
599 CaseOptionError::addError(doc,A_FUNCINFO,velem.xpathFullName(),
600 String::format("The function '{0}' requires a parameter of type 'bool'",
601 func_activation_name));
602 }
603 else {
604 m_p->m_activate_function = func;
605 tm->info() << "Use the function '" << func->name() << "' to activate the option "
606 << velem.xpathFullName();
607 }
608 }
609 // Vérifie que l'élément 'function' n'est pas présent
610 {
611 String func_name = velem.attrValue(cnn->function_ref);
612 if (!func_name.null())
613 CaseOptionError::addError(doc,A_FUNCINFO,velem.xpathFullName(),
614 String::format("Attribute <{0}> invalid.",
615 cnn->function_ref));
616 }
617 m_p->m_is_phase1_read = true;
618 }
619}

731
732void CaseOptionSimple::
733_search(bool is_phase1)
734{
735 if (!is_phase1)
736 return;
737
738 //ITraceMng* msg = caseMng()->traceMng();
739 const String& velem_name = name();
740 XmlNodeList velems = rootElement().children(velem_name);
741 XmlNode velem;
742 Integer nb_elem = velems.size();
743 ICaseDocument* doc = caseDocument();
744 if (nb_elem>=1){
745 velem = velems[0];
746 if (nb_elem>=2){
747 CaseOptionError::addWarning(doc,A_FUNCINFO,velem.xpathFullName(),
748 String::format("Only one token of the element is allowed (nb_occur={0})",
749 nb_elem));
750 }
751 }
752
753 m_element = velem;
754 m_function = 0;
755
756 _searchFunction(velem);
757
758 String physical_unit = m_element.attrValue("unit");
759 if (!physical_unit.null()){
760 _setPhysicalUnit(physical_unit);
761 if (_allowPhysicalUnit()){
762 //TODO: VERIFIER QU'IL Y A UNE DEFAULT_PHYSICAL_UNIT.
763 m_unit_converter = subDomain()->physicalUnitSystem()->createConverter(physical_unit,defaultPhysicalUnit());
764 }
765 else
766 CaseOptionError::addError(doc,A_FUNCINFO,velem.xpathFullName(),
767 String::format("Usage of a physic unit ('{0}') is not allowed for this kind of option",
768 physical_unit));
769 }
770}

791/*---------------------------------------------------------------------------*/
792
793void CaseOptionSimple::
794_searchFunction(XmlNode& velem)
795{
796 if (velem.null())
797 return;
798
799 // Recherche une éventuelle fonction associée
800 String fname = caseDocument()->caseNodeNames()->function_ref;
801 String func_name = velem.attrValue(fname);
802 if (func_name.null())
803 return;
804
805 ICaseFunction* func = caseMng()->findFunction(func_name);
806 ITraceMng* msg = caseMng()->traceMng();
807 if (!func){
808 msg->pfatal() << "In element <" << velem.name()
809 << ">: no function named <" << func_name << ">";
810 }
811
812 // Recherche s'il s'agit d'une fonction standard
813 IStandardFunction* sf = dynamic_cast<IStandardFunction*>(func);
814 if (sf){
815 msg->info() << "Use standard function: " << func_name;
816 m_standard_function = sf;
817 }
818 else{
819 msg->info() << "Use function: " << func_name;
820 m_function = func;
821 }
822}
