146void TimerMng::
147_setVirtualTime()
148{
149#ifdef ARCANE_TIMER_USE_CLOCK
150 current_clock_value = ::clock();
151#else
152 struct itimerval time_val;
153 struct itimerval otime_val;
154 time_val.it_value.tv_sec = 5000000;
155 time_val.it_value.tv_usec = 0;
156 time_val.it_interval.tv_sec = 0;
157 time_val.it_interval.tv_usec = 0;
158 int r = ::setitimer(ITIMER_VIRTUAL,&time_val,&otime_val);
159 if (r!=0)
160 _errorInTimer("setitimer()",r);
161#endif
162}

164/*---------------------------------------------------------------------------*/
165/*---------------------------------------------------------------------------*/
166
167Real TimerMng::
168_getVirtualTime()
169{
170#ifdef ARCANE_TIMER_USE_CLOCK
171 clock_t cv = ::clock();
172 Real diffv = static_cast<Real>(cv-current_clock_value);
173 return diffv / CLOCKS_PER_SEC;
174#else
175 struct itimerval time_val;
176 int r = ::getitimer(ITIMER_VIRTUAL,&time_val);
177 if (r!=0)
178 _errorInTimer("getitimer()",r);
179 double v = ((double)time_val.it_value.tv_sec) * 1.0
180 + ((double)time_val.it_value.tv_usec) * 1e-6;
181 return (5000000. - v);
182#endif
183}
