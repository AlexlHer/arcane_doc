70void
71splitString(const String& str,Array<String>& str_array,char c)
72{
73 const char* str_str = str.localstr();
74 Int64 offset = 0;
75 Int64 len = str.length();
76 for( Int64 i=0; i<len; ++i ){
77 if (str_str[i]==c && i!=offset){
78 str_array.add(std::string_view(str_str+offset,i-offset));
79 offset = i+1;
80 }
81 }
82 if (len!=offset)
83 str_array.add(std::string_view(str_str+offset,len-offset));
84}

157void HGroup::
158recursiveCreate(const Hid& loc_id,const Array<String>& bufs)
159{
160 close();
161 hid_t last_hid = loc_id.id();
162 Integer nb_create = bufs.size();
163 UniqueArray<hid_t> ref_ids(nb_create);
164 for( Integer i=0; i<nb_create; ++i ){
165 last_hid = _checkOrCreate(last_hid,bufs[i]);
166 ref_ids[i] = last_hid;
167 }
168 // Libere tous les groupes intermediaires crees
169 for( Integer i=0; i<nb_create-1; ++i )
170 H5Gclose(ref_ids[i]);
171 _setId(last_hid);
172}

177void HGroup::
178checkDelete(const Hid& loc_id,const String& var)
179{
180 UniqueArray<String> bufs;
181 splitString(var,bufs,'/');
182 hid_t last_hid = loc_id.id();
183 hid_t parent_hid = last_hid;
184 Integer i = 0;
185 Integer size = bufs.size();
186 for( ; i<size; ++i ){
187 parent_hid = last_hid;
188 last_hid = _checkExist(last_hid,bufs[i]);
189 if (last_hid==0)
190 break;
191 }
192 // Groupe trouvé, on le détruit.
193 if (last_hid>0 && parent_hid>0 && i==size){
194 //cerr << "** DELETE <" << bufs[size-1] << "\n";
195 H5Gunlink(parent_hid,bufs[size-1].localstr());
196 }
197}

224void HGroup::
225openIfExists(const Hid& loc_id,const Array<String>& paths)
226{
227 close();
228 hid_t last_hid = loc_id.id();
229 bool is_valid = true;
230 Integer nb_open = paths.size();
231 UniqueArray<hid_t> ref_ids;
232 ref_ids.reserve(nb_open);
233 for( Integer i=0; i<nb_open; ++i ){
234 if (HGroup::hasChildren(last_hid,paths[i].localstr())){
235 last_hid = _H5Gopen(last_hid,paths[i].localstr());
236 ref_ids.add(last_hid);
237 }
238 else{
239 is_valid = false;
240 break;
241 }
242 }
243 if (is_valid)
244 _setId(last_hid);
245 // Ferme tous les groupes intermediaires
246 for( Integer i=0; i<ref_ids.size(); ++i ){
247 if (ref_ids[i]!=last_hid)
248 H5Gclose(ref_ids[i]);
249 }
250}

285 //cerr << "** CHECK CREATE <" << group_name.str() << ">\n";
286 herr_t v = H5Giterate(loc_id,".",0,_ArcaneHdf5UtilsGroupIterateMe,&gs);
287
288 // Regarde si le groupe existe déjà
289 //herr_t he = H5Gget_objinfo(loc_id,group_name.str(),true,0);
290 //cerr << "** CHECK CREATE <" << group_name.str() << "> " << v << "\n";
291 //cerr << "** CHECK CREATE <" << group_name.str() << "> " << v << ' ' << he << "\n";
292 if (v>0){
293 return _H5Gopen(loc_id,group_name.localstr());
294 }
295 hid_t new_id = _H5Gcreate(loc_id,group_name.localstr());
296 //cerr << "** TRY TO CREATE <" << group_name.str() << "> " << new_id << "\n";
297 return new_id;

401void HDataset::
402recursiveCreate(const Hid& loc_id,const String& var,hid_t save_type,
403 const HSpace& space_id,hid_t plist)
404{
405 // Si le dataset existe déjà, il faut le supprimer
406 // car sinon il n'est pas toujours possible de modifer le space_id
407 UniqueArray<String> paths;
408 splitString(var,paths,'/');
409 Integer nb_path = paths.size();
410 if (nb_path==1){
411 if (HGroup::hasChildren(loc_id.id(),var)){
412 _remove(loc_id.id(),var);
413 }
414 create(loc_id,var,save_type,space_id,plist);
415 return;
416 }
417 String last_name = paths[nb_path-1];
418 paths.resize(nb_path-1);
419 HGroup group;
420 group.recursiveCreate(loc_id,paths);
421 if (group.hasChildren(last_name)){
422 _remove(group.id(),last_name);
423 }
424 create(group.id(),last_name,save_type,space_id,plist);
425}

450void HDataset::
451openIfExists(const Hid& loc_id,const String& var)
452{
453 UniqueArray<String> paths;
454 splitString(var,paths,'/');
455 Integer nb_path = paths.size();
456 HGroup parent_group;
457 String last_name = var;
458 if (nb_path>1){
459 last_name = paths[nb_path-1];
460 paths.resize(nb_path-1);
461 parent_group.openIfExists(loc_id,paths);
462 }
463 else{
464 parent_group.open(loc_id,".");
465 }
466 if (parent_group.isBad())
467 return;
468 if (parent_group.hasChildren(last_name))
469 open(loc_id.id(),var.localstr());
470}

655void StandardArray::
656readDim()
657{
658 if (m_is_init)
659 return;
660 m_is_init = true;
661 m_hdataset.open(m_hfile,m_hpath);
662 HSpace hspace(m_hdataset.getSpace());
663 {
664 const int max_dim = 256; // Nombre maxi de dimensions des tableaux HDF
665 hsize_t hdf_dims[max_dim];

811
812template<typename DataType> void StandardArrayT<DataType>::
813_writeSortedValues(ITraceMng* tm,StandardTypes& st,
814 ConstArrayView<DataType> buffer,
815 Int64ConstArrayView unique_ids)
816{
817 ARCANE_UNUSED(tm);
818
819 Integer total_size = buffer.size();
820 Integer nb_element = unique_ids.size();
821 Integer dim2_size = 1;
822 if (nb_element != total_size){
823 if (nb_element == 0)
824 ARCANE_THROW(ArgumentException,"unique_ids size is zero but not buffer size ({0})",
825 total_size);
826 dim2_size = total_size / nb_element;
827 if (dim2_size*nb_element != total_size)
828 ARCANE_THROW(ArgumentException,"buffer size ({0}) is not a multiple of unique_ids size ({1})",
829 total_size,nb_element);
830 }
831
832 UniqueArray<ValueWithUid> values_to_sort(nb_element);
833 UniqueArray<DataType> out_buffer(total_size);
834 //tm->info() << " WRITE total_size=" << total_size
835 //<< " uid_size=" << unique_ids.size();
836 for( Integer i=0; i<nb_element; ++i ){
837 values_to_sort[i].m_uid = unique_ids[i];
838 values_to_sort[i].m_index = i;
839 //values_to_sort[i].m_value = buffer[i];
840 //tm->info() << "BEFORE SORT i=" << i << " uid=" << unique_ids[i];
841 }
842 std::sort(std::begin(values_to_sort),std::end(values_to_sort));
843 for( Integer i=0; i<nb_element; ++i ){
844 Integer old_index = values_to_sort[i].m_index;
845 for( Integer j=0; j<dim2_size; ++j ){
846 Integer pos = (i*dim2_size)+j;
847 out_buffer[pos] = buffer[(old_index*dim2_size)+j];
848 //tm->info() << "AFTER SORT i=" << i << " uid=" << values_to_sort[i].m_uid
849 // << " j=" << j << " pos=" << pos
850 // << " value=" << out_buffer[pos];
851 }
852 }
853 write(st,out_buffer);
854}

858
859template<typename DataType> void StandardArrayT<DataType>::
860parallelWrite(IParallelMng* pm,StandardTypes& st,
861 ConstArrayView<DataType> buffer,Int64ConstArrayView unique_ids)
862{
863 //TODO:
864 // Pour l'instant, seul le proc maitre ecrit.
865 // Il recupère toutes les valeurs, les trie par uniqueId croissant
866 // et les écrit.
867 // Il est important que tout soit trié par ordre croissant car
868 // cela permet de garder le même ordre d'écriture même en présence
869 // de repartitionnement du maillage. La relecture considère
870 // que cette contrainte est respectée et ne relit les informations
871 // des uniqueId qu'au démarrage du cas.
872 bool is_parallel = pm->isParallel();
873 ITraceMng* tm = pm->traceMng();
874
875 if (!is_parallel){
876 _writeSortedValues(tm,st,buffer,unique_ids);
877 return;
878 }
879
880 bool is_master = pm->isMasterIO();
881 Integer master_rank = pm->masterIORank();
882 Integer nb_rank = pm->commSize();
883 Integer buf_size = buffer.size();
884 Integer unique_id_size = unique_ids.size();
885 IntegerUniqueArray rank_sizes(2*nb_rank);
886 // Le sous-domaine maitre récupère les infos de tous les autres.
887 // Si un sous-domaine n'a pas d'éléments à envoyer, il ne fait rien
888 // (on n'envoie pas de buffers vides)
889 if (is_master){
890 Integer buf[2];
891 buf[0] = buf_size;
892 buf[1] = unique_id_size;
893 IntegerArrayView iav(2,buf);
894 pm->allGather(iav,rank_sizes);
895
896 Integer buffer_total_size = 0;
897 Integer unique_id_total_size = 0;
898 IntegerUniqueArray buffer_rank_index(nb_rank);
899 IntegerUniqueArray unique_id_rank_index(nb_rank);
900
901 for( Integer i=0; i<nb_rank; ++i ){
902 buffer_rank_index[i] = buffer_total_size;
903 buffer_total_size += rank_sizes[(i*2)];
904 unique_id_rank_index[i] = unique_id_total_size;
905 unique_id_total_size += rank_sizes[(i*2)+1];
906 }
907
908 UniqueArray<DataType> full_buffer(buffer_total_size);
909 Int64UniqueArray full_unique_ids(unique_id_total_size);
910
911 for( Integer i=0; i<nb_rank; ++i ){
912 // Ne recoit pas de valeurs des processus n'ayant pas de valeurs
913 if (rank_sizes[(i*2)]==0)
914 continue;
915 ArrayView<DataType> local_buf(rank_sizes[(i*2)],&full_buffer[ buffer_rank_index[i] ]);
916 Int64ArrayView local_unique_ids(rank_sizes[(i*2)+1],&full_unique_ids[ unique_id_rank_index[i] ]);
917 if (i==master_rank){
918 local_buf.copy(buffer);
919 local_unique_ids.copy(unique_ids);
920 }
921 else{
922 pm->recv(local_buf,i);
923 pm->recv(local_unique_ids,i);
924 }
925 }
926 tm->info(5) << "PARALLEL WRITE path=" << m_hpath << " total_size=" << full_buffer.size();
927 _writeSortedValues(tm,st,full_buffer,full_unique_ids);
928 }
929 else{
930 Integer buf[2];
931 buf[0] = buf_size;
932 buf[1] = unique_id_size;
933 IntegerArrayView iav(2,buf);
934 pm->allGather(iav,rank_sizes);
935 // Pas la peine d'envoyer des buffers vides
936 if (buffer.size()>0){
937 pm->send(buffer,master_rank);
938 pm->send(unique_ids,master_rank);
939 }
940 }
941}
