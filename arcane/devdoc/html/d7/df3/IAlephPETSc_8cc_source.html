56// ****************************************************************************
57// * AlephVectorCreate
58// ****************************************************************************
59void AlephVectorPETSc::
60AlephVectorCreate(void)
61{
62 for( int iCpu=0;iCpu<m_kernel->size();++iCpu){
63 if (m_kernel->rank()!=m_kernel->solverRanks(m_index)[iCpu])
64 continue;
65 if (jLower==-1)
66 jLower=m_kernel->topology()->gathered_nb_row(iCpu);
67 jUpper=m_kernel->topology()->gathered_nb_row(iCpu+1);
68 }
69 // Mise à jour de la taille locale du buffer pour le calcul plus tard de la norme max, par exemple
70 jSize=jUpper-jLower;
71 VecCreateMPI(MPI_COMM_SUB,
72 jSize, // n
73 m_kernel->topology()->nb_row_size(), // N
74 &m_petsc_vector);
75 debug()<<"\t\t[AlephVectorPETSc::AlephVectorCreate] PETSC VectorCreate"
76 <<" of local size=" <<jSize<<"/"<<m_kernel->topology()->nb_row_size()
77 <<", done";
78}

153// ****************************************************************************
154// * AlephMatrixCreate
155// ****************************************************************************
156void AlephMatrixPETSc::
157AlephMatrixCreate(void)
158{
159 Integer ilower=-1;
160 Integer iupper=0;
161
162 for( int iCpu=0;iCpu<m_kernel->size();++iCpu ){
163 if (m_kernel->rank()!=m_kernel->solverRanks(m_index)[iCpu]) continue;
164 if (ilower==-1) ilower=m_kernel->topology()->gathered_nb_row(iCpu);
165 iupper=m_kernel->topology()->gathered_nb_row(iCpu+1);
166 }
167 Integer size = iupper-ilower;
168 Integer jlower=ilower;
169 Integer jupper=iupper;
170
171#if PETSC_VERSION_GE(3,3,0)
172 #define PETSC_VERSION_MatCreate MatCreateAIJ
173#elif PETSC_VERSION_(3,0,0)
174 #define PETSC_VERSION_MatCreate MatCreateMPIAIJ
175#endif
176 PETSC_VERSION_MatCreate(MPI_COMM_SUB,
177 iupper-ilower, // m = number of rows
178 jupper-jlower, // n = number of columns
179 m_kernel->topology()->nb_row_size(), // M = number of global rows
180 m_kernel->topology()->nb_row_size(), // N = number of global columns
181 0, // ignored (number of nonzeros per row in DIAGONAL portion of local submatrix)
182 // array containing the number of nonzeros in the various rows of the DIAGONAL portion of local submatrix
183 m_kernel->topology()->gathered_nb_row_elements().subView(ilower,size).unguardedBasePointer(),
184 0, // ignored (number of nonzeros per row in the OFF-DIAGONAL portion of local submatrix)
185 // array containing the number of nonzeros in the various rows of the OFF-DIAGONAL portion of local submatrix
186 m_kernel->topology()->gathered_nb_row_elements().subView(ilower,size).unguardedBasePointer(),
187 &m_petsc_matrix);
188 MatSetOption(m_petsc_matrix, MAT_NEW_NONZERO_LOCATION_ERR, PETSC_FALSE);
189 MatSetUp(m_petsc_matrix);
190 debug()<<"\t\t[AlephMatrixPetsc::AlephMatrixCreate] PETSC MatrixCreate idx:"<<m_index
191 <<", ("<<ilower<<"->"<<(iupper-1)<<")";
192}

268// ****************************************************************************
269// * AlephMatrixSolve
270// ****************************************************************************
271int AlephMatrixPETSc::
272AlephMatrixSolve(AlephVector* x,AlephVector* b,AlephVector* t,
273 Integer& nb_iteration,Real* residual_norm,
274 AlephParams* solver_param)
275{
276 ARCANE_UNUSED(t);
277
278 PC prec;
279 PetscInt its;
280 PetscReal norm;
281 KSPConvergedReason reason;
282
283 AlephVectorPETSc* x_petsc = dynamic_cast<AlephVectorPETSc*> (x->implementation());
284 AlephVectorPETSc* b_petsc = dynamic_cast<AlephVectorPETSc*> (b->implementation());
285
286 ARCANE_CHECK_POINTER(x_petsc);
287 ARCANE_CHECK_POINTER(b_petsc);
288
289 Vec solution = x_petsc->m_petsc_vector;
290 Vec RHS = b_petsc->m_petsc_vector;
291
292 debug()<<"[AlephMatrixSolve]";
293 KSPCreate(MPI_COMM_SUB,&m_ksp_solver);
294#if PETSC_VERSION_GE(3,6,1)
295 KSPSetOperators(m_ksp_solver,m_petsc_matrix,m_petsc_matrix);//SAME_PRECONDITIONER);
296#else
297 KSPSetOperators(m_ksp_solver,m_petsc_matrix,m_petsc_matrix,SAME_NONZERO_PATTERN);//SAME_PRECONDITIONER);
298#endif
299 // Builds KSP for a particular solver
300 switch(solver_param->method()){
301 // Preconditioned conjugate gradient (PCG) iterative method
302 case TypesSolver::PCG : KSPSetType(m_ksp_solver,KSPCG); break;
303 // BiCGStab (Stabilized version of BiConjugate Gradient Squared) method
304 case TypesSolver::BiCGStab : KSPSetType(m_ksp_solver,KSPBCGS); break;
305 // IBiCGStab (Improved Stabilized version of BiConjugate Gradient Squared) method
306 // in an alternative form to have only a single global reduction operation instead of the usual 3 (or 4)
307 case TypesSolver::BiCGStab2: KSPSetType(m_ksp_solver,KSPIBCGS); break;
308 // Generalized Minimal Residual method. (Saad and Schultz, 1986) with restart
309 case TypesSolver::GMRES : KSPSetType(m_ksp_solver,KSPGMRES); break;
310 default : throw ArgumentException("AlephMatrixPETSc::AlephMatrixSolve", "Unknown solver method");
311 }
312
313 KSPGetPC(m_ksp_solver,&prec);
314 switch(solver_param->precond()){
315 case TypesSolver::NONE : PCSetType(prec,PCNONE); break;
316 // Jacobi (i.e. diagonal scaling preconditioning)
317 case TypesSolver::DIAGONAL : PCSetType(prec,PCJACOBI); break;
318 // Incomplete factorization preconditioners.
319 case TypesSolver::ILU : PCSetType(prec,PCILU); break;
320 // Incomplete Cholesky factorization preconditioners.
321 case TypesSolver::IC : PCSetType(prec,PCICC); break;
322 // Sparse Approximate Inverse method of Grote and Barnard as a preconditioner (SIAM J. Sci. Comput.; vol 18, nr 3)
323 case TypesSolver::SPAIstat : PCSetType(prec,PCSPAI); break;
324 // Use multigrid preconditioning.
325 // This preconditioner requires you provide additional information
326 // about the coarser grid matrices and restriction/interpolation operators.
327 case TypesSolver::AMG:
328 // By default PCMG uses GMRES on the fine grid smoother so this should be used with KSPFGMRES
329 // or the smoother changed to not use GMRES
330 // Implements the Flexible Generalized Minimal Residual method. developed by Saad with restart
331 KSPSetType(m_ksp_solver,KSPFGMRES);
332 PCSetType(prec,PCMG);
333 //Sets the number of levels to use with MG. Must be called before any other MG routine.
334 PCMGSetLevels(prec,1,
335 (MPI_Comm*)(m_kernel->subParallelMng(m_index)->getMPICommunicator()));
336 // Determines the form of multigrid to use: multiplicative, additive, full, or the Kaskade algorithm.
337 PCMGSetType(prec,PC_MG_MULTIPLICATIVE); // PC_MG_MULTIPLICATIVE,PC_MG_ADDITIVE,PC_MG_FULL,PC_MG_KASKADE
338 // Sets the type cycles to use. Use PCMGSetCycleTypeOnLevel() for more complicated cycling.
339 PCMGSetCycleType(prec,PC_MG_CYCLE_V); // PC_MG_CYCLE_V or PC_MG_CYCLE_W
340 //Sets the number of pre-smoothing steps to use on all levels
341#if PETSC_VERSION_GE(3,10,2)
342 PCMGSetNumberSmooth(prec, 1);
343#else
344 PCMGSetNumberSmoothDown(prec, 1);
345 PCMGSetNumberSmoothUp(prec, 1);
346#endif
347 // PCMGSetResidual: Sets the function to be used to calculate the residual on the lth level
348 // PCMGSetInterpolation: Sets the function to be used to calculate the interpolation from l-1 to the lth level
349 // PCMGSetRestriction: Sets the function to be used to restrict vector from level l to l-1
350 // PCMGSetRhs: Sets the vector space to be used to store the right-hand side on a particular level
351 // PCMGSetX: Sets the vector space to be used to store the solution on a particular level
352 // PCMGSetR: Sets the vector space to be used to store the residual on a particular level
353 //PCMGSetRestriction, PCMGSetRhs, PCMGSetX, PCMGSetR, etc.
354 //PCMGSetLevels
355 break;
356 case TypesSolver::AINV : throw ArgumentException("AlephMatrixPETSc", "preconditionnement AINV indisponible");
357 case TypesSolver::SPAIdyn : throw ArgumentException("AlephMatrixPETSc", "preconditionnement SPAIdyn indisponible");
358 case TypesSolver::ILUp : throw ArgumentException("AlephMatrixPETSc", "preconditionnement ILUp indisponible");
359 case TypesSolver::POLY : throw ArgumentException("AlephMatrixPETSc", "preconditionnement POLY indisponible");
360 default : throw ArgumentException("AlephMatrixPETSc", "preconditionnement inconnu");
361 }
362
363 if (solver_param->xoUser())
364 KSPSetInitialGuessNonzero(m_ksp_solver,PETSC_TRUE);
365
366 KSPSetTolerances(m_ksp_solver,
367 solver_param->epsilon(),
368 PETSC_DEFAULT,
369 PETSC_DEFAULT,
370 solver_param->maxIter());
371 KSPSetUp(m_ksp_solver);
372 //KSPSetFromOptions(m_ksp_solver);
373 debug()<<"[AlephMatrixSolve] All set up, mow solving";
374 KSPSolve(m_ksp_solver,RHS,solution);
375 debug()<<"[AlephMatrixSolve] solved";
376 KSPGetConvergedReason(m_ksp_solver,&reason);
377
378 switch(reason){
379#if !PETSC_VERSION_(3,0,0)
380 case(KSP_CONVERGED_RTOL_NORMAL):{break;}
381 case(KSP_CONVERGED_ATOL_NORMAL):{break;}
382#endif
383 case(KSP_CONVERGED_RTOL):{break;}
384 case(KSP_CONVERGED_ATOL):{break;}
385 case(KSP_CONVERGED_ITS):{break;}
386 case(KSP_CONVERGED_CG_NEG_CURVE):{break;}
387 case(KSP_CONVERGED_CG_CONSTRAINED):{break;}
388 case(KSP_CONVERGED_STEP_LENGTH):{break;}
389 case(KSP_CONVERGED_HAPPY_BREAKDOWN):{break;}
390 /* diverged */
391 case(KSP_DIVERGED_NULL):{ throw Exception("AlephMatrixPETSc::Solve", "KSP_DIVERGED_NULL");}
392 case(KSP_DIVERGED_ITS):{ throw Exception("AlephMatrixPETSc::Solve", "KSP_DIVERGED_ITS");}
393 case(KSP_DIVERGED_DTOL):{ throw Exception("AlephMatrixPETSc::Solve", "KSP_DIVERGED_DTOL");}
394 case(KSP_DIVERGED_BREAKDOWN):{ throw Exception("AlephMatrixPETSc::Solve", "KSP_DIVERGED_BREAKDOWN");}
395 case(KSP_DIVERGED_BREAKDOWN_BICG):{ throw Exception("AlephMatrixPETSc::Solve", "KSP_DIVERGED_BREAKDOWN_BICG");}
396 case(KSP_DIVERGED_NONSYMMETRIC):{ throw Exception("AlephMatrixPETSc::Solve", "KSP_DIVERGED_NONSYMMETRIC");}
397 case(KSP_DIVERGED_INDEFINITE_PC):{ throw Exception("AlephMatrixPETSc::Solve", "KSP_DIVERGED_INDEFINITE_PC");}
398#if PETSC_VERSION_GE(3,6,1)
399 case(KSP_DIVERGED_NANORINF):{ throw Exception("AlephMatrixPETSc::Solve", "KSP_DIVERGED_NANORINF");}
400#else
401 case(KSP_DIVERGED_NAN):{ throw Exception("AlephMatrixPETSc::Solve", "KSP_DIVERGED_NAN");}
402#endif
403 case(KSP_DIVERGED_INDEFINITE_MAT):{ throw Exception("AlephMatrixPETSc::Solve", "KSP_DIVERGED_INDEFINITE_MAT");}
404
405 case(KSP_CONVERGED_ITERATING):{break;}
406 default: throw Exception("AlephMatrixPETSc::Solve", "");
407 }
408
409 KSPGetIterationNumber(m_ksp_solver,&its);
410 nb_iteration=its;
411
412 KSPGetResidualNorm(m_ksp_solver,&norm);
413 *residual_norm=norm;
414
415 return 0;
416}
