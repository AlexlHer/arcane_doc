69void FaceUniqueIdBuilder::
70computeFacesUniqueIds()
71{
72 IParallelMng* pm = m_mesh->parallelMng();
73 Real begin_time = platform::getRealTime();
74 Integer face_version = m_mesh->meshUniqueIdMng()->faceBuilderVersion();
75 bool is_parallel = pm->isParallel();
76 info() << "Using version=" << face_version << " to compute faces unique ids"
77 << " mesh=" << m_mesh->name() << " is_parallel=" << is_parallel;
78
79 if (face_version>4 || face_version<0)
80 ARCANE_FATAL("Invalid value '{0}' for compute face unique ids versions: v>=0 && v<=4");
81
82 if (face_version==4)
83 arcaneComputeCartesianFaceUniqueId(m_mesh);
84 else if (face_version==3)
85 _FaceUiDBuilderComputeNewVersion(m_mesh);
86 else{
87 if (is_parallel){
88 if (face_version==2){
89 //PAS ENCORE PAR DEFAUT
90 info() << "Use new mesh init in FaceUniqueIdBuilder";
91 _computeFacesUniqueIdsParallelV2();
92 }
93 else{
94 // Version par défaut.
95 _computeFacesUniqueIdsParallelV1();
96 }
97 }
98 else{
99 if (face_version==0){
100 pwarning() << "No face renumbering";
101 return;
102 }
103 _computeFacesUniqueIdsSequential();
104 }
105 }
106
107 Real end_time = platform::getRealTime();
108 Real diff = (Real)(end_time - begin_time);
109 info() << "TIME to compute face unique ids=" << diff;
110
111 _checkNoDuplicate();
112
113 ItemInternalMap& faces_map = m_mesh->facesMap();
114
115 // Il faut ranger à nouveau #m_faces_map car les uniqueId() des
116 // faces ont été modifiés
117 m_mesh->faceFamily()->notifyItemsUniqueIdChanged();
118
119 bool is_verbose = m_mesh_builder->isVerbose();
120 if (is_verbose){
121 info() << "NEW FACES_MAP after re-indexing";
122 ENUMERATE_ITEM_INTERNAL_MAP_DATA(nbid,faces_map){
123 ItemInternal* face = nbid->value();
124 info() << "Face uid=" << face->uniqueId() << " lid=" << face->localId();
125 }
126 }
127}

134void FaceUniqueIdBuilder::
135_checkNoDuplicate()
136{
137 info() << "Check no duplicate face uniqueId";
138 ItemInternalMap& faces_map = m_mesh->facesMap();
139 std::map<ItemUniqueId,ItemInternal*> checked_faces_map;
140 ENUMERATE_ITEM_INTERNAL_MAP_DATA(nbid,faces_map){
141 ItemInternal* face = nbid->value();
142 ItemUniqueId uid = face->uniqueId();
143 auto p = checked_faces_map.find(uid);
144 if (p!=checked_faces_map.end()){
145 pwarning() << "Duplicate Face UniqueId=" << uid;
146 ARCANE_FATAL("Duplicate Face uniqueId={0}",uid);
147 }
148 checked_faces_map.insert(std::make_pair(uid,face));
149 }
150}

207void FaceUniqueIdBuilder::
208_computeFacesUniqueIdsParallelV1()
209{
210 IParallelMng* pm = m_mesh->parallelMng();
211 Integer my_rank = pm->commRank();
212 Integer nb_rank = pm->commSize();
213
214 Integer nb_local_face = m_mesh_builder->oneMeshItemAdder()->nbFace();
215 Integer nb_local_cell = m_mesh_builder->oneMeshItemAdder()->nbCell();
216 bool is_verbose = m_mesh_builder->isVerbose();
217
218 UniqueArray<Int64> faces_opposite_cell_uid(nb_local_face);
219 faces_opposite_cell_uid.fill(NULL_ITEM_ID);
220 UniqueArray<Integer> faces_opposite_cell_index(nb_local_face);
221 UniqueArray<Integer> faces_opposite_cell_owner(nb_local_face);
222
223 // Pour vérification, s'assure que tous les éléments de ce tableau
224 // sont valides, ce qui signifie que toutes les faces ont bien été
225 // renumérotés
226 UniqueArray<Int64> faces_new_uid(nb_local_face);
227 faces_new_uid.fill(NULL_ITEM_ID);
228
229 Integer nb_recv_sub_domain_boundary_face = 0;
230
231 Int64UniqueArray faces_infos;
232 faces_infos.reserve(10000);
233 ItemInternalMap& cells_map = m_mesh->cellsMap();
234 ItemInternalMap& faces_map = m_mesh->facesMap();
235 ItemInternalMap& nodes_map = m_mesh->nodesMap();
236
237
238 // NOTE: ce tableau n'est pas utile sur toutes les mailles. Il
239 // suffit qu'il contienne les mailles dont on a besoin, c'est à dire
240 // les notres + celles connectées à une de nos faces. Une table
241 // de hashage sera plus appropriée.
242 HashTableMapT<Int64,Int64> cells_first_face_uid(m_mesh_builder->oneMeshItemAdder()->nbCell()*2,true);
243
244 // Rassemble les données des autres processeurs dans recv_cells;
245 // Pour éviter que les tableaux ne soient trop gros, on procède en plusieurs
246 // étapes.
247 // Chaque sous-domaine construit sa liste de faces frontières, avec pour
248 // chaque face:
249 // - son type
250 // - la liste de ses noeuds,
251 // - le numéro unique de sa maille
252 // - le propriétaire de sa maille
253 // - son indice dans sa maille
254 // Cette liste sera ensuite envoyée à tous les sous-domaines.
255 {
256 ItemTypeMng* itm = m_mesh->itemTypeMng();
257
258 UniqueArray<ItemInternal*> faces;
259
260 ENUMERATE_ITEM_INTERNAL_MAP_DATA(nbid,faces_map){
261 ItemInternal* face = nbid->value();
262 bool boundary_val = face->flags() & ItemInternal::II_Boundary;
263 if (boundary_val)
264 faces.add(face);
265 }
266
267 Integer nb_sub_domain_boundary_face = faces.size();
268 Int64 global_nb_boundary_face = pm->reduce(Parallel::ReduceSum,(Int64)nb_sub_domain_boundary_face);
269 debug() << "NB BOUNDARY FACE=" << nb_sub_domain_boundary_face
270 << " NB_FACE=" << nb_local_face
271 << " GLOBAL_NB_BOUNDARY_FACE=" << global_nb_boundary_face;
272
273
274 Int64UniqueArray faces_infos2;
275 Int64UniqueArray recv_faces_infos;
276
277 // Calcule la taille d'un bloc d'envoie
278 // La mémoire nécessaire pour une face est égale à nb_node + 4.
279 // Si on suppose qu'on a des quadrangles en général, cela
280 // fait 8 Int64 par face, soit 64 octets par face.
281 // La mémoire nécessaire pour tout envoyer est donc 64 * global_nb_boundary_face.
282 // step_size * nb_proc * 64 octets.
283 // Le step_size par défaut est calculé pour que la mémoire nécessaire
284 // soit de l'ordre de 100 Mo pour chaque message.
285 Int64 step_size = 1500000;
286 Integer nb_phase = CheckedConvert::toInteger((global_nb_boundary_face / step_size) + 1);
287 for( Integer i_phase=0; i_phase<nb_phase; ++i_phase ){
288 Integer nb_face_to_send = nb_sub_domain_boundary_face / nb_phase;
289 Integer first_face_to_send = nb_face_to_send * i_phase;
290 Integer last_face_to_send = first_face_to_send + nb_face_to_send;
291 if (i_phase+1==nb_phase)
292 last_face_to_send = nb_sub_domain_boundary_face;
293 Integer real_nb_face_to_send = last_face_to_send - first_face_to_send;
294
295 faces_infos2.clear();
296 for( Integer i_face=first_face_to_send; i_face<first_face_to_send+real_nb_face_to_send; ++i_face ){
297 Face face(faces[i_face]);
298 face.internal()->addFlags(ItemInternal::II_Shared | ItemInternal::II_SubDomainBoundary);
299 bool has_back_cell = face.internal()->flags() & ItemInternal::II_HasBackCell;
300 faces_infos2.add(face.type());
301 for( Node node : face.nodes() )
302 faces_infos2.add(node.uniqueId().asInt64());
303
304 //info() << " ADD FACE lid=" << face->localId();
305
306 Cell cell = face.cell(0);
307 faces_infos2.add(cell.uniqueId().asInt64());
308 faces_infos2.add(cell.owner());
309
310 Integer face_index_in_cell = 0;
311 if (has_back_cell){
312 for( Face current_face_in_cell : cell.faces() ){
313 if (current_face_in_cell==face)
314 break;
315 if (current_face_in_cell.backCell()==cell)
316 ++face_index_in_cell;
317 }
318 }
319 else
320 face_index_in_cell = NULL_ITEM_ID;
321 faces_infos2.add(face_index_in_cell);
322 }
323 faces_infos2.add(IT_NullType); // Pour dire que la liste s'arête
324
325 pm->allGatherVariable(faces_infos2,recv_faces_infos);
326
327 info() << "Number of face bytes received: " << recv_faces_infos.size()
328 << " phase=" << i_phase << "/" << nb_phase
329 << " first_face=" << first_face_to_send
330 << " last_face=" << last_face_to_send
331 << " nb=" << real_nb_face_to_send;
332
333 Integer recv_faces_infos_index = 0;
334
335 for( Integer i_sub_domain=0; i_sub_domain<nb_rank; ++i_sub_domain ){
336 bool is_end = false;
337 // Il ne faut pas lire les infos qui viennent de moi
338 if (i_sub_domain==my_rank){
339 recv_faces_infos_index += faces_infos2.size();
340 continue;
341 }
342 // Désérialise les infos de chaque sous-domaine
343 while (!is_end){
344 Integer face_type = CheckedConvert::toInteger(recv_faces_infos[recv_faces_infos_index]);
345 ++recv_faces_infos_index;
346 if (face_type==IT_NullType){
347 is_end = true;
348 break;
349 }
350 ItemTypeInfo* itt = itm->typeFromId(face_type);
351 Integer face_nb_node = itt->nbLocalNode();
352 ConstArrayView<Int64> faces_nodes_uid(face_nb_node,&recv_faces_infos[recv_faces_infos_index]);
353
354 recv_faces_infos_index += face_nb_node;
355 Int64 cell_uid = recv_faces_infos[recv_faces_infos_index];
356 ++recv_faces_infos_index;
357 Integer cell_owner = CheckedConvert::toInteger(recv_faces_infos[recv_faces_infos_index]);
358 ++recv_faces_infos_index;
359 Integer cell_face_index = CheckedConvert::toInteger(recv_faces_infos[recv_faces_infos_index]);
360 ++recv_faces_infos_index;
361
362 ItemInternalMapData* data = nodes_map.lookup(faces_nodes_uid[0]);
363 // Si la face n'existe pas dans mon sous-domaine, elle ne m'intéresse pas
364 if (!data)
365 continue;
366 ItemInternal* face = ItemTools::findFaceInNode(data->value(),face_type,faces_nodes_uid);
367 if (!face)
368 continue;
369 ++nb_recv_sub_domain_boundary_face;
370 faces_opposite_cell_uid[face->localId()] = cell_uid;
371 faces_opposite_cell_index[face->localId()] = cell_face_index;
372 faces_opposite_cell_owner[face->localId()] = cell_owner;
373 cells_first_face_uid.add(cell_uid,-1);
374 }
375 }
376 }
377 info() << "Number of faces on the subdomain interface: "
378 << nb_sub_domain_boundary_face << ' ' << nb_recv_sub_domain_boundary_face;
379 }
380
381 // Cherche le uniqueId max des mailles sur tous les sous-domaines.
382 Int64 max_cell_uid = 0;
383 Int32 max_cell_local_id = 0;
384 ENUMERATE_ITEM_INTERNAL_MAP_DATA(nbid,cells_map){
385 ItemInternal* cell = nbid->value();
386 Int64 cell_uid = cell->uniqueId().asInt64();
387 Int32 cell_local_id = cell->localId();
388 if (cell_uid>max_cell_uid)
389 max_cell_uid = cell_uid;
390 if (cell_local_id>max_cell_local_id)
391 max_cell_local_id = cell_local_id;
392 }
393 Int64 global_max_cell_uid = pm->reduce(Parallel::ReduceMax,max_cell_uid);
394 debug() << "GLOBAL MAX CELL UID=" << global_max_cell_uid;
395
396
397 UniqueArray<T_CellFaceInfo> my_cells_faces_info;
398 my_cells_faces_info.reserve(nb_local_cell);
399 IntegerUniqueArray my_cells_nb_back_face(max_cell_local_id+1);
400 my_cells_nb_back_face.fill(0);
401
402 ENUMERATE_ITEM_INTERNAL_MAP_DATA(nbid,cells_map){
403 Cell cell(nbid->value());
404 Int64 cell_uid = cell.uniqueId().asInt64();
405 Int32 cell_local_id = cell.localId();
406 Integer nb_back_face = 0;
407 Integer nb_true_boundary_face = 0;
408 for( Face face : cell.faces() ){
409 Int64 opposite_cell_uid = faces_opposite_cell_uid[face.localId()];
410 if (face.backCell()==cell)
411 ++nb_back_face;
412 else if (face->nbCell()==1 && opposite_cell_uid==NULL_ITEM_ID){
413 ++nb_true_boundary_face;
414 }
415 }
416 my_cells_nb_back_face[cell_local_id] = nb_back_face;
417 my_cells_faces_info.add(T_CellFaceInfo(cell_uid,nb_back_face,nb_true_boundary_face));
418 }
419 std::sort(std::begin(my_cells_faces_info),std::end(my_cells_faces_info));
420
421 {
422 Integer nb_phase = 16;
423 Integer first_cell_index_to_send = 0;
424 Int64 current_face_uid = 0;
425
426 for( Integer i_phase=0; i_phase<nb_phase; ++i_phase ){
427 Integer nb_uid_to_send = CheckedConvert::toInteger(global_max_cell_uid / nb_phase);
428 Integer first_uid_to_send = nb_uid_to_send * i_phase;
429 Int64 last_uid_to_send = first_uid_to_send + nb_uid_to_send;
430 if (i_phase+1==nb_phase)
431 last_uid_to_send = global_max_cell_uid;
432
433 //Integer last_cell_index_to_send = first_cell_index_to_send;
434 Integer nb_cell_to_send = 0;
435 for( Integer zz=first_cell_index_to_send, zs=my_cells_faces_info.size(); zz<zs; ++zz ){
436 if (my_cells_faces_info[zz].m_unique_id<=last_uid_to_send)
437 ++nb_cell_to_send;
438 else
439 break;
440 }
441 debug() << "FIRST TO SEND=" << first_cell_index_to_send
442 << " NB=" << nb_cell_to_send
443 << " first_uid=" << first_uid_to_send
444 << " last_uid=" << last_uid_to_send;
445
446 T_CellFaceInfo* begin_cell_array = my_cells_faces_info.data()+first_cell_index_to_send;
447 Int64* begin_array = reinterpret_cast<Int64*>(begin_cell_array);
448 Integer begin_size = nb_cell_to_send*3;
449
450 Int64ConstArrayView cells_faces_infos(begin_size,begin_array);
451
452 Int64UniqueArray recv_cells_faces_infos;
453 pm->allGatherVariable(cells_faces_infos,recv_cells_faces_infos);
454 first_cell_index_to_send += nb_cell_to_send;
455
456 info() << "Infos faces (received) nb_int64=" << recv_cells_faces_infos.size();
457 Integer recv_nb_cell = recv_cells_faces_infos.size() / 3;
458
459 // NOTE: comme on connait le uid min et max possible, on peut optimiser en
460 // creant un tableau dimensionne avec comme borne ce min et ce max et
461 // remplir directement les elements de ce tableau. Comme cela, on
462 // n'a pas besoin de tri.
463 UniqueArray<T_CellFaceInfo> global_cells_faces_info(recv_nb_cell);
464 for( Integer i=0; i<recv_nb_cell; ++i ){
465 Int64 cell_uid = recv_cells_faces_infos[i*3];
466 global_cells_faces_info[i].m_unique_id = cell_uid;
467 global_cells_faces_info[i].m_nb_back_face = recv_cells_faces_infos[(i*3) +1];
468 global_cells_faces_info[i].m_nb_true_boundary_face = recv_cells_faces_infos[(i*3) +2];
469 }
470 info() << "Sorting the faces nb=" << global_cells_faces_info.size();
471 std::sort(std::begin(global_cells_faces_info),std::end(global_cells_faces_info));
472
473 for( Integer i=0; i<recv_nb_cell; ++i ){
474 Int64 cell_uid = global_cells_faces_info[i].m_unique_id;
475 if (cells_map.hasKey(cell_uid) || cells_first_face_uid.hasKey(cell_uid))
476 cells_first_face_uid.add(cell_uid,current_face_uid);
477 current_face_uid += global_cells_faces_info[i].m_nb_back_face + global_cells_faces_info[i].m_nb_true_boundary_face;
478 }
479 }
480 }
481
482 ENUMERATE_ITEM_INTERNAL_MAP_DATA(nbid,cells_map){
483 Cell cell(nbid->value());
484 Int64 cell_uid = cell.uniqueId();
485 Int32 cell_local_id = cell.localId();
486 Integer num_local_face = 0;
487 Integer num_true_boundary_face = 0;
488 for( Face face : cell.faces() ){
489 Int64 opposite_cell_uid = faces_opposite_cell_uid[face.localId()];
490 Int64 face_new_uid = NULL_ITEM_UNIQUE_ID;
491 if (face.backCell()==cell){
492 if (!cells_first_face_uid.hasKey(cell_uid))
493 fatal() << "NO KEY 0 for cell_uid=" << cell_uid;
494 face_new_uid = cells_first_face_uid[cell_uid]+num_local_face;
495 face.internal()->setOwner(my_rank,my_rank);
496 ++num_local_face;
497 }
498 else if (face.nbCell()==1){
499 if (opposite_cell_uid==NULL_ITEM_UNIQUE_ID){
500 // Il s'agit d'une face frontière du domaine initial
501 if (!cells_first_face_uid.hasKey(cell_uid))
502 fatal() << "NO KEY 1 for cell_uid=" << cell_uid;
503 face_new_uid = cells_first_face_uid[cell_uid] + my_cells_nb_back_face[cell_local_id] + num_true_boundary_face;
504 ++num_true_boundary_face;
505 face.internal()->setOwner(my_rank,my_rank);
506 }
507 else{
508 if (!cells_first_face_uid.hasKey(opposite_cell_uid))
509 fatal() << "NO KEY 1 for cell_uid=" << cell_uid << " opoosite=" << opposite_cell_uid;
510 face_new_uid = cells_first_face_uid[opposite_cell_uid]+faces_opposite_cell_index[face.localId()];
511 face.internal()->setOwner(faces_opposite_cell_owner[face.localId()],my_rank);
512 }
513 }
514 if (face_new_uid!=NULL_ITEM_UNIQUE_ID){
515 faces_new_uid[face.localId()] = face_new_uid;
516 face.internal()->setUniqueId(face_new_uid);
517 }
518 }
519 }
520
521 // Vérifie que toutes les faces ont été réindéxées
522 {
523 Integer nb_error = 0;
524 for( Integer i=0, is=nb_local_face; i<is; ++i ){
525 if (faces_new_uid[i]==NULL_ITEM_UNIQUE_ID){
526 ++nb_error;
527 if (nb_error<10)
528 error() << "The face lid=" << i << " has not been re-indexed.";
529 }
530 }
531 if (nb_error!=0)
532 ARCANE_FATAL("Some ({0}) faces have not been reindexed",nb_error);
533 }
534
535 if (is_verbose){
536 OStringStream ostr;
537 ENUMERATE_ITEM_INTERNAL_MAP_DATA(nbid,cells_map){
538 Cell cell(nbid->value());
539 Int64 cell_uid = cell->uniqueId().asInt64();
540 Integer face_index = 0;
541 for( Face face : cell.faces() ){
542 Int64 opposite_cell_uid = faces_opposite_cell_uid[face.localId()];
543 bool shared = false;
544 bool true_boundary = false;
545 bool internal_other = false;
546 if (face.backCell()==cell){
547 }
548 else if (face.nbCell()==1){
549 if (opposite_cell_uid==NULL_ITEM_ID)
550 true_boundary = true;
551 else
552 shared = true;
553 }
554 else{
555 internal_other = true;
556 opposite_cell_uid = face.backCell().uniqueId().asInt64();
557 }
558 ostr() << "NEW UNIQUE ID FOR FACE"
559 << " lid=" << face->localId()
560 << " cell=" << cell_uid
561 << " face=" << face->uniqueId()
562 << " nbcell=" << face->nbCell()
563 << " cellindex=" << face_index << " (";
564 for( Node node : face.nodes() )
565 ostr() << ' ' << node.uniqueId();
566 ostr() << ")";
567 if (internal_other)
568 ostr() << " internal-other";
569 if (true_boundary)
570 ostr() << " true-boundary";
571 if (opposite_cell_uid!=NULL_ITEM_ID){
572 ostr() << " opposite " << opposite_cell_uid;
573 }
574 if (shared)
575 ostr() << " (shared)";
576 ostr() << "\n";
577 ++face_index;
578 }
579 }
580 info() << ostr.str();
581 String file_name("faces_uid.");
582 file_name = file_name + my_rank;
583 std::ofstream ofile(file_name.localstr());
584 ofile << ostr.str();
585 }

594void FaceUniqueIdBuilder::
595_exchangeData(IParallelExchanger* exchanger,BoundaryInfosMap& boundary_infos_to_send)
596{
597 for( BoundaryInfosMapEnumerator i_map(boundary_infos_to_send); ++i_map; ){
598 Int32 sd = i_map.data()->key();
599 exchanger->addSender(sd);
600 }
601 exchanger->initializeCommunicationsMessages();
602 Integer nb_sender = exchanger->nbSender();
603 Integer nb_receiver = exchanger->nbReceiver();
604 info() << "NB_SEND=" << nb_sender << " NB_RECV=" << nb_receiver;
605 Integer total = nb_sender+nb_receiver;
606 Integer global_total = exchanger->parallelMng()->reduce(Parallel::ReduceSum,total);
607 info() << "GLOBAL_NB_MESSAGE=" << global_total;
608
609 {
610 for( Integer i=0, ns=exchanger->nbSender(); i<ns; ++i ){
611 ISerializeMessage* sm = exchanger->messageToSend(i);
612 Int32 rank = sm->destination().value();
613 ISerializer* s = sm->serializer();

674void FaceUniqueIdBuilder::
675_computeFacesUniqueIdsParallelV2()
676{
677 IParallelMng* pm = m_mesh->parallelMng();
678 Integer my_rank = pm->commRank();
679 Integer nb_rank = pm->commSize();
680
681 Integer nb_local_face = m_mesh_builder->oneMeshItemAdder()->nbFace();
682 //Integer nb_local_cell = m_mesh_builder->nbCell();
683 //bool is_verbose = m_mesh_builder->isVerbose();
684
685 Int64UniqueArray faces_opposite_cell_uid(nb_local_face);
686 faces_opposite_cell_uid.fill(NULL_ITEM_ID);
687 IntegerUniqueArray faces_opposite_cell_index(nb_local_face);
688 IntegerUniqueArray faces_opposite_cell_owner(nb_local_face);
689
690 // Pour vérification, s'assure que tous les éléments de ce tableau
691 // sont valides, ce qui signifie que toutes les faces ont bien été
692 // renumérotés
693 Int64UniqueArray faces_new_uid(nb_local_face);
694 faces_new_uid.fill(NULL_ITEM_ID);
695
696 Int64UniqueArray faces_infos;
697 faces_infos.reserve(10000);
698 ItemInternalMap& faces_map = m_mesh->facesMap();
699 ItemInternalMap& nodes_map = m_mesh->nodesMap();
700
701
702 // NOTE: ce tableau n'est pas utile sur toutes les mailles. Il
703 // suffit qu'il contienne les mailles dont on a besoin, c'est à dire
704 // les notres + celles connectées à une de nos faces.
705 HashTableMapT<Int32,Int32> cell_first_face_uid(m_mesh_builder->oneMeshItemAdder()->nbCell()*2,true);
706
707 // Rassemble les données des autres processeurs dans recv_cells;
708 // Pour éviter que les tableaux ne soient trop gros, on procède en plusieurs
709 // étapes.
710 // Chaque sous-domaine construit sa liste de faces frontières, avec pour
711 // chaque face:
712 // - son type
713 // - la liste de ses noeuds,
714 // - le numéro unique de sa maille
715 // - le propriétaire de sa maille
716 // - son indice dans sa maille
717 // Cette liste sera ensuite envoyée à tous les sous-domaines.
718 ItemTypeMng* itm = m_mesh->itemTypeMng();
719
720 // Détermine le unique id max des noeuds
721 Int64 my_max_node_uid = NULL_ITEM_UNIQUE_ID;
722 ENUMERATE_ITEM_INTERNAL_MAP_DATA(nbid,nodes_map){
723 ItemInternal* node = nbid->value();
724 Int64 node_uid = node->uniqueId();
725 if (node_uid>my_max_node_uid)
726 my_max_node_uid = node_uid;
727 }
728 Int64 global_max_node_uid = pm->reduce(Parallel::ReduceMax,my_max_node_uid);
729 debug() << "NODE_UID_INFO: MY_MAX_UID=" << my_max_node_uid
730 << " GLOBAL=" << global_max_node_uid;
731
732 //TODO: choisir bonne valeur pour initialiser la table
733 BoundaryInfosMap boundary_infos_to_send(nb_rank,true);
734 NodeUidToSubDomain uid_to_subdomain_converter(global_max_node_uid,nb_rank);
735 info() << "NB_CORE modulo=" << uid_to_subdomain_converter.modulo();
736 HashTableMapT<Int64,SharedArray<Int64> > nodes_info(100000,true);
737 IItemFamily* node_family = m_mesh->nodeFamily();
738 UniqueArray<bool> is_boundary_nodes(node_family->maxLocalId(),false);
739
740 // Marque tous les noeuds frontieres car ce sont ceux qu'il faudra envoyer
741 ENUMERATE_ITEM_INTERNAL_MAP_DATA(nbid,faces_map){
742 Face face(nbid->value());
743 Integer face_nb_cell = face.nbCell();
744 if (face_nb_cell==1){
745 for( Node node : face.nodes() )
746 is_boundary_nodes[node.localId()] = true;
747 }
748 }
749
750 // Détermine la liste des faces frontières
751 ENUMERATE_ITEM_INTERNAL_MAP_DATA(nbid,faces_map){
752 Face face(nbid->value());
753 Node first_node = face.node(0);
754 Int64 first_node_uid = first_node.uniqueId();
755 SharedArray<Int64> v;
756 Int32 dest_rank = -1;
757 if (!is_boundary_nodes[first_node.localId()]){
758 v = nodes_info.lookupAdd(first_node_uid)->value();
759 }
760 else{
761 dest_rank = uid_to_subdomain_converter.uidToRank(first_node_uid);
762 v = boundary_infos_to_send.lookupAdd(dest_rank)->value();
763 }
764 v.add(first_node_uid); // 0
765 v.add(my_rank); // 1
766 v.add(face->uniqueId()); // 2
767 v.add(face->type()); // 3
768 Cell back_cell = face.backCell();
769 Cell front_cell = face.frontCell();
770 if (back_cell.null()) // 4 : only used for debug
771 v.add(NULL_ITEM_UNIQUE_ID);
772 else
773 v.add(back_cell.uniqueId());
774 if (front_cell.null()) // 5 : only used for debug
775 v.add(NULL_ITEM_UNIQUE_ID);
776 else
777 v.add(front_cell.uniqueId());
778 for( Integer z=0, zs=face.nbNode(); z<zs; ++z )
779 v.add(face.node(z)->uniqueId());
780 }
781
782 // Positionne la liste des envoies
783 Ref<IParallelExchanger> exchanger{ParallelMngUtils::createExchangerRef(pm)};
784 _exchangeData(exchanger.get(),boundary_infos_to_send);
785
786 {
787 Integer nb_receiver = exchanger->nbReceiver();
788 debug() << "NB RECEIVER=" << nb_receiver;
789 Int64UniqueArray received_infos;
790 for( Integer i=0; i<nb_receiver; ++i ){
791 ISerializeMessage* sm = exchanger->messageToReceive(i);
792 //Int32 orig_rank = sm->destSubDomain();
793 ISerializer* s = sm->serializer();
794 s->setMode(ISerializer::ModeGet);
795 Int64 nb_info = s->getInt64();
796 //info() << "RECEIVE NB_INFO=" << nb_info << " from=" << orig_rank;
797 received_infos.resize(nb_info);
798 s->getSpan(received_infos);
799 //if ((nb_info % 3)!=0)
800 //fatal() << "info size can not be divided by 3";
801 Integer z =0;
802 while(z<nb_info){
803 Int64 node_uid = received_infos[z+0];
804 Int32 face_type = (Int32)received_infos[z+3];
805 ItemTypeInfo* itt = itm->typeFromId(face_type);
806 Integer face_nb_node = itt->nbLocalNode();
807 Int64Array& a = nodes_info.lookupAdd(node_uid)->value();
808 a.addRange(Int64ConstArrayView(6+face_nb_node,&received_infos[z]));
809 z += 6;
810 z += face_nb_node;
811 }
812 }
813 Integer my_max_face_node = 0;
814 for( HashTableMapT<Int64,SharedArray<Int64> >::Enumerator inode(nodes_info); ++inode; ){
815 Int64ConstArrayView a = *inode;
816 Integer nb_info = a.size();
817 Integer z = 0;
818 Integer node_nb_face = 0;
819 while(z<nb_info){
820 ++node_nb_face;
821 Int32 face_type = (Int32)a[z+3];
822 ItemTypeInfo* itt = itm->typeFromId(face_type);
823 Integer face_nb_node = itt->nbLocalNode();
824 z += 6;
825 z += face_nb_node;
826 }
827 my_max_face_node = math::max(node_nb_face,my_max_face_node);
828 }
829 Integer global_max_face_node = pm->reduce(Parallel::ReduceMax,my_max_face_node);
830 debug() << "GLOBAL MAX FACE NODE=" << global_max_face_node;
831 // OK, maintenant donne comme uid de la face (node_uid * global_max_face_node + index)
832 IntegerUniqueArray indexes;
833 boundary_infos_to_send = BoundaryInfosMap(nb_rank,true);
834
835 for( HashTableMapT<Int64,SharedArray<Int64> >::Enumerator inode(nodes_info); ++inode; ){
836 Int64ConstArrayView a = *inode;
837 Integer nb_info = a.size();
838 Integer z = 0;
839 Integer node_nb_face = 0;
840 indexes.clear();
841 while(z<nb_info){
842 Int64 node_uid = a[z+0];
843 Int32 sender_rank = (Int32)a[z+1];
844 Int64 face_uid = a[z+2];
845 Int32 face_type = (Int32)a[z+3];
846 ItemTypeInfo* itt = itm->typeFromId(face_type);
847 Integer face_nb_node = itt->nbLocalNode();
848
849 // Regarde si la face est déjà dans la liste:
850 Integer face_index = node_nb_face;
851 Int32 face_new_owner = sender_rank;
852 for( Integer y=0; y<node_nb_face; ++y ){
853 if (memcmp(&a[indexes[y]+6],&a[z+6],sizeof(Int64)*face_nb_node)==0){
854 face_index = y;
855 face_new_owner = (Int32)a[indexes[y]+1];
856 }
857 }
858 Int64 face_new_uid = (node_uid * global_max_face_node) + face_index;
859 Int64Array& v = boundary_infos_to_send.lookupAdd(sender_rank)->value();
860 // Indique au propriétaire de cette face son nouvel uid
861 v.add(face_uid);
862 v.add(face_new_uid);
863 v.add(face_new_owner);
864 indexes.add(z);
865 z += 6;
866 z += face_nb_node;
867 ++node_nb_face;
868 }
869 my_max_face_node = math::max(node_nb_face,my_max_face_node);
870 }
871 }
872 exchanger = ParallelMngUtils::createExchangerRef(pm);
873
874 _exchangeData(exchanger.get(),boundary_infos_to_send);
875 {
876 Integer nb_receiver = exchanger->nbReceiver();
877 debug() << "NB RECEIVER=" << nb_receiver;
878 Int64UniqueArray received_infos;
879 for( Integer i=0; i<nb_receiver; ++i ){
880 ISerializeMessage* sm = exchanger->messageToReceive(i);
881 ISerializer* s = sm->serializer();
882 s->setMode(ISerializer::ModeGet);
883 Int64 nb_info = s->getInt64();
884 received_infos.resize(nb_info);
885 s->getSpan(received_infos);
886 if ((nb_info % 3)!=0)
887 ARCANE_FATAL("info size can not be divided by 3 v={0}",nb_info);;
888 Int64 nb_item = nb_info / 3;
889 for (Int64 z=0; z<nb_item; ++z ){
890 Int64 old_uid = received_infos[(z*3)];
891 Int64 new_uid = received_infos[(z*3)+1];
892 Int32 new_owner = (Int32)received_infos[(z*3)+2];
893 ItemInternalMapData* face_data = faces_map.lookup(old_uid);
894 if (!face_data)
895 fatal() << "Can not find own face uid=" << old_uid;
896 face_data->value()->setUniqueId(new_uid);
897 face_data->value()->setOwner(new_owner,my_rank);
898 }
899 }
900 }
901
902 traceMng()->flush();
903 pm->barrier();
904 debug() << "END OF TEST NEW FACE COMPUTE";
905 return;
906}

954 for( Integer i=0; i<nb_computed; ++i ){
955 cell_first_face_uid[i] = current_face_uid;
956 current_face_uid += cell_nb_num_back_face[i] + cell_true_boundary_face[i];

979 ++nb_num_back_face;
980 }
981 else if (face.nbCell()==1){
982 face_new_uid = cell_first_face_uid[cell_uid] + cell_nb_num_back_face[cell_uid] + nb_true_boundary_face;
983 ++nb_true_boundary_face;
984 }
985 if (face_new_uid!=NULL_ITEM_UNIQUE_ID){
986 face.internal()->setUniqueId(face_new_uid);

988 }

1000 if (face->backCell()==cell){
1001 }
1002 else if (face->nbCell()==1){
1003 true_boundary = true;
1004 }

1006 internal_other = true;
1007 opposite_cell_uid = face.backCell().uniqueId().asInt64();
1008 }
1009 ostr() << "NEW LOCAL ID FOR CELLFACE cell_uid=" << cell.uniqueId() << ' '
1010 << face_index << " uid=" << face.uniqueId() << " (";

1017 ostr() << " true-boundary";
1018 if (opposite_cell_uid!=NULL_ITEM_ID)
1019 ostr() << " opposite " << opposite_cell_uid;
1020 ostr() << '\n';
1021 ++face_index;
1022 }
1023 }
1024 info() << ostr.str();
1025 }
