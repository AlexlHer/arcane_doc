39template<class T> inline bool
40builtInGetValue(T& v,const String& s)
41{
42 T read_val = T();
43 std::istringstream sbuf(s.localstr());
44 sbuf >> read_val;
45 if (sbuf.fail() || sbuf.bad())
46 return true;
47 if (!sbuf.eof())
48 return true;
49 v = read_val;
50 return false;
51}

113inline bool
114builtInGetValue(bool& v,const String& s)
115{
116 if (s.null())
117 return true;
118 if (s=="false" || s=="faux" || s=="0"){
119 v = false;
120 return false;
121 }
122
123 if (s=="true" || s=="vrai" || s=="1"){
124 v = true;
125 return false;
126 }
127 return true;
128}
