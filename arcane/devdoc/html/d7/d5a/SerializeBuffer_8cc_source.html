30void SerializeBuffer::
31allGather(IParallelMng* pm,const SerializeBuffer& send_serializer)
32{
33 const SerializeBuffer* sbuf = &send_serializer;
34 SerializeBuffer* recv_buf = this;
35 SerializeBuffer::Impl2* sbuf_p2 = sbuf->m_p2;
36 SerializeBuffer::Impl2* recv_p2 = recv_buf->m_p2;
37
38 Span<const Real> send_real = sbuf_p2->realBytes();
39 Span<const Int16> send_int16 = sbuf_p2->int16Bytes();
40 Span<const Int32> send_int32 = sbuf_p2->int32Bytes();
41 Span<const Int64> send_int64 = sbuf_p2->int64Bytes();
42 Span<const Byte> send_byte = sbuf_p2->byteBytes();
43
44 Int64 sizes[5];
45 sizes[0] = send_real.size();
46 sizes[1] = send_int16.size();
47 sizes[2] = send_int32.size();
48 sizes[3] = send_int64.size();
49 sizes[4] = send_byte.size();
50
51 ITraceMng* msg = pm->traceMng();
52 msg->info(4) << "SBUF_GATHER SIZE real=" << sizes[0] << " int16=" << sizes[1]
53 << " int32=" << sizes[2] << " int64=" << sizes[3]
54 << " bytes=" << sizes[4];
55
56 pm->reduce(Parallel::ReduceSum,Int64ArrayView(5,sizes));
57
58 Int64 recv_nb_real = sizes[0];
59 Int64 recv_nb_int16 = sizes[1];
60 Int64 recv_nb_int32 = sizes[2];
61 Int64 recv_nb_int64 = sizes[3];
62 Int64 recv_nb_byte = sizes[4];
63
64 recv_p2->allocateBuffer(recv_nb_real,recv_nb_int16,recv_nb_int32,recv_nb_int64,recv_nb_byte);
65 auto recv_p = recv_buf->_p();
66 {
67 RealUniqueArray real_buf;
68 pm->allGatherVariable(send_real.smallView(),real_buf);
69 recv_p->getRealBuffer().copy(real_buf);
70 }
71
72 {

74 pm->allGatherVariable(send_int32.smallView(),int32_buf);
75 recv_p->getInt32Buffer().copy(int32_buf);
76 }
77
78 {
79 Int16UniqueArray int16_buf;
80 pm->allGatherVariable(send_int16.smallView(),int16_buf);
81 recv_p->getInt16Buffer().copy(int16_buf);

85 Int64UniqueArray int64_buf;

95}
