105void AllEnvData::
106forceRecompute(bool compute_all)
107{
108 m_material_mng->incrementTimestamp();
109
110 IMesh* mesh = m_material_mng->mesh();
111 IItemFamily* cell_family = mesh->cellFamily();
112 ItemGroup all_cells = cell_family->allItems();
113 ConstArrayView<MeshMaterialVariableIndexer*> vars_idx = m_material_mng->variablesIndexer();
114 Integer nb_var = vars_idx.size();
115 info(4) << "ForceRecompute NB_VAR_IDX=" << nb_var
116 << " compute_all?=" << compute_all;
117
118 //TODO: Utiliser celui de MeshMaterialMng
119 bool m_is_verbose = false;
120
121 // Il faut compter le nombre total de mailles par milieu et par matériau
122
123 ConstArrayView<MeshEnvironment*> true_environments(m_material_mng->trueEnvironments());
124
125 if (compute_all)
126 _computeNbEnvAndNbMatPerCell();
127
128 // Calcul le nombre de milieux par maille, et pour chaque
129 // milieu le nombre de matériaux par maille
130 Integer nb_env = true_environments.size();
131 Integer total_env_cell = 0;
132 Integer total_mat_cell = 0;
133 info(4) << "NB_ENV = " << nb_env;
134 for( Integer i=0; i<nb_env; ++i ){
135 MeshEnvironment* env = true_environments[i];
136 CellGroup cells = env->cells();
137 Integer env_nb_cell = cells.size();
138 info(4) << "NB_CELL=" << env_nb_cell << " env_name=" << cells.name();
139 total_env_cell += env_nb_cell;
140 total_mat_cell += env->totalNbCellMat();
141 }
142
143 // Il faut ajouter les infos pour les mailles de type AllEnvCell
144 Integer max_local_id = cell_family->maxLocalId();
145 info(4) << "TOTAL_ENV_CELL=" << total_env_cell
146 << " TOTAL_MAT_CELL=" << total_mat_cell;
147 //TODO:
148 // Le m_nb_mat_per_cell ne doit pas se faire sur les variablesIndexer().
149 // Il doit prendre etre different suivant les milieux et les materiaux.
150 // - Si un milieu est le seul dans la maille, il prend la valeur globale.
151 // - Si un materiau est le seul dans la maille, il prend la valeur
152 // de la maille milieu correspondante (globale ou partielle suivant le cas)
153
154 // Redimensionne les tableaux des infos
155 // ATTENTION: il ne doivent plus être redimensionnés par la suite sous peine
156 // de tout invalider.
157 m_all_env_items_internal.resize(max_local_id);
158 m_env_items_internal.resize(total_env_cell);
159 info(4) << "RESIZE all_env_items_internal size=" << max_local_id
160 << " total_env_cell=" << total_env_cell;
161
162 bool is_full_verbose = traceMng()->verbosityLevel()>=5;
163
164 if (compute_all){
165 for( Integer i=0; i<nb_env; ++i ){
166 MeshMaterialVariableIndexer* var_indexer = true_environments[i]->variableIndexer();
167 ComponentItemListBuilder list_builder(var_indexer,0);
168 CellGroup cells = var_indexer->cells();
169 Integer var_nb_cell = cells.size();
170 info(4) << "ENV_INDEXER (V2) i=" << i << " NB_CELL=" << var_nb_cell << " name=" << cells.name()
171 << " index=" << var_indexer->index();
172 if (is_full_verbose){
173 Int32UniqueArray my_array(cells.view().localIds());
174 info(5) << "ENV_INDEXER (V2) name=" << cells.name() << " cells=" << my_array;
175 }
176 ENUMERATE_CELL(icell,var_indexer->cells()){
177 if (m_nb_env_per_cell[icell]>1)
178 list_builder.addPartialItem(icell.itemLocalId());
179 else
180 // Je suis le seul milieu de la maille donc je prend l'indice global
181 list_builder.addPureItem(icell.itemLocalId());
182 }
183 if (is_full_verbose)
184 info() << "MAT_NB_MULTIPLE_CELL (V2) mat=" << var_indexer->name()
185 << " nb_in_global=" << list_builder.pureMatVarIndexes().size()
186 << " (ids=" << list_builder.pureMatVarIndexes() << ")"
187 << " nb_in_multiple=" << list_builder.partialMatVarIndexes().size()
188 << " (ids=" << list_builder.partialLocalIds() << ")";
189 var_indexer->endUpdate(list_builder);
190 }
191
192 for( Integer i=0; i<nb_env; ++i ){
193 true_environments[i]->computeItemListForMaterials(m_nb_env_per_cell);
194 }
195 }
196
197 for( Integer i=0; i<nb_env; ++i ){
198 MeshMaterialVariableIndexer* var_indexer = true_environments[i]->variableIndexer();
199 CellGroup cells = var_indexer->cells();
200 Integer var_nb_cell = cells.size();
201 info(4) << "FINAL_INDEXER i=" << i << " NB_CELL=" << var_nb_cell << " name=" << cells.name()
202 << " index=" << var_indexer->index();
203 if (is_full_verbose){
204 Int32UniqueArray my_array(cells.view().localIds());
205 info(5) << "FINAL_INDEXER (V2) name=" << cells.name() << " cells=" << my_array;
206 info(4) << "FINAL_MAT_NB_MULTIPLE_CELL (V2) mat=" << var_indexer->name()
207 << " ids=" << var_indexer->matvarIndexes();
208 }
209 }
210
211 // Initialise à des valeurs invalides pour détecter les
212 // erreurs.
213 {
214 for( Integer i=0; i<max_local_id; ++i ){
215 ComponentItemInternal& ref_ii = m_all_env_items_internal[i];
216 ref_ii.reset();
217 }
218 for( Integer i=0; i<total_env_cell; ++i ){
219 ComponentItemInternal& ref_ii = m_env_items_internal[i];
220 ref_ii.reset();
221 }
222 }
223
224
225 // Calcule pour chaque maille sa position dans le tableau des milieux
226 // en considérant que les milieux de chaque maille sont rangés consécutivement
227 // dans m_env_items_internal.
228 Int32UniqueArray env_cell_indexes(max_local_id);
229 {
230 Integer env_cell_index = 0;
231 ENUMERATE_CELL(icell,all_cells){
232 Int32 lid = icell.itemLocalId();
233 Int32 nb_env = m_nb_env_per_cell[icell];
234 env_cell_indexes[lid] = env_cell_index;
235 env_cell_index += nb_env;
236 }
237 }
238
239 // Positionne les infos pour les EnvCell
240 {
241 Int32UniqueArray current_pos(env_cell_indexes);
242 ItemInternalList items_internal(cell_family->itemsInternal());
243 for( Integer i=0; i<nb_env; ++i ){
244 MeshEnvironment* env = true_environments[i];
245 MeshMaterialVariableIndexer* var_indexer = env->variableIndexer();
246 CellGroup cells = env->cells();
247 Int32ConstArrayView nb_mat_per_cell = env->m_nb_mat_per_cell.asArray();
248
249 env->resizeItemsInternal(var_indexer->nbItem());
250
251 info(4) << "COMPUTE (V2) env_cells env=" << env->name() << " nb_cell=" << cells.size()
252 << " index=" << var_indexer->index()
253 << " max_multiple_index=" << var_indexer->maxIndexInMultipleArray();
254
255 ConstArrayView<MatVarIndex> matvar_indexes = var_indexer->matvarIndexes();
256
257 ArrayView<ComponentItemInternal*> env_items_internal = env->itemsInternalView();
258 Int32ConstArrayView local_ids = var_indexer->localIds();
259
260 for( Integer z=0, nb_id = matvar_indexes.size(); z<nb_id; ++z){
261 MatVarIndex mvi = matvar_indexes[z];
262
263 Int32 lid = local_ids[z];
264 Int32 pos = current_pos[lid];
265 ++current_pos[lid];
266 Int32 nb_mat = nb_mat_per_cell[lid];
267 ComponentItemInternal& ref_ii = m_env_items_internal[pos];
268 env_items_internal[z] = &m_env_items_internal[pos];
269 ref_ii.setSuperAndGlobalItem(&m_all_env_items_internal[lid],items_internal[lid]);
270 ref_ii.setNbSubItem(nb_mat);
271 ref_ii.setVariableIndex(mvi);
272 ref_ii.setLevel(LEVEL_ENVIRONMENT);
273 }
274 }
275 for( Integer i=0; i<nb_env; ++i ){
276 MeshEnvironment* env = true_environments[i];
277 env->computeMaterialIndexes();
278 }
279 }
280
281 // Positionne les infos pour les AllEnvCell.
282 {
283 ENUMERATE_CELL(icell,all_cells){
284 Cell c = *icell;
285 Int32 lid = icell.itemLocalId();
286 Int32 nb_env = m_nb_env_per_cell[icell];
287 ComponentItemInternal& ref_ii = m_all_env_items_internal[lid];
288 ref_ii.setSuperAndGlobalItem(0,c.internal());
289 ref_ii.setVariableIndex(MatVarIndex(0,lid));
290 ref_ii.setNbSubItem(nb_env);
291 ref_ii.setLevel(LEVEL_ALLENVIRONMENT);
292 if (nb_env!=0)
293 ref_ii.setFirstSubItem(&m_env_items_internal[env_cell_indexes[lid]]);
294 }
295 }
296
297
298 if (m_is_verbose){
299 printAllEnvCells(all_cells.view().localIds());
300 }
301
302 if (m_is_verbose){
303 for( IMeshMaterial* material : m_material_mng->materials().range() ){
304 //Integer nb_mat = m_materials.size();
305 //for( Integer i=0; i<nb_mat; ++i ){
306 //IMeshMaterial* material = m_materials[i];
307 info() << "MAT name=" << material->name();
308 ENUMERATE_COMPONENTITEM(MatCell,imatcell,material){
309 MatCell pmc = *imatcell;
310 info() << "CELL IN MAT vindex=" << pmc._varIndex();
311 }
312 }
313 }
314
315 for( MeshEnvironment* env : true_environments.range() ){
316 env->componentData()->rebuildPartData();
317 for( MeshMaterial* mat : env->trueMaterials().range() )
318 mat->componentData()->rebuildPartData();
319 }
320
321 m_material_mng->checkValid();
322
323 m_material_mng->syncVariablesReferences();
324
325 if (m_is_verbose){
326 OStringStream ostr;
327 m_material_mng->dumpInfos2(ostr());
328 info() << ostr.str();
329 }
330}

374 UniqueArray<Int32> pure_local_ids;
375 UniqueArray<Int32> partial_indexes;
376

399 partial_indexes,operation,false,is_verbose);

401 info(4) << "NB_MAT_TRANSFORM=" << pure_local_ids.size()
402 << " name=" << mat->name();
403
404 _copyBetweenPartialsAndGlobals(pure_local_ids,partial_indexes,
405 indexer->index(),operation);

503Integer AllEnvData::
504_checkMaterialPrescence(IMeshMaterial* mat,Int32ConstArrayView ids,eOperation operation)
505{
506 // TODO: faut-il vérifier la validité des \a ids
507 //(ils sont compris entre 0 et max_loca_id-1) ?
508
509 MeshMaterialVariableIndexer* indexer = mat->variableIndexer();
510 IItemFamily* item_family = mat->cells().itemFamily();
511 ItemInternalList items_internal = item_family->itemsInternal();
512 Integer max_local_id = item_family->maxLocalId();
513 UniqueArray<bool> presence_flags(max_local_id,false);
514 Int32ConstArrayView mat_local_ids = indexer->localIds();
515 Integer nb_error = 0;
516 String name = mat->name();
517
518 for( Integer i=0, n=ids.size(); i<n; ++i ){
519 Int32 lid = ids[i];
520 if (presence_flags[lid]){
521 info() << "ERROR: item " << ItemPrinter(items_internal[lid])
522 << " is present several times in add/remove list for material mat=" << name;
523 ++nb_error;
524 }
525 presence_flags[lid] = true;
526 }
527
528 if (operation==eOperation::Add){
529 for( Integer i=0, n=mat_local_ids.size(); i<n; ++i ){
530 Int32 lid = mat_local_ids[i];
531 if (presence_flags[lid]){
532 info() << "ERROR: item " << ItemPrinter(items_internal[lid])
533 << " is already in material mat=" << name;
534 ++nb_error;
535 }
536 }
537 }
538 else if (operation==eOperation::Remove){
539 for( Integer i=0, n=mat_local_ids.size(); i<n; ++i ){
540 Int32 lid = mat_local_ids[i];
541 presence_flags[lid]= false;
542 }
543
544 for( Integer i=0, n=ids.size(); i<n; ++i ){
545 Int32 lid = ids[i];
546 if (presence_flags[lid]){
547 info() << "ERROR: item " << ItemPrinter(items_internal[lid])
548 << " is not in material mat=" << name;
549 ++nb_error;
550 }
551 }
552 }
553 else
554 _throwBadOperation(operation);
555
556 return nb_error;
557}

664void AllEnvData::
665_updateMaterialDirect(IMeshMaterial* mat,Int32ConstArrayView ids,eOperation operation)
666{
667 if (operation!=eOperation::Add && operation!=eOperation::Remove)
668 _throwBadOperation(operation);
669 bool is_add = (operation==eOperation::Add);
670
671 MeshMaterial* true_mat = dynamic_cast<MeshMaterial*>(mat);
672 if (!true_mat)
673 throw NotImplementedException(A_FUNCINFO,"material is not an instance of MeshMaterial");
674
675 info(4) << "Using optimisation updateMaterialDirect operation=" << operation;
676
677 IMeshEnvironment* env = mat->environment();
678 MeshEnvironment* true_env = true_mat->trueEnvironment();
679 Integer nb_mat = env->nbMaterial();
680
681 Int32UniqueArray cells_changed_in_env;
682
683 if (nb_mat!=1){
684
685 // S'il est possible d'avoir plusieurs matériaux par milieu, il faut gérer
686 // pour chaque maille si le milieu évolue suite à l'ajout/suppression de matériau.
687 // les deux cas sont:
688 // - en cas d'ajout, le milieu évolue pour une maille s'il n'y avait pas
689 // de matériau avant. Dans ce cas le milieu est ajouté à la maille.
690 // - en cas de suppression, le milieu évolue dans la maille s'il y avait
691 // 1 seul matériau avant. Dans ce cas le milieu est supprimé de la maille.
692
693 Int32UniqueArray cells_unchanged_in_env;
694 Int32ArrayView cells_nb_mat = true_env->m_nb_mat_per_cell.asArray();
695 const Int32 ref_nb_mat = (is_add) ? 0 : 1;
696
697 info(4) << "Using optimisation updateMaterialDirect with multimat operation="
698 << operation;
699
700 for( Integer i=0, n=ids.size(); i<n; ++i ){
701 Int32 lid = ids[i];
702 if (cells_nb_mat[lid]!=ref_nb_mat){
703 info(5)<< "CELL i=" << i << " lid="<< lid << " unchanged in environment nb_mat=" << cells_nb_mat[lid];
704 cells_unchanged_in_env.add(lid);
705 }
706 else{
707 cells_changed_in_env.add(lid);
708 }
709 }
710
711 Integer nb_unchanged_in_env = cells_unchanged_in_env.size();
712 info(4) << "Cells unchanged in environment n=" << nb_unchanged_in_env;
713
714 if (is_add){
715 mat->cells().addItems(cells_unchanged_in_env);
716 }
717 else{
718 mat->cells().removeItems(cells_unchanged_in_env);
719 }
720 true_env->updateItemsDirect(m_nb_env_per_cell,true_mat,cells_unchanged_in_env,operation,false);
721
722 ids = cells_changed_in_env.view();
723 }
724
725 Int32ArrayView cells_nb_env = m_nb_env_per_cell.asArray();
726
727 // Met à jour le nombre de milieux de chaque maille.
728 if (is_add){
729 for( Integer i=0, n=ids.size(); i<n; ++i ){
730 ++cells_nb_env[ids[i]];
731 }
732 }
733 else{
734 for( Integer i=0, n=ids.size(); i<n; ++i ){
735 --cells_nb_env[ids[i]];
736 }
737 }
738
739 // Comme on a ajouté/supprimé des mailles matériau dans le milieu,
740 // il faut transformer les mailles pures en mailles partielles (en cas
741 // d'ajout) ou les mailles partielles en mailles pures (en cas de
742 // suppression).
743 info(4) << "Transform PartialPure for material name=" << true_mat->name();
744 _switchComponentItemsForMaterials(true_mat,operation);
745 info(4) << "Transform PartialPure for environemnt name=" << env->name();
746 _switchComponentItemsForEnvironments(env,operation);
747
748 // Si je suis mono-mat, alors mat->cells()<=>env->cells() et il ne faut
749 // mettre à jour que l'un des deux groupes.
750 bool need_update_env = (nb_mat!=1);
751
752 if (is_add){
753 mat->cells().addItems(ids);
754 if (need_update_env)
755 env->cells().addItems(ids);
756 }
757 else{
758 mat->cells().removeItems(ids);
759 if (need_update_env)
760 env->cells().removeItems(ids);
761 }
762 true_env->updateItemsDirect(m_nb_env_per_cell,true_mat,ids,operation,need_update_env);
763}
