169void MeshMaterialMng::
170build()
171{
172 {
173 auto* x = MeshMaterialVariableFactoryRegisterer::firstRegisterer();
174 while (x){
175 m_variable_factory_mng->registerFactory(x->createFactory());
176 x = x->nextRegisterer();
177 }
178 }
179
180 // Choix des optimisations.
181 {
182 int default_flags = 0;
183
184 //GG: Décommenter pour activer par défaut les optimisations.
185 //default_flags = (int)eModificationFlags::GenericOptimize | (int)eModificationFlags::OptimizeMultiAddRemove;
186
187 int opt_flag_value = 0;
188 String env_name = "ARCANE_MATERIAL_MODIFICATION_FLAGS";
189 String opt_flag_str = platform::getEnvironmentVariable(env_name);
190 if (!opt_flag_str.null()){
191 if (builtInGetValue(opt_flag_value,opt_flag_str)){
192 pwarning() << "Invalid value '" << opt_flag_str
193 << " 'for environment variable '" << env_name
194 << "'";
195 opt_flag_value = default_flags;
196 }
197 }
198 else{
199 opt_flag_value = default_flags;
200 }
201 m_modification_flags = opt_flag_value;
202 }
203
204 // Choix de la version de l'implémentation des synchronisations
205 {
206 String env_name = "ARCANE_MATSYNCHRONIZE_VERSION";
207 String env_value = platform::getEnvironmentVariable(env_name);
208 info() << "ENV_VALUE=" << env_value;
209 Integer version = m_synchronize_variable_version;
210 if (!env_value.null()){
211 if (builtInGetValue(version,env_value)){
212 pwarning() << "Invalid value '" << env_value
213 << " 'for environment variable '" << env_name
214 << "'";
215 }
216 else
217 m_synchronize_variable_version = version;
218 }
219 info() << "Set material variable synchronize version to "
220 << "'" << m_synchronize_variable_version << "'";
221 }
222
223 m_exchange_mng->build();
224 // Si les traces des énumérateurs sur les entités sont actives, active celles
225 // sur les matériaux.
226 // TODO: rendre ce code thread-safe en cas d'utilisation de IParallelMng via les threads
227 // et ne l'appeler qu'une fois.
228 IItemEnumeratorTracer* item_tracer = IItemEnumeratorTracer::singleton();
229 if (item_tracer){
230 info() << "Adding material enumerator tracing";
231 EnumeratorTracer::_setSingleton(new EnumeratorTracer(item_tracer->perfCounter(),traceMng()));
232 }
233}

253MeshMaterial* MeshMaterialMng::
254_createMaterial(MeshEnvironment* env,MeshMaterialInfo* infos,const String& name)
255{
256 _checkEndCreate();
257 if (infos->materialMng()!=this)
258 ARCANE_FATAL("Invalid materialMng() for material info");
259 if (env->materialMng()!=this)
260 ARCANE_FATAL("Invalid materialMng() for environment");
261 Integer var_index = m_variables_indexer.size();
262 //const String& name = infos->name();
263 Int32 mat_id = (Int32)m_materials.size();
264 MeshMaterial* mat = new MeshMaterial(infos,env,name,mat_id);
265 info() << "Create material name=" << name << "mat_id=" << mat_id << " var_index=" << var_index;
266 mat->build();
267 m_materials.add(mat);
268 m_materials_as_components.add(mat);
269 m_true_materials.add(mat);
270
271 _addVariableIndexer(mat->variableIndexer());
272 return mat;
273}

301IMeshEnvironment* MeshMaterialMng::
302createEnvironment(const MeshEnvironmentBuildInfo& infos)
303{
304 _checkEndCreate();
305 Int32 env_index = (Int32)m_environments.size();
306 // Vérifie qu'un milieu de même nom n'existe pas.
307 const String& env_name = infos.name();
308 MeshEnvironment* old_me = _findEnvironment(env_name);
309 if (old_me)
310 ARCANE_FATAL("Un milieu de nom '{0}' est déjà enregistré",env_name);
311
312 info() << "Creating environment name=" << env_name << " index=" << env_index;
313 // Créé le milieu
314 MeshEnvironment* me = new MeshEnvironment(this,env_name,env_index);
315 me->build();
316 m_true_environments.add(me);
317 m_environments.add(me);
318 m_environments_as_components.add(me);
319
320 // Créé et ajoute les matériaux
321 Integer nb_mat = infos.materials().size();
322 for( Integer i=0; i<nb_mat; ++i ){
323 const MeshEnvironmentBuildInfo::MatInfo& buildinfo = infos.materials()[i];
324 const String& mat_name = buildinfo.m_name;
325 String new_mat_name = env_name + "_" + mat_name;
326 MeshMaterialInfo* mat_info = _findMaterialInfo(mat_name);
327 if (!mat_info){
328 ARCANE_FATAL("Aucun matériau de nom '{0}' n'est défini",mat_name);
329 }
330 MeshMaterial* mm = _createMaterial(me,mat_info,new_mat_name);
331 const String& mat_var_name = buildinfo.m_var_name;
332 if (!mat_var_name.null()){
333 info() << "Adding material '" << mat_name << "' with variable name " << mat_var_name
334 << " to environment " << env_name;
335 mm->variableIndexer()->setWantedVariableName(mat_var_name);
336 }
337 me->addMaterial(mm);
338 mat_info->_addEnvironment(env_name);
339 }
340 // Si le milieu contient plusieurs matériaux, il faut lui allouer
341 // des valeurs partielles. Sinon, ses valeurs partielles sont celles
342 // de son unique matériau.
343 // EXCEPTION: en mode compatible, si m_is_use_global_cell_for_envcell
344 // est vrai alors on duplique toujours le milieu même s'il ne
345 // contient qu'un seul matériau.
346 {
347 MeshMaterialVariableIndexer* var_idx = 0;
348 bool duplicate_unique_env_cell = false;
349 if (nb_mat==1 && !duplicate_unique_env_cell){
350 var_idx = me->materials()[0]->variableIndexer();
351 }
352 else{
353 var_idx = new MeshMaterialVariableIndexer(traceMng(),me->name());
354 _addVariableIndexer(var_idx);
355 m_variables_indexer_to_destroy.add(var_idx);
356 }
357 me->setVariableIndexer(var_idx);
358 }
359 return me;
360}

365IMeshBlock* MeshMaterialMng::
366createBlock(const MeshBlockBuildInfo& infos)
367{
368 _checkEndCreate();
369
370 Int32 block_index = (Int32)m_blocks.size();
371 // Vérifie qu'un bloc de même nom n'existe pas.
372 const String& name = infos.name();
373 MeshBlock* old_mb = _findBlock(name);
374 if (old_mb)
375 ARCANE_FATAL("Un bloc de nom '{0}' est déjà enregistré",name);
376
377 info() << "Creating block name=" << name << " index=" << block_index
378 << " nb_env=" << infos.environments().size();
379 Integer nb_env = infos.environments().size();
380 for( Integer i=0; i<nb_env; ++i )
381 info() << " Adding environment name=" << infos.environments()[i]->name() << " to block";
382
383 // Créé le bloc
384 MeshBlock* mb = new MeshBlock(this,block_index,infos);
385 mb->build();
386 m_true_blocks.add(mb);
387 m_blocks.add(mb);
388
389 return mb;
390}

415void MeshMaterialMng::
416endCreate(bool is_continue)
417{
418 if (m_is_end_create)
419 return;
420 _saveInfosInProperties();
421 info() << "This version uses MatVarIndex for Indexer";
422 info() << "NOTE: this version use one loop all enumeration and remove support for two loop";
423 info() << "END CREATE MATERIAL_MNG is_continue=" << is_continue << " this=" << this;
424 m_modifier->initOptimizationFlags();
425 auto synchronizer = mesh()->cellFamily()->allItemsSynchronizer();
426 m_all_cells_mat_env_synchronizer =
427 new MeshMaterialVariableSynchronizer(this,synchronizer,MatVarSpace::MaterialAndEnvironment);
428 m_all_cells_env_only_synchronizer =
429 new MeshMaterialVariableSynchronizer(this,synchronizer,MatVarSpace::Environment);
430
431 // Détermine la liste de tous les composants.
432 {
433 Integer nb_component = m_environments_as_components.size() + m_materials_as_components.size();
434 m_components.reserve(nb_component);
435 m_components.addRange(m_environments_as_components);
436 m_components.addRange(m_materials_as_components);
437 }
438 // Il faut construire et initialiser les variables qui ont été
439 // créées avant cette allocation.
440 for( auto i : m_full_name_variable_map ){
441 IMeshMaterialVariable* mv = i.second;
442 info() << "BUILD FROM MANAGER name=" << mv->name() << " this=" << this;
443 mv->buildFromManager(is_continue);
444 }
445 if (is_continue)
446 _endUpdate();
447 m_is_end_create = true;
448
449 // Vérifie que les milieux sont valides.
450 // NOTE: on ne peut pas toujours appeler checkValid()
451 // (notamment au démarrage) car les groupes d'entités existent
452 // mais les infos matériaux associées ne sont pas
453 // forcément encore créés.
454 // (Il faudra regarder une si cela est du au mode compatible ou pas).
455 for( IMeshEnvironment* env : m_environments ){
456 env->checkValid();
457 }
458
459 // Maintenant que tout est créé il est valide d'enregistrer les mécanismes
460 // d'échange.
461 m_exchange_mng->registerFactory();
462}

615
616void MeshMaterialMng::
617checkValid()
618{
619 IItemFamily* cell_family = mesh()->cellFamily();
620 ItemGroup all_cells = cell_family->allItems();
621 const VariableCellInt32 nb_env_per_cell = m_all_env_data->nbEnvPerCell();
622 ENUMERATE_ALLENVCELL(iallenvcell,view(all_cells.view().localIds())){
623 AllEnvCell all_env_cell = *iallenvcell;
624 Integer cell_nb_env = all_env_cell.nbEnvironment();
625 Cell cell = all_env_cell.globalCell();
626 Int64 cell_uid = cell.uniqueId();
627 if (all_env_cell.level()!=LEVEL_ALLENVIRONMENT)
628 ARCANE_FATAL("Bad level for all_env_item");
629
630 if (all_env_cell.globalCell().internal()!=cell.internal())
631 ARCANE_FATAL("Bad corresponding globalCell() in all_env_item");
632 if (cell_nb_env!=nb_env_per_cell[cell])
633 ARCANE_FATAL("Bad value for nb_env direct='{0}' var='{1}'",
634 cell_nb_env,nb_env_per_cell[cell]);
635
636 for( Integer z=0; z<cell_nb_env; ++z ){
637 EnvCell ec = all_env_cell.cell(z);
638 Integer cell_nb_mat = ec.nbMaterial();
639 ComponentItemInternal* eii = ec.internal();
640 if (all_env_cell.internal()!=eii->superItem())
641 ARCANE_FATAL("Bad corresponding allEnvItem() in env_item uid={0}",cell_uid);
642 if (eii->globalItem()!=cell.internal())
643 ARCANE_FATAL("Bad corresponding globalItem() in env_item");
644 if (eii->level()!=LEVEL_ENVIRONMENT)
645 ARCANE_FATAL("Bad level '{0}' for in env_item",eii->level());
646 // Si la maille n'est pas pure, la variable milieu ne peut être équivalente à
647 // la variable globale.
648 if (cell_nb_env>1 && ec._varIndex().arrayIndex()==0)
649 ARCANE_FATAL("Global index for a partial cell env_item={0}",ec);
650
651 for( Integer k=0; k<cell_nb_mat; ++k ){
652 MatCell mc = ec.cell(k);
653 ComponentItemInternal* mci = mc.internal();
654 if (eii!=mci->superItem())
655 ARCANE_FATAL("Bad corresponding env_item in mat_item");
656 if (mci->globalItem()!=cell.internal())
657 ARCANE_FATAL("Bad corresponding globalItem() in mat_item");
658 if (mci->level()!=LEVEL_MATERIAL)
659 ARCANE_FATAL("Bad level '{0}' for in mat_item",mci->level());
660 // Si la maille n'est pas pure, la variable matériau ne peut être équivalente à
661 // la variable globale.
662 if ((cell_nb_env>1 || cell_nb_mat>1) && mc._varIndex().arrayIndex()==0){
663 ARCANE_FATAL("Global index for a partial cell matitem={0} name={1} nb_mat={2} nb_env={3}",
664 mc,mc.material()->name(),cell_nb_mat,cell_nb_env);
665 }
666 }
667 }
668 }
669
670 for( IMeshEnvironment* env : m_environments ){
671 env->checkValid();
672 }
673}

1082
1083void MeshMaterialMng::
1084recreateFromDump()
1085{
1086 if (m_is_end_create)
1087 ARCANE_FATAL("Can not recreate a created instance");
1088
1089 _checkCreateProperties();
1090
1091 info() << "Creating material infos from dump";
1092
1093 // Sauve le numéro de version pour être sur que c'est OK en reprise
1094 Int32 v = m_properties->getInt32("Version");
1095 if (v!=SERIALIZE_VERSION)
1096 ARCANE_FATAL("Bad serializer version: trying to read from incompatible checkpoint v={0} expected={1}",
1097 v,SERIALIZE_VERSION);
1098
1099 UniqueArray<String> material_info_names;
1100 m_properties->get("MaterialInfoNames",material_info_names);
1101 for( Integer i=0, n=material_info_names.size(); i<n; ++i )
1102 this->registerMaterialInfo(material_info_names[i]);
1103
1104 UniqueArray<String> env_names;
1105 UniqueArray<Int32> env_nb_mat;
1106 UniqueArray<String> env_mat_names;
1107 m_properties->get("EnvNames",env_names);
1108 m_properties->get("EnvNbMat",env_nb_mat);
1109 m_properties->get("EnvMatNames",env_mat_names);
1110
1111 Integer mat_index = 0;
1112 for( Integer ienv=0, nenv=env_names.size(); ienv<nenv; ++ienv ){
1113 Materials::MeshEnvironmentBuildInfo env_build(env_names[ienv]);
1114 Integer nb_mat = env_nb_mat[ienv];
1115 for( Integer imat=0; imat<nb_mat; ++imat ){
1116 env_build.addMaterial(env_mat_names[mat_index]);
1117 ++mat_index;
1118 }
1119 this->createEnvironment(env_build);
1120 }
1121
1122 // Recréé les blocs.
1123 // Pour chaque bloc, son nom et le nom du groupe de maille correspondant.
1124 UniqueArray<String> block_names;
1125 UniqueArray<String> block_cell_group_names;
1126 UniqueArray<String> block_env_names;
1127 UniqueArray<Int32> block_nb_env;
1128 m_properties->get("BlockNames",block_names);
1129 m_properties->get("BlockCellGroupNames",block_cell_group_names);
1130 m_properties->get("BlockNbEnv",block_nb_env);
1131 m_properties->get("BlockEnvNames",block_env_names);
1132 IItemFamily* cell_family = mesh()->cellFamily();
1133 Integer block_env_index = 0;
1134 for( Integer i=0, n=block_names.size(); i<n; ++i ){
1135 String name = block_names[i];
1136 String cell_group_name = block_cell_group_names[i];
1137 CellGroup cells = cell_family->findGroup(cell_group_name);
1138 if (cells.null())
1139 ARCANE_FATAL("Can not find cell group '{0}' for block creation",
1140 cell_group_name);
1141 MeshBlockBuildInfo mbbi(name,cells);
1142 if (block_nb_env.size()!=0){
1143 Integer nb_env = block_nb_env[i];
1144 for( Integer ienv=0; ienv<nb_env; ++ienv ){
1145 const String& name2 = block_env_names[block_env_index];
1146 ++block_env_index;
1147 IMeshEnvironment* env = findEnvironment(name2,false);
1148 if (!env)
1149 ARCANE_FATAL("Invalid environment name '{0}' for recreating blocks",name2);
1150 mbbi.addEnvironment(env);
1151 }
1152 }
1153 this->createBlock(mbbi);
1154 }
1155
1156 endCreate(true);
1157}
