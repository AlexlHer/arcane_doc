82template <int N>
83class ComplexLoopRanges
84{
85 public:
86 typedef typename ArrayBounds<N>::IndexType IndexType;
87 public:
88 ComplexLoopRanges(ArrayBounds<N> lower,ArrayBounds<N> extents)
89 : m_lower_bounds(lower.asStdArray()), m_extents(extents){}
90 ComplexLoopRanges(const SimpleLoopRanges<N>& bounds)
91 : m_extents(bounds.m_bounds){}
92 public:
93 constexpr Int32 lowerBound(int i) const { return m_lower_bounds[i]; }
94 constexpr Int32 upperBound(int i) const { return m_lower_bounds[i]+m_extents.extent(i); }
95 constexpr Int32 extent(int i) const { return m_extents.extent(i); }
96 constexpr Int64 nbElement() const { return m_extents.nbElement(); }
97 constexpr ArrayBoundsIndex<N> getIndices(Int32 i) const
98 {
99 auto x = m_extents.getIndices(i);
100 x.add(m_lower_bounds);
101 return x;
102 }
103 private:
104 ArrayBoundsIndex<N> m_lower_bounds;
105 ArrayBounds<N> m_extents;
106};
