171TEST(Span,Convert)
172{
173 using namespace Arccore;
174 std::vector<Int64> vector_values = { 5, 7, 11 };
175 Int32 vector_size = static_cast<Int32>(vector_values.size());
176 ArrayView<Int64> a_view(vector_size,vector_values.data());
177 ASSERT_EQ(a_view.size(),vector_size) << "Bad a_view size";
178 ASSERT_EQ(a_view[0],vector_values[0]) << "Bad a_view[0]";
179 ASSERT_EQ(a_view[1],vector_values[1]) << "Bad a_view[1]";
180 ASSERT_EQ(a_view[2],vector_values[2]) << "Bad a_view[2]";
181
182 ConstArrayView<Int64> a_const_view(vector_size,vector_values.data());
183 ASSERT_EQ(a_const_view.size(),vector_size) << "Bad a_const_view size";
184 ASSERT_EQ(a_const_view[0],vector_values[0]) << "Bad a_const_view[0]";
185 ASSERT_EQ(a_const_view[1],vector_values[1]) << "Bad a_const_view[1]";
186 ASSERT_EQ(a_const_view[2],vector_values[2]) << "Bad a_const_view[2]";
187
188 Span<const Int64> a_const_span(vector_values.data(),vector_values.size());
189 Span<Int64> a_span(vector_values.data(),vector_values.size());
190 ByteConstSpan a_const_bytes = asBytes(a_const_span);
191 ByteSpan a_bytes = asWritableBytes(a_span);
192 ASSERT_EQ(a_const_bytes.size(),24) << "Bad a_const_bytes_size (1)";
193 ASSERT_EQ(a_bytes.size(),24) << "Bad a_bytes_size (2)";
194 Span<Int64> span2(a_view);
195 Span<const Int64> span3(a_view);
196 Span<const Int64> span4(a_const_view);
197 ASSERT_EQ(span2.size(),a_view.size()) << "Bad span2 size";
198 ASSERT_EQ(span3.size(),a_view.size()) << "Bad span3 size";
199 ASSERT_EQ(span4.size(),a_const_view.size()) << "Bad span4 size";
200 span3 = a_const_view;
201 span3 = a_view;
202 span2 = a_view;
203 ASSERT_EQ(span2.size(),a_view.size()) << "Bad span2 (2) size";
204 ASSERT_EQ(span3.size(),a_view.size()) << "Bad span3 (2) size";
205 std::cout << "View=" << a_view << '\n';
206 std::cout << "ConstView=" << a_const_view << '\n';
207 std::cout << "Span3=" << span3 << '\n';
208}

213// Vérifie que \a a1 et \a a2 sont identiques
214template<typename A1,typename A2>
215void _checkSame(A1& a1,A2& a2,const char* message)
216{
217 using namespace Arccore;
218 using size_type = typename A1::size_type;
219 Int64 s1 = a1.size();
220 Int64 s2 = a2.size();
221 ASSERT_EQ(s1,s2) << "Bad size " << message;
222 for( size_type i=0, n=a1.size(); i<n; ++i )
223 ASSERT_EQ(a1[i],a2[i]) << "Bad value[" << i << "]" << message;
224}

229// Vérifie que \a a1 et \a a2 sont identiques
230template<typename A1,typename A2>
231void _checkSame2(A1& a1,A2& a2,const char* message)
232{
233 using namespace Arccore;
234 using size_type = typename A1::size_type;
235 const Int64 s1_dim1 = a1.dim1Size();
236 const Int64 s2_dim1 = a2.dim1Size();
237 ASSERT_EQ(s1_dim1,s2_dim1) << "Bad size " << message;
238 const Int64 s1_dim2 = a1.dim2Size();
239 const Int64 s2_dim2 = a2.dim2Size();
240 ASSERT_EQ(s1_dim2,s2_dim2) << "Bad size " << message;
241 for( size_type i=0; i<s1_dim1; ++i )
242 for( size_type j=0; j<s1_dim2; ++j )
243 ASSERT_EQ(a1[i][j],a2[i][j]) << "Bad value[" << i << ',' << j << "]" << message;
244}
