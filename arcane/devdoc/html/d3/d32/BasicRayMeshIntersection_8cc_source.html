137void RayTriangle3DIntersection::
138compute()
139{
140 Integer nb_ray = m_rays_origin.size();
141 m_distances.resize(nb_ray);
142 m_distances.fill(1.0e100);
143 m_intersected_triangle_indexes.resize(nb_ray);
144 m_intersected_triangle_indexes.fill(-1);
145 Integer nb_index = m_triangles_indexes.size();
146 if ((nb_index%3)!=0)
147 throw FatalErrorException(A_FUNCINFO,"bad triangle_index count (({0}) % 3)!=0");
148 Integer nb_triangle = nb_index / 3;
149 info() << "COMPUTE RAY INTERSECTION nb_ray=" << nb_ray
150 << " nb_triangle=" << nb_triangle;
151
152 for( Integer i=0; i<nb_triangle; ++i ){
153 Real3 p0 = m_triangles_coordinates[m_triangles_indexes[(i*3)]];
154 Real3 p1 = m_triangles_coordinates[m_triangles_indexes[(i*3)+1]];
155 Real3 p2 = m_triangles_coordinates[m_triangles_indexes[(i*3)+2]];
156 _compute2(i,p0,p1,p2);
157 }
158
159 for( Integer i=0; i<nb_ray; ++i ){
160 Int32 tid = m_intersected_triangle_indexes[i];
161 if (tid==(-1))
162 m_distances[i] = -1.0;
163 }
164}

169void RayTriangle3DIntersection::
170_compute2(Int32 triangle_id,Real3 p0,Real3 p1,Real3 p2)
171{
172 Integer nb_ray = m_rays_origin.size();
173 for( Integer i=0; i<nb_ray; ++i ){
174 Real t = checkIntersection(m_rays_origin[i],m_rays_direction[i],p0,p1,p2);
175 if (t>=0.0){
176 info() << "Segment " << i << " intersect triangle " << triangle_id << " T=" << t;
177 if (t<m_distances[i]){
178 m_distances[i] = t;
179 m_intersected_triangle_indexes[i] = triangle_id;
180 info() << "Get this triangle";
181 }
182 }
183 }

198Real RayTriangle3DIntersection::
199checkIntersection(Real3 origin,
200 Real3 direction,
201 Real3 p0,Real3 p1,Real3 p2)
202{
203 // Cette routine s'inspire du code de la bibliothèque WildMagic
204 // dont la licence est ci-dessous.
205
206 // Wild Magic Source Code
207 // David Eberly
208 // http://www.geometrictools.com
209 // Copyright (c) 1998-2009
210 //
211 // This library is free software; you can redistribute it and/or modify it
212 // under the terms of the GNU Lesser General Public License as published by
213 // the Free Software Foundation; either version 2.1 of the License, or (at
214 // your option) any later version. The license is available for reading at
215 // either of the locations:
216 // http://www.gnu.org/copyleft/lgpl.html
217 // http://www.geometrictools.com/License/WildMagicLicense.pdf
218 //
219
220
221 Real ZERO_TOLERANCE = 1.0e-14;
222 // compute the offset origin, edges, and normal
223 Real3 kDiff = origin - p0;
224 Real3 kEdge1 = p1 - p0;
225 Real3 kEdge2 = p2 - p0;
226 Real3 kNormal = math::vecMul(kEdge1,kEdge2);
227
228 Real fDdN = math::dot(direction,kNormal);
229 Real fSign;
230 if (fDdN > ZERO_TOLERANCE)
231 {
232 fSign = (Real)1.0;
233 }
234 else if (fDdN < -ZERO_TOLERANCE)
235 {
236 fSign = (Real)-1.0;
237 fDdN = -fDdN;
238 }
239 else
240 {
241 // Segment and triangle are parallel, call it a "no intersection"
242 // even if the segment does intersect.
243 return (-1.0);
244 }
245
246 Real fDdQxE2 = fSign*math::dot(direction,math::vecMul(kDiff,kEdge2));
247 if (fDdQxE2 >= (Real)0.0)
248 {
249 Real fDdE1xQ = fSign*math::dot(direction,math::vecMul(kEdge1,kDiff));
250 if (fDdE1xQ >= (Real)0.0)
251 {
252 Real diff = fDdN-(fDdQxE2 + fDdE1xQ);
253 // L'epsilon sert si le segment traverse par la face à proximité
254 // d'une des arêtes de la face. Dans ce cas, on considère qu'il
255 // y a intersection. Sans cela, à cause des arrondis numériques,
256 // un segment pourrait traverser le maillage en passant par une
257 // arête entre deux faces sans intersecter ces dernières.
258 if (diff>=0.0 || diff>-ZERO_TOLERANCE)
259 {
260 // line intersects triangle, check if segment does
261 Real fQdN = -fSign*math::dot(kDiff,kNormal);
262
263 // Comme il s'agit d'une demi-droite et pas d'un segment,
264 // il y a toujours intersection si \a t est positif.
265#if 0
266 Real fExtDdN = m_pkSegment->Extent*fDdN;
267 if (-fExtDdN <= fQdN && fQdN <= fExtDdN)
268 {
269 // segment intersects triangle
270 return true;
271 }
272#endif
273 Real fInv = ((Real)1.0)/fDdN;
274 Real t = fQdN*fInv;
275 //Real b1 = fDdQxE2*fInv;
276 //Real b2 = fDdE1xQ*fInv;
277 //Real b0 = (Real)1.0 - b1 - b2;
278 // P = origin + t*direction = b0*V0 + b1*V1 + b2*V2
279 //info() << "** INTERSECT POS=" << b0 << " y=" << b1 << " z=" << b2 << " T=" << t;
280 //Real intersect_p = origin + t*direction;
281 return t;
282
283 // else: |t| > extent, no intersection
284 }
285 // else: b1+b2 > 1, no intersection
286 }
287 // else: b2 < 0, no intersection
288 }
289 // else: b1 < 0, no intersection
290
291 return (-1.0);
292}

297bool RayTriangle3DIntersection::
298checkBoundingBox(Real3 origin,Real3 direction,Real3 box_min,Real3 box_max)
299{
300 Real3 box_center = (box_max+box_min) * 0.5;
301 Real afWdU[3], afAWdU[3], afDdU[3], afADdU[3], afAWxDdU[3], fRhs;
302
303 Real3 kDiff = origin - box_center;
304 Real3 Axis[3];
305 //Axis[0].x = box_max.x - box_min.x;
306 //Axis[1].y = box_max.y - box_min.y;
307 //Axis[2].z = box_max.z - box_min.z;
308 Axis[0] = Real3(1.0,0.0,0.0);
309 Axis[1] = Real3(0.0,1.0,0.0);
310 Axis[2] = Real3(0.0,0.0,1.0);
311 Real Extent[3];
312 Extent[0] = box_max.x - box_min.x;
313 Extent[1] = box_max.y - box_min.y;
314 Extent[2] = box_max.z - box_min.z;
315
316 afWdU[0] = math::dot(direction,Axis[0]);
317 afAWdU[0] = math::abs(afWdU[0]);
318 afDdU[0] = math::dot(kDiff,Axis[0]);
319 afADdU[0] = math::abs(afDdU[0]);
320 if (afADdU[0] > Extent[0] && afDdU[0]*afWdU[0] >= (Real)0.0)
321 {
322 return false;
323 }
324
325 afWdU[1] = math::dot(direction,Axis[1]);
326 afAWdU[1] = math::abs(afWdU[1]);
327 afDdU[1] = math::dot(kDiff,Axis[1]);
328 afADdU[1] = math::abs(afDdU[1]);
329 if (afADdU[1] > Extent[1] && afDdU[1]*afWdU[1] >= (Real)0.0)
330 {
331 return false;
332 }
333
334 afWdU[2] = math::dot(direction,Axis[2]);
335 afAWdU[2] = math::abs(afWdU[2]);
336 afDdU[2] = math::dot(kDiff,Axis[2]);
337 afADdU[2] = math::abs(afDdU[2]);
338 if (afADdU[2] > Extent[2] && afDdU[2]*afWdU[2] >= (Real)0.0)
339 {
340 return false;
341 }
342
343 Real3 kWxD = math::vecMul(direction,kDiff);
344
345 afAWxDdU[0] = math::abs(math::dot(kWxD,Axis[0]));
346 fRhs = Extent[1]*afAWdU[2] + Extent[2]*afAWdU[1];
347 if (afAWxDdU[0] > fRhs)
348 {
349 return false;
350 }
351
352 afAWxDdU[1] = math::abs(math::dot(kWxD,Axis[1]));
353 fRhs = Extent[0]*afAWdU[2] + Extent[2]*afAWdU[0];
354 if (afAWxDdU[1] > fRhs)
355 {
356 return false;
357 }
358
359 afAWxDdU[2] = math::abs(math::dot(kWxD,Axis[2]));
360 fRhs = Extent[0]*afAWdU[1] + Extent[1]*afAWdU[0];
361 if (afAWxDdU[2] > fRhs)
362 {
363 return false;
364 }
365
366 return true;
367}

372class BasicRayFaceIntersector
373: public IRayFaceIntersector
374{
375 public:
376 BasicRayFaceIntersector(ITraceMng* tm)
377 : m_triangle_intersector(tm)
378 {
379 }
380 public:
381 bool computeIntersection(Real3 origin,Real3 direction,
382 Int32 orig_face_local_id,
383 Int32 face_local_id,
384 Real3ConstArrayView face_nodes,
385 Int32* user_value,
386 Real* distance,Real3* position)
387 {
388 ARCANE_UNUSED(orig_face_local_id);
389 ARCANE_UNUSED(face_local_id);
390
391 Integer nb_node = face_nodes.size();
392 switch(nb_node){
393 case 4:
394 {
395 Real3 center = (face_nodes[0] + face_nodes[1] + face_nodes[2] + face_nodes[3]) / 4.0;
396 Real d0 = m_triangle_intersector.checkIntersection(origin,direction,center,face_nodes[0],face_nodes[1]);
397 Real d1 = m_triangle_intersector.checkIntersection(origin,direction,center,face_nodes[1],face_nodes[2]);
398 Real d2 = m_triangle_intersector.checkIntersection(origin,direction,center,face_nodes[2],face_nodes[3]);
399 Real d3 = m_triangle_intersector.checkIntersection(origin,direction,center,face_nodes[3],face_nodes[0]);
400 Real d = 1.0e100;
401 bool found = false;
402 if (d0>=0.0 && d0<d){
403 found = true;
404 d = d0;
405 }
406 if (d1>=0.0 && d1<d){
407 found = true;
408 d = d1;
409 }
410 if (d2>=0.0 && d2<d){
411 found = true;
412 d = d2;
413 }
414 if (d3>=0.0 && d3<d){
415 found = true;
416 d = d3;
417 }
418 if (!found)
419 d = -1.0;
420 *distance = d;
421 *position = origin + d * direction;
422 *user_value = 1;
423 return found;
424 }
425 break;
426 default:
427 throw NotImplementedException(A_FUNCINFO,"only quad face is implemented");
428 }
429 }
430 public:
431 RayTriangle3DIntersection m_triangle_intersector;
432};

518void BasicRayMeshIntersection::
519compute(Real3ConstArrayView segments_position,
520 Real3ConstArrayView segments_direction,
521 Int32ConstArrayView segments_orig_face,
522 Int32ArrayView user_values,
523 Real3ArrayView segments_intersection,
524 RealArrayView segments_distance,
525 Int32ArrayView faces_local_id)
526{
527 IMesh* mesh = this->mesh();
528
529 bool is_3d = mesh->dimension()==3;
530 info() << "COMPUTE INTERSECTION!!";
531 FaceGroup outer_faces = mesh->outerFaces();
532 Integer nb_face = outer_faces.size();
533 Integer nb_segment = segments_position.size();
534 info() << "NB OUTER FACE=" << nb_face << " NB_SEGMENT=" << nb_segment;
535 VariableNodeReal3 nodes_coordinates(mesh->nodesCoordinates());
536 const Real max_value = 1.0e100;
537 const Real3 max_bb(max_value,max_value,max_value);
538 const Real3 min_bb(-max_value,-max_value,-max_value);
539
540 Real3 mesh_min_bounding_box(max_bb);
541 Real3 mesh_max_bounding_box(min_bb);
542 Integer max_face_local_id = mesh->faceFamily()->maxLocalId();
543 Real3UniqueArray faces_min_bb(max_face_local_id);
544 Real3UniqueArray faces_max_bb(max_face_local_id);
545
546 // Calcule les bounding box
547 {
548 ENUMERATE_FACE(iface,outer_faces){
549 Int32 lid = iface.localId();
550 Real3 face_max_bb(max_bb);
551 Real3 face_min_bb(min_bb);
552 for( NodeEnumerator inode((*iface).nodes()); inode.hasNext(); ++inode )
553 _checkBoundingBox(nodes_coordinates[inode],&face_min_bb,&face_max_bb);
554 _checkBoundingBox(face_min_bb,&mesh_min_bounding_box,&mesh_max_bounding_box);
555 _checkBoundingBox(face_max_bb,&mesh_min_bounding_box,&mesh_max_bounding_box);
556 //TODO: peut-etre ajouter un epsilon autour de le BB pour eviter
557 // les erreurs d'arrondi dans la determination de l'intersection
558 faces_min_bb[lid] = face_min_bb;
559 faces_max_bb[lid] = face_max_bb;
560 }
561 }
562
563 //RealArray segments_distance;
564 segments_distance.fill(max_value);
565 faces_local_id.fill(NULL_ITEM_LOCAL_ID);
566
567 if (!m_face_intersector)
568 m_face_intersector = new BasicRayFaceIntersector(traceMng());
569
570 Real3UniqueArray face_nodes;
571 for( Integer i=0; i<nb_segment; ++i ){
572 Real3 position = segments_position[i];
573 Real3 direction = segments_direction[i];
574 Int32 orig_face_local_id = segments_orig_face[i];
575 Real3 intersection;
576 Real distance = 1.0e100;
577 Int32 min_face_local_id = NULL_ITEM_LOCAL_ID;
578 Int32 user_value = 0;
579 ENUMERATE_FACE(iface,outer_faces){
580 const Face& face = *iface;
581 // On ne traite que ses propres faces
582 if (!face.isOwn())
583 continue;
584 Int32 lid = face.localId();
585 // En 3D, cherche si intersection avec la bounding box.
586 // s'il n'y en a pas, inutile d'aller plus loin.
587 bool is_bb = true;
588 if (is_3d)
589 is_bb = RayTriangle3DIntersection::checkBoundingBox(position,direction,faces_min_bb[lid],faces_max_bb[lid]);
590 if (!is_bb)
591 continue;
592 Integer nb_node = face.nbNode();
593 face_nodes.resize(nb_node);
594 for( NodeEnumerator inode(face.nodes()); inode.hasNext(); ++inode )
595 face_nodes[inode.index()] = nodes_coordinates[inode];
596 Real d = 0.0;
597 Real3 local_intersection;
598 Int32 uv = 0;
599 bool is_found = m_face_intersector->computeIntersection(position,direction,orig_face_local_id,lid,
600 face_nodes,&uv,&d,&local_intersection);
601 //if (i==15){
602 // for( Integer z=0; z<nb_node; ++z )
603 // info() << "FACE NODE lid=" << lid << " I=" << i << " v=" << face_nodes[z] << " d=" << d;
604 //}
605 //if (!is_bb)
606 //info() << "CHECK INTERSECTION: is_bb=" << is_bb << " " << is_found << '\n';
607 if (is_found && !is_bb)
608 fatal() << "Intersection found but no bounding box intersection";
609 if (is_found && d<distance){
610 distance = d;
611 min_face_local_id = lid;
612 intersection = local_intersection;
613 user_value = uv;
614 }
615 }
616 if (min_face_local_id==NULL_ITEM_LOCAL_ID)
617 distance = -1.0;
618 segments_distance[i] = distance;
619 segments_intersection[i] = intersection;
620 faces_local_id[i] = min_face_local_id;
621 user_values[i] = user_value;
622 }
623}

641void BasicRayMeshIntersection::
642compute(IItemFamily* ray_family,
643 VariableParticleReal3& rays_position,
644 VariableParticleReal3& rays_direction,
645 VariableParticleInt32& rays_orig_face,
646 VariableParticleInt32& rays_user_value,
647 VariableParticleReal3& intersections,
648 VariableParticleReal& distances,
649 VariableParticleInt32& rays_face)
650{
651 // NOTE: rays_orig_face n'est pas utilisé.
652
653 IMesh* mesh = ray_family->mesh();
654 IParallelMng* pm = mesh->parallelMng();
655 Int32 my_rank = pm->commRank();
656 // Suppose que les rayons sont compactés
657 Integer nb_local_ray = ray_family->allItems().size();
658 //if (nb_local_ray!=ray_family->maxLocalId()){
659 //fatal() << "La famille de rayons doit être compactée nb=" << nb_local_ray
660 // << " max_id=" << ray_family->maxLocalId();
661 //}
662 Integer global_nb_ray = pm->reduce(Parallel::ReduceSum,nb_local_ray);
663 if (global_nb_ray==0)
664 return;
665 info() << "LOCAL_NB_RAY=" << nb_local_ray << " GLOBAL=" << global_nb_ray;
666
667 // Recopie dans un tableau local les informations d'entrée.
668 // Cela est toujours nécessaire car les particules ne sont pas forcément
669 // compactées et il peut y avoir des trous dans la numérotation.
670 Real3UniqueArray local_positions(nb_local_ray);
671 Real3UniqueArray local_directions(nb_local_ray);
672
673 ENUMERATE_PARTICLE(iitem,ray_family->allItems()){
674 Integer index = iitem.index();
675 //local_ids[index] = iitem.localId();
676 //unique_ids[index] = (*iitem).uniqueId();
677 local_positions[index] = rays_position[iitem];
678 local_directions[index] = rays_direction[iitem];
679 }
680
681 if (pm->isParallel()){
682
683 Int32UniqueArray local_ids(nb_local_ray);
684 Int64UniqueArray unique_ids(nb_local_ray);
685 ENUMERATE_ITEM(iitem,ray_family->allItems()){
686 Integer index = iitem.index();
687 local_ids[index] = iitem.localId();
688 unique_ids[index] = (*iitem).uniqueId();
689 }
690
691 // En parallèle, pour l'instant récupère les rayons des autres processeurs
692
693 Real3UniqueArray all_positions;
694 pm->allGatherVariable(local_positions,all_positions);
695 Real3UniqueArray all_directions;
696 pm->allGatherVariable(local_directions,all_directions);
697
698 Int32UniqueArray local_owners(nb_local_ray);
699 local_owners.fill(my_rank);
700
701 Int32UniqueArray all_owners;
702 pm->allGatherVariable(local_owners,all_owners);
703 Int32UniqueArray all_local_ids;
704 pm->allGatherVariable(local_ids,all_local_ids);
705 Int64UniqueArray all_unique_ids;

835_writeSegments(Int32 rank,
836 Real3ConstArrayView positions,
837 Real3ConstArrayView directions,
838 RealConstArrayView distances)
839{
840 Lima::Maillage lima("segments");
841 lima.dimension(Lima::D3);
842
843 Integer nb_segment = positions.size();
844 UniqueArray<Lima::Noeud> lm_nodes(nb_segment*2);
845 for( Integer i=0; i<nb_segment; ++i ){
846 Real t = distances[i];
847 if (t<0.0)
848 t = 10.0;
849 lm_nodes[i*2].set_x(positions[i].x);
850 lm_nodes[i*2].set_y(positions[i].y);
851 lm_nodes[i*2].set_z(positions[i].z);
852
853 lm_nodes[(i*2)+1].set_x(positions[i].x + t*directions[i].x);
854 lm_nodes[(i*2)+1].set_y(positions[i].y + t*directions[i].y);
855 lm_nodes[(i*2)+1].set_z(positions[i].z + t*directions[i].z);
856 lima.ajouter(lm_nodes[(i*2)]);
857 lima.ajouter(lm_nodes[(i*2)+1]);
858 lima.ajouter(Lima::Bras(lm_nodes[i*2],lm_nodes[(i*2)+1]));
859 }
860 StringBuilder sb("segments");
861 sb+=rank;
862 sb+=".unf";
863 std::string s(sb.toString().localstr());
864 lima.ecrire(s);
865}
