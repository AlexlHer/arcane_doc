685template<typename DataType,typename IntegerType,typename SizeType> inline void
686_sampleSpan(SpanImpl<const DataType,SizeType> values,
687 SpanImpl<const IntegerType,SizeType> indexes,
688 SpanImpl<DataType,SizeType> result)
689{
690 const Int64 result_size = indexes.size();
691 [[maybe_unused]] const Int64 my_size = values.size();
692 const DataType* ptr = values.data();
693 for( Int64 i=0; i<result_size; ++i) {
694 IntegerType index = indexes[i];
695 ARCCORE_CHECK_AT(index,my_size);
696 result[i] = ptr[index];
697 }
698}
