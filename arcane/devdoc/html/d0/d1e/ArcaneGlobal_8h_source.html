600// Support pour l'alignement.
601// le C++11 utilise le mot clé alignas pour spécifier l'alignement.
602// Cela fonctionne avec GCC 4.9+ et Visual Studio 2015. Cela ne fonctionne
603// pas avec Visual Studio 2013. Donc pour Visual Studio on utilise dans tous
604// les cas __declspec qui fonctionne toujours. Sous Linux, __attribute__ fonctionne
605// aussi toujours donc on utilise cela. A noter que les structures Simd ont besoin
606// de l'attribut 'packed' qui n'existe que avec GCC et Intel. Il ne semble pas y avoir
607// d'équivalent avec MSVC.
608#ifdef _MSC_VER
610# define ARCANE_ALIGNAS(value) __declspec(align(value))
612# define ARCANE_ALIGNAS_PACKED(value) __declspec(align(value))
613#else
615# define ARCANE_ALIGNAS_PACKED(value) __attribute__ ((aligned (value),packed))
617# define ARCANE_ALIGNAS(value) __attribute__ ((aligned (value)))
618#endif
