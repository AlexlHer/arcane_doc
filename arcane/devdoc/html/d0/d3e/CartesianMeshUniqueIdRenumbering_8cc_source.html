50void CartesianMeshUniqueIdRenumbering::
51renumber()
52{
53 IMesh* mesh = m_cartesian_mesh->mesh();
54 Int64 cartesian_global_nb_cell = m_generation_info->globalNbCell();
55 info() << "Apply UniqueId renumbering to mesh '" << mesh->name() << "'"
56 << " global_nb_cell=" << cartesian_global_nb_cell
57 << " global_nb_cell_by_dim=" << m_generation_info->globalNbCells();
58
59 if (mesh->dimension() != 2)
60 ARCANE_THROW(NotImplementedException, "Renumbering is only implemented for 2D mesh");
61
62 VariableCellInt64 cells_new_uid(VariableBuildInfo(mesh, "ArcaneRenumberCellsNewUid"));
63 VariableNodeInt64 nodes_new_uid(VariableBuildInfo(mesh, "ArcaneRenumberNodesNewUid"));
64 VariableFaceInt64 faces_new_uid(VariableBuildInfo(mesh, "ArcaneRenumberFacesNewUid"));
65
66 cells_new_uid.fill(-1);
67 nodes_new_uid.fill(-1);
68 faces_new_uid.fill(-1);
69
70 // Marque les entités issues du maillage cartésien comme étant de niveau 0
71 // Elles ne seront pas renumérotées
72 ICartesianMeshPatch* patch0 = m_cartesian_mesh->patch(0);
73 ENUMERATE_ (Cell, icell, patch0->cells()) {
74 Cell c{ *icell };
75 cells_new_uid[icell] = c.uniqueId().asInt64();
76 for (Node n : c.nodes())
77 nodes_new_uid[n] = n.uniqueId();
78 for (Face f : c.faces())
79 faces_new_uid[f] = f.uniqueId();
80 }
81
82 // Pour chaque maille de niveau 0, calcule son indice (i,j) dans le maillage cartésien
83
84 // Pour cela, on suppose que le maillage a été créé avec le 'CartesianMeshGenerator'
85 // (ou un générateur qui a la même numérotation) et que le uniqueId() d'une maille est:
86 // Int64 cell_unique_id = i + j * all_nb_cell_x;
87 // avec:
88 // all_nb_cell_x = m_generation_info->globalNbCells()[MD_DirX];
89
90 Int64ConstArrayView global_nb_cells_by_direction = m_generation_info->globalNbCells();
91 Int64 nb_cell_x = global_nb_cells_by_direction[MD_DirX];
92 if (nb_cell_x <= 0)
93 ARCANE_FATAL("Bad value '{0}' for globalNbCells()[MD_DirX] (should be >0)", nb_cell_x);
94
95 Int64 nb_cell_y = global_nb_cells_by_direction[MD_DirY];
96 if (nb_cell_y <= 0)
97 ARCANE_FATAL("Bad value '{0}' for globalNbCells()[MD_DirY] (should be >0)", nb_cell_y);
98
99 ENUMERATE_ (Cell, icell, patch0->cells()) {
100 Cell cell{ *icell };
101 Int64 uid = cell.uniqueId();
102 Int64 coord_i = uid % nb_cell_x;
103 Int64 coord_j = uid / nb_cell_x;
104 if (m_is_verbose)
105 info() << "Renumbering: PARENT: cell_uid=" << cell.uniqueId() << " I=" << coord_i
106 << " J=" << coord_j << " nb_cell_x=" << nb_cell_x;
107 _applyChildrenCell(cell, nodes_new_uid, faces_new_uid, cells_new_uid, coord_i, coord_j, nb_cell_x, nb_cell_y, 1);
108 }
109
110 // TODO: faire une classe pour cela.
111 //info() << "Change CellFamily";
112 //mesh->cellFamily()->notifyItemsUniqueIdChanged();
113
114 _applyFamilyRenumbering(mesh->cellFamily(), cells_new_uid);
115 _applyFamilyRenumbering(mesh->nodeFamily(), nodes_new_uid);
116 _applyFamilyRenumbering(mesh->faceFamily(), faces_new_uid);
117 mesh->checkValidMesh();
118}

123void CartesianMeshUniqueIdRenumbering::
124_applyFamilyRenumbering(IItemFamily* family, VariableItemInt64& items_new_uid)
125{
126 info() << "Change uniqueId() for family=" << family->name();
127 items_new_uid.synchronize();
128 ENUMERATE_ (Item, iitem, family->allItems()) {
129 Item item{ *iitem };
130 Int64 current_uid = item.uniqueId();
131 Int64 new_uid = items_new_uid[iitem];
132 if (new_uid >= 0 && new_uid != current_uid) {
133 if (m_is_verbose)
134 info() << "Change ItemUID old=" << current_uid << " new=" << new_uid;
135 item.internal()->setUniqueId(new_uid);
136 }
137 }
138 family->notifyItemsUniqueIdChanged();
139}

144void CartesianMeshUniqueIdRenumbering::
145_applyChildrenCell(Cell cell, VariableNodeInt64& nodes_new_uid, VariableFaceInt64& faces_new_uid,
146 VariableCellInt64& cells_new_uid,
147 Int64 coord_i, Int64 coord_j,
148 Int64 nb_cell_x, Int64 nb_cell_y, Int32 level)
149{
150 // TODO: pour pouvoir s'adapter à tous les raffinements, au lieu de 4,
151 // il faudrait prendre le max des nbHChildren()
152
153 // Suppose qu'on a un pattern 2x2
154 coord_i *= 2;
155 coord_j *= 2;
156 nb_cell_x *= 2;
157 nb_cell_y *= 2;
158 const Int64 nb_node_x = nb_cell_x + 1;
159 const Int64 nb_node_y = nb_cell_y + 1;
160 const Int64 cell_adder = nb_cell_x * nb_cell_y * level;
161 const Int64 nb_face_x = nb_cell_x + 1;
162 const Int64 node_adder = nb_node_x * nb_node_y * level;
163 const Int64 face_adder = node_adder * 2;
164
165 // Renumérote les noeuds de la maille courante.
166 // Suppose qu'on a 4 noeuds
167 // ATTENTION a priori on ne peut pas conserver facilement l'ordre
168 // des uniqueId() entre l'ancienne et la nouvelle numérotation.
169 // Cela invalide l'orientation des faces qu'il faudra refaire.
170 {
171 if (cell.nbNode() != 4)
172 ARCANE_FATAL("Invalid number of nodes N={0}, expected=4", cell.nbNode());
173 std::array<Int64, 4> new_uids;
174 new_uids[0] = (coord_i + 0) + ((coord_j + 0) * nb_node_x);
175 new_uids[1] = (coord_i + 1) + ((coord_j + 0) * nb_node_x);
176 new_uids[2] = (coord_i + 1) + ((coord_j + 1) * nb_node_x);
177 new_uids[3] = (coord_i + 0) + ((coord_j + 1) * nb_node_x);
178 for (Integer z = 0; z < 4; ++z) {
179 Node node = cell.node(z);
180 if (nodes_new_uid[node] < 0) {
181 new_uids[z] += node_adder;
182 if (m_is_verbose)
183 info() << "APPLY_NODE_CHILD: uid=" << node.uniqueId() << " parent_cell=" << cell.uniqueId()
184 << " I=" << z << " new_uid=" << new_uids[z];
185 nodes_new_uid[node] = new_uids[z];
186 }
187 }
188 }
189 // Renumérote les faces
190 // TODO: Vérifier la validité de cette méthode.
191 {
192 if (cell.nbFace() != 4)
193 ARCANE_FATAL("Invalid number of faces N={0}, expected=4", cell.nbFace());
194 std::array<Int64, 4> new_uids;
195 new_uids[0] = (coord_i + 0) + ((coord_j + 0) * nb_face_x);
196 new_uids[1] = (coord_i + 1) + ((coord_j + 0) * nb_face_x);
197 new_uids[2] = (coord_i + 1) + ((coord_j + 1) * nb_face_x);
198 new_uids[3] = (coord_i + 0) + ((coord_j + 1) * nb_face_x);
199 for (Integer z = 0; z < 4; ++z) {
200 Face face = cell.face(z);
201 if (faces_new_uid[face] < 0) {
202 new_uids[z] += face_adder;
203 if (m_is_verbose)
204 info() << "APPLY_FACE_CHILD: uid=" << face.uniqueId() << " parent_cell=" << cell.uniqueId()
205 << " I=" << z << " new_uid=" << new_uids[z];
206 faces_new_uid[face] = new_uids[z];
207 }
208 }
209 }
210 // Renumérote les sous-mailles
211 // Suppose qu'on a 4 mailles enfants comme suit par mailles
212 // -------
213 // | 2| 3|
214 // -------
215 // | 0| 1|
216 // -------
217 Int32 nb_child = cell.nbHChildren();
218 for (Int32 icell = 0; icell < nb_child; ++icell) {
219 Cell sub_cell = cell.hChild(icell);
220 Int64 my_coord_i = coord_i + icell % 2;
221 Int64 my_coord_j = coord_j + icell / 2;
222 Int64 new_uid = (my_coord_i + my_coord_j * nb_cell_x) + cell_adder;
223 if (m_is_verbose)
224 info() << "APPLY_CELL_CHILD: uid=" << sub_cell.uniqueId() << " I=" << my_coord_i << " J=" << my_coord_j
225 << " level=" << level << " new_uid=" << new_uid << " NodeAdder=" << node_adder;
226
227 _applyChildrenCell(sub_cell, nodes_new_uid, faces_new_uid, cells_new_uid, my_coord_i, my_coord_j,
228 nb_cell_x, nb_cell_y, level + 1);
229 if (cells_new_uid[sub_cell] < 0)
230 cells_new_uid[sub_cell] = new_uid;
231 }
232}
