77template <typename DataType>
78Ref<ISerializedData> ScalarDataT<DataType>::
79createSerializedDataRef(bool use_basic_type) const
80{
81 typedef typename DataTypeTraitsT<DataType>::BasicType BasicType;
82
83 Integer nb_count = 1;
84 eDataType data_type = dataType();
85 Integer type_size = sizeof(DataType);
86
87 if (use_basic_type) {
88 nb_count = DataTypeTraitsT<DataType>::nbBasicType();
89 data_type = DataTypeTraitsT<BasicType>::type();
90 type_size = sizeof(BasicType);
91 }
92
93 Integer nb_element = 1;
94 Integer nb_base_element = nb_element * nb_count;
95 Integer full_size = nb_base_element * type_size;
96 ByteConstArrayView base_values(full_size, reinterpret_cast<const Byte*>(&m_value));
97 UniqueArray<Int64> dimensions;
98 dimensions.add(nb_element);
99 auto sd = arcaneCreateSerializedDataRef(data_type, base_values.size(), 0, nb_element,
100 nb_base_element, false, dimensions);
101 sd->setConstBytes(base_values);
102 return sd;
103}

108template <typename DataType>
109void ScalarDataT<DataType>::
110allocateBufferForSerializedData(ISerializedData* sdata)
111{
112 typedef typename DataTypeTraitsT<DataType>::BasicType BasicType;
113
114 eDataType data_type = sdata->baseDataType();
115 eDataType base_data_type = DataTypeTraitsT<BasicType>::type();
116
117 if (data_type != dataType() && data_type != base_data_type)
118 throw ArgumentException(A_FUNCINFO, "Bad serialized type");
119
120 Span<Byte> byte_values(reinterpret_cast<Byte*>(&m_value), sdata->memorySize());
121 sdata->setWritableBytes(byte_values);
122}

137template <typename DataType>
138void ScalarDataT<DataType>::
139serialize(ISerializer* sbuf, IDataOperation*)
140{
141 Integer nb_count = DataTypeTraitsT<DataType>::nbBasicType();
142 typedef typename DataTypeTraitsT<DataType>::BasicType BasicType;
143 eDataType data_type = DataTypeTraitsT<BasicType>::type();
144
145 DataType ttmp = m_value;
146 ArrayView<BasicType> vtmp(1 * nb_count, reinterpret_cast<BasicType*>(&ttmp));
147
148 ISerializer::eMode mode = sbuf->mode();
149 if (mode == ISerializer::ModeReserve)
150 sbuf->reserveSpan(data_type, vtmp.size());
151 else if (mode == ISerializer::ModePut)
152 sbuf->putSpan(vtmp);
153 else if (mode == ISerializer::ModeGet)
154 sbuf->getSpan(vtmp);
155}
