106// Load and initialize .NET Core and get desired function pointer for scenario
107// (Note: pas utilisé pour l'instant).
108load_assembly_and_get_function_pointer_fn
109getDotnetLoadAssembly(const String& assembly)
110{
111 string_t assembly1 = _toString(assembly);
112 // Load .NET Core
113 void* load_assembly_and_get_function_pointer = nullptr;
114 hostfxr_handle cxt = nullptr;
115
116 hostfxr_initialize_parameters params;
117 params.size = sizeof(params);
118 // TODO: il s'agit du chemin de l'exécutable, pas de l'assembly.
119 params.host_path = assembly1.c_str(); //get_path_to_the_host_exe(); // Path to the current executable
120
121 int rc = init_fptr(assembly1.c_str(), &params, &cxt);
122 if (rc != 0 || cxt == nullptr) {
123 std::cerr << "Init failed: " << std::hex << std::showbase << rc << std::endl;
124 close_fptr(cxt);
125 return nullptr;
126 }
127
128 // Get the load assembly function pointer
129 rc = get_delegate_fptr(cxt, hdt_load_assembly_and_get_function_pointer,
130 &load_assembly_and_get_function_pointer);
131 if (rc != 0 || load_assembly_and_get_function_pointer == nullptr)
132 std::cerr << "Get delegate failed: " << std::hex << std::showbase << rc << std::endl;
133
134 close_fptr(cxt);
135 return (load_assembly_and_get_function_pointer_fn)load_assembly_and_get_function_pointer;
136}

143_execDirect(const CommandLineArguments& cmd_args,
144 const String& orig_assembly_name)
145{
146 int argc = *(cmd_args.commandLineArgc());
147 //const char** old_argv = (const char**)*(cmd_args.commandLineArgv());

149 Arcane::String root_path = Arcane::platform::getFileDirName(orig_assembly_name);
150 string_t root_path1 = _toString(root_path);

152 std::cerr << "ENTERING _execDirect root_path=" << root_path << "\n";
153
154 string_t dotnet_root = _toString(String(arcane_dotnet_root));
155 hostfxr_initialize_parameters params;
156 params.size = sizeof(params);
157 params.host_path = root_path1.c_str();
158 params.dotnet_root = dotnet_root.c_str();
159 typedef const char* const_char_t;
160 const_char_t* argv = new const_char_t[1];
161 argv[0] = (const_char_t)orig_assembly_name1.c_str(); //strdup(orig_assembly_name.localstr());
162 std::cerr << "_execDirect argv[0] =" << orig_assembly_name << "\n";
163 argc = 1;
164
165 hostfxr_handle host_context_handle;
166 int rc = init_command_line_fptr(argc, (const char_t**)argv, &params, &host_context_handle);
167 std::cerr << "_execDirect init_command_line R = " << rc << "\n";
168 if (rc!=0)
169 ARCANE_FATAL("Can not initialize runtime RC={0}",rc);
170
171#if TEST
172 size_t buffer_used = 0;
173 if (hostfxr_get_runtime_property(host_context_handle, "TEST_PROPERTY", nullptr, 0, &buffer_used) == HostApiMissingProperty) {
174 hostfxr_set_runtime_property(host_context_handle, "TEST_PROPERTY", "TRUE");
175 }
176#endif
177
178 std::cerr << "Launching '.Net'\n";
179 int r = run_app_fptr(host_context_handle);
180 std::cerr << "End '.Net': R=" << r << "\n";
181
182 close_fptr(host_context_handle);
183 return r;
184}

192extern "C" ARCANE_EXPORT int
193arcane_dotnet_coreclr_main(const Arcane::CommandLineArguments& cmd_args,
194 const Arcane::String& orig_assembly_name)
195{
196 String verbose_str = Arcane::platform::getEnvironmentVariable("ARCANE_DEBUG_DOTNET");
197 if (!verbose_str.null())
198 dotnet_verbose = 1;
199
200 // Si l'utilisateur a spécifié cette variable, alors on ne surcharge pas
201 // les appels avec la version configurée lors de la compilation.
202 // Le runtime 'coreclr' se charge d'utiliser cette variable d'environnement.
203
204 String dotnet_root_env = platform::getEnvironmentVariable("DOTNET_ROOT");
205 if (!dotnet_root_env.null()){
206 arcane_dotnet_root_env_variable = dotnet_root_env.toStdStringView();
207 arcane_dotnet_root = arcane_dotnet_root_env_variable.c_str();
208 }
209
210 // TODO: trouver un moyen d'utiliser 'cmd_args'
211 PRINT_FORMAT(1,"ARCANE_DOTNET_CORECLR_MAIN assembly_name={0}",orig_assembly_name);
212
213 if (orig_assembly_name.empty())
214 ARCANE_FATAL("No assembly name");
215
216 string_t orig_assembly_name1 = _toString(orig_assembly_name);
217
218 String root_path = Arcane::platform::getFileDirName(orig_assembly_name);
219
220 PRINT_FORMAT(1,"ENTERING CORECLR_MAIN root_path={0}",root_path);
221
222 //
223 // STEP 1: Load HostFxr and get exported hosting functions
224 //
225 if (!load_hostfxr(orig_assembly_name1))
226 ARCANE_FATAL("Failure: load_hostfxr()");
227
228 return _execDirect(cmd_args, orig_assembly_name);
229
230 // NOTE: Cette partie n'est pas utilisée pour l'instant.
231
232 //
233 // STEP 2: Initialize and start the .NET Core runtime
234 //
235 Arcane::String config_path = Arcane::Directory(root_path).file("Arcane.Main.runtimeconfig.json");
236 load_assembly_and_get_function_pointer_fn load_assembly_and_get_function_pointer = nullptr;
237 load_assembly_and_get_function_pointer = getDotnetLoadAssembly(config_path);
238 assert(load_assembly_and_get_function_pointer != nullptr && "Failure: get_dotnet_load_assembly()");
239
240 //
241 // STEP 3: Load managed assembly and get function pointer to a managed method
242 //
243 //const string_t dotnetlib_path = root_path + STR("Arcane.Main.dll");
244 const char_t* dotnet_type = STR("ArcaneMainExec, Arcane.Main");
245 const char_t* dotnet_type_method = STR("CoreClrComponentEntryPoint");
246 // <SnippetLoadAndGet>
247 // Function pointer to managed delegate
248 component_entry_point_fn dll_entry_point_func = nullptr;
249 // ATTENTION: si on passe 'nullptr' comme 'delegate_type_name', alors 'dotnet_type_method'
250 // doit être du type 'int (InpPtr args,int sizeBytes)'.
251 int rc = load_assembly_and_get_function_pointer(orig_assembly_name1.c_str(),
252 dotnet_type,
253 dotnet_type_method,
254 nullptr /*delegate_type_name*/,
255 nullptr,
256 (void**)&dll_entry_point_func);
257 // </SnippetLoadAndGet>
258 if (rc != 0)
259 ARCANE_FATAL("load_assembly_and_get_function_pointer: rc={0}", rc);
260 if (!dll_entry_point_func)
261 ARCANE_FATAL("Failure: load_assembly_and_get_function_pointer()");
262
263 //
264 // STEP 4: Run managed code
265 //
266 struct lib_args
267 {
268 const char_t* message;
269 int number;
270 };
271 lib_args args{ STR("from host!"), 0 };
272
273 return dll_entry_point_func(&args, sizeof(args));
274}

294{

297 return f;
298}

308 void* f = dlsym(h, name);
309 PRINT_FORMAT(1,"get_export name={0} f={1}",name,f);
310 if (!f)
311 PRINT_FORMAT(0,"Can not load symbol '{0}'",name);
312 assert(f != nullptr);
313 return f;
314}

317// <SnippetLoadHostFxr>
318// Using the nethost library, discover the location of hostfxr and get exports
319bool
320load_hostfxr(const string_t& assembly_name)
321{
322 // Si 'coreclr' n'est pas installé dans un chemin standard, il est possible
323 // qu'il faille le spécifier. La variable d'environnement DOTNET_ROOT permet
324 // de le faire. Si elle n'est pas positionnée, utilise le chemin trouvée
325 // lors de la configuration.
326 string_t dotnet_root = _toString(String(arcane_dotnet_root));
327 get_hostfxr_parameters hostfxr_parameters;
328 hostfxr_parameters.size = sizeof(get_hostfxr_parameters);
329 hostfxr_parameters.assembly_path = assembly_name.c_str();
330 hostfxr_parameters.dotnet_root = dotnet_root.c_str();
331
332 PRINT_FORMAT(1,"Entering load_hostfxr() dotnet_root={0}",arcane_dotnet_root);
333 // Pre-allocate a large buffer for the path to hostfxr
334 const int BUF_LEN = 12000;
335 char_t buffer[BUF_LEN];
336 size_t buffer_size = sizeof(buffer) / sizeof(char_t);
337 int rc = get_hostfxr_path(buffer, &buffer_size, &hostfxr_parameters);
338 PRINT_FORMAT(1,"Return value of 'get_hostfxr_path' = '{0}'",rc);
339 //if (rc != 0)
340 //return false;
341 // Load hostfxr and get desired exports
342 void* lib = load_library(buffer);
343 PRINT_FORMAT(1,"LIB_PTR={0} path={1}",lib,buffer);
344 init_fptr = (hostfxr_initialize_for_runtime_config_fn)get_export(lib, "hostfxr_initialize_for_runtime_config");
345 get_delegate_fptr = (hostfxr_get_runtime_delegate_fn)get_export(lib, "hostfxr_get_runtime_delegate");
346 close_fptr = (hostfxr_close_fn)get_export(lib, "hostfxr_close");
347 init_command_line_fptr = (hostfxr_initialize_for_dotnet_command_line_fn)get_export(lib, "hostfxr_initialize_for_dotnet_command_line");
348 run_app_fptr = (hostfxr_run_app_fn)get_export(lib, "hostfxr_run_app");
349 return (init_fptr && get_delegate_fptr && close_fptr && init_command_line_fptr && run_app_fptr);
350}
