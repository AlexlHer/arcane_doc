48class PolyhedralFamily : public ItemFamily{
49 IMesh* m_mesh;
50 ItemSharedInfo* m_shared_info = nullptr;
51 Int32UniqueArray m_empty_connectivity {0};
52 Int32UniqueArray m_empty_connectivity_indexes;
53 Int32UniqueArray m_empty_connectivity_nb_item;
54
55 public:
56 inline static const String m_arcane_item_lids_property_name {"Arcane_Item_Lids"}; // inline used to initialize within the declaration
57
58 public:
59 PolyhedralFamily(IMesh* mesh, eItemKind ik, String name)
60 : ItemFamily(mesh,ik,name)
61 , m_mesh(mesh) {
62 ItemFamily::build();
63 m_sub_domain_id = subDomain()->subDomainId();
64 ItemTypeMng* itm = m_mesh->itemTypeMng();
65 ItemTypeInfo* dof_type_info = itm->typeFromId(IT_NullType);
66 m_shared_info = _findSharedInfo(dof_type_info);
67 _updateEmptyConnectivity();
68 }
69
70 public:
71
72 void preAllocate(Integer nb_item)
73 {
74 Integer nb_hash = itemsMap().buckets().size();
75 Integer wanted_size = 2*(nb_item+infos().nbItem());
76 if (nb_hash<wanted_size)
77 itemsMap().resize(wanted_size,true);
78 Integer base_mem = 1+ItemSharedInfo::COMMON_BASE_MEMORY;
79 Integer mem = base_mem * (nb_item+1);
80 _reserveInfosMemory(mem);
81 m_empty_connectivity_indexes.resize(nb_item + nbItem(),0);
82 m_empty_connectivity_nb_item.resize(nb_item + nbItem(), 0);
83 _updateEmptyConnectivity();
84 }
85
86 ItemInternal* _allocItem(const Int64 uid)
87 {
88 bool need_alloc; // given by alloc
89 ItemInternal* item_internal = ItemFamily::_allocOne(uid,need_alloc);
90 if (!need_alloc)
91 item_internal->setUniqueId(uid);
92 else{
93 _allocateInfos(item_internal,uid,m_shared_info);
94 }
95 item_internal->setOwner(m_sub_domain_id,m_sub_domain_id);
96 return item_internal;
97 }
98
99 void addItems(Int64ConstArrayView uids, Int32ArrayView items)
100 {
101 if (uids.empty()) return ;
102 ARCANE_ASSERT((uids.size()==items.size()),("one must have items.size==uids.size()"));
103 m_mesh->traceMng()->info() << " PolyhedralFamily::ADDITEMS " ;
104 preAllocate(uids.size());
105 auto index {0};
106 for (auto uid : uids) {
107 ItemInternal* ii = _allocItem(uid);
108 items[index++] = ii->localId();
109 }
110 m_need_prepare_dump = true;
111 _updateItemInternalList();
112 }
113
114 void _updateItemInternalList()
115 {
116 switch (itemKind()) {
117 case IK_Cell:
118 m_item_internal_list->cells = _itemsInternal();
119 break;
120 case IK_Face:
121 m_item_internal_list->faces = _itemsInternal();
122 break;
123 case IK_Edge:
124 m_item_internal_list->edges = _itemsInternal();
125 break;
126 case IK_Node:
127 m_item_internal_list->nodes = _itemsInternal();
128 break;
129 case IK_DoF:
130 case IK_Particle:
131 case IK_Unknown:
132 break;
133 }
134 }
135
136 void _updateEmptyConnectivity() {
137 auto item_internal_connectivity_list = itemInternalConnectivityList();
138 for (auto item_kind = 0 ; item_kind < ItemInternalConnectivityList::MAX_ITEM_KIND; ++item_kind){
139 item_internal_connectivity_list->setConnectivityList(item_kind,m_empty_connectivity);
140 item_internal_connectivity_list->setConnectivityIndex(item_kind, m_empty_connectivity_indexes);
141 item_internal_connectivity_list->setConnectivityNbItem(item_kind, m_empty_connectivity_nb_item);
142 }
143 }
144};

245 void scheduleAddItems(PolyhedralFamily* arcane_item_family,
246 Int64ConstArrayView uids,
247 ItemLocalIds& item_local_ids) {
248 auto& added_items = item_local_ids.m_future_items;
249 auto& item_family = m_mesh.findFamily(itemKindArcaneToNeo(arcane_item_family->itemKind()),
250 arcane_item_family->name().localstr());
251 m_mesh.scheduleAddItems(item_family, std::vector<Int64>{uids.begin(), uids.end()}, added_items);
252 // add arcane items
253 auto & mesh_graph = m_mesh.internalMeshGraph();
254 item_family.addProperty<Neo::utils::Int32>(PolyhedralFamily::m_arcane_item_lids_property_name.localstr());
255 mesh_graph.addAlgorithm(Neo::InProperty{item_family,item_family.lidPropName()},
256 Neo::OutProperty{item_family,PolyhedralFamily::m_arcane_item_lids_property_name.localstr()},
257 [arcane_item_family,uids]
258 (Neo::ItemLidsProperty const& lids_property,
259 Neo::PropertyT<Neo::utils::Int32> &){
260 Int32UniqueArray arcane_items(uids.size());
261 arcane_item_family->addItems(uids,arcane_items);
262 arcane_item_family->traceMng()->info() << arcane_items;
263 // debug check lid matching. maybe to remove if too coostly
264 auto neo_lids = lids_property.values();
265 if (!std::equal(neo_lids.begin(),neo_lids.end(),arcane_items.begin()))
266 arcane_item_family->traceMng()->fatal()<< "Inconsistent item lids generation between Arcane and Neo.";
267 });
268 }

322 // template to handle nb_items_per_item type (an int or an array)
323 template <typename ConnectivitySizeType>
324 void _scheduleAddConnectivity(PolyhedralFamily* arcane_source_item_family,
325 ItemLocalIds& source_items,
326 ConnectivitySizeType&& nb_connected_items_per_item,
327 PolyhedralFamily* arcane_target_item_family,
328 Int64ConstArrayView target_item_uids,
329 String const& connectivity_name)
330 {
331 // add connectivity in Neo
332 auto& source_family = m_mesh.findFamily(itemKindArcaneToNeo(arcane_source_item_family->itemKind()),
333 arcane_source_item_family->name().localstr());
334 auto& target_family = m_mesh.findFamily(itemKindArcaneToNeo(arcane_target_item_family->itemKind()),
335 arcane_target_item_family->name().localstr());
336 // Remove connectivities with a null item
337 std::vector<Int64> target_item_uids_filtered;
338 target_item_uids_filtered.reserve(target_item_uids.size());
339 std::copy_if(target_item_uids.begin(),
340 target_item_uids.end(),
341 std::back_inserter(target_item_uids_filtered),
342 [](auto uid){return uid != NULL_ITEM_UNIQUE_ID; });
343 // Add connectivity (async)
344 m_mesh.scheduleAddConnectivity(source_family, source_items.m_future_items, target_family,
345 std::forward<ConnectivitySizeType>(nb_connected_items_per_item),
346 std::move(target_item_uids_filtered),
347 connectivity_name.localstr());
348 // Register connectivity in Arcane : via un algo !! todo : quelle prop out
349 auto& mesh_graph = m_mesh.internalMeshGraph();
350 source_family.addProperty <Int32>("NoOutProperty"); // todo remove : create noOutput algo in Neo
351 mesh_graph.addAlgorithm(Neo::InProperty{source_family, connectivity_name.localstr()},
352 Neo::OutProperty{source_family,"NoOutProperty"},
353 [arcane_source_item_family, arcane_target_item_family, &source_family, this, connectivity_name]
354 (Neo::Mesh::ConnectivityPropertyType const& ,
355 Neo::PropertyT<Neo::utils::Int32> & ){
356 this->m_subdomain->traceMng()->info() << "ADD CONNECTIVITY";
357 auto item_internal_connectivity_list = arcane_source_item_family->itemInternalConnectivityList();
358 // todo check if families are default families
359 auto& connectivity_values = source_family.getConcreteProperty<Neo::Mesh::ConnectivityPropertyType>(connectivity_name.localstr());
360 auto nb_item_data = connectivity_values.m_offsets.data();
361 auto nb_item_size = connectivity_values.m_offsets.size();
362 item_internal_connectivity_list->setConnectivityNbItem(arcane_target_item_family->itemKind(),
363 Int32ArrayView{Integer(nb_item_size),nb_item_data});
364 auto connectivity_values_data = connectivity_values.m_data.data();
365 auto connectivity_values_size = connectivity_values.m_data.size();
366 item_internal_connectivity_list->setConnectivityList(arcane_target_item_family->itemKind(),
367 Int32ArrayView{Integer(connectivity_values_size),connectivity_values_data});
368 auto connectivity_index_data = connectivity_values.m_indexes.data();
369 auto connectivity_index_size = connectivity_values.m_indexes.size();
370 item_internal_connectivity_list->setConnectivityIndex(arcane_target_item_family->itemKind(),
371 Int32ArrayView{ Integer(connectivity_index_size), connectivity_index_data });
372 });
373 }

449void Arcane::mesh::PolyhedralMesh::
450read(const String& filename)
451{
452 // First step: create manually a unit mesh
453// ARCANE_UNUSED(filename); // temporary
454// _createUnitMesh();
455 // Second step read a vtk polyhedral mesh
456 m_subdomain->traceMng()->info() << "--PolyhedralMesh : reading " << filename;
457 // First step create a single cell
458 // _createSingleCellTest();
459 // Second step : read a vtk polyhedral file
460 UniqueArray<String> splitted_filename{};
461 filename.split(splitted_filename,'.');
462 auto file_extension = splitted_filename.back();
463 if (file_extension != "vtk")
464 m_subdomain->traceMng()->fatal() << "Only vtk file format supported for polyhedral mesh";
465
466 createItemFamily(IK_Cell, "CellFamily");
467 createItemFamily(IK_Node, "NodeFamily");
468 createItemFamily(IK_Face, "FaceFamily");
469 createItemFamily(IK_Edge, "EdgeFamily");
470 [[maybe_unused]] auto cell_family = arcaneDefaultFamily(IK_Cell);
471 [[maybe_unused]] auto node_family = arcaneDefaultFamily(IK_Node);
472 [[maybe_unused]] auto face_family = arcaneDefaultFamily(IK_Face);
473 [[maybe_unused]] auto edge_family = arcaneDefaultFamily(IK_Edge);
474#ifdef ARCANE_HAS_VTKIO
475 PolyhedralMeshTools::VtkReader reader{filename};
476 PolyhedralMeshImpl::ItemLocalIds cell_lids,node_lids, face_lids, edge_lids;
477 // todo separate Link with Arcane and link with vtk structure
478 // Add items
479 m_mesh->scheduleAddItems(cell_family, reader.cellUids(), cell_lids);
480 m_mesh->scheduleAddItems(node_family, reader.nodeUids(), node_lids);
481 m_mesh->scheduleAddItems(face_family, reader.faceUids(), face_lids);
482 m_mesh->scheduleAddItems(edge_family, reader.edgeUids(), edge_lids);
483 // Add connectivities
484 m_mesh->scheduleAddConnectivity(cell_family,cell_lids,reader.cellNbNodes(),node_family,reader.cellNodes(),String{"CellToNodes"});
485 m_mesh->scheduleAddConnectivity(cell_family,cell_lids,reader.cellNbFaces(),face_family,reader.cellFaces(),String{"CellToFaces"});
486 m_mesh->scheduleAddConnectivity(face_family,face_lids,reader.faceNbNodes(),node_family,reader.faceNodes(),String{"FaceToNodes"});
487 m_mesh->scheduleAddConnectivity(edge_family,edge_lids,2,node_family,reader.edgeNodes(),String{"EdgeToNodes"});
488 m_mesh->scheduleAddConnectivity(face_family,face_lids,reader.faceNbCells(),cell_family,reader.faceCells(),String{"FaceToCells"});
489 m_mesh->scheduleAddConnectivity(edge_family,edge_lids,reader.edgeNbCells(),cell_family,reader.edgeCells(),String{"EdgeToCells"});
490 m_mesh->scheduleAddConnectivity(edge_family,edge_lids,reader.edgeNbFaces(),face_family,reader.edgeFaces(),String{"EdgeToFaces"});
491 m_mesh->scheduleAddConnectivity(face_family,face_lids,reader.faceNbEdges(),edge_family,reader.faceEdges(),String{"FaceToEdges"});
492 m_mesh->scheduleAddConnectivity(cell_family,cell_lids,reader.cellNbEdges(),edge_family,reader.cellEdges(),String{"CellToEdges"});
493 m_mesh->scheduleAddConnectivity(node_family,node_lids,reader.nodeNbCells(),cell_family,reader.nodeCells(),String{"NodeToCells"});
494 m_mesh->scheduleAddConnectivity(node_family,node_lids,reader.nodeNbFaces(),face_family,reader.nodeFaces(),String{"NodeToFaces"});
495 m_mesh->scheduleAddConnectivity(node_family,node_lids,reader.nodeNbEdges(),edge_family,reader.nodeEdges(),String{"NodeToEdges"});
496// m_mesh->scheduleAddConnectivity(node_family,node_lids,1,cell_family,
497// Int64UniqueArray{0,0,0,0,0,0},String{"NodeToCells"});
498 m_mesh->applyScheduledOperations();
499 cell_family->endUpdate();
500 node_family->endUpdate();
501 face_family->endUpdate();
502 edge_family->endUpdate();
503 endUpdate();
504 m_is_allocated = true;
505#else
506 ARCANE_FATAL("Need VTKIO to read polyhedral mesh");
507#endif
508}

640void mesh::PolyhedralMesh::
641_createUnitMesh()
642{
643 createItemFamily(IK_Cell, "CellFamily");
644 createItemFamily(IK_Node, "NodeFamily");
645 auto cell_family = m_default_arcane_families[IK_Cell];
646 auto node_family = m_default_arcane_families[IK_Node];
647 Int64UniqueArray cell_uids{0},node_uids{ 0, 1, 2, 3, 4, 5 };
648 // todo add a cell_lids struct (containing future)
649 PolyhedralMeshImpl::ItemLocalIds cell_lids,node_lids;
650 m_mesh->scheduleAddItems(cell_family, cell_uids, cell_lids);
651 m_mesh->scheduleAddItems(node_family, node_uids, node_lids);
652 int nb_node = 6;
653 m_mesh->scheduleAddConnectivity(cell_family,cell_lids,nb_node,node_family,node_uids,String{"CellToNodes"});
654 m_mesh->scheduleAddConnectivity(node_family,node_lids,1,cell_family,
655 Int64UniqueArray{0,0,0,0,0,0},String{"NodeToCells"});
656 m_mesh->applyScheduledOperations();
657 cell_family->endUpdate();
658 node_family->endUpdate();
659 endUpdate();
660 // Mimic what IMeshModifier::endUpdate would do => default families are completed.
661 // Families created after a first endUpdate call are not default families
662}
