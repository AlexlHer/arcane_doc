135void MEDMeshReaderService::
136_initMEDToArcaneTypes()
137{
138 m_med_to_arcane_types.clear();
139
140 // TODO: regarder la correspondance de connectivité entre
141 // Arcane et MED pour les éléments quadratiques
142 // Types 1D
143 _addTypeInfo(1,2,MED_SEG2,IT_Line2);
144 _addTypeInfo(1,3,MED_SEG3,IT_Line3); // Non supporté
145 _addTypeInfo(1,4,MED_SEG4,IT_NullType); // Non supporté
146
147 // Types 2D.
148 _addTypeInfo(2,3,MED_TRIA3,IT_Triangle3, Triangle3_indirection);
149 _addTypeInfo(2,4,MED_QUAD4,IT_Quad4, Quad4_indirection);
150 _addTypeInfo(2,6,MED_TRIA6,IT_NullType); // Non supporté
151 _addTypeInfo(2,7,MED_TRIA7,IT_NullType); // Non supporté
152 _addTypeInfo(2,8,MED_QUAD8,IT_Quad8); // Non supporté
153 _addTypeInfo(2,9,MED_QUAD9,IT_NullType); // Non supporté
154
155 // Types 3D
156 _addTypeInfo(3, 4,MED_TETRA4,IT_Tetraedron4);
157 _addTypeInfo(3, 5,MED_PYRA5,IT_Pyramid5, Pyramid5_indirection);
158 _addTypeInfo(3, 6,MED_PENTA6,IT_Pentaedron6);
159 _addTypeInfo(3, 8,MED_HEXA8,IT_Hexaedron8, Hexaedron8_indirection);
160 _addTypeInfo(3,10,MED_TETRA10,IT_Tetraedron10);
161 _addTypeInfo(3,12,MED_OCTA12,IT_Octaedron12);
162 _addTypeInfo(3,13,MED_PYRA13,IT_NullType); // Non supporté
163 _addTypeInfo(3,15,MED_PENTA15,IT_NullType); // Non supporté
164 _addTypeInfo(3,18,MED_PENTA18,IT_NullType); // Non supporté
165 _addTypeInfo(3,20,MED_HEXA20,IT_Hexaedron20);
166 _addTypeInfo(3,27,MED_HEXA27,IT_NullType); // Non supporté
167
168 // Mailles dont la géométrie à une connectivité variable.
169 // Pour l'instant on ne supporte aucun de ces types dans Arcane.
170 // On traite quand même ces éléments pour afficher une erreur s'ils sont
171 // présents dans le maillage. En mettant la valeur (0) pour le nombre
172 // de noeuds on signale à _readItems() qu'on ne sait pas traiter ces éléments.
174 _addTypeInfo(2,0,MED_POLYGON,IT_NullType);
175 _addTypeInfo(2,0,MED_POLYGON2,IT_NullType);
176 _addTypeInfo(3,0,MED_POLYHEDRON,IT_NullType);
177
178 // Mailles dont la géométrie est dynamique (découverte du modèle dans le fichier)
179 // TODO: regarder comment les traiter
180 //#define MED_STRUCT_GEO_INTERNAL 600
181 //#define MED_STRUCT_GEO_SUP_INTERNAL 700
182}

205IMeshReader::eReturnType MEDMeshReaderService::
206_readMesh(IPrimaryMesh* mesh,const String& filename)
207{
208 med_idt fid = MEDfileOpen(filename.localstr(),MED_ACC_RDONLY);
209 if (fid<0){
210 MESSAGE("ERROR: can not open MED file ");
211 error() << "ERROR: can not open MED file '" << filename << "'";
212 return RTError;
213 }
214 // Pour garantir la fermeture du fichier.
215 AutoCloseMED auto_close_med(fid);
216
217 int nb_mesh = MEDnMesh(fid);
218 if (nb_mesh<0){
219 error() << "Error reading number of meshes";
220 return RTError;
221 }
222 info() << "MED: nb_mesh=" << nb_mesh;
223 if (nb_mesh==0){
224 error() << "No mesh is present";
225 return RTError;
226 }
227
228 // Le maillage qu'on lit est toujours le premier
229 int mesh_index = 1;
230
231 // Récupère la dimension d'espace. Cela est nécessaire pour dimensionner axisname eet unitname
232 int nb_axis = MEDmeshnAxis(fid,mesh_index);
233 if (nb_axis<0){
234 error() << "Can not read number of axis (MEDmeshnAxis)";
235 return RTError;
236 }
237 info() << "MED: nb_axis=" << nb_axis;
238
239 UniqueArray<char> axisname(MED_SNAME_SIZE*nb_axis+1,'\0');
240 UniqueArray<char> unitname(MED_SNAME_SIZE*nb_axis+1,'\0');
241
242 char meshname[MED_NAME_SIZE+1];
243 meshname[0] = '\0';
244 char meshdescription[MED_COMMENT_SIZE+1];
245 meshdescription[0] = '\0';
246 char dtunit[MED_SNAME_SIZE+1];
247 dtunit[0] = '\0';
248 med_int spacedim = 0;
249 med_int meshdim = 0;
250 med_mesh_type meshtype = MED_UNDEF_MESH_TYPE;
251 med_sorting_type sortingtype = MED_SORT_UNDEF;
252 med_int nstep = 0;
253 med_axis_type axistype = MED_UNDEF_AXIS_TYPE;
254 int err = 0;
255 err = MEDmeshInfo(fid, mesh_index, meshname, &spacedim, &meshdim, &meshtype, meshdescription,
256 dtunit, &sortingtype, &nstep, &axistype, axisname.data(), unitname.data());
257 if (err < 0) {
258 error() << "Can not read mesh info (MEDmeshInfo) r=" << err;
259 return RTError;
260 }
261 if (meshtype!=MED_UNSTRUCTURED_MESH){
262 error() << "Arcane handle only MED unstructured mesh (MED_UNSTRUCTURED_MESH) type=" << meshtype;
263 return RTError;
264 }
265 Integer mesh_dimension = meshdim;
266 if (mesh_dimension!=2 && mesh_dimension!=3)
267 ARCANE_FATAL("MED reader handles only 2D or 3D meshes");
268
269 info() << "MED: name=" << meshname;
270 info() << "MED: description=" << meshdescription;
271 info() << "MED: spacedim=" << spacedim;
272 info() << "MED: meshdim=" << meshdim;
273 info() << "MED: dtunit=" << dtunit;
274 info() << "MED: meshtype=" << meshtype;
275 info() << "MED: sortingtype=" << sortingtype;
276 info() << "MED: axistype=" << axistype;
277 info() << "MED: nstep=" << nstep;
278
279 Int64 nb_node = 0;
280 // Lecture du nombre de noeuds.
281 {
282 med_bool coordinatechangement;
283 med_bool geotransformation;
284 // TODO: traiter les informations telles que coordinatechangement
285 // et geotransformation si besoin
286 med_int med_nb_node = MEDmeshnEntity(fid, meshname, MED_NO_DT, MED_NO_IT, MED_NODE, MED_NO_GEOTYPE,
287 MED_COORDINATE, MED_NO_CMODE,&coordinatechangement,
288 &geotransformation);
289 if (med_nb_node<0) {
290 error() << "Can not read number of nodes (MEDmeshnEntity) err=" << med_nb_node;
291 return RTError;
292 }
293 nb_node = med_nb_node;
294 }
295 info() << "MED: nb_node=" << nb_node;
296
297 mesh->setDimension(mesh_dimension);
298
299 Int64 cell_unique_id = 0;
300 Int32 cell_local_id = 0;
301 // Alloue les entités types par type.
302 for( const auto& iinfo : m_med_to_arcane_types ){
303 Integer item_dimension = iinfo.dimension();
304 // On ne traite que les entités de la dimension du maillage.
305 if (item_dimension!=mesh_dimension)
306 continue;
307 UniqueArray<med_int> med_connectivity;
308 Int32 nb_item = _readItems(fid,meshname,iinfo,med_connectivity);
309 if (nb_item==0)
310 continue;
311 Integer arcane_type = iinfo.arcaneType();
312 Integer nb_item_node = iinfo.nbNode();
313 if (arcane_type==IT_NullType){
314 // Indique un type supporté par MED mais pas par Arcane
315 ARCANE_FATAL("MED type '{0}' is not supported by Arcane",iinfo.medType());
316 }
317 Int64 cells_infos_index = 0;
318 Int64 med_connectivity_index = 0;
319 UniqueArray<Int64> cells_infos((2+nb_item_node)*nb_item);
320 info() << "CELL_INFOS size=" << cells_infos.size() << " nb_item=" << nb_item
321 << " type=" << arcane_type;
322 const Int32* indirection = iinfo.indirection();
323 for( Int32 i=0; i<nb_item; ++i ){
324 cells_infos[cells_infos_index] = arcane_type;
325 ++cells_infos_index;
326 cells_infos[cells_infos_index] = cell_unique_id;
327 ++cells_infos_index;
328 ++cell_unique_id;
329 // La connectivité dans MED commence à 1 et Arcane à 0.
330 // Il faut donc retrancher 1 de la connectivité donnée par MED.
331 Span<Int64> cinfo_span(cells_infos.span().subspan(cells_infos_index,nb_item_node));
332 Span<med_int> med_cinfo_span(med_connectivity.span().subspan(med_connectivity_index,nb_item_node));
333 if (indirection){
334 for( Integer k=0; k<nb_item_node; ++k ){
335 cinfo_span[k] = med_cinfo_span[indirection[k]] - 1;
336 }
337 }
338 else{
339 for( Integer k=0; k<nb_item_node; ++k )
340 cinfo_span[k] = med_cinfo_span[k] - 1;
341 }
342
343 med_connectivity_index += nb_item_node;
344 cells_infos_index += nb_item_node;
345
346 ++cell_local_id;
347 }
348 mesh->allocateCells(nb_item,cells_infos,false);
349 }
350 mesh->endAllocate();
351
352 // Lit les coordonnées des noeuds et positionne les coordonnées dans Arcane
353 UniqueArray<Real3> nodes_coordinates(nb_node);
354 {
355 UniqueArray<med_float> coordinates(nb_node*spacedim);
356 err = MEDmeshNodeCoordinateRd(fid, meshname, MED_NO_DT, MED_NO_IT, MED_FULL_INTERLACE,
357 coordinates.data());
358 if (err<0){
359 error() << "Can not read nodes coordinates err=" << err;
360 return RTError;
361 }
362
363 if (spacedim==3){
364 for( Int64 i=0; i<nb_node; ++i ){
365 Real3 xyz(coordinates[i*3],coordinates[(i*3)+1],coordinates[(i*3)+2]);
366 info() << "I=" << i << " XYZ=" << xyz;
367 nodes_coordinates[i] = xyz;
368 }
369 }
370 else if (spacedim==2){
371 for( Int64 i=0; i<nb_node; ++i ){
372 Real3 xyz(coordinates[i*2],coordinates[(i*2)+1],0.0);
373 info() << "I=" << i << " XYZ=" << xyz;
374 nodes_coordinates[i] = xyz;
375 }
376 }
377 else
378 ARCANE_THROW(NotImplementedException,"spacedim!=2 && spacedim!=3");
379 }
380
381 // Positionne les coordonnées
382 {
383 VariableNodeReal3& nodes_coord_var(mesh->nodesCoordinates());
384 ENUMERATE_NODE(inode,mesh->allNodes()){
385 Node node = *inode;
386 nodes_coord_var[inode] = nodes_coordinates[node.uniqueId()];
387 }
388 }
389 return RTOk;
390}

395Int32 MEDMeshReaderService::
396_readItems(med_idt fid,const char* meshname,const MEDToArcaneItemInfo& iinfo,
397 Array<med_int>& connectivity)
398{
399 int med_item_type = iinfo.medType();
400 med_bool coordinatechangement;
401 med_bool geotransformation;
402 med_int nb_med_item = ::MEDmeshnEntity(fid, meshname, MED_NO_DT, MED_NO_IT, MED_CELL,med_item_type,
403 MED_CONNECTIVITY, MED_NODAL, &coordinatechangement,
404 &geotransformation);
405 if (nb_med_item < 0) {
406 ARCANE_FATAL("Can not read MED med_item_type '{0}' error={1}",med_item_type,nb_med_item);
407 }
408 info() << "MED: type=" << med_item_type << " nb_item=" << nb_med_item;
409 if (nb_med_item==0)
410 return 0;
411 Int64 nb_node = iinfo.nbNode();
412 if (nb_node==0)
413 // Indique un élément qu'on ne sais pas traiter.
414 ARCANE_THROW(NotImplementedException,"Reading items with MED type '{0}'",med_item_type);
415
416 connectivity.resize(nb_node*nb_med_item);
417 int err = MEDmeshElementConnectivityRd(fid, meshname, MED_NO_DT, MED_NO_IT, MED_CELL,
418 med_item_type, MED_NODAL, MED_FULL_INTERLACE,
419 connectivity.data());
420 if (err<0){
421 ARCANE_FATAL("Can not read connectivity MED med_item_type '{0}' error={1}",
422 med_item_type,err);
423 }
424 info() << "CON: " << connectivity;
425 return nb_med_item;
426}
