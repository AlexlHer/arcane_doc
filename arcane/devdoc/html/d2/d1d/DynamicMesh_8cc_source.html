308void DynamicMesh::
309build()
310{
311 Trace::Setter mci(traceMng(),_className());
312
313 info() << "BUILDING DYNAMIC_MESH=" << this << " name=" << name() << '\n';
314
315 m_tied_interface_mng = new TiedInterfaceMng(this);
316
317 // IMPORTANT: les premiers éléments de m_item_families doivent
318 // correspondre aux familles associées aux valeurs de eItemKind
319 // (IK_Node, IK_Edge, IK_Face, IK_Cell)
320 for( IItemFamily* family : m_item_families ){
321 _buildAndInitFamily(family);
322 }
323
324 // Permet d'appeler la fonction '_prepareDump' lorsqu'une
325 // protection va être effectuée
326 IVariableMng* vm = subDomain()->variableMng();
327 m_observer_pool.addObserver(this,
328 &DynamicMesh::_prepareForDump,
329 vm->writeObservable());
330 // Permet d'appeler la fonction '_reloadFromDump' lorsqu'une
331 // protection est relue
332 m_observer_pool.addObserver(this,
333 &DynamicMesh::_readFromDump,
334 vm->readObservable());
335 IPropertyMng* pm = subDomain()->propertyMng();
336 m_observer_pool.addObserver(this,
337 &DynamicMesh::_saveProperties,
338 pm->writeObservable());
339 m_observer_pool.addObserver(this,
340 &DynamicMesh::_loadProperties,
341 pm->readObservable());
342
343 m_mesh_builder = new DynamicMeshIncrementalBuilder(this);
344 m_mesh_checker = new DynamicMeshChecker(this);
345 m_partition_constraint_mng = new MeshPartitionConstraintMng(this);
346
347 if (parentMesh()) {
348 // Cela peut induire des segfaults si le DynamicMesh n'est pas correctement utilisé
349 m_submesh_tools = new SubMeshTools(this, m_mesh_builder);
351 m_mesh_refinement = 0;
352
353 this->properties()->setBool("compact",true);
354 this->properties()->setBool("sort",true);
355
356 ItemGroupDynamicMeshObserver * obs = NULL;
357 m_parent_group->attachObserver(this, (obs = new ItemGroupDynamicMeshObserver(this)));
358
359 this->endAllocate();
360 Int32ConstArrayView localIds = m_parent_group->itemsLocalId();
361 obs->executeExtend(&localIds);
362 this->endUpdate();
363
364 } else {
365 m_submesh_tools = 0;
367
368 // GG: ne construit m_mesh_refinement qui si l'AMR est actif
369 // Cela évite de creer des variables inutile lorsque l'AMR n'est pas
370 // demandé. La création de m_mesh_refinement se fait maintenant
371 // dans readAmrActivator(). Ce n'est peut-être pas le bon endroit
372 // pour le faire et on peut éventuellement le déplacer. Il faut
373 // juste que m_mesh_refinement ne soit créé que si l'AMR est actif.
374 // SDC : OK. Remis car maintenant l'info AMR (in)actif est connue a
375 // la construction. Suppression de readAmrActivator.
376
377 if (m_is_amr_activated)
378 m_mesh_refinement = new MeshRefinement(this);
379 }
380}

595void DynamicMesh::
596initializeVariables(const XmlNode& init_node)
597{
598 // TEMPORAIRE: recopie pour chaque famille le champ owner() dans
599 // la variable correspondante.
600 for( IItemFamily* family : m_item_families ){
601 VariableItemInt32& items_owner(family->itemsNewOwner());
602 ENUMERATE_ITEM(iitem,family->allItems()){
603 Item item = *iitem;
604 items_owner[iitem] = item.owner();
605 }
606 }
607
608 ICaseDocument* doc = subDomain()->caseDocument();
609 XmlNode root = doc->rootElement();
610 if (root.null())
611 return;
612
613 bool has_error = false;
614 IVariableMng* vm = m_sub_domain->variableMng();
615 XmlNodeList child_list = init_node.children(String("variable"));
616 for( auto i : child_list ){
617 String var_name = i.attrValue("nom");
618 IVariable* var = vm->findMeshVariable(this,var_name);
619 if (!var){
620 error() << "Failed to initialize the variable `" << var_name
621 << "'\nNo variable with that name exists";
622 has_error = true;
623 continue;
624 }
625 // N'initialise pas les variables non utilisées.
626 if (!var->isUsed())
627 continue;
628
629 String grp_name = i.attrValue("groupe");
630 ItemGroup grp = findGroup(grp_name);
631
632 if (grp.null()){
633 error() << "Failed to initialize the variable `" << var_name
634 << "' on the group `" << grp_name << "'\n"
635 << "No group with that name exists";
636 has_error = true;
637 continue;
638 }
639 debug() << "Read value variable `" << grp_name
640 << "' `" << var_name << "' " << var;
641 String val_str = i.attrValue("valeur");
642 bool ret = var->initialize(grp,val_str);
643 if (ret){
644 error() << "Failed to initialized the variable `" << var_name
645 << "' on the group `" << grp_name << "'";
646 has_error = true;
647 continue;
648 }
649 }
650 if (has_error)
651 fatal() << "Variable initialization failed";
652}

703endAllocate()
704{
705 Trace::Setter mci(traceMng(),_className());
706 if (m_is_allocated)
707 ARCANE_FATAL("endAllocate() has already been called");
708 _checkDimension(); // HP: add control if endAllocate is called

729
730 _compactItems(true,false);
731 _compactItemInternalReferences();

734

736 ts->dumpTimeAndMemoryUsage(pm);
737
738#ifdef ARCANE_DEBUG_DYNAMIC_MESH
739 {

742 file_name += "-";
743 file_name += m_mesh_rank;

745 mesh_utils::writeMeshConnectivity(this,file_name);
746 }
747#endif
748
749 _computeSynchronizeInfos();
750 m_mesh_checker->checkMeshFromReferenceFile();
751
752 // Positionne les proprietaires pour que cela soit comme apres
753 // un echange. Ce n'est a priori pas indispensable
754 // mais cela permet d'assurer la coherence avec
755 // d'autres appels a setOwnersFromCells()

764 // Indique aux familles qu'on vient de mettre à jour le maillage

766 _notifyEndUpdateForFamilies();
767
768 _prepareForDump();
769
770 m_is_allocated = true;
771}

871_serializeItems(ISerializer* buffer,Int32ConstArrayView item_local_ids, IItemFamily* item_family)
872{
873 // 1-Get item lids for family and downard dependencies.
874 // Rk downard relations are not taken here => automatically added in addItems(ItemData) :
875 // this will change. Todo add relations when addItems has been updated
876 using FamilyLidMap = std::map<String, Int32UniqueArray>;
877 FamilyLidMap serialized_items;
878 serialized_items[item_family->name()] = item_local_ids;
879 for (auto connectivity : mesh()->itemFamilyNetwork()->getChildDependencies(item_family)){
880 ENUMERATE_ITEM(item, item_family->view(item_local_ids)) {
881 auto connectivity_accessor = ConnectivityItemVector{connectivity};
882 auto& connected_family_serialized_items = serialized_items[connectivity->targetFamily()->name()];
883 connected_family_serialized_items.addRange(connectivity_accessor.connectedItems(ItemLocalId(item)).localIds());
884 }
885 }
886 // 2-Serialize each family in the buffer. The order is important: use the family graph from leaves to root
887 buffer->setMode(ISerializer::ModeReserve);
888 _fillSerializer(buffer, serialized_items);
889 buffer->allocateBuffer();
890 buffer->setMode(ISerializer::ModePut);
891 _fillSerializer(buffer, serialized_items);

1303void DynamicMesh::
1304_prepareForDump()
1305{
1306 bool want_dump = m_properties->getBool("dump");
1307 info(4) << "DynamicMesh::prepareForDump() name=" << name()
1308 << " need_compact?=" << m_need_compact
1309 << " want_dump?=" << want_dump;
1310
1311 // Si le maillage n'est pas sauvé, ne fait rien. Cela évite de compacter
1312 // et trier le maillage ce qui n'est pas souhaitable si les propriétés
1313 // 'sort' et 'compact' sont à 'false'.
1314 if (!want_dump)
1315 return;
1316
1317 if (m_need_compact){
1318 // Pour l'instant, il faut trier et compacter les entites
1319 // avant une sauvegarde
1320 _compactItems(true,true);
1321 }
1322
1323 // Préparation des connexions maillage/sous-maillage
1324 {
1325 if (m_parent_mesh) {
1326 ARCANE_ASSERT((m_parent_group != NULL),("Unexpected NULL parent group"));
1327 m_parent_mesh_name = m_parent_mesh->name();
1328 m_parent_group_name = m_parent_group->name();
1329 } else {
1330 ARCANE_ASSERT((m_parent_group == NULL),("Unexpected non-NULL parent group"));
1331 m_parent_mesh_name = String();
1332 m_parent_group_name = String();
1333 }
1334 const Integer n_sub_mesh = m_child_meshes.size();
1335 m_child_meshes_name.resize(n_sub_mesh);
1336 for(Integer i=0;i<n_sub_mesh; ++i)
1337 m_child_meshes_name[i] = m_child_meshes[i]->name();
1338 }
1339
1340 // Sauve les infos sur les familles d'entités
1341 {
1342 Integer nb_item_family = m_item_families.count();
1343 m_item_families_name.resize(nb_item_family);
1344 m_item_families_kind.resize(nb_item_family);
1345 Integer index = 0;
1346 for( IItemFamily* family : m_item_families ){
1347 m_item_families_kind[index] = family->itemKind();
1348 m_item_families_name[index] = family->name();
1349 ++index;
1350 }
1351 }
1352
1353 for( IItemFamily* family : m_item_families ){
1354 family->prepareForDump();
1355 }
1356
1357 if (m_tied_interface_need_prepare_dump){
1358 m_tied_interface_mng->prepareTiedInterfacesForDump();
1359 m_tied_interface_need_prepare_dump = false;
1360 }
1361}

1591void DynamicMesh::
1592_multipleExchangeItems(Integer nb_exchange,Integer version,bool do_compact)
1593{
1594 if (version<1 || version>2)
1595 ARCANE_FATAL("Invalid value '{0}' for version. Valid values are 1 or 2",version);
1596
1597 info() << "** ** MULTIPLE EXCHANGE ITEM version=" << version << " nb_exchange=" << nb_exchange;
1598 UniqueArray<UniqueArray<Int32>> cells_to_exchange_new_owner(nb_exchange);
1599 // Il faut stocker le uid car suite a un equilibrage les localId vont changer
1600 UniqueArray<UniqueArray<Int64>> cells_to_exchange_uid(nb_exchange);
1601
1602 IItemFamily* cell_family = cellFamily();
1603 VariableItemInt32& cells_new_owner = cell_family->itemsNewOwner();
1604
1605 Integer nb_cell= ownCells().size();
1606 ENUMERATE_CELL(icell,ownCells()){
1607 Cell cell = *icell;
1608 Int32 current_owner = cell.owner();
1609 Int32 new_owner = cells_new_owner[icell];
1610 if (current_owner==new_owner)
1611 continue;
1612 Integer phase = 0;
1613 if (version==2)
1614 phase = (new_owner % nb_exchange);
1615 else if (version==1)
1616 phase = icell.index() / nb_cell;
1617 cells_to_exchange_new_owner[phase].add(new_owner);
1618 cells_to_exchange_uid[phase].add(cell.uniqueId().asInt64());
1619 }
1620
1621 // Remet comme si la maille ne changeait pas de propriétaire pour
1622 // éviter de l'envoyer.
1623 ENUMERATE_CELL(icell,ownCells()){
1624 Cell cell = *icell;
1625 cells_new_owner[icell] = cell.owner();
1626 }
1627
1628 // A partir d'ici, le cells_new_owner est identique au cell.owner()
1629 // pour chaque maille.
1630 Int32UniqueArray uids_to_lids;
1631 for( Integer i=0; i<nb_exchange; ++i ){
1632 Int32ConstArrayView new_owners = cells_to_exchange_new_owner[i];
1633 Int64ConstArrayView new_uids = cells_to_exchange_uid[i];
1634 Integer nb_cell = new_uids.size();
1635 info() << "MultipleExchange current_exchange=" << i << " nb_cell=" << nb_cell;
1636 uids_to_lids.resize(nb_cell);
1637 cell_family->itemsUniqueIdToLocalId(uids_to_lids,new_uids);
1638 ItemInternalList cells = cell_family->itemsInternal();
1639 // Pour chaque maille de la partie en cours d'echange, positionne le new_owner
1640 // a la bonne valeurs
1641 for( Integer z=0; z<nb_cell; ++z )
1642 cells_new_owner[cells[uids_to_lids[z]]] = new_owners[z];
1643 cells_new_owner.synchronize();
1644 mesh()->utilities()->changeOwnersFromCells();
1645 _exchangeItemsNew();
1646 }
1647
1648 if (do_compact){
1649 Timer::Action ts_action1(m_sub_domain,"CompactItems",true);
1650 bool do_sort = m_properties->getBool("sort");
1651 _compactItems(do_sort,true);
1652 }
1653}

2391_synchronizeVariables()
2392{
2393 // On ne synchronise que les variables sur les items du maillage courant
2394 // - Les items de graphe ne sont pas traités ici (était déjà retiré de
2395 // la version précédente du code)
2396 // - Les particules et ceux sans genre (Unknown) ne pas sujet à synchronisation
2397 // La synchronisation est ici factorisée par collection de variables de même
2398 // synchroniser (même pour les synchronisers sur groupes != famille)
2399 // Pour préserver un ordre consistent de synchronisation une structure
2400 // auxiliaire OrderedSyncList est utilisée.
2401
2402 // Peut on le faire avec une structure plus compacte ?
2403 typedef UniqueArray<IVariableSynchronizer*> OrderedSyncList;
2404 typedef std::map<IVariableSynchronizer*, VariableCollection> SyncList;
2405 OrderedSyncList ordered_sync_list;
2406 SyncList sync_list;
2407
2408 VariableCollection used_vars(subDomain()->variableMng()->usedVariables());
2409 for( VariableCollection::Enumerator i_var(used_vars); ++i_var; ) {
2410 IVariable* var = *i_var;
2411 switch(var->itemKind()) {
2412 case IK_Node:
2413 case IK_Edge:
2414 case IK_Face:
2415 case IK_Cell:
2416 case IK_DoF:{
2417 IVariableSynchronizer * synchronizer = 0;
2418 if (var->isPartial())
2419 synchronizer = var->itemGroup().synchronizer();
2420 else
2421 synchronizer = var->itemFamily()->allItemsSynchronizer();
2422 IMesh * sync_mesh = synchronizer->itemGroup().mesh();
2423 if (sync_mesh != this) continue; // on ne synchronise que sur le maillage courant
2424 std::pair<SyncList::iterator,bool> inserter = sync_list.insert(std::make_pair(synchronizer,VariableCollection()));
2425 if (inserter.second) { // nouveau synchronizer
2426 ordered_sync_list.add(synchronizer);
2427 }
2428 VariableCollection & collection = inserter.first->second;
2429 collection.add(var);
2430 } break;
2431 case IK_Particle:
2432 case IK_Unknown:
2433 break;
2434 }
2435 }
2436
2437 for(Integer i_sync = 0; i_sync < ordered_sync_list.size(); ++i_sync) {
2438 IVariableSynchronizer * synchronizer = ordered_sync_list[i_sync];
2439 VariableCollection & collection = sync_list[synchronizer];
2440 synchronizer->synchronize(collection);
2441 }
