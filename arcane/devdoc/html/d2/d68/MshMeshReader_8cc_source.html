261Integer MshMeshReader::
262_switchMshType(Integer mshElemType, Integer& nNodes)
263{
264 switch (mshElemType) {
265 case (IT_NullType): // used to decode IT_NullType: IT_HemiHexa7|IT_Line9
266 switch (nNodes) {
267 case (7):
268 return IT_HemiHexa7;
269 default:
270 info() << "Could not decode IT_NullType with nNodes=" << nNodes;
271 throw IOException("_switchMshType", "Could not decode IT_NullType with nNodes");
272 }
273 break;
274 case (MSH_PNT):
275 nNodes = 1;
276 return IT_Vertex; //printf("1-node point");
277 case (MSH_LIN_2):
278 nNodes = 2;
279 return IT_Line2; //printf("2-node line");
280 case (MSH_TRI_3):
281 nNodes = 3;
282 return IT_Triangle3; //printf("3-node triangle");
283 case (MSH_QUA_4):
284 nNodes = 4;
285 return IT_Quad4; //printf("4-node quadrangle");
286 case (MSH_TET_4):
287 nNodes = 4;
288 return IT_Tetraedron4; //printf("4-node tetrahedron");
289 case (MSH_HEX_8):
290 nNodes = 8;
291 return IT_Hexaedron8; //printf("8-node hexahedron");
292 case (MSH_PRI_6):
293 nNodes = 6;
294 return IT_Pentaedron6; //printf("6-node prism");
295 case (MSH_PYR_5):
296 nNodes = 5;
297 return IT_Pyramid5; //printf("5-node pyramid");
298 case (MSH_TRI_10):
299 nNodes = 10;
300 return IT_Heptaedron10; //printf("10-node second order tetrahedron")
301 case (MSH_TRI_12):
302 nNodes = 12;
303 return IT_Octaedron12; //printf("Unknown MSH_TRI_12");
304
305 case (MSH_TRI_6):
306 case (MSH_QUA_9):
307 case (MSH_HEX_27):
308 case (MSH_PRI_18):
309 case (MSH_PYR_14):
310 case (MSH_QUA_8):
311 case (MSH_HEX_20):
312 case (MSH_PRI_15):
313 case (MSH_PYR_13):
314 case (MSH_TRI_9):
315 case (MSH_TRI_15):
316 case (MSH_TRI_15I):
317 case (MSH_TRI_21):
318 default:
319 ARCANE_THROW(NotSupportedException, "Unknown GMSH element type '{0}'", mshElemType);
320 }
321 return IT_NullType;
322}

327IMeshReader::eReturnType MshMeshReader::
328_readNodesFromAsciiMshV2File(IosFile& ios_file, Array<Real3>& node_coords)
329{
330 // number-of-nodes & coords
331 info() << "[_readNodesFromAsciiMshV2File] Looking for number-of-nodes";
332 Integer nb_node = ios_file.getInteger();
333 if (nb_node < 0)
334 throw IOException(A_FUNCINFO, String("Invalid number of nodes: n=") + nb_node);
335 info() << "[_readNodesFromAsciiMshV2File] nb_node=" << nb_node;
336 for (Integer i = 0; i < nb_node; ++i) {
337 // Il faut lire l'id même si on ne s'en sert pas.
338 [[maybe_unused]] Int32 id = ios_file.getInteger();
339 Real nx = ios_file.getReal();
340 Real ny = ios_file.getReal();
341 Real nz = ios_file.getReal();
342 node_coords.add(Real3(nx, ny, nz));
343 //info() << "id_" << id << " xyz(" << nx << "," << ny << "," << nz << ")";
344 }
345 ios_file.getNextLine(); // Skip current \n\r
346 return RTOk;
347}

370IMeshReader::eReturnType MshMeshReader::
371_readNodesFromAsciiMshV4File(IosFile& ios_file, MeshInfo& mesh_info)
372{
373 // Première ligne du fichier
374 Integer nb_entity = ios_file.getInteger();
375 Integer total_nb_node = ios_file.getInteger();
376 Integer min_node_tag = ios_file.getInteger();
377 Integer max_node_tag = ios_file.getInteger();
378 ios_file.getNextLine(); // Skip current \n\r
379 if (total_nb_node < 0)
380 ARCANE_THROW(IOException,"Invalid number of nodes : '{0}'",total_nb_node);
381 info() << "[Nodes] nb_entity=" << nb_entity
382 << " total_nb_node=" << total_nb_node
383 << " min_tag=" << min_node_tag
384 << " max_tag=" << max_node_tag;
385
386 UniqueArray<Int32> nodes_uids;
387 for( Integer i_entity=0; i_entity<nb_entity; ++i_entity ){
388 // Dimension de l'entité (pas utile)
389 [[maybe_unused]] Integer entity_dim = ios_file.getInteger();
390 // Tag de l'entité (pas utile)
391 [[maybe_unused]] Integer entity_tag = ios_file.getInteger();
392 Integer parametric_coordinates = ios_file.getInteger();
393 Integer nb_node2 = ios_file.getInteger();
394 ios_file.getNextLine();
395
396 info(4) << "[Nodes] index=" << i_entity << " entity_dim=" << entity_dim << " entity_tag=" << entity_tag
397 << " parametric=" << parametric_coordinates
398 << " nb_node2=" << nb_node2;
399 if (parametric_coordinates!=0)
400 ARCANE_THROW(NotSupportedException,"Only 'parametric coordinates' value of '0' is supported (current={0})",parametric_coordinates);
401 // Il est possible que le nombre de noeuds soit 0.
402 // Dans ce cas, il faut directement passer à la ligne suivante
403 if (nb_node2==0)
404 continue;
405 nodes_uids.resize(nb_node2);
406 for (Integer i = 0; i < nb_node2; ++i) {
407 // Conserve le uniqueId() du noeuds.
408 nodes_uids[i] = ios_file.getInteger();
409 //info() << "I=" << i << " ID=" << id;
410 }
411 for (Integer i = 0; i < nb_node2; ++i) {
412 Real nx = ios_file.getReal();
413 Real ny = ios_file.getReal();
414 Real nz = ios_file.getReal();
415 Real3 xyz(nx, ny, nz);
416 mesh_info.node_coords_map.add(nodes_uids[i],xyz);
417 //info() << "I=" << i << " COORD=" << xyz;
418 //info() << "id_" << id << " xyz(" << nx << "," << ny << "," << nz << ")";
419 }
420 ios_file.getNextLine(); // Skip current \n\r
421 }
422 return RTOk;
423}

428Integer MshMeshReader::
429_readElementsFromAsciiMshV2File(IosFile& ios_file, MeshInfo& mesh_info)
430{
431 Integer number_of_elements = ios_file.getInteger(); // is an integer equal to 0 in the ASCII file format, equal to 1 for the binary format
432 if (number_of_elements < 0)
433 ARCANE_THROW(IOException, String("Error with the number_of_elements=") + number_of_elements);
434
435 info() << "nb_elements=" << number_of_elements;
436
437 //elm-number elm-type number-of-tags < tag > ... node-number-list
438 // Lecture des infos des mailles & de la connectivité
439 // bool pour voir si on est depuis 0 ou 1
440 bool it_starts_at_zero = false;
441 for (Integer i = 0; i < number_of_elements; ++i) {
442 [[maybe_unused]] Integer elm_number = ios_file.getInteger(); // Index
443 Integer elm_type = ios_file.getInteger(); // Type
444 // info() << elm_number << " " << elm_type;
445 // Now get tags in the case the number of nodes is encoded
446 Integer number_of_tags = ios_file.getInteger();
447 Integer lastTag = 0;
448 for (Integer j = 0; j < number_of_tags; ++j)
449 lastTag = ios_file.getInteger();
450 Integer number_of_nodes = 0; // Set the number of nodes from the discovered type
451 if (elm_type == IT_NullType) {
452 number_of_nodes = lastTag;
453 info() << "We hit the case the number of nodes is encoded (number_of_nodes=" << number_of_nodes << ")";
454 }
455 Integer cell_type = _switchMshType(elm_type, number_of_nodes);
456 //#warning Skipping 2D lines & points
457 // We skip 2-node lines and 1-node points
458 if (number_of_nodes < 3) {
459 for (Integer j = 0; j < number_of_nodes; ++j)
460 ios_file.getInteger();
461 continue;
462 }
463
464 mesh_info.cells_type.add(cell_type);
465 mesh_info.cells_nb_node.add(number_of_nodes);
466 mesh_info.cells_uid.add(i);
467 info() << elm_number << " " << elm_type << " " << number_of_tags << " number_of_nodes=" << number_of_nodes;
468 // printf("%i %i %i %i %i (", elm_number, elm_type, reg_phys, reg_elem, number_of_nodes);
469 for (Integer j = 0; j < number_of_nodes; ++j) {
470 // printf("%i ", node_number);
471 Integer id = ios_file.getInteger();
472 if (id == 0)
473 it_starts_at_zero = true;
474 mesh_info.cells_connectivity.add(id);
475 }
476 // printf(")\n");
477 }
478 if (!it_starts_at_zero)
479 for (Integer j = 0, max = mesh_info.cells_connectivity.size(); j < max; ++j)
480 mesh_info.cells_connectivity[j] = mesh_info.cells_connectivity[j] - 1;
481
482 ios_file.getNextLine(); // Skip current \n\r
483
484 // On ne supporte que les maillage de dimension 3 dans ce vieux format
485 return 3;
486}

507Integer MshMeshReader::
508_readElementsFromAsciiMshV4File(IosFile& ios_file, MeshInfo& mesh_info)
509{
510 Integer nb_block = ios_file.getInteger();
511
512 Integer number_of_elements = ios_file.getInteger(); // is an integer equal to 0 in the ASCII file format, equal to 1 for the binary format
513 Integer min_element_tag = ios_file.getInteger();
514 Integer max_element_tag = ios_file.getInteger();
515 ios_file.getNextLine(); // Skip current \n\r
516 info() << "[Elements] nb_block=" << nb_block
517 << " nb_elements=" << number_of_elements
518 << " min_element_tag=" << min_element_tag
519 << " max_element_tag=" << max_element_tag;
520
521 if (number_of_elements < 0)
522 ARCANE_THROW(IOException,"Invalid number of elements: {0}",number_of_elements);
523
524 UniqueArray<MeshV4ElementsBlock>& blocks = mesh_info.blocks;
525 blocks.resize(nb_block);
526
527 {
528 // Numérote les blocs (pour le débug)
529 Integer index = 0;
530 for( MeshV4ElementsBlock& block : blocks ){
531 block.index = index;
532 ++index;
533 }
534 }
535 for( MeshV4ElementsBlock& block : blocks ){
536 Integer entity_dim = ios_file.getInteger();
537 Integer entity_tag = ios_file.getInteger(); // is an integer equal to 0 in the ASCII file format, equal to 1 for the binary format
538 Integer entity_type = ios_file.getInteger();
539 Integer nb_entity_in_block = ios_file.getInteger();
540
541 Integer item_nb_node = 0;
542 Integer item_type = _switchMshType(entity_type, item_nb_node);
543
544 info(4) << "[Elements] index=" << block.index << " entity_dim=" << entity_dim
545 << " entity_tag=" << entity_tag
546 << " entity_type=" << entity_type << " nb_in_block=" << nb_entity_in_block
547 << " item_type=" << item_type << " item_nb_node=" << item_nb_node;
548
549 block.nb_entity = nb_entity_in_block;
550 block.item_type = item_type;
551 block.item_nb_node = item_nb_node;
552 block.dimension = entity_dim;
553 block.entity_tag = entity_tag;
554
555 for (Integer i = 0; i < nb_entity_in_block; ++i) {
556 Int64 item_unique_id = ios_file.getInt64();
557 block.uids.add(item_unique_id);
558
559 for ( Integer j = 0; j < item_nb_node; ++j )
560 block.connectivity.add(ios_file.getInt64());
561 }
562 ios_file.getNextLine(); // Skip current \n\r
563 }
564 // Maintenant qu'on a tous les blocks, la dimension du maillage est
565 // la plus grande dimension des blocks
566 Integer mesh_dimension = -1;
567 for( MeshV4ElementsBlock& block : blocks )
568 mesh_dimension = math::max(mesh_dimension,block.dimension);
569 if (mesh_dimension<0)
570 ARCANE_FATAL("Invalid computed mesh dimension '{0}'",mesh_dimension);
571 if (mesh_dimension!=2 && mesh_dimension!=3)
572 ARCANE_THROW(NotSupportedException,"mesh dimension '{0}'. Only 2D or 3D meshes are supported",mesh_dimension);
573 info() << "Computed mesh dimension = " << mesh_dimension;
574
575 // On ne conserve que les blocs de notre dimension
576 // pour créér les mailles
577 for( MeshV4ElementsBlock& block : blocks ){
578 if (block.dimension!=mesh_dimension)
579 continue;
580
581 info(4) << "Keeping block index=" << block.index;
582
583 Integer item_type = block.item_type;
584 Integer item_nb_node = block.item_nb_node;
585
586 for (Integer i = 0; i < block.nb_entity; ++i) {
587 mesh_info.cells_type.add(item_type);
588 mesh_info.cells_nb_node.add(item_nb_node);
589 mesh_info.cells_uid.add(block.uids[i]);
590 auto v = block.connectivity.subView(i*item_nb_node,item_nb_node);
591 mesh_info.cells_connectivity.addRange(v);
592 }
593 }
594
595 return mesh_dimension;
596}

617void MshMeshReader::
618_allocateCells(IMesh* mesh, MeshInfo& mesh_info, bool is_read_items)
619{
620 Integer nb_elements = mesh_info.cells_type.size();
621 info() << "nb_of_elements=cells_type.size()=" << nb_elements;
622 Integer nb_cell_node = mesh_info.cells_connectivity.size();
623 info() << "nb_cell_node=cells_connectivity.size()=" << nb_cell_node;
624
625 // Création des mailles
626 info() << "Building cells, nb_cell=" << nb_elements << " nb_cell_node=" << nb_cell_node;
627 // Infos pour la création des mailles
628 // par maille: 1 pour son unique id,
629 // 1 pour son type,
630 // 1 pour chaque noeud
631 UniqueArray<Int64> cells_infos;
632 Integer connectivity_index = 0;
633 UniqueArray<Real3> local_coords;
634 for (Integer i = 0; i < nb_elements; ++i) {
635 Integer current_cell_nb_node = mesh_info.cells_nb_node[i];
636 Integer cell_type = mesh_info.cells_type[i];
637 Int64 cell_uid = mesh_info.cells_uid[i];
638 cells_infos.add(cell_type);
639 cells_infos.add(cell_uid); //cell_unique_id
640
641 ArrayView<Int64> local_info(current_cell_nb_node,&mesh_info.cells_connectivity[connectivity_index]);
642 cells_infos.addRange(local_info);
643 connectivity_index += current_cell_nb_node;
644 }
645
646 IPrimaryMesh* pmesh = mesh->toPrimaryMesh();
647 info() << "## Allocating ##";
648
649 if (is_read_items)
650 pmesh->allocateCells(nb_elements, cells_infos, false);
651 else
652 pmesh->allocateCells(0, UniqueArray<Int64>(0), false);
653
654 info() << "## Ending ##";
655 pmesh->endAllocate();
656 info() << "## Done ##";
657
658 // Positionne les coordonnées
659 {
660 VariableNodeReal3& nodes_coord_var(pmesh->nodesCoordinates());
661 bool has_map = mesh_info.node_coords.empty();
662 if (has_map){
663 ENUMERATE_NODE (i, mesh->ownNodes()) {
664 Node node = *i;
665 nodes_coord_var[node] = mesh_info.node_coords_map.lookupValue(node.uniqueId().asInt64());

670 ENUMERATE_NODE (node, mesh->allNodes()) {

680void MshMeshReader::
681_allocateGroups(IMesh* mesh, MeshInfo& mesh_info, bool is_read_items)
682{
683 Int32 mesh_dim = mesh->dimension();
684 Int32 face_dim = mesh_dim - 1;
685 for( MeshV4ElementsBlock& block : mesh_info.blocks ){
686 Int32 block_index = block.index;
687 Int32 block_dim = block.dimension;
688 // On alloue un groupe s'il a un nom physique associé.
689 // Pour cela, il faut déjà qu'il soit associé à une entité.
690 Int32 block_entity_tag = block.entity_tag;
691 if (block_entity_tag<0){
692 info(5) << "[Groups] Skipping block index=" << block_index << " because it has no entity";
693 continue;
694 }
695 // Pour l'instant on ne traite pas les nuages
696 if (block_dim==0){
697 info(5) << "[Groups] Skipping block index=" << block_index
698 << " because NodeGroup is not yet supported for this format";
699 continue;
700 }
701 MeshV4EntitiesWithNodes* entity = mesh_info.findEntities(block_dim,block_entity_tag);
702 if (!entity){
703 info(5) << "[Groups] Skipping block index=" << block_index
704 << " because entity tag is invalid";
705 continue;
706 }
707 Int32 entity_physical_tag = entity->physical_tag;
708 MeshPhysicalName physical_name = mesh_info.physical_name_list.find(block_dim,entity_physical_tag);
709 if (physical_name.isNull()){
710 info(5) << "[Groups] Skipping block index=" << block_index
711 << " because entity physical tag is invalid";
712 continue;
713 }
714 info(4) << "[Groups] Block index=" << block_index << " dim=" << block_dim
715 << " name='" << physical_name.name << "'";
716 if (block_dim==mesh_dim){
717 if (is_read_items)
718 _addCellGroup(mesh,block,physical_name.name);
719 else
720 mesh->cellFamily()->findGroup(physical_name.name,true);
721 }
722 else if (block_dim==face_dim){
723 if (is_read_items)
724 _addFaceGroup(mesh,block,physical_name.name);
725 else
726 mesh->faceFamily()->findGroup(physical_name.name,true);
727 }
728 else{
729 info(4) << "[Groups] Skipping block index=" << block_index
730 << " because its dimension is not yet supported";
731 }
732 }
733}

738void MshMeshReader::
739_addFaceGroup(IMesh* mesh, MeshV4ElementsBlock& block, const String& group_name)
740{
741 const Int32 nb_entity = block.nb_entity;
742
743 // Il peut y avoir plusieurs blocs pour le même groupe.
744 // On récupère le groupe s'il existe déjà.
745 FaceGroup face_group = mesh->faceFamily()->findGroup(group_name,true);
746
747 UniqueArray<Int32> faces_id(nb_entity); // Numéro de la face dans le maillage \a mesh
748
749 const Int32 item_nb_node = block.item_nb_node;
750 const Int32 face_nb_node = nb_entity * item_nb_node;
751
752 UniqueArray<Int64> faces_first_node_unique_id(nb_entity);
753 UniqueArray<Int32> faces_first_node_local_id(nb_entity);
754 UniqueArray<Int64> faces_nodes_unique_id(face_nb_node);
755 Integer faces_nodes_unique_id_index = 0;
756
757 UniqueArray<Int64> orig_nodes_id(item_nb_node);
758 UniqueArray<Integer> face_nodes_index(item_nb_node);
759
760 IItemFamily* node_family = mesh->nodeFamily();
761 ItemInternalList mesh_nodes(node_family->itemsInternal());
762
763 // Réordonne les identifiants des faces pour se conformer à Arcane et retrouver
764 // la face dans le maillage
765 for( Integer i_face=0; i_face<nb_entity; ++i_face ){
766 for( Integer z=0; z<item_nb_node; ++z )
767 orig_nodes_id[z] = block.connectivity[faces_nodes_unique_id_index+z];
768
769 mesh_utils::reorderNodesOfFace2(orig_nodes_id,face_nodes_index);
770 for( Integer z=0; z<item_nb_node; ++z )
771 faces_nodes_unique_id[faces_nodes_unique_id_index+z] = orig_nodes_id[face_nodes_index[z]];
772 faces_first_node_unique_id[i_face] = orig_nodes_id[face_nodes_index[0]];
773 faces_nodes_unique_id_index += item_nb_node;
774 }
775
776 node_family->itemsUniqueIdToLocalId(faces_first_node_local_id,faces_first_node_unique_id);
777
778 faces_nodes_unique_id_index = 0;
779 for( Integer i_face=0; i_face<nb_entity; ++i_face ){
780 const Integer n = item_nb_node;
781 Int64ConstArrayView face_nodes_id(item_nb_node,&faces_nodes_unique_id[faces_nodes_unique_id_index]);
782 Node current_node(mesh_nodes[faces_first_node_local_id[i_face]]);
783 Face face = mesh_utils::getFaceFromNodesUnique(current_node,face_nodes_id);
784
785 if (face.null()){
786 OStringStream ostr;
787 ostr() << "(Nodes:";
788 for( Integer z=0; z<n; ++z )
789 ostr() << ' ' << face_nodes_id[z];
790 ostr() << " - " << current_node.localId() << ")";
791 ARCANE_FATAL("INTERNAL: MeshMeshReader face index={0} with nodes '{1}' is not in node/face connectivity",
792 i_face,ostr.str());
793 }
794 faces_id[i_face] = face.localId();
795
796 faces_nodes_unique_id_index += n;
797 }
798 info(4) << "Adding " << faces_id.size() << " faces from block index=" << block.index
799 << " to group '" << face_group.name() << "'";
800 face_group.addItems(faces_id);
801}

827/*
828 * $PhysicalNames // same as MSH version 2
829 * numPhysicalNames(ASCII int)
830 * dimension(ASCII int) physicalTag(ASCII int) "name"(127 characters max)
831 * ...
832 * $EndPhysicalNames
833 */
834void MshMeshReader::
835_readPhysicalNames(IosFile& ios_file,MeshInfo& mesh_info)
836{
837 String quote_mark = "\"";
838 Int32 nb_name = ios_file.getInteger();
839 info() << "nb_physical_name=" << nb_name;
840 ios_file.getNextLine();
841 for( Int32 i=0; i<nb_name; ++i ){
842 Int32 dim = ios_file.getInteger();
843 Int32 tag = ios_file.getInteger();
844 String s = ios_file.getNextLine();
845 if (dim<0 || dim>3)
846 ARCANE_FATAL("Invalid value for physical name dimension dim={0}",dim);
847 // Les noms des groupes peuvent commencer par des espaces et contiennent
848 // des guillemets qu'il faut supprimer.
849 s = String::collapseWhiteSpace(s);
850 if (s.startsWith(quote_mark))
851 s = s.substring(1);
852 if (s.endsWith(quote_mark))
853 s = s.substring(0,s.length()-1);
854 mesh_info.physical_name_list.add(dim,tag,s);
855 info(4) << "[PhysicalName] index=" << i << " dim=" << dim << " tag=" << tag << " name='" << s << "'";
856 }
857 StringView s = ios_file.getNextLine();
858 if (s!="$EndPhysicalNames")
859 ARCANE_FATAL("found '{0}' and expected '$EndPhysicalNames'",s);
860}

895void MshMeshReader::
896_readEntitiesV4(IosFile& ios_file,MeshInfo& mesh_info)
897{
898 Int32 nb_dim_item[4];
899 nb_dim_item[0]= ios_file.getInteger();
900 nb_dim_item[1] = ios_file.getInteger();
901 nb_dim_item[2] = ios_file.getInteger();
902 nb_dim_item[3] = ios_file.getInteger();
903 info(4) << "[Entities] nb_0d=" << nb_dim_item[0] << " nb_1d=" << nb_dim_item[1]
904 << " nb_2d=" << nb_dim_item[2] << " nb_3d=" << nb_dim_item[3];
905 // Après le format, on peut avoir les entités mais cela est optionnel
906 // Si elles sont présentes, on lit le fichier jusqu'à la fin de cette section.
907 StringView next_line = ios_file.getNextLine();
908 for( Int32 i=0; i<nb_dim_item[0]; ++i ){
909 Int32 tag = ios_file.getInteger();
910 Real x = ios_file.getReal();
911 Real y = ios_file.getReal();
912 Real z = ios_file.getReal();
913 Int32 num_physical_tag = ios_file.getInteger();
914 if (num_physical_tag>1)
915 ARCANE_FATAL("NotImplemented numPhysicalTag>1 (n={0})",num_physical_tag);
916 Int32 physical_tag = -1;
917 if (num_physical_tag==1)
918 physical_tag = ios_file.getInteger();
919 info(4) << "[Entities] point tag=" << tag << " x=" << x << " y=" << y << " z=" << z << " phys_tag=" << physical_tag;
920 mesh_info.entities_nodes_list.add(MeshV4EntitiesNodes(tag,physical_tag));
921 next_line = ios_file.getNextLine();
922 }
923
924 for( Int32 dim=1; dim<=3; ++dim ){
925 for( Int32 i=0; i<nb_dim_item[dim]; ++i ){
926 Int32 tag = ios_file.getInteger();
927 Real min_x = ios_file.getReal();
928 Real min_y = ios_file.getReal();
929 Real min_z = ios_file.getReal();
930 Real max_x = ios_file.getReal();
931 Real max_y = ios_file.getReal();
932 Real max_z = ios_file.getReal();
933 Int32 num_physical_tag = ios_file.getInteger();
934 if (num_physical_tag>1)
935 ARCANE_FATAL("NotImplemented numPhysicalTag>1 (n={0})",num_physical_tag);
936 Int32 physical_tag = -1;
937 if (num_physical_tag==1)
938 physical_tag = ios_file.getInteger();
939 Int32 num_bounding_group = ios_file.getInteger();
940 for( Int32 k=0; k<num_bounding_group; ++k ){
941 [[maybe_unused]] Int32 group_tag = ios_file.getInteger();
942 }
943 mesh_info.entities_with_nodes_list[dim-1].add(MeshV4EntitiesWithNodes(dim,tag,physical_tag));
944 info(4) << "[Entities] dim=" << dim << " tag=" << tag
945 << " min_x=" << min_x << " min_y=" << min_y << " min_z=" << min_z
946 << " max_x=" << max_x << " max_y=" << max_y << " max_z=" << max_z
947 << " phys_tag=" << physical_tag;
948 next_line = ios_file.getNextLine();
949 }
950 }
951 StringView s = ios_file.getNextLine();
952 if (s!="$EndEntities")
953 ARCANE_FATAL("found '{0}' and expected '$EndEntities'",s);
954}

989IMeshReader::eReturnType MshMeshReader::
990_readMeshFromNewMshFile(IMesh* mesh, IosFile& ios_file)
991{
992 const char* func_name = "MshMeshReader::_readMeshFromNewMshFile()";
993 info() << "[_readMeshFromNewMshFile] New native mesh file format detected";
994 MeshInfo mesh_info;
995#define MSH_BINARY_TYPE 1
996
997 Real version = ios_file.getReal();
998 if (version==2.0)
999 m_version = 2;
1000 else if (version==4.1)
1001 m_version = 4;
1002 else
1003 ARCANE_THROW(IOException, "Wrong msh file version '{0}'. Only versions '2.0' or '4.1' are supported",version);
1004 info() << "Msh mesh_major_version=" << m_version;
1005 Integer file_type = ios_file.getInteger(); // is an integer equal to 0 in the ASCII file format, equal to 1 for the binary format
1006 if (file_type == MSH_BINARY_TYPE)
1007 ARCANE_THROW(IOException, "Binary mode is not supported!");
1008
1009 Integer data_size = ios_file.getInteger(); // is an integer equal to the size of the floating point numbers used in the file
1010 ARCANE_UNUSED(data_size);
1011
1012 if (file_type == MSH_BINARY_TYPE) {
1013 (void)ios_file.getInteger(); // is an integer of value 1 written in binary form
1014 }
1015 ios_file.getNextLine(); // Skip current \n\r
1016
1017 // $EndMeshFormat
1018 if (!ios_file.lookForString("$EndMeshFormat"))
1019 ARCANE_THROW(IOException, "$EndMeshFormat not found");
1020
1021 // TODO: Les différentes sections ($Nodes, $Entitites, ...) peuvent
1022 // être dans n'importe quel ordre (à part $Nodes qui doit être avant $Elements)
1023 // Faire un méthode qui gère cela.
1024
1025 StringView next_line = ios_file.getNextLine();
1026 // Noms des groupes
1027 if (next_line=="$PhysicalNames"){
1028 _readPhysicalNames(ios_file,mesh_info);
1029 next_line = ios_file.getNextLine();
1030 }
1031 // Après le format, on peut avoir les entités mais cela est optionnel
1032 // Si elles sont présentes, on lit le fichier jusqu'à la fin de cette section.
1033 if (next_line=="$Entities"){
1034 _readEntitiesV4(ios_file,mesh_info);
1035 next_line = ios_file.getNextLine();
1036 }
1037 // $Nodes
1038 if (next_line!="$Nodes")
1039 ARCANE_THROW(IOException,"Unexpected string '{0}'. Valid values are '$Nodes'",next_line);
1040
1041 // Fetch nodes number and the coordinates
1042 if (file_type != MSH_BINARY_TYPE) {
1043 if (m_version==2){
1044 if (_readNodesFromAsciiMshV2File(ios_file, mesh_info.node_coords) != RTOk)
1045 ARCANE_THROW(IOException, "Ascii nodes coords error");
1046 }
1047 else if (m_version==4){
1048 if (_readNodesFromAsciiMshV4File(ios_file, mesh_info) != RTOk)
1049 ARCANE_THROW(IOException, "Ascii nodes coords error");
1050 }
1051 }
1052 else {
1053 if (_readNodesFromBinaryMshFile(ios_file, mesh_info.node_coords) != RTOk)
1054 ARCANE_THROW(IOException, "Binary nodes coords error");
1055 }
1056 // $EndNodes
1057 if (!ios_file.lookForString("$EndNodes"))
1058 ARCANE_THROW(IOException, "$EndNodes not found");
1059
1060 // $Elements
1061 Integer mesh_dimension = -1;
1062 {
1063 if (!ios_file.lookForString("$Elements"))
1064 ARCANE_THROW(IOException, "$Elements not found");
1065
1066 if (m_version==2)
1067 mesh_dimension = _readElementsFromAsciiMshV2File(ios_file,mesh_info);
1068 else if (m_version==4)
1069 mesh_dimension = _readElementsFromAsciiMshV4File(ios_file,mesh_info);
1070
1071 // $EndElements
1072 if (!ios_file.lookForString("$EndElements"))
1073 throw IOException(func_name, "$EndElements not found");
1074 }
1075
1076 info() << "Computed mesh dimension = " << mesh_dimension;
1077
1078 IPrimaryMesh* pmesh = mesh->toPrimaryMesh();
1079 pmesh->setDimension(mesh_dimension);
1080
1081 IParallelMng* pm = mesh->parallelMng();
1082 bool is_parallel = pm->isParallel();
1083 Int32 rank = mesh->meshPartInfo().partRank();
1084 // En parallèle, seul le rang 0 lit le maillage
1085 bool is_read_items = !(is_parallel && rank != 0);
1086 _allocateCells(mesh, mesh_info, is_read_items);
1087 _allocateGroups(mesh, mesh_info, is_read_items);
1088 return RTOk;
1089}
