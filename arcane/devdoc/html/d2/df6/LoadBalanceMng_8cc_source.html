186 Real _computeMemContrib(const ItemVectorViewT<ItemKind> list)
187 {
188 Real contrib = 0;

190 for ( ; iterator() ; ++iterator) {

236// Calcul de la consommation mémoire pour chaque type d'entité.
237// Les mailles bénéficient ensuite des contributions des autres entités adjacentes.
238void CriteriaMng::
239computeMemory()
240{
241 int lenght = m_familyName.size();
242 m_overallMem.resize(lenght);
243 m_residentMem.resize(lenght);
244 m_buffer.id = -1;
245 m_overallMem.fill(0);
246 m_residentMem.fill(0);
247
248 // For each variable, compute the size for one object.
249 for (VariableCollectionEnumerator vc(m_varMng->usedVariables()) ; ++vc ; ) {
250 IVariable* var = *vc;
251 Integer memory = 0;
252 try {
253 if (var->dataType()!=DT_String)
254 memory = dataTypeSize(var->dataType());
255 }
256 catch (const ArgumentException&) {
257 memory = 0; // Cannot know memory used for that ...
258 continue;
259 }
260 Integer family = -1;
261 Integer kind = var->itemKind();
262 if (kind == IK_Particle) { // Not the same counter for all items
263 family = _findEntity(var->itemFamilyName());
264 if (family >= 0) {
265 m_overallMem[family] += memory;
266 }
267 }
268 m_overallMem[kind] += memory;
269
270 int properties = var->property();
271 if ((properties & IVariable::PNoExchange) ||
272 (properties & IVariable::PNoNeedSync) ||
273 (properties & IVariable::PTemporary) ||
274 (properties & IVariable::PSubDomainPrivate)) {
275 continue;
276 }
277 m_residentMem[kind] += memory;
278 if (family >= 0) {
279 m_residentMem[family] += memory;
280 }
281 }
282}

325// Pour les mailles, on calcule la contribution mémoire des noeuds, aretes et faces.
326void CriteriaMng::
327_computeMemCell(const Cell& cell) {
328 Real contrib;
329 if (cell.localId() == m_buffer.id) // already computed
330 return;
331 m_buffer.id = cell.localId();
332 m_buffer.overallMem = m_overallMem[IK_Cell];
333 m_buffer.residentMem = m_residentMem[IK_Cell];
334
335 contrib = _computeMemContrib<Node>(cell.nodes());
336 m_buffer.overallMem += contrib*m_overallMem[IK_Node];
337 m_buffer.residentMem += contrib*m_residentMem[IK_Node];
338
339 contrib = _computeMemContrib<Face>(cell.faces());
340 m_buffer.overallMem += contrib*m_overallMem[IK_Face];
341 m_buffer.residentMem += contrib*m_residentMem[IK_Face];
342
343 contrib = _computeMemContrib<Edge>(cell.edges());
344 m_buffer.overallMem += contrib*m_overallMem[IK_Edge];
345 m_buffer.residentMem += contrib*m_residentMem[IK_Edge];
346
347}

394void LoadBalanceMng::
395initAccess(IMesh* mesh)
396{
397 m_cell_new_owner = std::make_unique<VariableCellInt32>(VariableBuildInfo(m_mesh_handle, "CellFamilyNewOwner", IVariable::PExecutionDepend | IVariable::PNoDump));
398 if (!mesh)
399 m_mesh_handle = mesh->handle();
400 int vflags = IVariable::PExecutionDepend|IVariable::PNoDump|IVariable::PTemporary;
401 m_comm_costs = new VariableFaceReal(VariableBuildInfo(m_mesh_handle, "LbMngCommCost", vflags));
402 m_mass_over_weigth = new VariableCellReal(VariableBuildInfo(m_mesh_handle, "LbMngOverallMass",vflags));
403 m_mass_res_weight = new VariableCellReal(VariableBuildInfo(m_mesh_handle, "LbMngResidentMass",vflags));
404 m_event_weights = new VariableCellArrayReal(VariableBuildInfo(m_mesh_handle, "LbMngMCriteriaWgt", vflags));
405 m_comm_costs->fill(1);
406 m_mass_over_weigth->fill(1);
407 m_mass_res_weight->fill(1);
408
409 if (m_compute_comm || m_mass_criterion) {// Memory useful only for communication cost or mass lb criterion
410 m_criteria->computeMemory();
411 _computeResidentMass();
412 }
413 if (m_compute_comm)
414 _computeComm();
415 if (m_mass_criterion)
416 _computeOverallMass();
417 _computeEvents();
418}

522
523void LoadBalanceMng::
524_computeComm()
525{
526 IMesh* mesh = m_mesh_handle.mesh();
527 Integer penalty=2; // How many times we do synchronization ?
528
529 if (m_comm_vars.size())
530 m_comm_costs->fill(0);
531
532 for (int i = 0 ; i < m_comm_vars.size() ; ++i) {
533 StoreIProxyItemVariable& commvar = m_comm_vars[i];
534 ENUMERATE_FACE(iface, mesh->ownFaces()) {
535 (*m_comm_costs)[iface] += commvar[iface]*m_criteria->getResidentMemory(commvar.getPos());
536 }
537 }
538 if (cellCommContrib()) {
539 ENUMERATE_CELL(icell, mesh->ownCells()) {
540 Real mem = (*m_mass_res_weight)[icell];
541 ENUMERATE_FACE(iface, icell->faces()) {
542 (*m_comm_costs)[iface] += mem*penalty;
543 }
544 }
545 }
546
547 IVariable *ivar;
548 // Make sure that ghosts contribution is used
549 ivar = m_comm_costs->variable();
550 ivar->itemFamily()->reduceFromGhostItems(ivar,Parallel::ReduceSum);
551 m_comm_costs->synchronize();
552}

555/*---------------------------------------------------------------------------*/
556
557void LoadBalanceMng::
558_computeEvents()
559{
560 m_event_weights->resize(nbCriteria());
561
562 ArrayView<StoreIProxyItemVariable> eventVars = m_event_vars;
563 if (m_mass_criterion) {
564 StoreIProxyItemVariable cvar(m_mass_over_weigth->variable());
565 m_event_vars[0] = cvar;
566 }
567 else {
568 eventVars = m_event_vars.subView(1, m_event_vars.size());
569 }
570
571 for (Integer i = 0 ; i < eventVars.size() ; ++i) {
572 ENUMERATE_CELL(icell, m_mesh_handle.mesh()->ownCells()) {
573 Integer count=i;
574 if (m_nb_criterion) {
575 count += 1;
576 (*m_event_weights)[icell][0] = 1;
577 }
578 (*m_event_weights)[icell][count] = eventVars[i][icell];
579 }
580 }
581}
