99void CellDirectionMng::
100_internalComputeInnerAndOuterItems(const ItemGroup& items)
101{
102 Int32UniqueArray inner_lids;
103 Int32UniqueArray outer_lids;
104 IItemFamily* family = items.itemFamily();
105 ENUMERATE_ITEM(iitem,items){
106 Int32 lid = iitem.itemLocalId();
107 ItemInternal* i1 = m_infos[lid].m_next_item;
108 ItemInternal* i2 = m_infos[lid].m_previous_item;
109 if (i1->null() || i2->null())
110 outer_lids.add(lid);
111 else
112 inner_lids.add(lid);
113 }
114 int dir = (int)m_direction;
115 String base_group_name = String("Direction")+dir;
116 if (m_p->m_patch_index>=0)
117 base_group_name = base_group_name + String("AMRPatch")+m_p->m_patch_index;
118 m_p->m_inner_all_items = family->createGroup(String("AllInner")+base_group_name,inner_lids,true);
119 m_p->m_outer_all_items = family->createGroup(String("AllOuter")+base_group_name,outer_lids,true);
120 m_p->m_all_items = items;
121 m_items = family->itemsInternal();
122}

154void CellDirectionMng::
155setNodesIndirection(Int32ConstArrayView nodes_indirection)
156{
157 for( Integer i=0; i<MAX_NB_NODE; ++i )
158 m_nodes_indirection[i] = nodes_indirection[i];
159
160 ITraceMng* tm = m_p->m_cartesian_mesh->traceMng();
161
162 tm->info() << "Set computed indirection dir=" << (int)m_direction;
163 for( Integer i=0; i<MAX_NB_NODE; ++i ){
164 tm->info() << "Indirection i=" << i << " v=" << m_nodes_indirection[i];
