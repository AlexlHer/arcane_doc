144
145void SerializedData::
146_copyExtentsToDimensions()
147{
148 Integer n = m_extents.size();
149 m_dimensions.resize(n);
150 // Il ne faut pas lever d'exceptions si on dépasse les bornes sinon
151 // le code lèvera une exception dès que le nombre d'éléments du tableau
152 // dépasse 32 bits. Cela n'est pas très grave si les valeurs de 'm_dimensions'
153 // ne sont pas valide car ce n'est plus utilisé que dans computeHash() pour
154 // garder la valeur compatible.
155 for( Integer i=0; i<n; ++i )
156 m_dimensions[i] = static_cast<Int32>(m_extents[i]);
157}

217void SerializedData::
218computeHash(IHashAlgorithm* algo,ByteArray& output) const
219{
220 // TODO: faire avec le support 64 bits mais cela change le hash.
221 algo->computeHash(m_const_buffer.constSmallView(),output);
222 const Byte* ptr = reinterpret_cast<const Byte*>(m_dimensions.data());
223 Integer msize = CheckedConvert::multiply(m_dimensions.size(),(Integer)sizeof(Integer));
224 ByteConstArrayView dim_bytes(msize,ptr);
225 algo->computeHash(dim_bytes,output);
226}

245void SerializedData::
246serialize(ISerializer* sbuf)
247{
248 ISerializer::eMode mode = sbuf->mode();
249
250 switch(mode){
251 case ISerializer::ModeReserve:
252 _serialize(sbuf);
253 break;
254 case ISerializer::ModePut:
255 _serialize(sbuf);
256 break;
257 case ISerializer::ModeGet:
258 switch(sbuf->readMode()){
259 case ISerializer::ReadReplace:
260 {
261 m_base_data_type = (eDataType)sbuf->getInteger(); // Pour le m_base_data_type
262 m_memory_size = sbuf->getInt64(); // Pour le m_memory_size
263 m_nb_dimension = sbuf->getInteger(); // Pour le m_nb_dimension
264 m_nb_element = sbuf->getInt64(); // Pour le m_nb_element
265 m_nb_base_element = sbuf->getInt64(); // Pour le m_nb_base_element
266 m_is_multi_size = sbuf->getInteger(); // Pour le m_is_multi_size
267 m_element_size = sbuf->getInt64(); // Pour le m_element_size
268
269 Int64 dimensions_size = sbuf->getInt64();
270 m_extents.resize(dimensions_size);
271 sbuf->getSpan(m_extents);
272 _copyExtentsToDimensions();
273
274 Int64 buffer_size = sbuf->getInt64();
275 m_stored_buffer.resize(buffer_size);
276 sbuf->getSpan(m_stored_buffer); // Pour les données
277 m_buffer = m_stored_buffer;
278 m_const_buffer = m_buffer;
279 }
280 break;
281 case ISerializer::ReadAdd:
282 ARCANE_THROW(NotImplementedException,"ReadAdd");
283 break;
284 }
285 break;
286 }
287}

297void SerializedData::
298_serialize(ISerializer* sbuf) const
299{
300 ISerializer::eMode mode = sbuf->mode();
301 if (m_extents.size()!=m_dimensions.size())
302 ARCANE_FATAL("Incoherence between extents ({0}) and dimensions ({1})",
303 m_extents.size(),m_dimensions.size());
304 switch(mode){
305 case ISerializer::ModeReserve:
306 sbuf->reserveInteger(1); // Pour le m_base_data_type
307 sbuf->reserve(DT_Int64,1); // Pour le m_memory_size
308 sbuf->reserveInteger(1); // Pour le m_nb_dimension
309 sbuf->reserve(DT_Int64,1); // Pour le m_nb_element
310 sbuf->reserve(DT_Int64,1); // Pour le m_nb_base_element
311 sbuf->reserveInteger(1); // Pour le m_is_multi_size
312 sbuf->reserve(DT_Int64,1); // Pour le m_element_size
313
314 sbuf->reserve(DT_Int64,1); // Pour le m_extents.size()
315 sbuf->reserveSpan(DT_Int64,m_extents.size()); // Pour les dimensions
316
317 sbuf->reserve(DT_Int64,1); // Pour le m_const_buffer.size()
318 sbuf->reserveSpan(DT_Byte,m_const_buffer.size()); // Pour les données
319 break;
320 case ISerializer::ModePut:
321 sbuf->putInteger(m_base_data_type); // Pour le m_base_data_type
322 sbuf->putInt64(m_memory_size); // Pour le m_memory_size
323 sbuf->putInteger(m_nb_dimension); // Pour le m_nb_dimension
324 sbuf->putInt64(m_nb_element); // Pour le m_nb_element
325 sbuf->putInt64(m_nb_base_element); // Pour le m_nb_base_element
326 sbuf->putInteger(m_is_multi_size); // Pour le m_is_multi_size
327 sbuf->putInt64(m_element_size); // Pour le m_element_size
328
329 sbuf->putInt64(m_extents.size()); // Pour le m_extents.size()
330 sbuf->putSpan(m_extents); // Pour les dimensions
331
332 sbuf->putInt64(m_const_buffer.size()); // Pour le m_const_buffer.size()
333 sbuf->putSpan(m_const_buffer); // Pour les données
334 break;
335 case ISerializer::ModeGet:
336 ARCANE_FATAL("This methode does not handle ModeGet");
337 }
338}
