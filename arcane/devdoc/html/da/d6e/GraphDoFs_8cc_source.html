58void GraphDoFs::
59_allocateGraph()
60{
61 if (m_graph_allocated)
62 return;
63
64 auto connectivity_index = 0;
65 m_connectivity_indexes_per_type.resize(NB_BASIC_ITEM_TYPE, -1);
66 m_connectivity_indexes_per_type[IT_DualNode] = connectivity_index++;
67 m_connectivity_indexes_per_type[IT_DualEdge] = connectivity_index++;
68 m_connectivity_indexes_per_type[IT_DualFace] = connectivity_index++;
69 m_connectivity_indexes_per_type[IT_DualCell] = connectivity_index++;
70 m_connectivity_indexes_per_type[IT_DualParticle] = connectivity_index++;
71
72 m_item_family_network = m_mesh->itemFamilyNetwork();
73 if (m_item_family_network == nullptr)
74 traceMng()->fatal() << "ARCANE_GRAPH_CONNECTIVITY_POLICY need to be activated";
75
76 m_links_incremental_connectivity =
77 new IncrementalItemConnectivity(linkFamily(),
78 dualNodeFamily(),
79 mesh::connectivityName(linkFamily(), dualNodeFamily()));
80 if (m_item_family_network)
81 //m_item_family_network->addRelation(linkFamily(),dualNodeFamily(),m_links_incremental_connectivity);
82 m_item_family_network->addDependency(linkFamily(), dualNodeFamily(), m_links_incremental_connectivity, false);
83
84 m_incremental_connectivities.resize(NB_DUAL_ITEM_TYPE, nullptr);
85 std::array<int, NB_DUAL_ITEM_TYPE> dual_node_kinds = { IT_DualCell, IT_DualFace, IT_DualEdge, IT_DualNode, IT_DualParticle };
86 for (auto dual_node_kind : dual_node_kinds) {
87 IItemFamily* dual_item_family = _dualItemFamily(dualItemKind(dual_node_kind));
88 if (dual_item_family) {
89 auto dof2dual_incremental_connectivity =
90 new IncrementalItemConnectivity(dualNodeFamily(),
91 dual_item_family,
92 mesh::connectivityName(dualNodeFamily(), dual_item_family));
93 m_incremental_connectivities[_connectivityIndex(dual_node_kind)] = dof2dual_incremental_connectivity;
94 if (m_item_family_network)
95 m_item_family_network->addDependency(dualNodeFamily(), dual_item_family, dof2dual_incremental_connectivity, false);
96 }
97 }
98 m_graph_connectivity.reset(new GraphIncrementalConnectivity(dualNodeFamily(),
99 linkFamily(),
100 m_links_incremental_connectivity,
101 m_incremental_connectivities,
102 m_dual_node_to_connectivity_index));
103 m_graph_allocated = true;
104}

109void GraphDoFs::
110addLinks(Integer nb_link,
111 Integer nb_dual_nodes_per_link,
112 Int64ConstArrayView links_infos)
113{
114 Trace::Setter mci(traceMng(), _className());
115 if (!m_graph_allocated)
116 _allocateGraph();
117
118 mesh::DoFFamily& link_family = m_dof_mng.family(GraphDoFs::linkFamilyName());
119 mesh::DoFFamily& dual_node_family = m_dof_mng.family(GraphDoFs::dualNodeFamilyName());
120
121 // Extract link infos
122 Int64UniqueArray link_uids, connected_dual_node_uids;
123 link_uids.reserve(nb_link);
124 connected_dual_node_uids.reserve(nb_link * nb_dual_nodes_per_link);
125 for (auto links_infos_index = 0; links_infos_index < links_infos.size();) {
126 auto link_uid = links_infos[links_infos_index++];
127 link_uids.add(link_uid);
128 connected_dual_node_uids.addRange(
129 links_infos.subConstView(links_infos_index, nb_dual_nodes_per_link));
130 links_infos_index += nb_dual_nodes_per_link;
131 }
132
133 Int32UniqueArray link_lids(link_uids.size());
134 link_family.addDoFs(link_uids, link_lids);
135 link_family.endUpdate();
136
137 // resize connectivity
138 // fill connectivity
139 Int32UniqueArray connected_dual_nodes_lids(nb_link * nb_dual_nodes_per_link);
140 dual_node_family.itemsUniqueIdToLocalId(
141 connected_dual_nodes_lids.view(), connected_dual_node_uids.constView(), true);
142
143 auto link_index = 0;
144 ENUMERATE_DOF (inewlink, link_family.view(link_lids)) {
145 m_links_incremental_connectivity->notifySourceItemAdded(ItemLocalId(*inewlink));
146 for (auto lid : connected_dual_nodes_lids.subConstView(link_index, nb_dual_nodes_per_link))
147 m_links_incremental_connectivity->addConnectedItem(ItemLocalId(*inewlink), ItemLocalId(lid));
148 link_index += nb_dual_nodes_per_link;
149 }
150 m_update_sync_info = true;
151}

156void GraphDoFs::
157addDualNodes(Integer graph_nb_dual_node,
158 Integer dual_node_kind,
159 Int64ConstArrayView dual_nodes_infos)
160{
161
162 Trace::Setter mci(traceMng(), _className());
163 if (!m_graph_allocated)
164 _allocateGraph();
165
166 // Size m_connecitivities if not yet done
167
168 Int64UniqueArray dual_node_uids, dual_item_uids;
169 dual_node_uids.reserve(graph_nb_dual_node);
170 dual_item_uids.reserve(graph_nb_dual_node);
171 for (auto infos_index = 0; infos_index < dual_nodes_infos.size();) {
172 dual_node_uids.add(dual_nodes_infos[infos_index++]);
173 dual_item_uids.add(dual_nodes_infos[infos_index++]);
174 }
175
176 Int32UniqueArray dual_node_lids(dual_node_uids.size());
177 auto& dual_node_family = m_dof_mng.family(GraphDoFs::dualNodeFamilyName());
178 dual_node_family.addDoFs(dual_node_uids, dual_node_lids);
179 dual_node_family.endUpdate();
180
181 IItemFamily* dual_item_family = _dualItemFamily(dualItemKind(dual_node_kind));
182
183 auto incremental_dual_item_connectivity = m_incremental_connectivities[_connectivityIndex(dual_node_kind)];
184 Int32UniqueArray dual_item_lids(dual_item_uids.size());
185 dual_item_family->itemsUniqueIdToLocalId(dual_item_lids, dual_item_uids);
186
187 ENUMERATE_DOF (idual_node, dual_node_family.view(dual_node_lids)) {
188 incremental_dual_item_connectivity->notifySourceItemAdded(ItemLocalId(*idual_node));
189 incremental_dual_item_connectivity->addConnectedItem(ItemLocalId(*idual_node), ItemLocalId(dual_item_lids[idual_node.index()]));
190 }
191
192 m_dual_node_to_connectivity_index.resize(&dual_node_family, _connectivityIndex(dual_node_kind));
193
194 m_update_sync_info = true;
195}

197void GraphDoFs::
198addDualNodes(Integer graph_nb_dual_node,
199 Int64ConstArrayView dual_nodes_infos)
200{
201
202 Trace::Setter mci(traceMng(), _className());
203 if (!m_graph_allocated)
204 _allocateGraph();
205
206 // Size m_connecitivities if not yet done
207
208 std::map<Int64, std::pair<Int64UniqueArray, Int64UniqueArray>> dual_info_per_kind;
209 for (auto infos_index = 0; infos_index < dual_nodes_infos.size();) {
210 Int64 dual_node_kind = dual_nodes_infos[infos_index++];
211 auto& info = dual_info_per_kind[dual_node_kind];
212 auto& dual_node_uids = info.first;
213 auto& dual_item_uids = info.second;
214 if (dual_node_uids.size() == 0) {
215 dual_node_uids.reserve(graph_nb_dual_node);
216 dual_item_uids.reserve(graph_nb_dual_node);
217 }
218 dual_node_uids.add(dual_nodes_infos[infos_index++]);
219 dual_item_uids.add(dual_nodes_infos[infos_index++]);
220 }
221
222 for (Integer index = 0; index < NB_DUAL_ITEM_TYPE; ++index) {
223 Integer dual_node_kind = m_dualnode_kinds[index];
224 auto& info = dual_info_per_kind[dual_node_kind];
225 auto& dual_node_uids = info.first;
226 auto& dual_item_uids = info.second;
227
228 Int32UniqueArray dual_node_lids(dual_node_uids.size());
229 auto& dual_node_family = m_dof_mng.family(GraphDoFs::dualNodeFamilyName());
230 dual_node_family.addDoFs(dual_node_uids, dual_node_lids);
231 dual_node_family.endUpdate();
232
233 auto incremental_dual_item_connectivity = m_incremental_connectivities[index];
234 IItemFamily* dual_item_family = _dualItemFamily(dualItemKind(dual_node_kind));
235 if (dual_item_family) {
236 Int32UniqueArray dual_item_lids(dual_item_uids.size());
237 dual_item_family->itemsUniqueIdToLocalId(dual_item_lids, dual_item_uids);
238
239 ENUMERATE_DOF (idual_node, dual_node_family.view(dual_node_lids)) {
240 incremental_dual_item_connectivity->notifySourceItemAdded(ItemLocalId(*idual_node));
241 incremental_dual_item_connectivity->addConnectedItem(ItemLocalId(*idual_node), ItemLocalId(dual_item_lids[idual_node.index()]));
242 }
243
244 m_dual_node_to_connectivity_index.resize(&dual_node_family, _connectivityIndex(dual_node_kind));
245 }
246 }
247 m_update_sync_info = true;
248}

293void GraphDoFs::
294updateAfterMeshChanged()
295{
296 if (!m_graph_allocated)
297 return;
298
299 auto& dual_node_family = m_dof_mng.family(GraphDoFs::dualNodeFamilyName());
300 m_dual_node_to_connectivity_index.resize(&dual_node_family, -1);
301 ENUMERATE_DOF (idof, dual_node_family.allItems()) {
302 for (Integer index = 0; index < m_incremental_connectivities.size(); ++index) {
303 auto connectivity = m_incremental_connectivities[index];
304 if (connectivity) {
305 ConnectivityItemVector accessor(connectivity);
306 if (accessor.connectedItems(*idof).size() > 0) {
307 m_dual_node_to_connectivity_index[*idof] = index;
308 }
309 }
310 }
311 }
312
313 {
314 // TODO: GG: appeler endUpdate() à la place pour éviter une duplication de code
315 auto* x = new GraphIncrementalConnectivity(dualNodeFamily(),
316 linkFamily(),
317 m_links_incremental_connectivity,
318 m_incremental_connectivities,
319 m_dual_node_to_connectivity_index);
320 m_graph_connectivity.reset(x);
321 }
322}
