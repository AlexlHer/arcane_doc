370extern "C++" ISubDomain*
371arcaneCreateSubDomain(ISession* session,const SubDomainBuildInfo& sdbi)
372{
373 Ref<IParallelMng> pm = sdbi.parallelMng();
374 Ref<IParallelMng> all_replica_pm = sdbi.allReplicaParallelMng();
375 String case_file_name = sdbi.caseFileName();
376 ByteConstArrayView bytes = sdbi.caseBytes();
377
378 ITraceMng* tm = pm->traceMng();
379 StringBuilder trace_id;
380 trace_id += String::fromNumber(pm->commRank());
381 if (all_replica_pm!=pm){
382 trace_id += ",r";
383 trace_id += pm->replication()->replicationRank();
384 }
385 trace_id += ",";
386 trace_id += platform::getHostName();
387 tm->setTraceId(trace_id.toString());
388
389 SubDomain* sd = new SubDomain(session,pm,all_replica_pm,case_file_name,bytes);
390 sd->build();
391 //GG: l'init se fait par l'appelant
392 //mng->initialize();
393 global_sub_domain = sd;
394 return sd;
395}

457void SubDomain::
458initialize()
459{
460 // Initialisation du module parallèle
461 // TODO: a supprimer car plus utile
462 m_parallel_mng->initialize();
463
464 {
465 // Initialise le runner par défaut en fonction des paramètres donnés par
466 // l'utilisateur.
467 IApplication* app = application();
468 m_accelerator_mng->initialize(app->acceleratorRuntimeInitialisationInfo());
469 }
470
471 IMainFactory* mf = m_application->mainFactory();
472
473 m_time_loop_mng->build();
474
475 m_has_mesh_service = false;
476
477 // Lit le jeu de données.
478 // Il s'agit uniquement de s'assurer de la validité syntaxique
479 // du jeu de données. Les options ne sont pas lues.
480 // Cela doit être fait avant la création du maillage
481 // car l'instance du maillage dépend du jeu de données
482 ICaseDocument* case_document = nullptr;
483 if (!m_case_bytes.empty()){
484 case_document = m_case_mng->readCaseDocument(m_case_full_file_name,m_case_bytes);
485 if (!case_document)
486 ARCANE_FATAL("Can not read case options");
487 // Ajoute à la configuration celle présente dans le jeu de données.
488 {
489 ConfigurationReader cr(traceMng(),m_configuration.get());
490 cr.addValuesFromXmlNode(case_document->configurationElement(),ConfigurationReader::P_CaseDocument);
491 }
492 }
493
494 properties::readFromConfiguration(m_configuration.get(),*this);
495
496 if (case_document){
497 // Regarde s'il y a un tag <meshes> indiquant que la création et
498 // la lecture du maillage sera gérée par un service
499 XmlNode meshes_elem = case_document->meshesElement();
500 if (!meshes_elem.null()){
501 info() << "Using mesh service to create and allocate meshes";
502 m_has_mesh_service = true;
503 }
504 // TODO: dans ce cas vérifier qu'on n'a pas l'élément 'maillage' historique.
505 }
506 // Créé le maillage par défaut. Il faut le faire avant la création
507 // des services car ces derniers peuvent en avoir besoin.
508 if (!m_has_mesh_service)
509 if (!m_is_create_default_mesh_v2)
510 m_legacy_mesh_builder->createDefaultMesh();
511
512 ScopedPtrT<IServiceLoader> service_loader(mf->createServiceLoader());
513 service_loader->loadSubDomainServices(this);
514
515 // Le module maître doit toujours être créé avant les autres pour que
516 // ses points d'entrées soient appelés en premier et dernier
517 m_module_master = mf->createModuleMaster(this);
518 m_time_history_mng = mf->createTimeHistoryMng(this);
519
520 service_loader->initializeModuleFactories(this);
521 m_lb_mng = mf->createLoadBalanceMng(this);
522
523 m_property_mng_checkpoint = new PropertyMngCheckpoint(this);
524
525 m_observers.addObserver(this,
526 &SubDomain::_notifyWriteCheckpoint,
527 m_checkpoint_mng->writeObservable());
528 }

543{

545
546 m_property_mng_checkpoint = nullptr;
547
548 m_on_destroy_observable.notifyAllObservers();
549 m_on_destroy_observable.detachAllObservers();
550
551 platform::callDotNETGarbageCollector();
552
553 // Normalement on devrait pouvoir supprimer ce test car il ne devrait plus
554 // rester de références sur des services ou module. Cela est le cas
555 // avec l'implémentation 'coreclr' mais pas avec 'mono'. Du coup on
556 // laisse pour l'instant ce test.
557 if (m_application->hasGarbageCollector())
558 return;
559
560 m_module_master = nullptr;
561
562 m_module_mng->removeAllModules();
563 m_service_mng = nullptr;
564 m_time_history_mng = nullptr;
565 m_mesh_mng->destroyMeshes();
566
567 m_time_loop_mng = nullptr;
568 m_case_mng = nullptr;
569 m_entry_point_mng = nullptr;
570 m_module_mng = nullptr;
571 m_io_mng = nullptr;
572
573 m_lb_mng = nullptr;
574 m_property_mng.reset();
575
576 // Comme tous les objets peuvent contenir des variables, il faut être
577 // certain de détruire le gestionnaire de variables en dernier.
578 m_variable_mng->removeAllVariables();
579 m_variable_mng = nullptr;
580
581 delete this;
582}

606allocateMeshes()
607{
608 Timer::Action ts_action2(this,"AllocateMesh");
609 Trace::Setter mci(traceMng(),_msgClassName());
610
611 // Vérifie que personne n'a fait d'appel à addMesh() avant cette méthode
612 // car les maillages seront déréférencés.
613 // NOTE: on garde cela pour des raisons de compatibilité mais il serait
614 // préférable d'empêcher les appels à addMesh() avant d'avoir alloué
615 // les maillages.
616 // NOTE (2): depuis l'utilisation des MeshHandle (juin 2019), les maillages
617 // déjà créés ne sont plus supprimés lors de l'appel à cette fonction
618 // donc il n'y a pas besoin de faire un fatal si des maillages existent déjà.
619 // On laisse le code en cas de problème mais si tout est OK on le supprimera

764void SubDomain::
765_doInitialPartition()
766{
767 // NOTE: Ce code n'est utilisé que par le mécanisme historique de création
768 // de maillage. Lorsqu'on utilise les maillages pour les services, le code
769 // correspondant est dans ArcaneCaseMeshService.

774 bool is_mesh_allocated = mesh->isAllocated();
775 info() << "InitialPartitioning mesh=" << mesh->name() << " is_allocated?=" << is_mesh_allocated;
776 if (!is_mesh_allocated)
777 continue;
778 // Comme 'parmetis' n'aime pas repartitionner un maillage si un des sous-domaines
779 // est vide (ce qui est le cas si un seul processeur créé les mailles), il
780 // faut d'abord utiliser le partitionneur basique qui répartit les mailles
781 // entre les sous-domaines, puis 'parmetis' s'il est présent
782 if (m_legacy_mesh_builder->m_use_partitioner_tester) {
783 Int64 nb_cell = mesh->nbCell();
784 Int64 min_nb_cell = parallelMng()->reduce(Parallel::ReduceMin,nb_cell);
785 info() << "Min nb cell=" << min_nb_cell;
786 if (min_nb_cell==0)
787 _doInitialPartitionForMesh(mesh,test_service,true);
788 else
789 info() << "Mesh name=" << mesh->name() << " have cells. Do not use " << test_service;
790 }
791 else
792 info() << "No basic partition first needed";
793 _doInitialPartitionForMesh(mesh,m_legacy_mesh_builder->m_internal_partitioner_name,false);
794 }

799
800void SubDomain::
801_doInitialPartitionForMesh(IMesh* mesh,const String& service_name,bool is_required)
802{
803 info() << "DoInitialPartition. mesh=" << mesh->name() << " service=" << service_name;
804
805 String lib_name = service_name;
806
807 IMeshPartitionerBase* mesh_partitioner_base = nullptr;
808 Ref<IMeshPartitionerBase> mesh_partitioner_base_ref;
809 Ref<IMeshPartitioner> mesh_partitioner_ref;
810
811 ServiceBuilder<IMeshPartitionerBase> sbuilder(this);
812 mesh_partitioner_base_ref = sbuilder.createReference(service_name,mesh,SB_AllowNull);
813 mesh_partitioner_base = mesh_partitioner_base_ref.get();
814
815 if (!mesh_partitioner_base){
816 // Si pas trouvé, recherche avec l'ancienne interface 'IMeshPartitioner' pour des
817 // raisons de compatibilité
818 pwarning() << "No implementation for 'IMeshPartitionerBase' interface found. "
819 << "Searching implementation for legacy 'IMeshPartitioner' interface";
820 ServiceBuilder<IMeshPartitioner> sbuilder_legacy(this);
821 mesh_partitioner_ref = sbuilder_legacy.createReference(service_name,mesh,SB_AllowNull);
822 if (mesh_partitioner_ref.get())
823 mesh_partitioner_base = mesh_partitioner_ref.get();
824 }
825
826 if (!mesh_partitioner_base){
827 // Si pas trouvé, récupère la liste des valeurs possibles et les affiche.
828 StringUniqueArray valid_names;
829 sbuilder.getServicesNames(valid_names);
830 String valid_values = String::join(", ",valid_names);
831 String msg = String::format("The specified service for the initial mesh partitionment ({0}) "
832 "is not available (valid_values={1}). This service has to implement "
833 "interface Arcane::IMeshPartitionerBase",
834 lib_name,valid_values);
835 if (is_required){
836 pfatal() << msg;
837 }
838 else{
839 pwarning() << msg;
840 }
841 return;
842 }
843
844 bool is_dynamic = mesh->isDynamic();
845 mesh->modifier()->setDynamic(true);
846 mesh->utilities()->partitionAndExchangeMeshWithReplication(mesh_partitioner_base,true);
847 mesh->modifier()->setDynamic(is_dynamic);
848}
