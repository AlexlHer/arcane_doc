35void
36_testArraySwap(bool use_own_swap)
37{
38 std::cout << "** TestArraySwap is_own=" << use_own_swap << "\n";
39
40 UniqueArray<IntSubClass> c1(7);
41 IntSubClass* x1 = c1.unguardedBasePointer();
42 std::cout << "** C1_this = " << &c1 << "\n";
43 std::cout << "** C1_BASE = " << x1 << "\n";
44 UniqueArray<IntSubClass> c2(3);
45 IntSubClass* x2 = c2.unguardedBasePointer();
46 std::cout << "** C2_this = " << &c2 << "\n";
47 std::cout << "** C2_BASE = " << x2 << "\n";
48
49 if (use_own_swap){
50 swap(c1,c2);
51 }
52 else
53 std::swap(c1,c2);
54
55 IntSubClass* after_x1 = c1.data();
56 IntSubClass* after_x2 = c2.data();
57 std::cout << "** C1_BASE_AFTER = " << after_x1 << " size=" << c1.size() << "\n";
58 std::cout << "** C2_BASE_AFTER = " << after_x2 << " size=" << c2.size() << "\n";
59
60 ASSERT_TRUE(x1==after_x2) << "Bad value after swap [1]";
61 ASSERT_TRUE(x2==after_x1) << "Bad value after swap [2]";
62}
63}
