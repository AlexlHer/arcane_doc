148Application::
149~Application()
150{
151 TaskFactory::terminate();
152 m_task_implementation.reset();
153
154 // Supprime les services que l'instance a positionné
155 if (platform::getProcessorAffinityService()==m_processor_affinity_service.get())
156 platform::setProcessorAffinityService(nullptr);
157
158 if (platform::getStackTraceService()==m_stack_trace_service.get())
159 platform::setStackTraceService(nullptr);
160
161 if (platform::getSymbolizerService()==m_symbolizer_service.get())
162 platform::setSymbolizerService(nullptr);
163
164 if (platform::getProfilingService()==m_profiling_service.get())
165 platform::setProfilingService(nullptr);
166
167 delete m_service_and_module_factory_mng;
168
169 m_sessions.each(Deleter());
170
171 delete m_configuration_mng;
172 delete m_ressource_mng;
173
174 m_owned_sequential_parallel_super_mng.reset();
175 m_parallel_super_mng.reset();
176
177 m_data_factory_mng.reset();
178 delete m_io_mng;
179 delete m_service_mng;
180
181 m_trace = nullptr;
182 // Il faut détruire le m_trace_policy après m_trace car ce dernier peut
183 // l'utiliser.
184 delete m_trace_policy;
185
186 // Supprime la référence au gestionnaire de thread. Il faut le faire en dernier car
187 // les autres gestionnaires peuvent l'utiliser.
188 if (platform::getThreadImplementationService()==m_thread_implementation.get())
189 platform::setThreadImplementationService(nullptr);
190}

213template<typename InterfaceType> Ref<InterfaceType> Application::
214_tryCreateService(const StringList& names,String* found_name)
215{
216 if (found_name)
217 (*found_name) = String();
218 ServiceBuilder<InterfaceType> sf(this);
219 for( String s : names ){
220 auto t = sf.createReference(s,SB_AllowNull);
221 if (t.get()){
222 if (found_name)
223 (*found_name) = s;
224 return t;
225 }
226 }
227 return {};
228}

235
236void Application::
237build()
238{
239 const ApplicationBuildInfo& build_info = applicationBuildInfo();
240
241 // Création du TraceMngPolicy. L'initialisation complète se fera plus tard
242 // car on a besoin d'informations supplémentaires comme le rang MPI et
243 // d'avoir lu des fichiers de configuration.
244 m_trace_policy = m_main_factory->createTraceMngPolicy(this);
245
246 // Toujours le premier après TraceMngPolicy
247 m_trace = m_main_factory->createTraceMng();
248
249 {
250 // TODO: positionner ces informations dans ApplicationBuildInfo.
251 Int32 output_level = build_info.outputLevel();
252 if (output_level!=Trace::UNSPECIFIED_VERBOSITY_LEVEL){
253 m_trace_policy->setVerbosityLevel(output_level);
254 m_trace_policy->setStandardOutputVerbosityLevel(output_level);
255 }
256 Int32 verbosity_level = build_info.verbosityLevel();
257 if (verbosity_level!=Trace::UNSPECIFIED_VERBOSITY_LEVEL){
258 m_trace_policy->setVerbosityLevel(verbosity_level);
259 }
260
261 bool has_output_file = build_info.isMasterHasOutputFile();
262 m_trace_policy->setIsMasterHasOutputFile(has_output_file);
263
264 // Positionne le niveau de verbosité en laissant au minimum le niveau
265 // par défaut. Sans cela, certains messages d'initialisation peuvent ne
266 // pas s'afficher ce qui peut être génant en cas de problèmes ou de plantage.
267 Int32 minimal_verbosity_level = build_info.minimalVerbosityLevel();
268 if (minimal_verbosity_level==Trace::UNSPECIFIED_VERBOSITY_LEVEL)
269 minimal_verbosity_level = Trace::DEFAULT_VERBOSITY_LEVEL;
270 m_trace_policy->setDefaultVerboseLevel(m_trace.get(),minimal_verbosity_level);
271 }
272
273 arcaneGlobalMemoryInfo()->setTraceMng(traceMng());
274
275 {
276 // Affiche quelques informations à l'initialisation dès le niveau 4
277 m_trace->info(4) << "*** Initialization informations:";
278 m_trace->info(4) << "*** PID: " << platform::getProcessId();
279 m_trace->info(4) << "*** Host: " << platform::getHostName();
280
281 // TODO: Avec MPC, cette séquence ne doit être appelée qu'une seule fois.
282 IDynamicLibraryLoader* dynamic_library_loader = platform::getDynamicLibraryLoader();
283 if (dynamic_library_loader){
284 String os_dir(m_exe_info.dataOsDir());
285 for( StringCollection::Enumerator i(m_exe_info.dynamicLibrariesName()); ++i; ){
286 String name = *i;
287 m_trace->info(4) << "*** Trying to load dynamic library: " << name;
288 IDynamicLibrary* dl = dynamic_library_loader->open(os_dir,name);
289 if (!dl)
290 m_trace->info(4) << "WARNING: Can not load library '" << name << "'";
291 }
292 }
293
294#ifdef ARCANE_OS_WIN32
295 if (dynamic_library_loader){
296 String os_dir(m_exe_info.dataOsDir());
297 String dyn_lib_names[5] = { "arcane_mpi", "arcane_std", "arcane_mesh",
298 "arcane_thread", "arcane_mpithread",
299 };
300 for( Integer i=0; i<5; ++i )
301 dynamic_library_loader->open(os_dir,dyn_lib_names[i]);
302 }
303#endif
304
305 m_configuration_mng = arcaneCreateConfigurationMng(traceMng());
306
307 {
308 m_service_and_module_factory_mng = arcaneCreateServiceAndModuleFactoryMng(traceMng());
309 for( ServiceFactoryInfoCollection::Enumerator i(m_main_service_factory_infos); ++i; )
310 m_service_and_module_factory_mng->addGlobalFactory(*i);
311 for( ModuleFactoryInfoCollection::Enumerator i(m_main_module_factory_infos); ++i; )
312 m_service_and_module_factory_mng->addGlobalFactory(*i);
313
314 m_service_and_module_factory_mng->createAllServiceRegistererFactories();
315 }
316
317 m_service_mng = m_main_factory->createServiceMng(this);
318
319 String pause_on_error = platform::getEnvironmentVariable("ARCANE_PAUSE_ON_ERROR");
320 if (!pause_on_error.null()){
321 arcaneSetPauseOnError(true);
322 }
323
324 // Recherche le service utilisé pour connaitre la pile d'appel
325 bool has_dbghelp = false;
326 {
327 ServiceBuilder<IStackTraceService> sf(this);
328 Ref<IStackTraceService> sv = sf.createReference("LibUnwind",SB_AllowNull);
329 if (!sv.get()){
330 sv = sf.createReference("DbgHelpStackTraceService", SB_AllowNull);
331 if (sv.get())
332 has_dbghelp = true;
333 }
334 if (sv.get()) {
335 m_stack_trace_service = sv;
336 platform::setStackTraceService(sv.get());
337 }
338 }
339
340 // Recherche le service utilisé pour connaitre les infos sur les symboles
341 // du code source. Pour l'instant on ne supporte que LLVM et on n'active ce service
342 // que si la variable d'environnement ARCANE_LLVMSYMBOLIZER_PATH est définie.
343 {
344 Ref<ISymbolizerService> s;
345 ServiceBuilder<ISymbolizerService> sf(this);
346 if (!platform::getEnvironmentVariable("ARCANE_LLVMSYMBOLIZER_PATH").null()){
347 s = sf.createReference("LLVMSymbolizer", SB_AllowNull);
348 }
349 if (!s.get() && has_dbghelp){
350 s = sf.createReference("DbgHelpSymbolizerService", SB_AllowNull);
351 }
352 if (s.get()) {
353 m_symbolizer_service = s;
354 platform::setSymbolizerService(s.get());
355 }
356 }
357
358 if (arcaneHasThread()){
359 StringList names = build_info.threadImplementationServices();
360 String found_name;
361 auto sv = _tryCreateService<IThreadImplementationService>(names,&found_name);
362 if (sv.get()){
363 m_thread_implementation_service = sv;
364 m_thread_implementation = makeRef(sv->createImplementation());
365 platform::setThreadImplementationService(m_thread_implementation.get());
366 m_thread_implementation->initialize();
367 m_used_thread_service_name = found_name;
368 }
369 else{
370 m_trace->info() << "Can not find thread implementation service "
371 << "(names=" << _stringListToArray(names) << "). Threads are disabled.";
372 arcaneSetHasThread(false);
373 }
374 }
375
376 // Le gestionnaire de thread a pu changer et il faut donc
377 // reinitialiser le gestionnaire de trace.
378 m_trace->resetThreadStatus();
379
380 // Recherche le service utilisé pour gérer les tâches
381 if (arcaneHasThread()){
382 Integer nb_task_thread = build_info.nbTaskThread();
383 if (nb_task_thread>=0){
384 StringList names = build_info.taskImplementationServices();
385 String found_name;
386 auto sv = _tryCreateService<ITaskImplementation>(names,&found_name);
387 if (sv.get()){
388 TaskFactory::setImplementation(sv.get());
389 //m_trace->info() << "Initialize task with nb_thread=" << nb_thread;
390 sv->initialize(nb_task_thread);
391 m_used_task_service_name = found_name;
392 m_task_implementation = sv;
393 }
394 else
395 ARCANE_FATAL("Can not find task implementation service (names='{0}')."
396 " Please check if Arcane is configured with Intel TBB library",
397 _stringListToArray(names));
398 }
399 }
400
401 // Recherche le service utilisé pour le profiling
402 {
403 String profile_str = platform::getEnvironmentVariable("ARCANE_PROFILING");
404 if (!profile_str.null()){
405 ServiceBuilder<IProfilingService> sf(this);
406 auto sv = sf.createReference(profile_str+"ProfilingService",SB_AllowNull);
407 if (sv.get()){
408 m_profiling_service = sv;
409 platform::setProfilingService(sv.get());
410 }
411 else
412 ARCANE_FATAL("Can not find profiling service (name='{0}')",profile_str);
413 }
414 }
415
416 // Par défaut, on accroche le service Hyoda
417 {
418 ServiceBuilder<IOnlineDebuggerService> sf(this);
419 auto sv = sf.createReference("Hyoda",SB_AllowNull);
420 if (sv.get()){
421 m_online_debugger = sv;
422 platform::setOnlineDebuggerService(sv.get());
423 }
424 }
425
426 // Recherche le service utilisé pour la gestion de l'affinité processeur
427 {
428 ServiceBuilder<IProcessorAffinityService> sf(this);
429 auto sv = sf.createReference("HWLoc",SB_AllowNull);
430 if (sv.get()){
431 m_processor_affinity_service = sv;
432 platform::setProcessorAffinityService(sv.get());
433 }
434 }
435
436 // Recherche le service utilisé pour le parallélisme
437 String message_passing_service = build_info.messagePassingService();
438 if (message_passing_service.null())
439 message_passing_service = build_info.internalDefaultMessagePassingService();
440 ServiceBuilder<IParallelSuperMng> sf(this);
441 auto sm = sf.createReference(message_passing_service,SB_AllowNull);
442 if (!sm)
443 ARCANE_FATAL("Can not find message passing service '{0}'",message_passing_service);
444
445 m_parallel_super_mng = sm;
446 m_parallel_super_mng->initialize();
447
448 IParallelSuperMng* seq_sm = nullptr;
449 if (sm->isParallel()){
450 m_owned_sequential_parallel_super_mng = sf.createReference("SequentialParallelSuperMng",SB_AllowNull);
451 seq_sm = m_owned_sequential_parallel_super_mng.get();
452 if (!seq_sm)
453 ARCANE_FATAL("Can not find service 'SequentialParallelSuperMng'");
454 seq_sm->initialize();
455 }
456 else
457 seq_sm = m_parallel_super_mng.get();
458
459 m_sequential_parallel_super_mng = seq_sm;
460
461 m_ressource_mng = IRessourceMng::createDefault(this);
462 m_io_mng = m_main_factory->createIOMng(this);
463 m_data_factory_mng = m_main_factory->createDataFactoryMngRef(this);
464 }
465
466 {
467 VersionInfo version_info = m_exe_info.codeVersion();
468 int vmajor = version_info.versionMajor();
469 int vminor = version_info.versionMinor();
470 int vpatch = version_info.versionPatch();
471 m_main_version_str = String::format("{0}.{1}.{2}",vmajor,vminor,vpatch);
472 m_major_and_minor_version_str = String::format("{0}.{1}",vmajor,vminor);
473 m_version_str = m_major_and_minor_version_str;
474 if (vpatch!=0)
475 m_version_str = m_major_and_minor_version_str + "." + vpatch;
476 }
477
478 m_targetinfo_str = m_exe_info.targetFullName();
479 m_application_name = m_exe_info.applicationName();
480 m_code_name = m_exe_info.codeName();
481
482 // Récupère le nom de l'utilisateur
483 m_user_name = platform::getUserName();
484
485 // Récupère le chemin du répertoire de la configuration utilisateur
486 // TODO: il faut changer car dans les nouvelles recommandations POSIX,
487 // le répertoire de configuration est '.config/arcane'.
488 m_user_config_path = platform::getEnvironmentVariable("ARCANE_CONFIG_PATH");
489 if (m_user_config_path.null()) {
490 Directory user_home_env(platform::getHomeDirectory());
491 m_user_config_path = Directory(user_home_env, ".arcane").path();
492 }
493
494 {
495 bool is_parallel = parallelSuperMng()->isParallel();
496 bool is_debug = applicationInfo().isDebug();
497 // Création et initialisation du TraceMngPolicy.
498 m_trace_policy->setIsParallel(is_parallel);
499 m_trace_policy->setIsDebug(is_debug);
500 bool is_parallel_output = is_parallel && is_debug;
501 // Permet de forcer les sorties meme en mode optimisé
502 {
503 String s = platform::getEnvironmentVariable("ARCANE_PARALLEL_OUTPUT");
504 if (!s.null())
505 is_parallel_output = true;
506 if (s=="0")
507 is_parallel_output = false;
508 }
509 m_trace_policy->setIsParallelOutput(is_parallel_output);
510 }
511
512 m_is_master = m_parallel_super_mng->commRank() == 0;
513
514 m_trace->info(4) << "*** UserName: " << m_user_name;
515 m_trace->info(4) << "*** HomeDirectory: " << platform::getHomeDirectory();
516
517#ifdef ARCANE_CHECK_MEMORY
518 arcaneGlobalMemoryInfo()->setTraceMng(m_trace);
519#endif
520}

525void Application::
526initialize()
527{
528 if (m_is_init){
529 m_trace->warning() << "Application is already initialised";
530 return;
531 }
532
533 bool is_debug = m_exe_info.isDebug();
534
535 // Analyse le fichier de configuration de l'utilisateur
536 _openUserConfig();
537
538 //m_trace->info() << "Application init trace mng rank=" << m_parallel_super_mng->traceRank();
539 m_trace_policy->setDefaultClassConfigXmlBuffer(userConfigBuffer());
540 m_trace_policy->initializeTraceMng(m_trace.get(),m_parallel_super_mng->traceRank());
541
542 m_trace->logdate() << "Begin execution.";
543
544 if (is_debug)
545 m_trace->info() << "WARNING: Execution in DEBUG mode!";
546
547#ifdef ARCANE_CHECK
548 m_trace->info() << "WARNING: Compilation in CHECK mode !";
549#endif
550
551 // Active ou désactive un mode vérification partiel si la variable d'environnement
552 // correspondante est positionnée.
553 String check_str = platform::getEnvironmentVariable("ARCANE_CHECK");
554 if (!check_str.null()){
555 bool is_check = check_str != "0";
556 m_trace->info() << "WARNING: Setting CHECK mode to " << is_check;
557 arcaneSetCheck(is_check);
558 }
559 if (arcaneIsCheck()){
560 m_trace->info() << "WARNING: Execution in CHECK mode!";
561 }
562
563#ifdef ARCANE_TRACE
564 m_trace->info() << "WARNING: Execution in TRACE mode !";
565#endif
566#ifdef ARCANE_64BIT
567 m_trace->info() << "Using 64bits version!";
568#endif
569
570 m_trace->info() << "** -- ** -- ** -- ** -- ** -- ** -- ** -- ** -- ** -- **";
571 m_trace->info() << "** -- ** -- ** -- ** -- ** -- ** -- ** -- ** -- ** -- **";
572 m_trace->info() << " ";
573 m_trace->info() << " "
574 << m_exe_info.applicationName();
575 m_trace->info() << " ";
576 VersionInfo version_info = m_exe_info.codeVersion();
577 m_trace->info() << " Version "
578 << version_info.versionMajor() << "."
579 << version_info.versionMinor() << "."
580 << version_info.versionPatch();
581
582 m_trace->info() << " ";
583 m_trace->info() << "** -- ** -- ** -- ** -- ** -- ** -- ** -- ** -- ** -- **";
584 m_trace->info() << "** -- ** -- ** -- ** -- ** -- ** -- ** -- ** -- ** -- **";
585 m_trace->info() << "Date: " << platform::getCurrentDateTime();
586 m_trace->info() << "ArccoreVersion: " << ARCCORE_VERSION_STR_FULL;
587 m_trace->info() << "ArcaneVersion: " << ARCANE_VERSION_STR_FULL
588 << " (Compiler: " << platform::getCompilerId() << ")";
589 m_trace->info() << "Current process pid=" << platform::getProcessId()
590 << " machine=" << platform::getHostName();
591 m_trace->info() << "MessagePassing service=" << applicationBuildInfo().messagePassingService();
592 if (platform::getStackTraceService()){
593 m_trace->info() << "Stack trace service is available";
594 }
595 if (platform::getSymbolizerService()){
596 m_trace->info() << "Symbolizer service is available";
597 }
598 m_trace->info() << "Array size storage = " << sizeof(Array<Int32>::size_type)
599 << " sizeof(ArrayImplBase)=" << sizeof(ArrayImplBase);
600
601#ifdef ARCANE_USE_LIBXML2
602 m_trace->info() << "Using 'libxml2' for XML parsing";
603#endif
604
605 // Affiche les infos sur les processeurs
606 {
607 IProcessorAffinityService* pas = platform::getProcessorAffinityService();
608 if (pas){
609 pas->printInfos();
610 }
611 }
612
613 // Affiche si l'on a un service de debug
614 if (platform::getOnlineDebuggerService()){
615 m_trace->info() << "Hyoda service is now hooked";
616 }
617 else{
618 m_trace->info() << "Unknown online debugger service";
619 }
620
621 m_is_init = true;
622
623 // Analyse le fichier de configuration du code.
624 _readCodeConfigurationFile();
625
626 {
627 // Construction des types internes
628 ItemTypeMng::_singleton()->build(m_parallel_super_mng.get(),traceMng());
629 }
630
631 {
632 ByteConstSpan runtime_config = m_exe_info.runtimeConfigFileContent();
633 if (!runtime_config.empty()){
634 m_trace->info() << "Reading configuration parameters from runtime config file";
635 JSONDocument jdoc;
636 jdoc.parse(runtime_config);
637 JSONValue config = jdoc.root().child("configuration");
638 ConfigurationReader cr(m_trace.get(),m_configuration_mng->defaultConfiguration());
639 cr.addValuesFromJSON(config,ConfigurationReader::P_GlobalRuntime);
640 }
641 }
642 {
643 if (!m_config_root_element.null()){
644 XmlNode configuration_elem = m_config_root_element.child("configuration");
645 if (!configuration_elem.null()){
646 m_trace->info() << "Reading configuration parameters from code config file";
647 ConfigurationReader cr(m_trace.get(),m_configuration_mng->defaultConfiguration());
648 cr.addValuesFromXmlNode(configuration_elem,ConfigurationReader::P_Global);
649 }
650 }
651 }
652
653 _initDataInitialisationPolicy();
654
655 {
656 if (!m_used_thread_service_name.null())
657 m_trace->info() << "Service used for thread management : '" << m_used_thread_service_name << "'";
658 else
659 m_trace->info() << "No thread management active";
660

666 m_trace->info() << "TaskManagement infos:" << ostr.str();
667 }
668 else
669 m_trace->info() << "No task management active";
670 }

672 // Recherche le service utilisé pour gérer le système d'unité.
673 {

678 m_trace->info() << "UnitSystem service found name=" << service_name;

714void Application::
715_readCodeConfigurationFile()
716{
717 // Analyse le fichier de configuration du code.
718 const ApplicationBuildInfo& build_info = applicationBuildInfo();
719
720 // Récupère le nom du fichier de configuration.
721 // Si nul, cela indique qu'il n'y a pas de fichier de configure.
722 // Si vide (le défaut), on récupère le nom à partir du nom du code.
723 // Sinon, on utilise le nom spécifié dans la configuration.
724
725 // A noter que cette valeur doit être la même pour tous les PE sinon cela
726 // va bloquer (TODO: faire une éventuellem réduction)
727 String config_file_name = build_info.configFileName();
728
729 bool use_config_file = true;
730 if (config_file_name.null()){
731 use_config_file = false;
732 }
733 else if (config_file_name.empty()){
734 // Regarde d'abord dans le répertoire courant, sinon dans le répertoire
735 // des données partagées (share).
736 // Pour des raisons de performances en parallèle, seul le processeur maitre
737 // fait le test.
738 StringBuilder buf;
739 if (m_is_master){
740 buf = m_exe_info.codeName();
741 buf += ".config";
742 if (!platform::isFileReadable(buf.toString())){
743 buf = m_exe_info.dataDir();
744 buf += "/";
745 buf += m_exe_info.codeName();
746 buf += ".config";
747 }
748 else{
749 m_trace->info() << "Using configuration file in current directory.";
750 }
751 }
752 config_file_name = buf.toString();
753 }
754 m_trace->info() << "Using configuration file: '" << config_file_name << "'";
755
756 if (use_config_file){
757 bool bad_file = m_io_mng->collectiveRead(config_file_name,m_config_bytes);
758 if (bad_file)
759 ARCANE_FATAL("Can not read configuration file '{0}'",config_file_name);
760 m_config_document = m_io_mng->parseXmlBuffer(m_config_bytes,config_file_name);
761 if (!m_config_document.get())
762 ARCANE_FATAL("Can not parse configuration file '{0}'",config_file_name);
763 m_config_root_element = m_config_document->documentNode().documentElement();
764 }
765}

815
816Ref<ICodeService> Application::
817getCodeService(const String& u_file_name)
818{
819 ServiceBuilder<ICodeService> builder(this);
820 auto services = builder.createAllInstances();
821
822 // Regarde si un service gere l'extension '.arc'.
823 // S'il n'y en a pas, utilise ArcaneCodeService pour
824 // cette extension.
825 {
826 bool has_arc_extension = false;
827 for( Integer i=0, n=services.size(); i<n; ++i ){
828 ICodeService* code_service = services[i].get();
829 if (_hasExtension(code_service,"arc")){
830 has_arc_extension = true;
831 break;
832 }
833 }
834 if (!has_arc_extension){
835 services.add(createArcaneCodeService(this));
836 }
837 }
838
839 // Cherche l'extension du fichier et la conserve dans \a case_ext
840 std::string_view fview = u_file_name.toStdStringView();
841 std::size_t extension_pos = fview.find_last_of('.');
842 if (extension_pos==std::string_view::npos)
843 return {};
844 fview.remove_prefix(extension_pos+1);
845 String case_ext(fview);
846
847 Ref<ICodeService> found_service;
848 for( auto code_service : services ){
849 StringCollection extensions = code_service->validExtensions();
850 for( StringCollection::Enumerator j(extensions); ++j; ){
851 if (case_ext==(*j)){
852 found_service = code_service;
853 break;
854 }
855 }
856 if (found_service.get())
857 break;
858 }
859 // TODO: retourner une référence.
860 return found_service;
861}
