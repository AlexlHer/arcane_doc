111void Stat::
112print(ITraceMng* msg)
113{
114 for (auto i : stats()) {
115 OneStat* os = i.second;
116 Real total_time = os->cumulativeTotalTime();
117 Int64 div_time = static_cast<Int64>(total_time * 1000.0);
118 Int64 nb_message = os->cumulativeNbMessage();
119 Int64 total_size = os->cumulativeTotalSize();
120 const String& name = os->name();
121 if (div_time == 0)
122 div_time = 1;
123 if (nb_message > 0) {
124 Int64 average_time = Convert::toInt64(total_time / (Real)nb_message);
125 msg->info() << " MPIStat " << name << " :" << nb_message << " messages";
126 msg->info() << " MPIStat " << name << " :" << total_size << " bytes ("
127 << total_size / div_time << " Kb/s) (average size "
128 << total_size / nb_message << " bytes)";
129 msg->info() << " MPIStat " << name << " Time: " << total_time << " seconds"
130 << " (avg=" << average_time << ")";
131 }
132 }
133}

152 std::string _formatToString(Real v)
153 {
154 v = v * 10000;
155 Int64 v_as_int = static_cast<Int64>(std::round(v));
156 Int64 a = v_as_int / 10000;
157 Int64 b = v_as_int % 10000;
158 std::ostringstream ostr;
159 ostr << std::setfill(' ') << std::right << std::setw(6) << a << '.';
160 ostr << std::setfill('0') << std::right << std::setw(4) << b;
161 return ostr.str();
162 }

165extern "C++" void
166printStatsCollective(IStat* s, IParallelMng* pm)
167{
168 // Les instances \a s de tous les rangs peuvent ne pas avoir les mêmes
169 // statistiques. Pour éviter des blocages, on ne garde que les statistiques
170 // communes à tout le monde.
171 UniqueArray<String> input_strings;
172 const auto& stat_map = s->toArccoreStat()->stats();
173 for (auto x : stat_map)
174 input_strings.add(x.first);
175 UniqueArray<String> common_strings;
176 MessagePassing::filterCommonStrings(pm, input_strings, common_strings);
177 Int32 nb_rank = pm->commSize();
178 ITraceMng* tm = pm->traceMng();
179 tm->info() << "Message passing Stats (unit is second) "
180 << Trace::Width(48) << "min"
181 << Trace::Width(7) << "max";
182 tm->info() << Trace::Width(55) << "average"
183 << Trace::Width(10) << "min"
184 << Trace::Width(12) << "max"
185 << Trace::Width(10) << "rank"
186 << Trace::Width(7) << "rank";
187 for (String name : common_strings) {
188 auto i = stat_map.find(name);
189 if (i == stat_map.end())
190 ARCANE_FATAL("Internal error: string '{0}' not in stats", name);
191 Real my_time = i->second->cumulativeTotalTime();
192 Real sum_time = 0.0;
193 Real min_time = 0.0;
194 Real max_time = 0.0;
195 Int32 min_time_rank = 0;
196 Int32 max_time_rank = 0;
197 pm->computeMinMaxSum(my_time, min_time, max_time, sum_time, min_time_rank, max_time_rank);
198 Real average_time = sum_time / static_cast<Real>(nb_rank);
199 tm->info() << " MPIStatAllRanks " << Trace::Width(25) << name << " :"
200 << " " << _formatToString(average_time)
201 << " " << _formatToString(min_time)
202 << " " << _formatToString(max_time)
203 << " " << Trace::Width(6) << min_time_rank
204 << " " << Trace::Width(6) << max_time_rank;
205 }
206}
