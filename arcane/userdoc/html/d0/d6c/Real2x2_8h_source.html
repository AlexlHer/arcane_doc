37/*!
38 * \brief Classe gérant une matrice de réel de dimension 2x2.
39
40 La matrice comprend deux composantes \a x et \a y qui sont du
41 type \b Real2. Par exemple:
42
43 * \code
44 * Real2x2 matrix;
45 * matrix.x.y = 2.;
46 * matrix.y.y = 3.;
47 * \endcode
48*/

51 public:
52
53 //! Construit la matrice nulle
54 constexpr ARCCORE_HOST_DEVICE Real2x2()
55 : x(Real2::null()), y(Real2::null()) {}
56 //! Construit le couple (ax,ay)
57 constexpr ARCCORE_HOST_DEVICE Real2x2(Real2 ax,Real2 ay)
58 : x(ax), y(ay) {}

60 * \brief Construit le couple ((ax,bx),(ay,by)).
61 * \deprecated Utiliser Real2x2(Real2 a,Real2 b) à la place.

64 : x(ax,bx), y(ay,by) {}
65 //! Construit un couple identique à \a f

67 : x(f.x), y(f.y) {}

74 return (*this);

113 /*!
114 * \brief Compare la matrice avec la matrice nulle.
115 *
116 * La matrice est nulle si et seulement si chacune de ses composant
117 * est inférieure à un espilon donné. La valeur de l'epsilon utilisée est celle
118 * de float_info<value_type>::nearlyEpsilon():
119 * \f[A=0 \Leftrightarrow |A.x|<\epsilon,|A.y|<\epsilon\f]
120 *
121 * \retval true si la matrice est égale à la matrice nulle,
122 * \retval false sinon.
123 */
124 constexpr ARCCORE_HOST_DEVICE bool isNearlyZero() const
125 { return x.isNearlyZero() && y.isNearlyZero(); }
126 /*!
127 * \brief Lit la matrice sur le flot \a i
128 * La matrice est lue sous la forme de trois Real2.

130 std::istream& assign(std::istream& i);

138 //! Soustrait \a b au couple

156 //! Multiple chaque composante du couple par la composant correspondant de \a b

159 constexpr ARCCORE_HOST_DEVICE void operator*=(Real b) { x*=b; y*=b; }
160 //! Divise chaque composante du couple par la composant correspondant de \a b

163 constexpr ARCCORE_HOST_DEVICE void operator/= (Real b) { x/=b; y/=b; }
164 //! Créé un couple qui vaut ce couple ajouté à \a b
165 constexpr ARCCORE_HOST_DEVICE Real2x2 operator+(Real2x2 b) const { return Real2x2(x+b.x,y+b.y); }
166 //! Créé un couple qui vaut \a b soustrait de ce couple
167 constexpr ARCCORE_HOST_DEVICE Real2x2 operator-(Real2x2 b) const { return Real2x2(x-b.x,y-b.y); }
168 //! Créé un tenseur opposé au tenseur actuel
169 constexpr ARCCORE_HOST_DEVICE Real2x2 operator-() const { return Real2x2(-x,-y); }

171 /*!
172 * \brief Compare composant pas composante l'instance courante à \a b.
173 *
174 * \retval true si this.x==b.x et this.y==b.y.
175 * \retval false sinon.
176 */
177 constexpr ARCCORE_HOST_DEVICE bool operator==(Real2x2 b) const
178 {
179 return (x==b.x) && (y==b.y);
180 }
181 /*!
182 * \brief Compare deux couples.
183 * Pour la notion d'égalité, voir operator==()
184 * \retval true si les deux couples sont différents,
185 * \retval false sinon.
186 */
187 constexpr ARCCORE_HOST_DEVICE bool operator!=(Real2x2 b) const
188 { return !operator==(b); }
189
190 /**
191 * Accès en lecture seule à la @a i-me (entre 0 et 2 inclus) ligne de l'instance.
192 * \param i numéro de la ligne à retourner
193 */
194 ARCCORE_HOST_DEVICE Real2 operator[](Integer i) const
195 {
196 ARCCORE_CHECK_AT(i,2);
197 return (&x)[i];
198 }
199
200 /**
201 * Accès à la @a i-ème ligne (entre 0 et 2 inclus) de l'instance.
202 * \param i numéro de la ligne à retourner
203 */
204 ARCCORE_HOST_DEVICE Real2& operator[](Integer i)
205 {
206 ARCCORE_CHECK_AT(i,2);
207 return (&x)[i];
208 }

213 * \brief Compare les valeurs de \a a et \a b avec le comparateur TypeEqualT

216 */
217 constexpr ARCCORE_HOST_DEVICE static bool _eq(Real a,Real b)
218 { return TypeEqualT<Real>::isEqual(a,b); }
