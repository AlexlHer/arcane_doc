32/**
33 * @brief Classe représentant une graine.
34 */
35class ARCANE_CORE_EXPORT RandomNumberGeneratorSeed
36{
37 public:
38 /**
39 * @brief Constructeur avec graine de type T.
40 *
41 * @tparam T Le type de graine (Int64 par ex).
42 * @param seed La valeur de la graine.
43 */
44 template <class T>
45 explicit RandomNumberGeneratorSeed(T seed)
46 {
47 setSeed(seed);
48 }
49
50 /**
51 * @brief Constructeur copie.
52 *
53 * @param seed La graine source.
54 */
55 RandomNumberGeneratorSeed(const RandomNumberGeneratorSeed& seed)
56 {
57 m_seed.resize(seed.sizeOfSeed());
58 seed.seed(m_seed.data());
59 }
60
61 /**
62 * @brief Constructeur par défaut.
63 */
64 RandomNumberGeneratorSeed()
65 : m_seed(0)
66 {}
67
68 virtual ~RandomNumberGeneratorSeed() {}
69
70 public:
71 /**
72 * @brief Méthode permettant de définir une graine.
73 *
74 * @tparam T Le type de graine.
75 * @param seed La valeur de la graine.
76 * @return true Si la graine a pu être construite.
77 * @return false Si la graine n'a pas pu être construite.
78 */
79 template <class T>
80 bool setSeed(T seed)
81 {
82 Integer size = sizeof(T);
83 m_seed.resize(size);
84 memcpy(m_seed.data(), &seed, size);
85 return true;
86 }
87
88 /**
89 * @brief Méthode permettant de récupérer la valeur de la graine.
90 *
91 * @tparam T Le type de la graine.
92 * @param seed [OUT] La valeur de la graine.
93 * @return true Si la graine a pu être placé.
94 * @return false Si la graine n'a pas pu être placé.
95 */
96 template <class T>
97 bool seed(T& seed) const
98 {
99 if (m_seed.size() == 0) {
100 return false;
101 }
102 memcpy(&seed, dataPtr(), m_seed.size());
103 return true;
104 }
105
106 /**
107 * @brief Méthode permettant de récupérer la valeur de la graine.
108 *
109 * @tparam T Le type de la graine.
110 * @param seed [OUT] La valeur de la graine.
111 * @return true Si la graine a pu être placé.
112 * @return false Si la graine n'a pas pu être placé.
113 */
114 template <class T>
115 bool seed(T* seed) const
116 {
117 if (m_seed.size() == 0) {
118 return false;
119 }
120 memcpy(seed, dataPtr(), m_seed.size());
121 return true;
122 }
123
124 /**
125 * @brief Méthode permettant de récupérer la taille de la graine.
126 *
127 * @return Integer La taille de la graine (en octet).
128 */
129 Integer sizeOfSeed() const
130 {
131 return m_seed.size();
132 }
133
134 /**
135 * @brief Méthode permettant de récupérer un pointeur vers le tableau de Bytes (hors toute protection).
136 *
137 * @return const Byte* Un pointeur vers le début du tableau.
138 */
139 const Byte* dataPtr() const
140 {
141 return m_seed.data();
142 }
143
144 /**
145 * @brief Opérateur de copie.
146 *
147 * @param other La graine source.
148 * @return RandomNumberGeneratorSeed& La graine destination.
149 */
150 RandomNumberGeneratorSeed& operator=(const RandomNumberGeneratorSeed& other)
151 {
152 if (this == &other) {
153 return *this;
154 }
155
156 m_seed.resize(other.sizeOfSeed());
157 other.seed(m_seed.data());
158 return *this;
159 }
160
161 /**
162 * @brief Opérateur de copie depuis une valeur de graine.
163 *
164 * @tparam T Le type de la graine.
165 * @param other La valeur de la graine.
166 * @return RandomNumberGeneratorSeed& La graine destination.
167 */
168 template <class T>
169 RandomNumberGeneratorSeed& operator=(T other)
170 {
171 setSeed(other);
172 return *this;
173 }
174
175 /**
176 * @brief Opérateur de comparaison.
177 *
178 * @param other La graine à comparer.
179 * @return true Si les deux graines sont identiques.
180 * @return false Si les deux graines ne sont pas identiques.
181 */
182 bool operator==(RandomNumberGeneratorSeed& other)
183 {
184 if (sizeOfSeed() != other.sizeOfSeed()) {
185 return false;
186 }
187
188 const Byte* other_data = other.dataPtr();
189 const Byte* my_data = dataPtr();
190
191 for (Integer i = 0; i < sizeOfSeed(); i++) {
192 if (my_data[i] != other_data[i]) {
193 return false;
194 }
195 }
196 return true;
197 }
198
199 /**
200 * @brief Opérateur de comparaison.
201 *
202 * @param other La graine à comparer.
203 * @return true Si les deux graines ne sont pas identiques.
204 * @return false Si les deux graines sont identiques.
205 */
206 bool operator!=(RandomNumberGeneratorSeed& other)
207 {
208 return !operator==(other);
209 }
210
211 protected:
212 ByteUniqueArray m_seed;
213};

215/**
216 * @brief Interface pour un générateur de nombre aléatoire.
217 *
218 */
219class ARCANE_CORE_EXPORT IRandomNumberGenerator
220{
221 public:
222 virtual ~IRandomNumberGenerator() = default;
223
224 public:
225 /**
226 * @brief Méthode permettant d'initialiser le service.
227 *
228 * Avec la graine en option (ou la graine par défaut si l'on est
229 * en mode singleton).
230 *
231 */
232 virtual void initSeed() = 0;
233
234 /**
235 * @brief Méthode permettant d'initialiser le service.
236 *
237 * @param seed La graine d'origine.
238 */
239 virtual void initSeed(RandomNumberGeneratorSeed seed) = 0;
240
241 /**
242 * @brief Méthode permettant de récupérer la graine actuelle.
243 *
244 * @return Int64 La graine.
245 */
246 virtual RandomNumberGeneratorSeed seed() = 0;
247
248 /**
249 * @brief Méthode permettant de savoir si les sauts sont permis sur le
250 * générateur de graines.
251 *
252 * @return true Si oui.
253 * @return false Si non.
254 */
255 virtual bool isLeapSeedSupported() = 0;
256
257 /**
258 * @brief Méthode permettant de générer une graine "enfant" à partir d'une
259 * graine "parent".
260 *
261 * @param leap Le saut à effectuer (0 = la graine n+1+0 / 1 = la graine n+1+1).
262 * @return Int64 La nouvelle graine généré à partir de la graine en mémoire.
263 */
264 virtual RandomNumberGeneratorSeed generateRandomSeed(Integer leap = 0) = 0;
265
266 /**
267 * @brief Méthode permettant de générer une graine "enfant" à partir d'une
268 * graine "parent".
269 *
270 * Cette méthode n'utilise pas la graine en mémoire.
271 *
272 * @param parent_seed La graine "parent".
273 * @param leap Le saut à effectuer (0 = la graine n+1+0 / 1 = la graine n+1+1).
274 * @return Int64 La nouvelle graine généré à partir de la graine "parent".
275 */
276 virtual RandomNumberGeneratorSeed generateRandomSeed(RandomNumberGeneratorSeed* parent_seed, Integer leap = 0) = 0;
277
278 /**
279 * @brief Méthode permettant de savoir si les sauts sont permis sur le
280 * générateur de nombres.
281 *
282 * @return true Si oui.
283 * @return false Si non.
284 */
285 virtual bool isLeapNumberSupported() = 0;
286
287 /**
288 * @brief Méthode permettant de générer un nombre aléatoire avec
289 * la graine en mémoire.
290 *
291 * @param leap Le saut à effectuer (0 = le nombre n+1+0 / 1 = le nombre n+1+1).
292 * @return Real Le nombre généré (entre 0 et 1).
293 */
294 virtual Real generateRandomNumber(Integer leap = 0) = 0;
295
296 /**
297 * @brief Méthode permettant de générer un nombre aléatoire avec
298 * la graine transmise en paramètre.
299 *
300 * Cette méthode n'utilise pas la graine en mémoire.
301 *
302 * @param seed La graine.
303 * @param leap Le saut à effectuer (0 = le nombre n+1+0 / 1 = le nombre n+1+1).
304 * @return Real Le nombre généré (entre 0 et 1).
305 */
306 virtual Real generateRandomNumber(RandomNumberGeneratorSeed* seed, Integer leap = 0) = 0;
307};
