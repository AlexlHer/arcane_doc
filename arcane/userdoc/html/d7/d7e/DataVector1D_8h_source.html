31/*!
32 * \internal
33 * \brief Vecteur de données 1D d'un type \a T.
34 *
35 * Cette classe s'utilise par référence, chaque instance gardant
36 * une réference sur les autres par l'intermédiaire d'une liste doublement
37 * chaînée.
38 *
39 * Les opérations add() et resize() obligent à remettre à jour toutes
40 * les instances qui font références au même tableau. Elle doivent donc
41 * être utilisées le moins possible.
42 *
43 */
44template<typename T>
45class DataVector1D
46: public DataVectorCommon1D<T>
47{
48 public:
49
50 //! Type de la classe de base
51 typedef DataVectorCommon1D<T> BaseClassType;
52
53 //! Type de l'allocateur
54 typedef typename BaseClassType::AllocatorType AllocatorType;
55
56 //! Type de cette classe
57 typedef DataVector1D<T> ThatClassType;
58
59 static AllocatorType* _defaultAllocator()
60 { return DefaultAllocatorT<T>::globalInstance(); }
61
62 public:
63
64
65 public:
66
67 //! Créé un vecteur sans éléments
68 DataVector1D(AllocatorType* allocator = _defaultAllocator())
69 : BaseClassType(allocator), m_prev(0), m_next(0) {}
70
71 //! Créé un vecteur de \a nb_elements
72 DataVector1D(Integer nb_element,AllocatorType* allocator = _defaultAllocator())
73 : BaseClassType(nb_element,allocator), m_prev(0), m_next(0) {}
74
75 //! Constructeur de copie
76 DataVector1D(ThatClassType& from)
77 : BaseClassType(from), m_prev(0), m_next(0)
78 {
79 _addReference(from);
80 }
81
82 const ThatClassType& operator=(ThatClassType& from)
83 {
84 if (&from!=this){
85 _removeReference();
86 _addReference(from);
87 BaseClassType* base = this;
88 (*base) = from;
89 }
90 return (*this);
91 }
92
93 //! Supprime cette instance
94 ~DataVector1D()
95 {
96 _removeReference();
97 _checkFreeMemory();
98 }
99
100 public:
101
102 //! Ajoute l'élément \a element en fin de liste
103 void add(const T& element)
104 {
105 this->_add(element);
106 _updateReferences();
107 }
108
109 //! Change le nombre d'éléments du tableau en \a new_size
110 void resize(Integer new_size)
111 {
112 this->_resize(new_size);
113 _updateReferences();
114 }
115
116 /*! \brief Réserve de la mémoire.
117 *
118 * Si la capacité actuelle est supérieure à \a new_capacity, la méthode
119 * ne fait rien. Sinon, alloue de la mémoire pour \a max_capacity
120 * éléments.
121 */
122 void reserve(Integer new_capacity)
123 {
124 this->_reserve(new_capacity);
125 _updateReferences();
126 }
127
128 private:
129
130 ThatClassType* m_prev; //!< Référence précédente dans la liste chaînée
131 ThatClassType* m_next; //!< Référence suivante dans la liste chaînée
132
133 private:
134
135 //! Mise à jour des références
136 void _updateReferences()
137 {
138 BaseClassType& that_ref = *this;
139 for( ThatClassType* i = m_prev; i; i = m_prev->m_prev ){
140 BaseClassType* bs = i;
141 (*bs) = that_ref;
142 }
143 for( ThatClassType* i = m_next; i; i = m_next->m_next ){
144 BaseClassType* bs = i;
145 (*bs) = that_ref;
146 }
147 }
148
149 /*! \brief Insère cette instance dans la liste chaînée.
150 * L'instance est insérée à la position de \a new_ref.
151 * \pre m_prev==0
152 * \pre m_next==0;
153 */
154 void _addReference(ThatClassType& new_ref)
155 {
156 ThatClassType* prev = new_ref.m_prev;
157 new_ref.m_prev = this;
158 m_prev = prev;
159 m_next = &new_ref;
160 }
161
162 //! Supprime cette instance de la liste chaînée des références
163 void _removeReference()
164 {
165 if (m_prev)
166 m_prev->m_next = m_next;
167 if (m_next)
168 m_next->m_prev = m_prev;
169 }
170
171 //! Détruit l'instance si plus personne ne la référence
172 void _checkFreeMemory()
173 {
174 if (!m_prev && !m_next)
175 this->_destroy();
176 }
177};
