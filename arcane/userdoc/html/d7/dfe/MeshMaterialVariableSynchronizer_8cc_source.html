88/*!
89 * \brief Remplit \a items avec la liste de tous les MatVarIndex des
90 * mailles de \a view.
91 */
92void MeshMaterialVariableSynchronizer::
93_fillCells(Array<MatVarIndex>& items,AllEnvCellVectorView view)
94{
95 bool has_mat = m_var_space==MatVarSpace::MaterialAndEnvironment;
96 // NOTE: il est possible d'optimiser en regardant les milieux qui n'ont
97 // qu'un seul matériau car dans ce cas la valeur milieu et la valeur
98 // matériau est la même. De la même manière, s'il n'y a qu'un milieu
99 // alors la valeur globale et milieu est la même. Dans ces cas, il n'est
100 // pas nécessaire d'ajouter la deuxième MatCell dans la liste.
101 items.clear();
102 ENUMERATE_ALLENVCELL(iallenvcell,view){
103 AllEnvCell all_env_cell = *iallenvcell;
104 ENUMERATE_CELL_ENVCELL(ienvcell,all_env_cell){
105 EnvCell env_cell = *ienvcell;
106 items.add(ienvcell._varIndex());
107 if (has_mat){
108 ENUMERATE_CELL_MATCELL(imatcell,env_cell){
109 items.add(imatcell._varIndex());
110 }
111 }
112 }
113 // A priori ajouter cette information n'est pas nécessaire car il
114 // est possible de récupérer l'info de la variable globale.
115 items.add(MatVarIndex(0,view.localId(iallenvcell.index())));
116 }
117}

133
134void MeshMaterialVariableSynchronizer::
135recompute()
136{
137 IVariableSynchronizer* var_syncer = m_variable_synchronizer;
138
139 // Calcul des informations de synchronisation pour les mailles matériaux.
140 // NOTE: Cette version nécessite que les matériaux soient correctement
141 // synchronisés entre les sous-domaines.
142
143 IItemFamily* family = var_syncer->itemGroup().itemFamily();
144 IParallelMng* pm = var_syncer->parallelMng();
145 if (!pm->isParallel())
146 return;
147 ItemGroup all_items = family->allItems();
148
149 Int32ConstArrayView ranks = var_syncer->communicatingRanks();
150 Integer nb_rank = ranks.size();
151
152 m_commun_buffer->setNbRank(nb_rank);
153
154 m_shared_items.resize(nb_rank);
155 m_ghost_items.resize(nb_rank);
156
157 for( Integer i=0; i<nb_rank; ++i ){
158
159 {
160 Int32ConstArrayView shared_ids = var_syncer->sharedItems(i);
161 CellVectorView shared_cells(family->view(shared_ids));
162 AllEnvCellVectorView view = m_material_mng->view(shared_cells);
163 Array<MatVarIndex>& items = m_shared_items[i];
164 _fillCells(items,view);
165 info(4) << "SIZE SHARED FOR rank=" << ranks[i] << " n=" << items.size();
166 }
167
168 {
169 Int32ConstArrayView ghost_ids = var_syncer->ghostItems(i);
170 CellVectorView ghost_cells(family->view(ghost_ids));
171 AllEnvCellVectorView view = m_material_mng->view(ghost_cells);
172 Array<MatVarIndex>& items = m_ghost_items[i];
173 _fillCells(items,view);
174 info(4) << "SIZE GHOST FOR rank=" << ranks[i] << " n=" << items.size();
175 }
176 }
177}
