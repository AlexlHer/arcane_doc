69/*!
70 * \brief Classe gérant un vecteur de réel de dimension 2.
71
72 Le vecteur comprend deuxs composantes \a x et \a y qui sont du
73 type \b Real.
74
75 \code
76 Real2 value (1.0,2.3); // Créé un couple (x=1.0, y=2.3)
77 cout << value.x; // Imprime la composante x
78 value.y += 3.2; // Ajoute 3.2 à la composante \b y
79 \endcode
80 */
81class ARCANE_UTILS_EXPORT Real2
82: public Real2POD
83{
84 public:
85
86 //! Construit le vecteur nul.
87 constexpr ARCCORE_HOST_DEVICE Real2() : Real2POD() { x=0.; y=0.; }
88 //! Construit le couplet (ax,ay)
89 constexpr ARCCORE_HOST_DEVICE Real2(Real ax,Real ay) : Real2POD() { x=ax; y=ay; }
90 //! Construit un couple identique à \a f
91 constexpr ARCCORE_HOST_DEVICE Real2(const Real2& f) : Real2POD() { x=f.x; y=f.y; }
92 //! Construit un coupe identique à \a f
93 constexpr ARCCORE_HOST_DEVICE explicit Real2(const Real2POD& f) : Real2POD() { x=f.x; y=f.y; }
94 constexpr ARCCORE_HOST_DEVICE Real2& operator=(Real2 f)
95 { x=f.x; y=f.y; return (*this); }
96 //! Affecte à l'instance le couple (v,v).
97 constexpr ARCCORE_HOST_DEVICE Real2& operator= (Real v)
98 { x = y = v; return (*this); }
99
100 public:
101
102 constexpr ARCCORE_HOST_DEVICE static Real2 null() { return Real2(0.,0.); }
103
104 public:
105
106 //! Retourne une copie du couple.
107 constexpr ARCCORE_HOST_DEVICE Real2 copy() const { return (*this); }
108 //! Réinitialise le couple avec les constructeurs par défaut.
109 constexpr ARCCORE_HOST_DEVICE Real2& reset() { x = y = 0.; return (*this); }
110 //! Affecte à l'instance le couple (ax,ay,az)
111 constexpr ARCCORE_HOST_DEVICE Real2& assign(Real ax,Real ay) { x = ax; y = ay; return (*this); }
112 //! Copie le couple \a f
113 constexpr ARCCORE_HOST_DEVICE Real2& assign(Real2 f) { x = f.x; y = f.y; return (*this); }
114 /*!
115 * \brief Indique si l'instance est proche de l'instance nulle.
116 *
117 * \retval true si math::isNearlyZero() est vrai pour chaque composante.
118 * \retval false sinon.
119 */
120 constexpr ARCCORE_HOST_DEVICE bool isNearlyZero() const
121 {
122 return math::isNearlyZero(x) && math::isNearlyZero(y);
123 }
124
125 //! Retourne la norme au carré du couple \f$x^2+y^2+z^2\f$
126 constexpr ARCCORE_HOST_DEVICE Real squareNormL2() const { return x*x + y*y; }
127 //! Retourne la norme du couple \f$\sqrt{x^2+y^2+z^2}\f$
128 ARCCORE_HOST_DEVICE Real normL2() const { return _sqrt(squareNormL2()); }
129
130 //! Retourne la norme au carré du couple \f$x^2+y^2+z^2\f$
131 ARCCORE_DEPRECATED_2021("Use squareNormL2() instead")
132 ARCCORE_HOST_DEVICE Real abs2() const { return x*x + y*y; }
133 //! Retourne la norme du couple \f$\sqrt{x^2+y^2+z^2}\f$
134 ARCCORE_DEPRECATED_2021("Use normL2() instead")
135 ARCCORE_HOST_DEVICE Real abs() const { return _sqrt(squareNormL2()); }
136 //! Valeur absolue composante par composante.
137 ARCCORE_HOST_DEVICE Real2 absolute() const { return Real2(math::abs(x),math::abs(y)); }
138
139 /*!
140 * \brief Lit un couple sur le flot \a i
141 * Le couple est lu sous la forme de trois valeur de type #value_type.
142 */
143 std::istream& assign(std::istream& i);
144 //! Ecrit le couple sur le flot \a o lisible par un assign()
145 std::ostream& print(std::ostream& o) const;
146 //! Ecrit le couple sur le flot \a o sous la forme (x,y)
147 std::ostream& printXy(std::ostream& o) const;
148
149 //! Ajoute \a b au couple
150 constexpr ARCCORE_HOST_DEVICE Real2& add(Real2 b) { x+=b.x; y+=b.y; return (*this); }
151 //! Soustrait \a b au couple
152 constexpr ARCCORE_HOST_DEVICE Real2& sub(Real2 b) { x-=b.x; y-=b.y; return (*this); }
153 //! Multiple chaque composante du couple par la composant correspondant de \a b
154 constexpr ARCCORE_HOST_DEVICE Real2& mul(Real2 b) { x*=b.x; y*=b.y; return (*this); }
155 //! Divise chaque composante du couple par la composant correspondant de \a b
156 constexpr ARCCORE_HOST_DEVICE Real2& div(Real2 b) { x/=b.x; y/=b.y; return (*this); }
157 //! Ajoute \a b à chaque composante du couple
158 constexpr ARCCORE_HOST_DEVICE Real2& addSame(Real b) { x+=b; y+=b; return (*this); }
159 //! Soustrait \a b à chaque composante du couple
160 constexpr ARCCORE_HOST_DEVICE Real2& subSame(Real b) { x-=b; y-=b; return (*this); }
161 //! Multiplie chaque composante du couple par \a b
162 constexpr ARCCORE_HOST_DEVICE Real2& mulSame(Real b) { x*=b; y*=b; return (*this); }
163 //! Divise chaque composante du couple par \a b
164 constexpr ARCCORE_HOST_DEVICE Real2& divSame(Real b) { x/=b; y/=b; return (*this); }
165 //! Ajoute \a b au couple.
166 constexpr ARCCORE_HOST_DEVICE Real2& operator+=(Real2 b) { return add(b); }
167 //! Soustrait \a b au couple
168 constexpr ARCCORE_HOST_DEVICE Real2& operator-=(Real2 b) { return sub(b); }
169 //! Multiplie chaque composante du couple par la composant correspondant de \a b
170 constexpr ARCCORE_HOST_DEVICE Real2& operator*=(Real2 b) { return mul(b); }
171 //! Multiplie chaque composante du couple par le réel \a b
172 constexpr ARCCORE_HOST_DEVICE void operator*=(Real b) { x*=b; y*=b; }
173 //! Divise chaque composante du couple par la composant correspondant de \a b
174 constexpr ARCCORE_HOST_DEVICE Real2& operator/=(Real2 b) { return div(b); }
175 //! Divise chaque composante du couple par le réel \a b
176 constexpr ARCCORE_HOST_DEVICE void operator/=(Real b) { x/=b; y/=b; }
177 //! Créé un couple qui vaut ce couple ajouté à \a b
178 constexpr ARCCORE_HOST_DEVICE Real2 operator+(Real2 b) const { return Real2(x+b.x,y+b.y); }
179 //! Créé un couple qui vaut \a b soustrait de ce couple
180 constexpr ARCCORE_HOST_DEVICE Real2 operator-(Real2 b) const { return Real2(x-b.x,y-b.y); }
181 //! Créé un couple opposé au couple actuel
182 constexpr ARCCORE_HOST_DEVICE Real2 operator-() const { return Real2(-x,-y); }
183 /*!
184 * \brief Créé un couple qui vaut ce couple dont chaque composant a été
185 * multipliée par la composante correspondante de \a b.
186 */
187 constexpr ARCCORE_HOST_DEVICE Real2 operator*(Real2 b) const { return Real2(x*b.x,y*b.y); }
188 /*!
189 * \brief Créé un couple qui vaut ce couple dont chaque composant a été divisée
190 * par la composante correspondante de \a b.
191 */
192 constexpr ARCCORE_HOST_DEVICE Real2 operator/(Real2 b) const { return Real2(x/b.x,y/b.y); }
193
194 /*!
195 * \brief Normalise le couple.
196 *
197 * Si le couple est non nul, divise chaque composante par la norme du couple
198 * (abs()), de telle sorte qu'après l'appel à cette méthode, abs() valent \a 1.
199 * Si le couple est nul, ne fait rien.
200 */
201 Real2& normalize()
202 {
203 Real d = normL2();
204 if (!math::isZero(d))
205 divSame(d);
206 return (*this);
207 }
208
209 /*!
210 * \brief Compare composant pas composante l'instance courante à \a b.
211 *
212 * \retval true si this.x==b.x et this.y==b.y.
213 * \retval false sinon.
214 */
215 constexpr ARCCORE_HOST_DEVICE bool operator==(Real2 b) const
216 {
217 return _eq(x,b.x) && _eq(y,b.y);
218 }
219
220 /*!
221 * \brief Compare deux couples.
222 * Pour la notion d'égalité, voir operator==()
223 * \retval true si les deux couples sont différents,
224 * \retval false sinon.
225 */
226 constexpr ARCCORE_HOST_DEVICE bool operator!=(Real2 b) const { return !operator==(b); }
227
228 private:
229
230 /*!
231 * \brief Compare les valeurs de \a a et \a b avec le comparateur TypeEqualT
232 * \retval true si \a a et \a b sont égaux,
233 * \retval false sinon.
234 */
235 constexpr ARCCORE_HOST_DEVICE static bool _eq(Real a,Real b) { return math::isEqual(a,b); }
236 //! Retourne la racine carrée de \a a
237 ARCCORE_HOST_DEVICE static Real _sqrt(Real a) { return math::sqrt(a); }
238};
