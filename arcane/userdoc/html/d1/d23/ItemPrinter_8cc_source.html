57void ItemPrinter::
58print(std::ostream& o) const
59{
60 if (m_item == NULL){
61 o << "(null pointer)";
62 }else if (m_item->null()){
63 if (m_has_item_kind){
64 o << "(null " << itemKindName(m_item_kind) << ")";
65 }
66 else{
67 o << "(null Item)";
68 }
69 }
70 else {
71 o << "(";
72 ItemPrinter::Internal::_printBasics(o,m_item);
73 ItemPrinter::Internal::_printFlags(o,m_item->flags());
74 ItemPrinter::Internal::_printParents(o,m_item);
75 ItemPrinter::Internal::_printErrors(o,m_item);
76 o << ")";
77 }
78}

83void FullItemPrinter::
84print(std::ostream& ostr) const
85{
86 if (!m_item) {
87 ostr << "(null pointer)";
88 }
89 else if (m_item->null()) {
90 ostr << "(null Item)";
91 }
92 else {
93 eItemKind ik = m_item->kind();
94 ostr << "(";
95 ItemPrinter::Internal::_printBasics(ostr,m_item);
96 ItemPrinter::Internal::_printFlags(ostr,m_item->flags());
97 ItemPrinter::Internal::_printParents(ostr,m_item);
98 ItemPrinter::Internal::_printErrors(ostr,m_item);
99 ostr << ")";
100 ostr << "\n\t";
101 if (ik!=IK_Node)
102 if (m_item->nbNode()!=0)
103 ItemPrinter::Internal::_printItemSubItems(ostr,"Nodes",m_item->internalNodes());
104 ostr << "\n\t";
105 if (m_item->nbEdge()!=0)
106 ItemPrinter::Internal::_printItemSubItems(ostr,"Edges",m_item->internalEdges());
107 ostr << "\n\t";
108 if (m_item->nbFace()!=0)
109 ItemPrinter::Internal::_printItemSubItems(ostr,"Faces",m_item->internalFaces());
110 ostr << "\n\t";
111 if (ik!=IK_Cell)
112 if (m_item->nbCell()!=0)
113 ItemPrinter::Internal::_printItemSubItems(ostr,"Cells",m_item->internalCells());
114 }
115}

120void NeighborItemPrinter::
121_printSubItems(std::ostream& ostr,Integer level, Integer levelmax,
122 ItemInternalVectorView sub_items,const char* name)
123{
124 indent(ostr,levelmax-level) << String::plural(sub_items.size(),name) << ":\n";
125 for( ItemInternal* sub_item : sub_items){
126 indent(ostr,levelmax-level) << "\t" << name;
127 print(ostr,sub_item,level-1,levelmax);
128 ostr << "\n";
129 }
130}

135void NeighborItemPrinter::
136print(std::ostream& ostr, ItemInternal* item, Integer level, Integer levelmax)
137{
138 if (!item) {
139 ostr << "(null pointer)";
140 }
141 else if (item->null()) {
142 ostr << "(null Item)";
143 }
144 else {
145 eItemKind ik = item->kind();
146 ostr << "(";
147 ItemPrinter::Internal::_printBasics(ostr,item);
148 ItemPrinter::Internal::_printFlags(ostr,item->flags());
149 ItemPrinter::Internal::_printParents(ostr,item);
150 ItemPrinter::Internal::_printErrors(ostr,item);
151 ostr << ")";
152 if (level > 0) {
153 ostr << "\n";
154 if (ik!=IK_Node)
155 if (item->nbNode()!=0)
156 _printSubItems(ostr,level,levelmax,item->internalNodes(),"node");
157 if (item->nbEdge()!=0)
158 _printSubItems(ostr,level,levelmax,item->internalEdges(),"edge");
159 if (item->nbFace()!=0)
160 _printSubItems(ostr,level,levelmax,item->internalFaces(),"face");
161 if (ik!=IK_Cell)
162 if (item->nbCell()!=0)
163 _printSubItems(ostr,level,levelmax,item->internalCells(),"cell");
164 }
165 }

195void ItemPrinter::Internal::
196_printFlags(std::ostream& o, Integer flags)
197{
198 Integer position = 0;
199 o << ",flags=";
200 if (flags & ItemInternal::II_Boundary)
201 o << ((position++)?"|":"") << "Boundary";
202 if (flags & ItemInternal::II_HasFrontCell)
203 o << ((position++)?"|":"") << "HasFrontCell";
204 if (flags & ItemInternal::II_HasBackCell)
205 o << ((position++)?"|":"") << "HasBackCell";
206 if (flags & ItemInternal::II_FrontCellIsFirst)
207 o << ((position++)?"|":"") << "FrontCellIsFirst";
208 if (flags & ItemInternal::II_BackCellIsFirst)
209 o << ((position++)?"|":"") << "BackCellIsFirst";
210 if (flags & ItemInternal::II_Own)
211 o << ((position++)?"|":"") << "Own";
212 if (flags & ItemInternal::II_Added)
213 o << ((position++)?"|":"") << "Added";
214 if (flags & ItemInternal::II_Suppressed)
215 o << ((position++)?"|":"") << "Suppressed";
216 if (flags & ItemInternal::II_Shared)
217 o << ((position++)?"|":"") << "Shared";
218 if (flags & ItemInternal::II_SubDomainBoundary)
219 o << ((position++)?"|":"") << "SubDomainBoundary";
220// if (flags & ItemInternal::II_JustRemoved)
221// o << ((position++)?"|":"") << "JustRemoved";
222 if (flags & ItemInternal::II_JustAdded)
223 o << ((position++)?"|":"") << "JustAdded";
224 if (flags & ItemInternal::II_NeedRemove)
225 o << ((position++)?"|":"") << "NeedRemove";
226 if (flags & ItemInternal::II_SlaveFace)
227 o << ((position++)?"|":"") << "SlaveFace";
228 if (flags & ItemInternal::II_MasterFace)
229 o << ((position++)?"|":"") << "MasterFace";
230 if (flags & ItemInternal::II_Detached)
231 o << ((position++)?"|":"") << "Detached";

236 if (flags & ItemInternal::II_UserMark2)

261
262void ItemPrinter::Internal::
263_printErrors(std::ostream& o, ItemInternal * item)
264{
265 if (item->isSuppressed())
266 return;
267 const char * str = ", error=";
268 Integer nerror = 0;
269 Int32 mesh_rank = item->family()->mesh()->meshPartInfo().partRank();
270 if (item->isOwn() != (mesh_rank == item->owner()))
271 o << ((nerror++==0)?str:"|") << "WRONG ISOWN";
272 if (item->nbParent() > 0){
273 // Les erreurs sont ici exclusives car invalidantes pour l'item
274 ItemInternal * parent = item->parent(0);
275 if (parent->uniqueId() != item->uniqueId())
276 o << ((nerror++==0)?str:"|") << "PARENT UID MISMATCH";
277 else if (parent->isSuppressed())
278 o << ((nerror++==0)?str:"|") << "SUPPRESSED PARENT";
279 else if (parent->owner() != item->owner())
280 o << ((nerror++==0)?str:"|") << "PARENT OWNER MISMATCH";
281 }
282}

287void ItemPrinter::Internal::
288_printItemSubItems(std::ostream& ostr, String name,const ItemInternalVectorView& enumerator)
289{
290 ostr << " " << name << " count=" << enumerator.size();
291 ostr << " (uids=";
292 for( ItemInternal* item : enumerator){
293 if (item){
294 if (item->localId() != NULL_ITEM_ID) {
295 if (item && !item->null())
296 ostr << " " << item->uniqueId();
297 else
298 ostr << " (null)";
299 }
300 else
301 ostr << " (null)";
302 }
303 else
304 ostr << " (nullptr)";
305 }
306 ostr << ", lids=";
307 for( ItemInternal* item : enumerator){
308 if (item){
309 if (item->localId() != NULL_ITEM_ID) {
310 if (item && !item->null())
311 ostr << " " << item->localId();
312 else
313 ostr << " (null)";
314 }
315 else
316 ostr << " (null)";
317 }
318 else
319 ostr << " (nullptr)";
320 }
321 ostr << ")";
322}
