125ItemInternalVectorView ItemInternal::
126activeCells(Int32Array& local_ids) const
127{
128 const Integer nbcell = this->nbCell();
129 for(Integer icell = 0 ; icell < nbcell ; ++icell) {
130 ItemInternal* cell = this->internalCell(icell);
131 if (cell->isActive()){
132 const Int32 local_id= cell->localId();
133 local_ids.add(local_id);
134 }
135 }
136 return ItemInternalVectorView(m_shared_info->m_items->cells,local_ids);
137}

141
142ItemInternalVectorView ItemInternal::
143activeFaces(Int32Array& local_ids) const
144{
145 const Integer nbface = this->nbFace();
146 for(Integer iface = 0 ; iface < nbface ; ++iface) {
147 ItemInternal* face = this->internalFace(iface);
148 if (!face->isBoundary()){
149 ItemInternal* bcell = face->backCell();
150 ItemInternal* fcell = face->frontCell();
151 if ( (bcell && bcell->isActive()) && (fcell && fcell->isActive()) )
152 local_ids.add(face->localId());
153 }
154 else {
155 ItemInternal* bcell = face->boundaryCell();
156 if ( (bcell && bcell->isActive()) )
157 local_ids.add(face->localId());
158 }
159 }
160 return ItemInternalVectorView(m_shared_info->m_items->faces,local_ids);
161}

188void ItemInternal::
189_setFaceBackAndFrontCells(Int32 back_cell_lid,Int32 front_cell_lid)
190{
191 if (front_cell_lid==NULL_ITEM_LOCAL_ID){
192 // Reste uniquement la back_cell ou aucune maille.
193 Int32 mod_flags = (back_cell_lid!=NULL_ITEM_LOCAL_ID) ? (II_Boundary | II_HasBackCell | II_BackCellIsFirst) : 0;
194 _setFaceInfos(mod_flags);
195 }
196 else if (back_cell_lid==NULL_ITEM_LOCAL_ID){
197 // Reste uniquement la front cell
198 _setFaceInfos(II_Boundary | II_HasFrontCell | II_FrontCellIsFirst);
199 }
200 else{
201 // Il y a deux mailles connectées.
202 _setFaceInfos(II_HasFrontCell | II_HasBackCell | II_BackCellIsFirst);
203 }
204}

209void ItemInternal::
210_internalCopyAndChangeSharedInfos(ItemSharedInfo* old_isi,ItemSharedInfo* new_isi,Integer new_data_index)
211{
212 ItemInternal* item = this;
213
214 Integer old_flags = item->flags();
215 Integer old_owner = item->owner();
216 Int32* old_data = item->dataPtr();
217
218 item->setSharedInfo(new_isi);
219 item->setDataIndex(new_data_index);
220
221 new_isi->setFlags(new_data_index,old_flags);
222 new_isi->setOwner(new_data_index,old_owner);
223
224 Int32* new_data = item->dataPtr();
225
226 ::memcpy(new_data+new_isi->firstNode(),old_data+old_isi->firstNode(),old_isi->nbNode()*sizeof(Int32));
227 ::memcpy(new_data+new_isi->firstParent(),old_data+old_isi->firstParent(),old_isi->nbParent()*sizeof(Int32));
228}
