173/*!
174 * \ingroup ArcaneCartesianMesh
175 * \brief Infos sur les mailles d'une direction spécifique X,Y ou Z d'un maillage structuré.
176 *
177 * Cette classe contient les informations pour retourner la liste
178 * des mailles dans une direction donnée et pour ces mailles
179 * connaitre la maille avant et après dans cette direction.
180 *
181 * Les instances de cette classe sont gérées par un ICartesianMesh et sont
182 * temporaires. Il ne faut pas les conserver d'une itération à l'autre
183 * car elles sont invalidées si le maillage change.
184 *
185 * Cette classe à une sémantique par référence.
186 *
187 * Par exemple, pour itérer sur les mailles de la direction X:
188
189 \code
190 * ICartesianMesh* cartesian_mesh = ...;
191 * CellDirectionMng cdm(cartesian_mesh->cellDirection(MD_DirX));
192 * ENUMERATE_CELL(icell,cdm.allCells()){
193 * DirCell dir_cell(cdm[icell]);
194 * Cell next = dir_cell.next();
195 * Cell prev = dir_cell.previous();
196 * }
197 \endcode
198 *
199 */
200class ARCANE_CARTESIANMESH_EXPORT CellDirectionMng
201{
202 friend CartesianMeshImpl;
203 friend CartesianMeshPatch;
204 class Impl;
205 static const int MAX_NB_NODE = 8;
