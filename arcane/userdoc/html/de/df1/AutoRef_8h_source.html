26/*!
27 * \brief Encapsulation d'un pointeur avec compteur de référence.
28 *
29 Cette classe renferme un pointeur d'un type qui implémente les méthodes
30 de la classe abstraite ISharedReference (le paramètre template n'a pas
31 besoin de dériver de cette classe) et incrémente (_addRef()) ou décrémente
32 (_removeRef()) le compteur de référence de l'élément pointé lors des
33 affectations succéssives. Cette classe n'effectue aucune action basée
34 sur la valeur de compteur de référence; la destruction éventuelle de l'objet
35 lorsque le compteur de référence arrive à zéro est gérée par l'objet lui même.
36
37 \ingroup Core
38 \since 0.2.9
39 \author Gilles Grospellier
40 \date 06/10/2000
41 */
42template<class T>
43class AutoRefT
44: public PtrT<T>
45{
46 public:
47
48 //! Type de la classe de base
49 typedef PtrT<T> BaseClass;
50
51 using BaseClass::m_value;
52
53 public:
54
55 //! Construit une instance sans référence
56 AutoRefT() : BaseClass(0) {}
57 //! Construit une instance référant \a t
58 explicit AutoRefT(T* t) : BaseClass(0) { _changeValue(t); }
59 //! Construit une référence référant \a from
60 AutoRefT(const AutoRefT<T>& from)
61 : BaseClass(0) { _changeValue(from.m_value); }
62
63 //! Opérateur de copie
64 const AutoRefT<T>& operator=(const AutoRefT<T>& from)
65 { _changeValue(from.m_value); return (*this); }
66
67 //! Affecte à l'instance la value \a new_value
68 const AutoRefT<T>& operator=(T* new_value)
69 { _changeValue(new_value); return (*this); }
70
71 //! Destructeur. Décrément le compteur de référence de l'objet pointé
72 ~AutoRefT() { _removeRef(); }
73
74 private:
75
76 //! Ajoute une référence à l'objet encapsulé si non nul
77 void _addRef()
78 {
79 if (m_value)
80 m_value->addRef();
81 }
82 //! Supprimer une référence à l'objet encapsulé si non nul
83 void _removeRef()
84 {
85 if (m_value)
86 m_value->removeRef();
87 }
88 //! Change l'objet référencé en \a new_value
89 void _changeValue(T* new_value)
90 {
91 if (m_value==new_value)
92 return;
93 _removeRef();
94 m_value = new_value;
95 _addRef();
96 }
97};
