27/*!
28 * \brief Interface d'un allocateur pour la mémoire.
29 *
30 * Cette classe définit une interface pour l'allocation mémoire utilisée
31 * par les classes tableaux de Arccore (Array, UniqueArray).
32 *
33 * Une instance de cette classe doit rester valide tant qu'il existe
34 * des tableaux l'utilisant.
35 */
36class ARCCORE_COLLECTIONS_EXPORT IMemoryAllocator
37{
38 public:
39
40 /*!
41 * \brief Détruit l'allocateur.
42 *
43 * Les objets alloués par l'allocateur doivent tous avoir été désalloués.
44 */
45 virtual ~IMemoryAllocator() = default;
46
47 public:
48
49 /*!
50 * \brief Indique si l'allocateur supporte la sémantique de realloc.
51 *
52 * Les allocateurs par défaut du C (malloc/realloc/free) supportent
53 * évidemment le realloc mais ce n'est pas forcément le cas
54 * des allocateurs spécifiques avec alignement mémoire (comme
55 * par exemple posix_memalign).
56 */
57 virtual bool hasRealloc() const =0;
58
59 /*!
60 * \brief Alloue de la mémoire pour \a new_size octets et retourne le pointeur.
61 *
62 * La sémantique est équivalent à malloc():
63 * - \a new_size peut valoir zéro et dans ce cas le pointeur retourné
64 * est soit nul, soit une valeur spécifique
65 * - le pointeur retourné peut être nul si la mémoire n'a pas pu être allouée.
66 */
67 virtual void* allocate(size_t new_size) =0;
68
69 /*!
70 * \brief Réalloue de la mémoire pour \a new_size octets et retourne le pointeur.
71 *
72 * Le pointeur \a current_ptr doit avoir été alloué via l'appel à
73 * allocate() ou reallocate() de cette instance.
74 *
75 * La sémantique de cette méthode est équivalente à realloc():
76 * - \a current_ptr peut-être nul auquel cas cet appel est équivalent
77 * à allocate().
78 * - le pointeur retourné peut être nul si la mémoire n'a pas pu être allouée.
79 */
80 virtual void* reallocate(void* current_ptr,size_t new_size) =0;
81
82 /*!
83 * \brief Libère la mémoire dont l'adresse de base est \a ptr.
84 *
85 * Le pointeur \a ptr doit avoir été alloué via l'appel à
86 * allocate() ou reallocate() de cette instance.
87 *
88 * La sémantique de cette méthode équivalente à free() et donc \a ptr
89 * peut être nul auquel cas aucune opération n'est effectuée.
90 */
91 virtual void deallocate(void* ptr) =0;
92
93 /*!
94 * \brief Ajuste la capacité suivant la taille d'élément.
95 *
96 * Cette méthode est utilisée pour éventuellement modifié le nombre
97 * d'éléments alloués suivant leur taille. Cela permet par exemple
98 * pour les allocateurs alignés de garantir que le nombre d'éléments
99 * alloués est un multiple de cet alignement.
100 *
101 */
102 virtual size_t adjustCapacity(size_t wanted_capacity,size_t element_size) =0;
103
104 /*!
105 * \brief Valeur de l'alignement garanti par l'allocateur.
106 *
107 * Cette méthode permet de s'assurer qu'un allocateur a un alignement suffisant
108 * pour certaines opérations comme la vectorisation par exemple.
109 *
110 * S'il n'y a aucune garantie, retourne 0.
111 */
112 virtual size_t guarantedAlignment() =0;
113};

117/*!
118 * \brief Allocateur mémoire via malloc/realloc/free.
119 *
120 * TODO: marquer les méthodes comme 'final'.
121 */
122class ARCCORE_COLLECTIONS_EXPORT DefaultMemoryAllocator
123: public IMemoryAllocator
124{
125 friend class ArrayImplBase;
126 friend class ArrayMetaData;
127
128 private:
129
130 static DefaultMemoryAllocator shared_null_instance;
131
132 public:
133
134 bool hasRealloc() const override;
135 void* allocate(size_t new_size) override;
136 void* reallocate(void* current_ptr,size_t new_size) override;
137 void deallocate(void* ptr) override;
138 size_t adjustCapacity(size_t wanted_capacity,size_t element_size) override;
139 size_t guarantedAlignment() override { return 0; }
140};

144/*!
145 * \brief Allocateur mémoire avec alignement mémoire spécifique.
146 *
147 * Cette classe s'utilise via les deux méthodes publiques Simd()
148 * et CacheLine() qui retournent repectivement un allocateur avec
149 * un alignement adéquat pour autoriser la vectorisation et un allocateur
150 * aligné sur une ligne de cache.
151 */
152class ARCCORE_COLLECTIONS_EXPORT AlignedMemoryAllocator
153: public IMemoryAllocator
154{
155 private:
156
157 static AlignedMemoryAllocator SimdAllocator;
158 static AlignedMemoryAllocator CacheLineAllocator;
159
160 public:
161
162 // TODO: essayer de trouver les bonnes valeurs en fonction de la cible.
163 // 64 est OK pour toutes les architectures x64 à la fois pour le SIMD
164 // et la ligne de cache.
165
166 // IMPORTANT: Si on change la valeur ici, il faut changer la taille de
167 // l'alignement de ArrayImplBase.
168
169 // TODO Pour l'instant seul un alignement sur 64 est autorisé. Pour
170 // autoriser d'autres valeurs, il faut modifier l'implémentation dans
171 // ArrayImplBase.
172
173 // TODO marquer les méthodes comme 'final'.
174
175 //! Alignement pour les structures utilisant la vectorisation
176 static constexpr Integer simdAlignment() { return 64; }
177 //! Alignement pour une ligne de cache.
178 static constexpr Integer cacheLineAlignment() { return 64; }
179
180 /*!
181 * \brief Allocateur garantissant l'alignement pour utiliser
182 * la vectorisation sur la plateforme cible.
183 *
184 * Il s'agit de l'alignement pour le type plus restrictif et donc il
185 * est possible d'utiliser cet allocateur pour toutes les structures vectorielles.
186 */
187 static AlignedMemoryAllocator* Simd()
188 {
189 return &SimdAllocator;
190 }
191
192 /*!
193 * \brief Allocateur garantissant l'alignement sur une ligne de cache.
194 */
195 static AlignedMemoryAllocator* CacheLine()
196 {
197 return &CacheLineAllocator;
198 }
199
200 protected:
201
202 AlignedMemoryAllocator(Integer alignment)
203 : m_alignment((size_t)alignment){}
204
205 public:
206
207 bool hasRealloc() const override;
208 void* allocate(size_t new_size) override;
209 void* reallocate(void* current_ptr,size_t new_size) override;
210 void deallocate(void* ptr) override;
211 size_t adjustCapacity(size_t wanted_capacity,size_t element_size) override;
212 size_t guarantedAlignment() override { return m_alignment; }
213
214 private:
215
216 size_t m_alignment;
217};
