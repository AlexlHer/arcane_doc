137void NodeDirectionMng::
138_internalComputeInfos(const CellDirectionMng& cell_dm,const NodeGroup& all_nodes,
139 const VariableCellReal3& cells_center)
140{
141 Node null_node;
142 m_infos.fill(NodeDirectionMng::ItemDirectionInfo(null_node.internal(),null_node.internal()));
143
144 Integer mesh_dim = m_p->m_cartesian_mesh->mesh()->dimension();
145 //TODO: ne garder que les noeuds de notre patch
146
147 // Calcul les infos de direction pour les noeuds
148 ENUMERATE_CELL(icell,cell_dm.allCells()){
149 Cell cell = *icell;
150 DirCellNode cn(cell_dm.cellNode(cell));
151
152 Node node_next_left = cn.nextLeft();
153 Node node_next_right = cn.nextRight();
154
155 Node node_previous_left = cn.previousLeft();
156 Node node_previous_right = cn.previousRight();
157
158 m_infos[node_previous_left.localId()].m_next_item = node_next_left.internal();
159 m_infos[node_next_left.localId()].m_previous_item = node_previous_left.internal();
160
161 m_infos[node_previous_right.localId()].m_next_item = node_next_right.internal();
162 m_infos[node_next_right.localId()].m_previous_item = node_previous_right.internal();
163
164 if (mesh_dim==3){
165 Node top_node_next_left = cn.topNextLeft();
166 Node top_node_next_right = cn.topNextRight();
167
168 Node top_node_previous_left = cn.topPreviousLeft();
169 Node top_node_previous_right = cn.topPreviousRight();
170
171 m_infos[top_node_previous_left.localId()].m_next_item = top_node_next_left.internal();
172 m_infos[top_node_next_left.localId()].m_previous_item = top_node_previous_left.internal();
173
174 m_infos[top_node_previous_right.localId()].m_next_item = top_node_next_right.internal();
175 m_infos[top_node_next_right.localId()].m_previous_item = top_node_previous_right.internal();
176 }
177 }
178
179 Int32UniqueArray inner_lids;
180 Int32UniqueArray outer_lids;
181 IItemFamily* family = all_nodes.itemFamily();
182 ENUMERATE_ITEM(iitem,all_nodes){
183 Int32 lid = iitem.itemLocalId();
184 ItemInternal* i1 = m_infos[lid].m_next_item;
185 ItemInternal* i2 = m_infos[lid].m_previous_item;
186 if (i1->null() || i2->null())
187 outer_lids.add(lid);
188 else
189 inner_lids.add(lid);
190 }
191 int dir = (int)m_direction;
192 String base_group_name = String("Direction")+dir;
193 if (m_p->m_patch_index>=0)
194 base_group_name = base_group_name + String("AMRPatch")+m_p->m_patch_index;
195 m_p->m_inner_all_items = family->createGroup(String("AllInner")+base_group_name,inner_lids,true);
196 m_p->m_outer_all_items = family->createGroup(String("AllOuter")+base_group_name,outer_lids,true);
197 m_p->m_all_items = all_nodes;
198
199 _filterNodes();
200 _computeNodeCellInfos(cell_dm,cells_center);
201}

205/*!
206 * \brief Filtre les noeuds devant/derrière pour ne garder que les
207 * noeuds de notre patch.
208 */
209void NodeDirectionMng::
210_filterNodes()
211{
212 // Ensemble contenant uniquement les noeuds de notre patch
213 std::set<NodeLocalId> nodes_set;
214 ENUMERATE_NODE(inode,allNodes()){
215 nodes_set.insert(NodeLocalId(inode.itemLocalId()));
216 }
217
218 Node null_node;
219
220 for( ItemDirectionInfo& idi : m_infos ){
221 {
222 ItemInternal* next = idi.m_next_item;
223 if (!next->null())
224 if (nodes_set.find(NodeLocalId(next->localId()))==nodes_set.end())
225 idi.m_next_item = null_node.internal();
226 }
227 {
228 ItemInternal* prev = idi.m_previous_item;
229 if (!prev->null())
230 if (nodes_set.find(NodeLocalId(prev->localId()))==nodes_set.end())
231 idi.m_previous_item = null_node.internal();
232 }
233 }
234}

238/*!
239 * \brief Calcul des connectivités noeuds/mailles par direction.
240 */
241void NodeDirectionMng::
242_computeNodeCellInfos(const CellDirectionMng& cell_dm,const VariableCellReal3& cells_center)
243{
244 // TODO: ne traiter que les mailles de notre patch.
245 IndexType indexes_ptr[8];
246 ArrayView<IndexType> indexes(8,indexes_ptr);
247
248 NodeDirectionMng& node_dm = *this;
249 NodeGroup dm_all_nodes = node_dm.allNodes();
250 eMeshDirection dir = m_direction;
251 IMesh* mesh = m_p->m_cartesian_mesh->mesh();
252 Integer mesh_dim = mesh->dimension();
253 VariableNodeReal3& nodes_coord = mesh->nodesCoordinates();
254 if (mesh_dim!=2 && mesh_dim!=3)
255 ARCANE_FATAL("Invalid mesh dimension '{0}'. Valid dimensions are 2 or 3",mesh_dim);
256
257 // Ensemble contenant uniquement les mailles de notre patch
258 // Cela sert à filtrer pour ne garder que ces mailles là dans la connectivité
259 std::set<CellLocalId> inside_cells;
260 ENUMERATE_CELL(icell,cell_dm.allCells()){
261 inside_cells.insert(CellLocalId(icell.itemLocalId()));
262 }
263
264 ENUMERATE_NODE(inode,dm_all_nodes){
265 Node node = *inode;
266 Integer nb_cell = node.nbCell();
267 Real3 node_pos = nodes_coord[node];
268 indexes.fill(DirNode::NULL_CELL);
269 for( Integer i=0; i<nb_cell; ++i ){
270 const IndexType bi = (IndexType)i;
271 Cell cell = node.cell(i);
272 if (inside_cells.find(CellLocalId(cell.localId()))==inside_cells.end())
273 continue;
274
275 Real3 center = cells_center[cell];
276 Real3 wanted_cell_pos;
277 Real3 wanted_node_pos;
278 if (dir==MD_DirX){
279 wanted_cell_pos = center;
280 wanted_node_pos = node_pos;
281 } else if (dir==MD_DirY){
282 wanted_cell_pos = Real3(center.y, -center.x, center.z);
283 wanted_node_pos = Real3(node_pos.y, -node_pos.x, node_pos.z);
284 } else if (dir==MD_DirZ){
285 // TODO: à vérifier pour Y et Z
286 wanted_cell_pos = Real3(center.z, -center.y, center.x);
287 wanted_node_pos = Real3(node_pos.z, -node_pos.y, node_pos.x);
288 }
289 bool is_top = ((wanted_cell_pos.z > wanted_node_pos.z) && mesh_dim==3);
290 if (!is_top){
291 if (wanted_cell_pos.x > wanted_node_pos.x ){
292 if (wanted_cell_pos.y > wanted_node_pos.y )
293 indexes_ptr[CNP_NextLeft] = bi;
294 else
295 indexes_ptr[CNP_NextRight] = bi;
296 }
297 else{
298 if (wanted_cell_pos.y > wanted_node_pos.y )
299 indexes_ptr[CNP_PreviousLeft] = bi;
300 else
301 indexes_ptr[CNP_PreviousRight] = bi;
302 }
303 }
304 else{
305 if (wanted_cell_pos.x > wanted_node_pos.x ){
306 if (wanted_cell_pos.y > wanted_node_pos.y )
307 indexes_ptr[CNP_TopNextLeft] = bi;
308 else
309 indexes_ptr[CNP_TopNextRight] = bi;
310 }
311 else{
312 if (wanted_cell_pos.y > wanted_node_pos.y )
313 indexes_ptr[CNP_TopPreviousLeft] = bi;
314 else
315 indexes_ptr[CNP_TopPreviousRight] = bi;
316 }
317 }
318 }
319 m_infos[node.localId()].setCellIndexes(indexes_ptr);
320 }
321}
