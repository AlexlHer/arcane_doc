98template<class ItemType,class ValueType> inline void
99_writeInfo(ISubDomain* mng,const VariableCollection& variables,const ItemType& item)
100{
101 typedef typename MeshVariableInfoT<ItemType,ValueType,0>::PrivateType VariableScalarTrueType;
102 typedef typename MeshVariableInfoT<ItemType,ValueType,1>::PrivateType VariableArrayTrueType;
103
104 ITraceMng* trace = mng->traceMng();
105 eItemKind item_kind = ItemTraitsT<ItemType>::kind();
106
107 for( VariableCollection::Enumerator i(variables); ++i; ){
108 IVariable* vp = *i;
109 // Pour l'instant, n'affiche pas les variables partielles
110 if (vp->isPartial())
111 continue;
112 // Vérifie qu'il s'agit bien d'une variable du maillage et du bon genre
113 ItemGroup group = vp->itemGroup();
114 if (group.itemKind()!=item_kind)
115 continue;
116
117 const String& name = vp->name();
118 auto* v = dynamic_cast<VariableScalarTrueType*>(vp);
119 if (v){
120 ConstArrayView<ValueType> values = v->valueView();
121 if (values.size() < item.localId()){
122 trace->error() << "Invalid dimensions for variable '" << name << "' "
123 << "(size=" << values.size() << " index=" << item.localId();
124 continue;
125 }
126 _writeValue(trace,name,values[item.localId()]);
127 }
128
129 auto* v2 = dynamic_cast<VariableArrayTrueType*>(vp);
130 if (v2){
131 Array2View<ValueType> values = v2->valueView();
132 Integer lid = item.localId();
133 if (values.dim1Size() < lid){
134 trace->error() << "Invalid dimensions for variable '" << name << "' "
135 << "(size=" << values.dim1Size() << " index=" << lid;
136 continue;
137 }
138 Integer n = values[lid].size();
139 for( Integer z=0; z<n; ++z ){
140 _writeValue(trace,name+"["+z+"]",values[lid][z]);
141 }
142 }
143
144 }
145}

395class _CompareItemWithNodes
396{
397 public:
398 bool operator()(const ItemWithNodes& cell1,const ItemWithNodes& cell2)
399 {
400 Integer nb1 = cell1.nbNode();
401 Integer nb2 = cell2.nbNode();
402 if (nb1<nb2)
403 return true;
404 if (nb2<nb1)
405 return false;
406 Integer n = nb1;
407 for( Integer i=0; i<n; ++i ){
408 Integer n1 = m_nodes_sorted_id[cell1.node(i).localId()];
409 Integer n2 = m_nodes_sorted_id[cell2.node(i).localId()];
410 if (n1!=n2)
411 return n1<n2;
412 }
413 return false;
414 //return n1<n2;
415 }
416 public:
417 _CompareItemWithNodes(IMesh* mesh,eItemKind ik,Int32ConstArrayView nodes_sorted_id)
418 : m_items(mesh->itemsInternal(ik)), m_nodes_sorted_id(nodes_sorted_id)
419 {
420 }
421 ItemInternalList m_items;
422 Int32ConstArrayView m_nodes_sorted_id;
423};

509void mesh_utils::
510writeMeshInfosSorted(IMesh* mesh,const String& file_name)
511{
512 if (!mesh)
513 return;
514 std::ofstream ofile(file_name.localstr());
515 ofile.precision(FloatInfo<Real>::maxDigit()+1);
516 Integer nb_node = mesh->nbNode();
517 Integer nb_edge = mesh->nbEdge();
518 Integer nb_face = mesh->nbFace();
519 Integer nb_cell = mesh->nbCell();
520 ofile << "** Mesh Sorted --> "
521 << " Nodes " << nb_node
522 << " Edges " << nb_edge
523 << " Faces " << nb_face
524 << " Cells " << nb_cell
525 << "\n";
526 UniqueArray<Node> sorted_nodes(nb_node);
527 UniqueArray<Edge> sorted_edges(nb_edge);
528 UniqueArray<Face> sorted_faces(nb_face);
529 UniqueArray<Cell> sorted_cells(nb_cell);
530
531 Int32UniqueArray nodes_sorted_id(nb_node);
532 Int32UniqueArray edges_sorted_id(nb_edge);
533 Int32UniqueArray faces_sorted_id(nb_face);
534 Int32UniqueArray cells_sorted_id(nb_cell);
535 {
536 // Trie les noeuds par ordre croissant de leur coordonnées
537 ItemInternalList nodes(mesh->itemsInternal(IK_Node));
538 for( Integer i=0; i<nb_node; ++i )
539 sorted_nodes[i] = nodes[i];
540 {
541 SharedVariableNodeReal3 shared_coords(mesh->sharedNodesCoordinates());
542 _CompareNodes compare_nodes(mesh,shared_coords);
543 std::sort(std::begin(sorted_nodes),std::end(sorted_nodes),compare_nodes);
544 }
545 for( Integer i=0; i<nb_node; ++i )
546 nodes_sorted_id[sorted_nodes[i].localId()] = i;
547
548 // Trie les arêtes
549 ItemInternalList edges(mesh->itemsInternal(IK_Edge));
550 for( Integer i=0; i<nb_edge; ++i )
551 sorted_edges[i] = edges[i];
552 {
553 _CompareItemWithNodes compare_edges(mesh,IK_Edge,edges_sorted_id);
554 std::sort(std::begin(sorted_edges),std::end(sorted_edges),compare_edges);
555 }
556 for( Integer i=0; i<nb_edge; ++i )
557 edges_sorted_id[sorted_edges[i].localId()] = i;
558
559 // Trie les faces
560 ItemInternalList faces(mesh->itemsInternal(IK_Face));
561 for( Integer i=0; i<nb_face; ++i )
562 sorted_faces[i] = faces[i];
563 {
564 _CompareItemWithNodes compare_faces(mesh,IK_Face,nodes_sorted_id);
565 std::sort(std::begin(sorted_faces),std::end(sorted_faces),compare_faces);
566 }
567 for( Integer i=0; i<nb_face; ++i )
568 faces_sorted_id[sorted_faces[i].localId()] = i;
569
570 // Trie les mailles
571 ItemInternalList cells(mesh->itemsInternal(IK_Cell));
572 for( Integer i=0; i<nb_cell; ++i )
573 sorted_cells[i] = cells[i];
574 {
575 _CompareItemWithNodes compare_cells(mesh,IK_Cell,nodes_sorted_id);
576 std::sort(std::begin(sorted_cells),std::end(sorted_cells),compare_cells);
577 }
578 for( Integer i=0; i<nb_cell; ++i )
579 cells_sorted_id[sorted_cells[i].localId()] = i;
580
581 SharedVariableNodeReal3 coords(mesh->sharedNodesCoordinates());
582
583 ofile << "** Nodes\n";
584 Int32UniqueArray lids;
585 for( Integer i=0; i<nb_node; ++i ){
586 const Node& node = sorted_nodes[i];
587 ofile << "Node: " << i << " Coord: " << coords[node];
588
589 _fillSorted(node,lids,edges_sorted_id,EdgeFiller());
590 std::sort(std::begin(lids),std::end(lids));
591 _writeItems(ofile,"Edges",lids);
592
593 _fillSorted(node,lids,faces_sorted_id,FaceFiller());
594 std::sort(std::begin(lids),std::end(lids));
595 _writeItems(ofile,"Faces",lids);
596
597 _fillSorted(node,lids,cells_sorted_id,CellFiller());
598 std::sort(std::begin(lids),std::end(lids));
599 _writeItems(ofile,"Cells",lids);
600
601 ofile << '\n';
602 }

1348
1349bool mesh_utils::
1350reorderNodesOfFace(Int64ConstArrayView before_ids,Int64ArrayView after_ids)
1351{
1352 // \a true s'il faut réorienter les faces pour que leur orientation
1353 // soit indépendante du partitionnement du maillage initial.
1354 bool need_swap_orientation = false;
1355 Integer min_node_index = 0;
1356 Integer nb_node = before_ids.size();
1357
1358 // Traite directement le cas des arêtes
1359 if (nb_node==2){
1360 if (before_ids[0]<before_ids[1]) {
1361 after_ids[0] = before_ids[0];
1362 after_ids[1] = before_ids[1];
1363 return false;
1364 }
1365 after_ids[0] = before_ids[1];
1366 after_ids[1] = before_ids[0];
1367 return true;
1368 }
1369
1370 // L'algorithme suivant oriente les faces en tenant compte uniquement
1371 // de l'ordre de la numérotation de ces noeuds. Si cet ordre est
1372 // conservé lors du partitionnement, alors l'orientation des faces
1373 // sera aussi conservée.
1374
1375 // L'algorithme est le suivant:
1376 // - Recherche le noeud n de plus petit indice.
1377 // - Recherche n-1 et n+1 les indices de ses 2 noeuds voisins.
1378 // - Si (n+1) est inférieur à (n-1), l'orientation n'est pas modifiée.
1379 // - Si (n+1) est supérieur à (n-1), l'orientation est inversée.
1380
1381 // Recherche le noeud de plus petit indice
1382
1383 Int64 min_node = INT64_MAX;
1384 for( Integer k=0; k<nb_node; ++k ){
1385 Int64 id = before_ids[k];
1386 if (id<min_node){
1387 min_node = id;
1388 min_node_index = k;
1389 }
1390 }
1391 Int64 next_node = before_ids[(min_node_index+1) % nb_node];
1392 Int64 prev_node = before_ids[(min_node_index+(nb_node-1)) % nb_node];
1393 if (next_node>prev_node)
1394 need_swap_orientation = true;
1395 if (need_swap_orientation){
1396 for( Integer k=0; k<nb_node; ++k ){
1397 Integer index = (nb_node-k+min_node_index) % nb_node;
1398 after_ids[k] = before_ids[index];
1399 }
1400 }
1401 else{
1402 for( Integer k=0; k<nb_node; ++k ){
1403 Integer index = (k+min_node_index) % nb_node;
1404 after_ids[k] = before_ids[index];
1405 }
1406 }
1407 return need_swap_orientation;
1408}
