30/*!
31 * \brief Implémentation d'un buffer de sérialisation contigu en mémoire.
32 *
33 * Cette implémentation permet de sérialiser les données dans une zone
34 * contigue en mémoire et ainsi de l'envoyer en une seule fois
35 * via MPI par exemple.
36 * Le buffer est composé d'un première partie contenant NB_SIZE_ELEM
37 * objets de type Integer. Cette partie contient le nombre d'élément
38 * sérialisé de chaque type (Real, Int64, Int32, Int16, Byte). La deuxième
39 * partie contient les données sérialisées proprement dites.
40 *
41 * Lorsque cette classe est utilisée dans le cadre d'un appel MPI,
42 * on utilise le tableau \a m_size_copy_buffer pour envoyer les tailles.
43 * lors du premier message lorsque le message complet est envoyé en plusieurs fois.
44 * La norme MPI indique effectivement qu'un buffer utilisé lors d'un appel MPI ne doit plus être
45 * utilisé tant que cet appel n'est pas terminé.
46 */
47class BasicSerializerNewImpl
48: public BasicSerializer::Impl
49{

51 static constexpr int IDX_TAG = 0;
52 //! Tag identifiant la sérialisation
53 static constexpr Int64 SERIALIZE_TAG = 0x7a9b3cd0;
54 //! Version de la sérialisation
55 static constexpr int IDX_VERSION = 1;
56 //! Champ réservé pour des informations supplémentaires (par exemple compression)
57 static constexpr int IDX_RESERVED1 = 2;
58

60 static constexpr int IDX_TOTAL_SIZE = 3;
61
62 static constexpr int IDX_NB_BYTE = 4;

64 static constexpr int IDX_NB_FLOAT32 = 6;

66 static constexpr int IDX_NB_FLOAT128 = 8;

68 static constexpr int IDX_NB_INT32 = 10;
69 static constexpr int IDX_NB_INT64 = 11;
70 static constexpr int IDX_NB_INT128 = 12;

72 // Laisse de la place pour de nouveaux types.
73 static constexpr int IDX_POS_BYTE = 32;

124 Span<Real> getRealBuffer() override { return m_real_view; }
125 Span<Int16> getInt16Buffer() override { return m_int16_view; }

129 void allocateBuffer(Int64 nb_real,Int64 nb_int16,Int64 nb_int32,
130 Int64 nb_int64,Int64 nb_byte) override
131 {
132 SizeInfo size_info = getPaddingSize(NB_SIZE_ELEM,sizeof(Int64));
133 Int64 total = size_info.m_padded_size;
134
135 Int64 real_position = total;
136 SizeInfo padded_real_size = getPaddingSize(nb_real,sizeof(Real));
137 total += padded_real_size.m_padded_size;
138
139 Int64 int16_position = total;
140 SizeInfo padded_int16_size = getPaddingSize(nb_int16,sizeof(Int16));
141 total += padded_int16_size.m_padded_size;
142
143 Int64 int32_position = total;
144 SizeInfo padded_int32_size = getPaddingSize(nb_int32,sizeof(Int32));
145 total += padded_int32_size.m_padded_size;
146
147 Int64 int64_position = total;
148 SizeInfo padded_int64_size = getPaddingSize(nb_int64,sizeof(Int64));
149 total += padded_int64_size.m_padded_size;
150
151 Int64 byte_position = total;
152 SizeInfo padded_byte_size = getPaddingSize(nb_byte,sizeof(Byte));
153 total += padded_byte_size.m_padded_size;
154
155 _allocBuffer(total);
156
157 _fillPadding(0,size_info);
158 _fillPadding(real_position,padded_real_size);
159 _fillPadding(int16_position,padded_int16_size);
160 _fillPadding(int32_position,padded_int32_size);
161 _fillPadding(int64_position,padded_int64_size);
162 _fillPadding(byte_position,padded_byte_size);
163
164 m_sizes_view = Int64ArrayView(NB_SIZE_ELEM,(Int64*)&m_buffer_view[0]);
165 m_sizes_view.fill(0);
166
167 m_sizes_view[IDX_TAG] = SERIALIZE_TAG;
168 m_sizes_view[IDX_VERSION] = 1;
169 m_sizes_view[IDX_RESERVED1] = 0;
170
171 m_sizes_view[IDX_TOTAL_SIZE] = total;
172 m_sizes_view[IDX_NB_FLOAT64] = nb_real;
173 m_sizes_view[IDX_NB_INT64] = nb_int64;
174 m_sizes_view[IDX_NB_INT32] = nb_int32;
175 m_sizes_view[IDX_NB_INT16] = nb_int16;
176 m_sizes_view[IDX_NB_BYTE] = nb_byte;
177
178 m_sizes_view[IDX_POS_FLOAT64] = real_position;
179 m_sizes_view[IDX_POS_INT64] = int64_position;
180 m_sizes_view[IDX_POS_INT32] = int32_position;
181 m_sizes_view[IDX_POS_INT16] = int16_position;
182 m_sizes_view[IDX_POS_BYTE] = byte_position;
183
184 m_real_view = Span<Real>((Real*)&m_buffer_view[real_position],nb_real);
185 m_int16_view = Span<Int16>((Int16*)&m_buffer_view[int16_position],nb_int16);
186 m_int32_view = Span<Int32>((Int32*)&m_buffer_view[int32_position],nb_int32);
187 m_int64_view = Span<Int64>((Int64*)&m_buffer_view[int64_position],nb_int64);
188 m_byte_view = Span<Byte>((Byte*)&m_buffer_view[byte_position],nb_byte);
189
190 _checkAlignment();
191 }

289 SizeInfo getPaddingSize(Int64 nb_elem,Int64 elem_size)
290 {
291 if (nb_elem<0)
292 ARCCORE_FATAL("Bad number of element '{0}' (should be >=0)",nb_elem);
293 if (elem_size<=0)
294 ARCCORE_FATAL("Bad elem_size '{0}'",elem_size);
295 Int64 s = nb_elem * elem_size;
296 Int64 pad = s % ALIGN_SIZE;
297 if (pad==0)
298 pad = ALIGN_SIZE;
299 Int64 new_size = s + (ALIGN_SIZE-pad);
300 if ( (new_size%ALIGN_SIZE)!=0 )
301 ARCCORE_FATAL("Bad padding {0}",new_size);
302 //std::cout << " nb_elem=" << nb_elem << " elem_size=" << elem_size << " s=" << s << " new_size=" << new_size << '\n';
303 return { s, new_size };
304 }
305
306 /*!
307 * \brief Remplit avec une valeur fixe les zones correspondantes au padding.
308 * Cela permet d'éviter d'avoir des valeurs non initialisées.
309 *
310 * Il faut avoir appeler _allocBuffer() avant
311 */
312 void _fillPadding(Int64 position,SizeInfo size_info)
313 {
314 Int64 begin = position + size_info.m_original_size;
315 Int64 end = position + size_info.m_padded_size;
316 _fillPadding(m_buffer_view.subspan(begin,end-begin));
317 }
318
319 void _fillPadding(Span<Byte> buf)
320 {
321 // Utilise une valeur non nulle pour repérer plus facilement
322 // les zones de padding si besoin.
323 constexpr Byte v = (Byte)(250);
324 for( Int64 i=0, s=buf.size(); i<s; ++i )
325 buf[i] = v;
326 }
327
328 void _checkAlignment()
329 {
330 _checkAddr(m_real_view.data());
331 _checkAddr(m_int16_view.data());
332 _checkAddr(m_int32_view.data());
333 _checkAddr(m_int64_view.data());
334 _checkAddr(m_byte_view.data());
335 }
336
337 void _checkAddr(void* ptr)
338 {
339 Int64 addr = (Int64)ptr;
340 if ((addr%ALIGN_SIZE)!=0){
341 _printAlignment();
342 ARCCORE_FATAL("Bad alignment addr={0} - {1}",addr,(addr % ALIGN_SIZE));
343 }
344 }
345
346 void _printAlignment()
347 {
348 for( Integer i=0, n=m_sizes_view.size(); i<n; ++i )
349 std::cout << " Size i=" << i << " v=" << m_sizes_view[i] << " pad=" << (m_sizes_view[i] % ALIGN_SIZE) << '\n';
350 _printAddr(m_buffer_view.data(),"Buffer");
351 _printAddr(m_real_view.data(),"Real");
352 _printAddr(m_int16_view.data(),"Int16");
353 _printAddr(m_int32_view.data(),"Int32");
354 _printAddr(m_int64_view.data(),"Int64");
355 _printAddr(m_byte_view.data(),"Byte");
356 }
357
358 void _printAddr(void* ptr,const String& name)
359 {
360 Int64 addr = (Int64)ptr;
361 std::cout << "Align type=" << name << " addr=" << addr << " offset=" << (addr % ALIGN_SIZE) << '\n';
362 }
363
364 void _allocBuffer(Int64 size)
365 {
366 if (size<1024)
367 size = 1024;
368 m_buffer.resize(size+ALIGN_SIZE*4);
369 Int64 addr = (Int64)(&m_buffer[0]);
370 Int64 padding = addr % ALIGN_SIZE;
371 Int64 position = 0;
372 if (padding!=0){
373 position = ALIGN_SIZE - padding;
374 }
375 // La taille doit être un multiple de ALIGN_SIZE;
376 Int64 new_size = (size + ALIGN_SIZE) - (size%ALIGN_SIZE);
377 m_buffer_view = m_buffer.span().subspan(position,new_size);
378
379 // Initialise les valeurs de la zone tampon
380 auto padding_view = m_buffer.span().subspan(position+size,new_size-size);
381 _fillPadding(padding_view);
382 }

411void BasicSerializer::Impl2::
412reserve(eDataType dt,Int64 n,Int64 nb_put)
413{
414 ARCCORE_ASSERT((m_mode==ModeReserve),("Bad mode"));
415 switch(dt){
416 case DT_Real: m_real.m_reserved_size += n; break;
417 case DT_Int64: m_int64.m_reserved_size += n; break;
418 case DT_Int32: m_int32.m_reserved_size += n; break;
419 case DT_Int16: m_int16.m_reserved_size += n; break;
420 case DT_Byte: m_byte.m_reserved_size += n; break;
421 default:
422 ARCCORE_THROW(ArgumentException,"bad datatype v={0}",(int)dt);
423 }
424 if (m_is_serialize_typeinfo)
425 // Pour le type de la donnée.
426 m_byte.m_reserved_size += nb_put;
427}
