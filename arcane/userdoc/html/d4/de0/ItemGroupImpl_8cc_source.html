98 exemple le groupe des entités propres) et ne sont donc pas

101 pas besoin d'être sauvée lors d'une protection.
102
103 \todo ajouter notion de groupe générée, avec les propriétés suivantes:

406/*!
407 * \brief Vérifie que les localIds() sont contigüs.
408 */
409void ItemGroupImplPrivate::
410checkIsContigous()
411{
412 m_is_contigous = false;
413 Int32ConstArrayView lids = itemsLocalId();
414 if (lids.empty()){
415 m_is_contigous = false;
416 return;
417 }
418 Int32 first_lid = lids[0];
419
420 bool is_bad = false;
421 for( Integer i=0, n=lids.size(); i<n; ++i ){
422 if (lids[i]!=(first_lid+i)){
423 is_bad = true;
424 break;
425 }
426 }
427 if (!is_bad)
428 m_is_contigous = true;
429}

1123void ItemGroupImpl::
1124addItems(Int32ConstArrayView items_local_id,bool check_if_present)
1125{
1126 ARCANE_ASSERT(( (!m_p->m_need_recompute && !isAllItems()) || (m_p->m_transaction_mode && isAllItems()) ),
1127 ("Operation on invalid group"));
1128 if (m_p->m_compute_functor && !m_p->m_transaction_mode)
1129 ARCANE_FATAL("Cannot add items on computed group ({0})", name());
1130 IMesh* amesh = mesh();
1131 if (!amesh)
1132 throw ArgumentException(A_FUNCINFO,"null group");
1133 ITraceMng* trace = amesh->traceMng();
1134
1135 Integer nb_item_to_add = items_local_id.size();
1136 if (nb_item_to_add==0)
1137 return;
1138
1139 Int32Array& items_lid = m_p->mutableItemsLocalId();
1140 const Integer current_size = items_lid.size();
1141 Integer nb_added = 0;
1142
1143 if(isAllItems()) {
1144 // Ajoute les nouveaux items à la fin
1145 Integer nb_items_id = current_size;
1146 m_p->m_items_index_in_all_group.resize(m_p->maxLocalId());
1147 for( Integer i=0, is=nb_item_to_add; i<is; ++i ){
1148 Int32 local_id = items_local_id[i];
1149 items_lid.add(local_id);
1150 m_p->m_items_index_in_all_group[local_id] = nb_items_id+i;
1151 }
1152 }
1153 else if (check_if_present) {
1154 // Vérifie que les entités à ajouter ne sont pas déjà présentes
1155 UniqueArray<bool> presence_checks(m_p->maxLocalId());
1156 presence_checks.fill(false);
1157 for( Integer i=0, is=items_lid.size(); i<is; ++i ){
1158 presence_checks[items_lid[i]] = true;
1159 }
1160
1161 for( Integer i=0; i<nb_item_to_add; ++i ) {
1162 const Integer lid = items_local_id[i];
1163 if (!presence_checks[lid]){
1164 items_lid.add(lid);
1165 // Met à \a true comme cela si l'entité est présente plusieurs fois
1166 // dans \a items_local_id cela fonctionne quand même.
1167 presence_checks[lid] = true;
1168 ++nb_added;
1169 }
1170 }
1171 }
1172 else {
1173 nb_added = nb_item_to_add;
1174 items_lid.resize(current_size+nb_added);
1175 Int32ArrayView ptr = items_lid;
1176 memcpy(&ptr[current_size],items_local_id.data(),sizeof(Int32)*nb_added);
1177 }
1178
1179 if (arcaneIsCheck()){
1180 trace->debug(Trace::High) << "ItemGroupImpl::addItems() group <" << name() << "> "
1181 << " checkpresent=" << check_if_present
1182 << " nb_current=" << current_size
1183 << " want_to_add=" << nb_item_to_add
1184 << " effective_added=" << nb_added;
1185 checkValid();
1186 }
1187
1188 if (nb_added!=0) {
1189 m_p->updateTimestamp();
1190 Int32ConstArrayView observation_info(nb_added, items_lid.unguardedBasePointer()+current_size);
1191 m_p->notifyExtendObservers(&observation_info);
1192 }

1197
1198void ItemGroupImpl::
1199removeItems(Int32ConstArrayView items_local_id,bool check_if_present)
1200{
1201 // TODO: check_if_present n'est pas pris en compte
1202 ARCANE_UNUSED(check_if_present);
1203
1204 ARCANE_ASSERT(( (!m_p->m_need_recompute && !isAllItems()) || (m_p->m_transaction_mode && isAllItems()) ),
1205 ("Operation on invalid group"));
1206 if (m_p->m_compute_functor && !m_p->m_transaction_mode)
1207 ARCANE_FATAL("Cannot remove items on computed group ({0})", name());
1208 IMesh* amesh = mesh();
1209 if (!amesh)
1210 throw ArgumentException(A_FUNCINFO,"null group");
1211 ITraceMng* trace = amesh->traceMng();
1212 if (isOwn() && amesh->meshPartInfo().nbPart()!=1){
1213 ARCANE_THROW(NotSupportedException,"Cannot remove items if isOwn() is true");
1214 }
1215
1216 Integer nb_item_to_remove = items_local_id.size();
1217 Int32UniqueArray removed_ids, removed_lids;
1218
1219 if (nb_item_to_remove!=0) { // on ne peut tout de même pas faire de retour anticipé à cause des observers
1220
1221 Int32Array& items_lid = m_p->mutableItemsLocalId();
1222 const Integer old_size = items_lid.size();
1223 bool has_removed = false;
1224
1225 if(isAllItems()) {
1226 // Algorithme anciennement dans DynamicMeshKindInfo
1227 // Supprime des items du groupe all_items par commutation avec les
1228 // éléments de fin de ce groupe
1229 // ie memoire persistante O(size groupe), algo O(remove items)
1230 has_removed = true;
1231 Integer nb_item = old_size;
1232 for( Integer i=0, is=nb_item_to_remove; i<is; ++i ){
1233 Int32 removed_local_id = items_local_id[i];
1234 Int32 index = m_p->m_items_index_in_all_group[removed_local_id];
1235 --nb_item;
1236 Int32 moved_local_id = items_lid[nb_item];
1237 items_lid[index] = moved_local_id;
1238 m_p->m_items_index_in_all_group[moved_local_id] = index;
1239 }
1240 items_lid.resize(nb_item);
1241 }
1242 else {
1243 // Algorithme pour les autres groupes
1244 // Décalage de tableau
1245 // ie mémoire locale O(size groupe), algo O(size groupe)
1246 // Marque un tableau indiquant les entités à supprimer
1247 BoolUniqueArray remove_flags(m_p->maxLocalId(),false);
1248 for( Integer i=0; i<nb_item_to_remove; ++i )
1249 remove_flags[items_local_id[i]] = true;
1250
1251 {
1252 Integer next_index = 0;
1253 for( Integer i=0; i<old_size; ++i ){
1254 Int32 lid = items_lid[i];
1255 if (remove_flags[lid]) {
1256 removed_ids.add(i);
1257 removed_lids.add(lid);
1258 continue;
1259 }
1260 items_lid[next_index] = lid;
1261 ++next_index;
1262 }
1263 if (next_index!=old_size) {
1264 has_removed = true;
1265 items_lid.resize(next_index);
1266 }
1267 }
1268 }
1269
1270 m_p->updateTimestamp();
1271 if(arcaneIsCheck()){
1272 trace->debug(Trace::High) << "ItemGroupImpl::removeItems() group <" << name() << "> "
1273 << " old_size=" << old_size
1274 << " new_size=" << size()
1275 << " removed?=" << has_removed;
1276 checkValid();
1277 }
1278 }
1279
1280 Int32ConstArrayView observation_info(removed_lids.size(), removed_lids.unguardedBasePointer());
1281 m_p->notifyReduceObservers(&observation_info);
1282}

1647/*---------------------------------------------------------------------------*/
1648/*---------------------------------------------------------------------------*/
1649/*!
1650 * \brief Remplit les derniers éléments du groupe pour avoir un vecteur
1651 * SIMD complet.
1652 *
1653 * Pour que la vectorisation fonctionne il faut que le nombre d'éléments
1654 * du groupe soit un multiple de la taille d'un vecteur SIMD. Si ce n'est
1655 * pas le cas, on remplit les dernières valeurs du tableau des localId()
1656 * avec le dernier élément.
1657 *
1658 * Par exemple, on supporse une taille d'un vecteur SIMD de 8 (ce qui est le maximum
1659 * actuellement avec l'AVX512) et un groupe \a grp de 13 éléments. Il faut donc
1660 * remplit le groupe comme suit:
1661 \code
1662 Int32 last_local_id = grp[12];
1663 grp[13] = grp[14] = grp[15] = last_local_id.
1664 \endcode
1665 *
1666 * A noter que la taille du groupe reste effectivement de 13 éléments. Le
1667 * padding supplémentaire n'est que pour les itérations via ENUMERATE_SIMD.
1668 * Comme le tableau des localId() est alloué avec l'allocateur d'alignement
1669 * il est garanti que la mémoire allouée est suffisante pour faire le padding.
1670 *
1671 * \todo Ne pas faire cela dans tous les checkNeedUpdate() mais mettre
1672 * en place une méthode qui retourne un énumérateur spécifique pour
1673 * la vectorisation.
1674 */
1675void ItemGroupImpl::
1676_checkUpdateSimdPadding()
1677{
1678 if (m_p->m_simd_timestamp >= m_p->timestamp())
1679 return;
1680 // Fait un padding des derniers éléments du tableau en recopiant la
1681 // dernière valeurs.
1682 m_p->m_simd_timestamp = m_p->timestamp();
1683 applySimdPadding(m_p->mutableItemsLocalId());
1684}
