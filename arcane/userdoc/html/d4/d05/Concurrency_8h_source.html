32/*!
33 * \ingroup Concurrency
34 *
35 * \brief Applique en concurrence la méthode \a function de l'instance
36 * \a instance sur la vue \a items_view avec les options \a options.
37 */
38template<typename InstanceType,typename ItemType> inline void
39arcaneParallelForeach(const ItemVectorView& items_view,const ParallelLoopOptions& options,
40 InstanceType* instance,void (InstanceType::*function)(ItemVectorViewT<ItemType> items))
41{
42 ItemRangeFunctorT<InstanceType,ItemType> ipf(items_view,instance,function,options.grainSize());
43 // Recopie \a options et utilise la valeur de 'grain_size' retournée par \a ifp
44 ParallelLoopOptions loop_opt(options);
45 loop_opt.setGrainSize(ipf.blockGrainSize());
46 TaskFactory::executeParallelFor(0,ipf.nbBlock(),loop_opt,&ipf);
47}
