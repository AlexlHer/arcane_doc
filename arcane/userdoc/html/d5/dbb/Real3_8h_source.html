70/*!
71 * \brief Classe gérant un vecteur de réel de dimension 3.
72
73 Le vecteur comprend trois composantes \a x, \a y et \a z qui sont du
74 type \b Real.
75
76 \code
77 Real3 value (1.0,2.3,4.5); // Créé un triplet (x=1.0, y=2.3, z=4.5)
78 cout << value.x; // Imprime le composant x
79 value.y += 3.2; // Ajoute 3.2 à la composante \b y
80 \endcode
81
82 ou de maniere equivalente
83
84 \code
85 Real3 value (1.0,2.3,4.5); // Créé un triplet (x=1.0, y=2.3, z=4.5)
86 cout << value[0]; // Imprime le composant x
87 value[1] += 3.2; // Ajoute 3.2 à la composante \b y
88 \endcode
89 */
90
91class ARCANE_UTILS_EXPORT Real3
92: public Real3POD
93{
94 public:
95
96 //! Construit le vecteur nul.
97 constexpr ARCCORE_HOST_DEVICE Real3() : Real3POD() { x=0.0; y=0.0; z=0.0; }
98 //! Construit le triplet (ax,ay,az)
99 constexpr ARCCORE_HOST_DEVICE Real3(Real ax,Real ay,Real az) : Real3POD() { x=ax; y=ay; z=az; }
100 //! Construit un triplet identique à \a f
101 constexpr ARCCORE_HOST_DEVICE Real3(const Real3& f) : Real3POD() { x=f.x; y=f.y; z=f.z; }
102 //! Construit un triplet identique à \a f
103 constexpr ARCCORE_HOST_DEVICE explicit Real3(const Real3POD& f) : Real3POD() { x=f.x; y=f.y; z=f.z; }
104 //! Opérateur de recopie.
105 constexpr ARCCORE_HOST_DEVICE Real3& operator=(Real3 f)
106 {
107 x=f.x; y=f.y; z=f.z;
108 return (*this);
109 }
110
111 //! Affecte à l'instance le triplet (v,v,v).
112 constexpr ARCCORE_HOST_DEVICE Real3& operator=(Real v)
113 {
114 x = y = z = v; return (*this);
115 }
116
117 public:
118
119 constexpr ARCCORE_HOST_DEVICE static Real3 null() { return Real3(0.,0.,0.); }
120 constexpr ARCCORE_HOST_DEVICE static Real3 zero() { return Real3(0.,0.,0.); }
121
122 public:
123
124 //! Retourne une copie du triplet.
125 constexpr ARCCORE_HOST_DEVICE Real3 copy() const { return (*this); }
126 //! Réinitialise le triplet avec les constructeurs par défaut.
127 constexpr ARCCORE_HOST_DEVICE Real3& reset() { x = y = z = 0.; return (*this); }
128 //! Affecte à l'instance le triplet (ax,ay,az)
129 constexpr ARCCORE_HOST_DEVICE Real3& assign(Real ax,Real ay,Real az)
130 { x = ax; y = ay; z = az; return (*this); }
131 //! Copie le triplet \a f
132 constexpr ARCCORE_HOST_DEVICE Real3& assign(Real3 f) { x = f.x; y = f.y; z = f.z; return (*this); }
133 /*!
134 * \brief Indique si l'instance est proche de l'instance nulle.
135 *
136 * \retval true si math::isNearlyZero() est vrai pour chaque composante.
137 * \retval false sinon.
138 */
139 constexpr ARCCORE_HOST_DEVICE bool isNearlyZero() const
140 {
141 return math::isNearlyZero(x) && math::isNearlyZero(y) && math::isNearlyZero(z);
142 }
143 //! Retourne la norme L2 au carré du triplet \f$x^2+y^2+z^2\f$
144 constexpr ARCCORE_HOST_DEVICE Real squareNormL2() const { return x*x + y*y + z*z; }
145 //! Retourne la norme L2 du triplet \f$\sqrt{x^2+y^2+z^2}\f$
146 ARCCORE_HOST_DEVICE Real normL2() const { return _sqrt(squareNormL2()); }
147 //! Retourne la norme au carré du triplet \f$x^2+y^2+z^2\f$
148 ARCCORE_DEPRECATED_2021("Use squareNormL2() instead")
149 constexpr ARCCORE_HOST_DEVICE Real abs2() const { return x*x + y*y + z*z; }
150 //! Retourne la norme du triplet \f$\sqrt{x^2+y^2+z^2}\f$
151 ARCCORE_DEPRECATED_2021("Use normL2() instead")
152 ARCCORE_HOST_DEVICE Real abs() const { return _sqrt(squareNormL2()); }
153 //! Valeur absolue composante par composante.
154 ARCCORE_HOST_DEVICE Real3 absolute() const { return Real3(math::abs(x),math::abs(y),math::abs(z)); }
155
156 /*!
157 * \brief Lit un triplet sur le flot \a i
158 * Le triplet est lu sous la forme de trois valeur de type #value_type.
159 */
160 std::istream& assign(std::istream& i);
161 //! Ecrit le triplet sur le flot \a o lisible par un assign()
162 std::ostream& print(std::ostream& o) const;
163 //! Ecrit le triplet sur le flot \a o sous la forme (x,y,z)
164 std::ostream& printXyz(std::ostream& o) const;
165
166 //! Ajoute \a b au triplet
167 constexpr ARCCORE_HOST_DEVICE Real3& add(Real3 b) { x+=b.x; y+=b.y; z+=b.z; return (*this); }
168 //! Soustrait \a b au triplet
169 constexpr ARCCORE_HOST_DEVICE Real3& sub(Real3 b) { x-=b.x; y-=b.y; z-=b.z; return (*this); }
170 //! Multiple chaque composante du triplet par la composant correspondant de \a b
171 constexpr ARCCORE_HOST_DEVICE Real3& mul(Real3 b) { x*=b.x; y*=b.y; z*=b.z; return (*this); }
172 //! Divise chaque composante du triplet par la composant correspondant de \a b
173 constexpr ARCCORE_HOST_DEVICE Real3& div(Real3 b) { x/=b.x; y/=b.y; z/=b.z; return (*this); }
174 //! Ajoute \a b à chaque composante du triplet
175 constexpr ARCCORE_HOST_DEVICE Real3& addSame(Real b) { x+=b; y+=b; z+=b; return (*this); }
176 //! Soustrait \a b à chaque composante du triplet
177 constexpr ARCCORE_HOST_DEVICE Real3& subSame(Real b) { x-=b; y-=b; z-=b; return (*this); }
178 //! Multiplie chaque composante du triplet par \a b
179 constexpr ARCCORE_HOST_DEVICE Real3& mulSame(Real b) { x*=b; y*=b; z*=b; return (*this); }
180 //! Divise chaque composante du triplet par \a b
181 constexpr ARCCORE_HOST_DEVICE Real3& divSame(Real b) { x/=b; y/=b; z/=b; return (*this); }
182 //! Ajoute \a b au triplet.
183 constexpr ARCCORE_HOST_DEVICE Real3& operator+= (Real3 b) { return add(b); }
184 //! Soustrait \a b au triplet
185 constexpr ARCCORE_HOST_DEVICE Real3& operator-= (Real3 b) { return sub(b); }
186 //! Multiple chaque composante du triplet par la composant correspondant de \a b
187 constexpr ARCCORE_HOST_DEVICE Real3& operator*= (Real3 b) { return mul(b); }
188 //! Multiple chaque composante du triplet par le réel \a b
189 constexpr ARCCORE_HOST_DEVICE void operator*= (Real b) { x*=b; y*=b; z*=b; }
190 //! Divise chaque composante du triplet par la composant correspondant de \a b
191 constexpr ARCCORE_HOST_DEVICE Real3& operator/= (Real3 b) { return div(b); }
192 //! Divise chaque composante du triplet par le réel \a b
193 constexpr ARCCORE_HOST_DEVICE void operator/= (Real b) { x/=b; y/=b; z/=b; }
194 //! Créé un triplet qui vaut ce triplet ajouté à \a b
195 constexpr ARCCORE_HOST_DEVICE Real3 operator+ (Real3 b) const { return Real3(x+b.x,y+b.y,z+b.z); }
196 //! Créé un triplet qui vaut \a b soustrait de ce triplet
197 constexpr ARCCORE_HOST_DEVICE Real3 operator- (Real3 b) const { return Real3(x-b.x,y-b.y,z-b.z); }
198 //! Créé un triplet opposé au triplet actuel
199 constexpr ARCCORE_HOST_DEVICE Real3 operator-() const { return Real3(-x,-y,-z); }
200 /*!
201 * \brief Créé un triplet qui vaut ce triplet dont chaque composant a été
202 * multipliée par la composante correspondante de \a b.
203 */
204 constexpr ARCCORE_HOST_DEVICE Real3 operator*(Real3 b) const { return Real3(x*b.x,y*b.y,z*b.z); }
205 /*!
206 * \brief Créé un triplet qui vaut ce triplet dont chaque composant a été divisée
207 * par la composante correspondante de \a b.
208 */
209 constexpr ARCCORE_HOST_DEVICE Real3 operator/(Real3 b) const { return Real3(x/b.x,y/b.y,z/b.z); }
210
211 /*!
212 * \brief Normalise le triplet.
213 *
214 * Si le triplet est non nul, divise chaque composante par la norme du triplet
215 * (abs()), de telle sorte qu'après l'appel à cette méthode, normL2() vaux \a 1.
216 * Si le triplet est nul, ne fait rien.
217 */
218 Real3& normalize()
219 {
220 Real d = normL2();
221 if (!math::isZero(d))
222 divSame(d);
223 return (*this);
224 }
225
226 /*!
227 * \brief Compare composant pas composante l'instance courante à \a b.
228 *
229 * \retval true si this.x==b.x et this.y==b.y et this.z==b.z.
230 * \retval false sinon.
231 */
232 constexpr ARCCORE_HOST_DEVICE bool operator==(Real3 b) const
233 {
234 return _eq(x,b.x) && _eq(y,b.y) && _eq(z,b.z);
235 }
236 /*!
237 * \brief Compare deux triplets.
238 * Pour la notion d'égalité, voir operator==()
239 * \retval true si les deux triplets sont différents,
240 * \retval false sinon.
241 */
242 constexpr ARCCORE_HOST_DEVICE bool operator!=(Real3 b) const { return !operator==(b); }
243
244 private:
245
246 /*!
247 * \brief Compare les valeurs de \a a et \a b avec le comparateur TypeEqualT
248 * \retval true si \a a et \a b sont égaux,
249 * \retval false sinon.
250 */
251 constexpr ARCCORE_HOST_DEVICE static bool _eq(Real a,Real b)
252 {
253 return math::isEqual(a,b);
254 }
255 //! Retourne la racine carrée de \a a
256 ARCCORE_HOST_DEVICE static Real _sqrt(Real a)
257 {
258 return math::sqrt(a);
259 }
260};

297/*!
298 * \brief Opérateur de comparaison.
299 *
300 * Cet opérateur permet de trier les Real3 pour les utiliser par exemple
301 * dans les std::set
302 */
303constexpr ARCCORE_HOST_DEVICE inline bool
304operator<(Real3 v1,Real3 v2)
305{
306 if (v1.x==v2.x){
307 if (v1.y==v2.y)
308 return v1.z<v2.z;
309 else
310 return v1.y<v2.y;
311 }
312 return (v1.x<v2.x);
313}
