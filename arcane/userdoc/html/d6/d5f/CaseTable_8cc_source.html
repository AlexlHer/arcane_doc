165CaseTable::eError CaseTable::
166_setValue(Integer index,const String& value_str)
167{
168 SmallVariant variant(value_str);
169 eValueType value_type(valueType());
170 if (value_type==ValueReal){
171 Real v = 0.;
172 if (builtInGetValue(v,value_str)){
173 return ErrCanNotConvertValueToRightType;
174 }
175 variant.setValueAll(v);
176 }
177 else if (value_type==ValueInteger){
178 Integer v = 0;
179 if (builtInGetValue(v,value_str)){
180 return ErrCanNotConvertValueToRightType;
181 }
182 variant.setValueAll(v);
183 }
184 else if (value_type==ValueBool){
185 bool v = 0;
186 if (builtInGetValue(v,value_str)){
187 return ErrCanNotConvertValueToRightType;
188 }
189 variant.setValueAll(v);
190 }
191
192 if (index>=m_value_list.size()){
193 m_value_list.add(variant);
194 }
195 else
196 m_value_list[index] = variant;
197
198 return ErrNo;
199}

233void CaseTable::
234insertElement(Integer id)
235{
236 // Ajoute un élément à la fin.
237 Integer n = nbElement();
238 if (n==0)
239 return;
240 if (id>=n)
241 id = n-1;
242
243
244 String param_str;
245 m_param_list->toString(n-1,param_str);
246 m_param_list->appendValue(param_str);
247
248 //SmallVariant value_str(m_value_list[n-1]);
249 m_value_list.add(m_value_list[n-1]);
250
251 for( Integer i=n; i>id; --i ){
252 m_param_list->toString(i-1,param_str);
253 m_param_list->setValue(i,param_str);
254 m_value_list[i] = m_value_list[i-1];
255 }
256}

310{
311 return (Real)(current_value + (next_value-current_value)*t );

336/*!
337 * \brief Calcul la valeur de la courbe pour le paramètre \a param.
338 * La valeur est stockée dans \a value.
339 * \retval false en cas de succès
340 * \retval true si erreur
341 */
342template<typename ParamType,typename ValueType> void CaseTable::
343_findValue(ParamType param,ValueType& avalue) const
344{
345 _applyParamTransform(param);
346
347 // On suppose que les éléments de la table sont rangés par paramètres
348 // croissants (begin[i+1]>begin[i])
349 Integer nb_elem = nbElement();
350
351 for( Integer i=0; i<nb_elem; ++i ){
352 const String& current_value_str = m_value_list[i].asString();
353 ParamType current_begin_value;
354 m_param_list->value(i,current_begin_value);
355
356 // Si dernier élément, on prend la valeur de celui-ci
357 if ((i+1)==nb_elem){
358 _verboseBuiltInGetValue(this,i,avalue,current_value_str);
359 return;
360 }
361
362 // Tout d'abord, regarde si le paramètre n'est pas
363 // égal ou inférieur à la borne actuelle.
364 if (math::isEqual(current_begin_value,param) || param<current_begin_value ){
365 _verboseBuiltInGetValue(this,i,avalue,current_value_str);
366 return;
367 }
368
369 const String& next_value_str = m_value_list[i+1].asString();
370 ParamType next_begin_value;
371 m_param_list->value(i+1,next_begin_value);
372
373 // Regarde si le paramètre n'est pas égal à la borne de l'élément suivant.
374 if (math::isEqual(next_begin_value,param)){
375 _verboseBuiltInGetValue(this,i,avalue,next_value_str);
376 return;
377 }
378
379 // Regarde si le paramètre est compris entre l'élément courant et le suivant.
380 if (param>current_begin_value && param<next_begin_value){
381 ValueType current_value = ValueType();
382 ValueType next_value = ValueType();
383 _verboseBuiltInGetValue(this,i,current_value,current_value_str);
384 if (m_curve_type==CurveConstant){
385 avalue = current_value;
386 return;
387 }
388 if (m_curve_type==CurveLinear){
389 _verboseBuiltInGetValue(this,i,next_value,next_value_str);
390 ParamType diff = next_begin_value - current_begin_value;
391 if (math::isZero(diff))
392 ARCANE_FATAL("Table '{0}' index={1} : DeltaX==0.0",name(),i);
393 ParamType t = (param - current_begin_value) / diff;
394 avalue = _curveLinear<ParamType>(current_value,next_value,t);
395 return;
396 }
397 }
398 }
399}
